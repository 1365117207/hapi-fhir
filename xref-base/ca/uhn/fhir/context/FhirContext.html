57  /**
58   * The FHIR context is the central starting point for the use of the HAPI FHIR API. It should be created once, and then
59   * used as a factory for various other types of objects (parsers, clients, etc.).
60   * 
61   * <p>
62   * Important usage notes:
63   * </p>
64   * <ul>
65   * <li>Thread safety: <b>This class is thread safe</b> and may be shared between multiple processing threads.</li>
66   * <li>
67   * Performance: <b>This class is expensive</b> to create, as it scans every resource class it needs to parse or encode
68   * to build up an internal model of those classes. For that reason, you should try to create one FhirContext instance
69   * which remains for the life of your application and reuse that instance. Note that it will not cause problems to
70   * create multiple instances (ie. resources originating from one FhirContext may be passed to parsers originating from
71   * another) but you will incur a performance penalty if a new FhirContext is created for every message you parse/encode.
72   * </li>
73   * </ul>
74   */
75  public class FhirContext {

115 	private FhirContext(FhirVersionEnum theVersion, Collection<Class<? extends IBaseResource>> theResourceTypes) {
116 		if (theVersion != null) {
117 			if (!theVersion.isPresentOnClasspath()) {
118 				throw new IllegalStateException(getLocalizer().getMessage(FhirContext.class, "noStructuresForSpecifiedVersion", theVersion.name()));
119 			}
120 			myVersion = theVersion.getVersionImplementation();
121 		} else if (FhirVersionEnum.DSTU1.isPresentOnClasspath()) {
122 			myVersion = FhirVersionEnum.DSTU1.getVersionImplementation();
123 		} else if (FhirVersionEnum.DSTU2.isPresentOnClasspath()) {
124 			myVersion = FhirVersionEnum.DSTU2.getVersionImplementation();
125 		} else if (FhirVersionEnum.DSTU2_HL7ORG.isPresentOnClasspath()) {
126 			myVersion = FhirVersionEnum.DSTU2_HL7ORG.getVersionImplementation();
127 		} else if (FhirVersionEnum.DEV.isPresentOnClasspath()) {
128 			myVersion = FhirVersionEnum.DEV.getVersionImplementation();
129 		} else {
130 			throw new IllegalStateException(getLocalizer().getMessage(FhirContext.class, "noStructures"));
131 		}
132 
133 		if (theVersion == null) {
134 			ourLog.info("Creating new FhirContext with auto-detected version [{}]. It is recommended to explicitly select a version for future compatibility by invoking FhirContext.forDstuX()", myVersion.getVersion().name());
135 		} else {
136 			ourLog.info("Creating new FHIR context for FHIR version [{}]", myVersion.getVersion().name());
137 		}
138 
139 		scanResourceTypes(toElementList(theResourceTypes));
140 	}

198 	@SuppressWarnings("unchecked")
199 	public RuntimeResourceDefinition getResourceDefinition(Class<? extends IBaseResource> theResourceType) {
200 		if (theResourceType == null) {
201 			throw new NullPointerException("theResourceType can not be null");
202 		}
203 		if (Modifier.isAbstract(theResourceType.getModifiers())) {
204 			throw new IllegalArgumentException("Can not scan abstract or interface class (resource definitions must be concrete classes): " + theResourceType.getName());
205 		}
206 
207 		RuntimeResourceDefinition retVal = (RuntimeResourceDefinition) myClassToElementDefinition.get(theResourceType);
208 		if (retVal == null) {
209 			retVal = scanResourceType((Class<? extends IResource>) theResourceType);
210 		}
211 		return retVal;
212 	}

214 	public RuntimeResourceDefinition getResourceDefinition(FhirVersionEnum theVersion, String theResourceName) {
215 		Validate.notNull(theVersion, "theVersion can not be null");
216 
217 		if (theVersion.equals(myVersion.getVersion())) {
218 			return getResourceDefinition(theResourceName);
219 		}
220 
221 		Map<String, Class<? extends IBaseResource>> nameToType = myVersionToNameToResourceType.get(theVersion);
222 		if (nameToType == null) {
223 			nameToType = new HashMap<String, Class<? extends IBaseResource>>();
224 			ModelScanner.scanVersionPropertyFile(null, nameToType, theVersion);
225 
226 			Map<FhirVersionEnum, Map<String, Class<? extends IBaseResource>>> newVersionToNameToResourceType = new HashMap<FhirVersionEnum, Map<String, Class<? extends IBaseResource>>>();
227 			newVersionToNameToResourceType.putAll(myVersionToNameToResourceType);
228 			newVersionToNameToResourceType.put(theVersion, nameToType);
229 			myVersionToNameToResourceType = newVersionToNameToResourceType;
230 		}
231 
232 		Class<? extends IBaseResource> resourceType = nameToType.get(theResourceName.toLowerCase());
233 		if (resourceType == null) {
234 			throw new DataFormatException(createUnknownResourceNameError(theResourceName, theVersion));
235 		}
236 
237 		return getResourceDefinition(resourceType);
238 	}

249 	 * Returns the scanned runtime model for the given type. This is an advanced feature which is generally only needed
250 	 * for extending the core library.
251 	 */
252 	@SuppressWarnings("unchecked")
253 	public RuntimeResourceDefinition getResourceDefinition(String theResourceName) {
254 		Validate.notBlank(theResourceName, "theResourceName must not be blank");
255 		
256 		String resourceName = theResourceName;
257 
258 		/*
259 		 * TODO: this is a bit of a hack, really we should have a translation table based on a property file or
260 		 * something so that we can detect names like diagnosticreport
261 		 */
262 		if (Character.isLowerCase(resourceName.charAt(0))) {
263 			resourceName = WordUtils.capitalize(resourceName);
264 		}
265 
266 		Validate.notBlank(resourceName, "Resource name must not be blank");
267 
268 		RuntimeResourceDefinition retVal = myNameToResourceDefinition.get(resourceName);
269 
270 		if (retVal == null) {
271 			Class<? extends IBaseResource> clazz = myNameToResourceType.get(resourceName.toLowerCase());
272 			if (clazz == null) {
273 				throw new DataFormatException(createUnknownResourceNameError(resourceName, myVersion.getVersion()));
274 			}
275 			if (IBaseResource.class.isAssignableFrom(clazz)) {
276 				retVal = scanResourceType((Class<? extends IResource>) clazz);
277 			}
278 		}
279 
280 		return retVal;
281 	}

415 	private BaseRuntimeElementDefinition<?> scanDatatype(Class<? extends IElement> theResourceType) {
416 		ArrayList<Class<? extends IElement>> resourceTypes = new ArrayList<Class<? extends IElement>>();
417 		resourceTypes.add(theResourceType);
418 		Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> defs = scanResourceTypes(resourceTypes);
419 		return defs.get(theResourceType);
420 	}

422 	private RuntimeResourceDefinition scanResourceType(Class<? extends IResource> theResourceType) {
423 		ArrayList<Class<? extends IElement>> resourceTypes = new ArrayList<Class<? extends IElement>>();
424 		resourceTypes.add(theResourceType);
425 		Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> defs = scanResourceTypes(resourceTypes);
426 		return (RuntimeResourceDefinition) defs.get(theResourceType);
427 	}

429 	private Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> scanResourceTypes(Collection<Class<? extends IElement>> theResourceTypes) {
430 		ModelScanner scanner = new ModelScanner(this, myVersion.getVersion(), myClassToElementDefinition, theResourceTypes);
431 		if (myRuntimeChildUndeclaredExtensionDefinition == null) {
432 			myRuntimeChildUndeclaredExtensionDefinition = scanner.getRuntimeChildUndeclaredExtensionDefinition();
433 		}
434 
435 		Map<String, BaseRuntimeElementDefinition<?>> nameToElementDefinition = new HashMap<String, BaseRuntimeElementDefinition<?>>();
436 		nameToElementDefinition.putAll(myNameToElementDefinition);
437 		nameToElementDefinition.putAll(scanner.getNameToElementDefinitions());
438 
439 		Map<String, RuntimeResourceDefinition> nameToResourceDefinition = new HashMap<String, RuntimeResourceDefinition>();
440 		nameToResourceDefinition.putAll(myNameToResourceDefinition);
441 		nameToResourceDefinition.putAll(scanner.getNameToResourceDefinition());
442 		
443 		Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> classToElementDefinition = new HashMap<Class<? extends IBase>, BaseRuntimeElementDefinition<?>>();
444 		classToElementDefinition.putAll(myClassToElementDefinition);
445 		classToElementDefinition.putAll(scanner.getClassToElementDefinitions());
446 
447 		Map<String, RuntimeResourceDefinition> idToElementDefinition = new HashMap<String, RuntimeResourceDefinition>();
448 		idToElementDefinition.putAll(myIdToResourceDefinition);
449 		idToElementDefinition.putAll(scanner.getIdToResourceDefinition());
450 
451 		myNameToElementDefinition = nameToElementDefinition;
452 		myClassToElementDefinition = classToElementDefinition;
453 		myIdToResourceDefinition = idToElementDefinition;
454 		myNameToResourceDefinition = nameToResourceDefinition;
455 
456 		myNameToResourceType = scanner.getNameToResourceType();
457 
458 		return classToElementDefinition;
459 	}

487 	private List<Class<? extends IElement>> toElementList(Collection<Class<? extends IBaseResource>> theResourceTypes) {
488 		if (theResourceTypes == null) {
489 			return null;
490 		}
491 		List<Class<? extends IElement>> resTypes = new ArrayList<Class<? extends IElement>>();
492 		for (Class<? extends IBaseResource> next : theResourceTypes) {
493 			resTypes.add((Class<? extends IElement>) next);
494 		}
495 		return resTypes;
496 	}

532 	@SuppressWarnings("unchecked")
533 	private static List<Class<? extends IBaseResource>> toCollection(Class<?>[] theResourceTypes) {
534 		ArrayList<Class<? extends IBaseResource>> retVal = new ArrayList<Class<? extends IBaseResource>>(1);
535 		for (Class<?> clazz : theResourceTypes) {
536 			if (!IResource.class.isAssignableFrom(clazz)) {
537 				throw new IllegalArgumentException(clazz.getCanonicalName() + " is not an instance of " + IResource.class.getSimpleName());
538 			}
539 			retVal.add((Class<? extends IResource>) clazz);
540 		}
541 		return retVal;
542 	}
