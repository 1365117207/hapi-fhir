23  import ca.uhn.fhir.model.api.IResource;
24  import ca.uhn.fhir.model.api.annotation.ProvidesResources;
25  
26  /**
27   * Scans a class tagged with {@code ProvidesResources} and adds any resources listed to its FhirContext's resource
28   * definition list. This makes the profile generator find the classes.
29   *
30   * @see ca.uhn.fhir.model.api.annotation.ProvidesResources
31   */
32  public class ProvidedResourceScanner {
33      private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(ModelScanner.class);
34      private FhirContext myContext;
35  
36      /**
37       * Constructor
38       * @param theContext - context whose resource definition list is to be updated by the scanner
39       */
40      public ProvidedResourceScanner(FhirContext theContext) {
41          myContext = theContext;
42      }
43  
44      /**
45       * If {@code theProvider} is tagged with the {@code ProvidesResources} annotation, this method will add every resource listed
46       * by the {@code resources} method.
47       * <p>
48       * Notes:
49       * </p>
50       * <ul>
51       * <li>if {@code theProvider} isn't annotated with {@code resources} nothing is done; it's expected that most RestfulServers and
52       * ResourceProviders won't be annotated.</li>
53       * <li>any object listed in {@code resources} that doesn't implement {@code IResource} will generate a warning in the log.</li>
54       * </ul>
55       *
56       * @param theProvider - Normally, either a {@link ca.uhn.fhir.rest.server.RestfulServer} or a {@link ca.uhn.fhir.rest.server.IResourceProvider}
57       *                    that might be annotated with {@link ca.uhn.fhir.model.api.annotation.ProvidesResources}
58       */
59      public void scanForProvidedResources(Object theProvider) {
60          ProvidesResources annotation = theProvider.getClass().getAnnotation(ProvidesResources.class);
61          if (annotation == null)
62              return;
63          for (Class clazz : annotation.resources()) {
64              if (IResource.class.isAssignableFrom(clazz)) {
65                  myContext.getResourceDefinition(clazz);
66              } else {
67                  ourLog.warn(clazz.getSimpleName() + "is not assignable from IResource");
68              }
69          }
70      }
71  }
