140 		if (theCommentsToAdd.size() > 0) {
141 			theListToAddTo.ensureCapacity(valueIdx);
142 			while (theListToAddTo.size() <= valueIdx) {
143 				theListToAddTo.add(null);
144 			}
145 			if (theListToAddTo.get(valueIdx) == null) {
146 				theListToAddTo.set(valueIdx, new ArrayList<String>());
147 			}
148 			theListToAddTo.get(valueIdx).addAll(theCommentsToAdd);
149 			return true;
150 		} else {
151 			return false;
152 		}
153 	}

213 	public <T extends IBaseResource> T doParseResource(Class<T> theResourceType, Reader theReader) {
214 			JsonObject object = parse(theReader);
215 
216 			JsonPrimitive resourceTypeObj = object.getAsJsonPrimitive("resourceType");
217 			if (resourceTypeObj == null || isBlank(resourceTypeObj.getAsString())) {
218 				throw new DataFormatException("Invalid JSON content detected, missing required element: 'resourceType'");
219 			}
220 			
221 			String resourceType = resourceTypeObj.getAsString();
222 
223 			ParserState<? extends IBaseResource> state = ParserState.getPreResourceInstance(this, theResourceType, myContext, true, getErrorHandler());
224 			state.enteringNewElement(null, resourceType);
225 
226 			parseChildren(object, state);
227 
228 			state.endingElement();
229 			state.endingElement();
230 
231 			@SuppressWarnings("unchecked")
232 			T retVal = (T) state.getObject();
233 
234 			return retVal;
235 	}

237 	private JsonObject parse(Reader theReader) {
238 
239 		PushbackReader pbr = new PushbackReader(theReader);
240 		JsonObject object;
241 		try {
242 			while(true) {
243 				int nextInt;
244 					nextInt = pbr.read();
245 				if (nextInt == -1) {
246 					throw new DataFormatException("Did not find any content to parse");
247 				}
248 				if (nextInt == '{') {
249 					pbr.unread('{');
250 					break;
251 				}
252 				if (Character.isWhitespace(nextInt)) {
253 					continue;
254 				}
255 				throw new DataFormatException("Content does not appear to be FHIR JSON, first non-whitespace character was: '" + (char)nextInt + "' (must be '{')");
256 			}
257 		
258 			Gson gson = new GsonBuilder().disableHtmlEscaping().create();
259 		
260 			object = gson.fromJson(pbr, JsonObject.class);
261 		} catch (Exception e) {
262 			throw new DataFormatException("Failed to parse JSON content, error was: " + e.getMessage(), e);
263 		}
264 		
265 		return object;
266 	}

268 	private void encodeBundleToWriterInDstu1Format(Bundle theBundle, JsonWriter theEventWriter) throws IOException {
269 		theEventWriter.beginObject();
270 
271 		write(theEventWriter, "resourceType", "Bundle");
272 
273 		writeTagWithTextNode(theEventWriter, "title", theBundle.getTitle());
274 		writeTagWithTextNode(theEventWriter, "id", theBundle.getBundleId());
275 		writeOptionalTagWithTextNode(theEventWriter, "updated", theBundle.getUpdated());
276 
277 		boolean linkStarted = false;
278 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "self", theBundle.getLinkSelf(), linkStarted);
279 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "first", theBundle.getLinkFirst(), linkStarted);
280 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "previous", theBundle.getLinkPrevious(), linkStarted);
281 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "next", theBundle.getLinkNext(), linkStarted);
282 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "last", theBundle.getLinkLast(), linkStarted);
283 		linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "fhir-base", theBundle.getLinkBase(), linkStarted);
284 		if (linkStarted) {
285 			theEventWriter.endArray();
286 		}
287 
288 		writeCategories(theEventWriter, theBundle.getCategories());
289 
290 		writeOptionalTagWithTextNode(theEventWriter, "totalResults", theBundle.getTotalResults());
291 
292 		writeAuthor(theBundle, theEventWriter);
293 
294 		beginArray(theEventWriter, "entry");
295 		for (BundleEntry nextEntry : theBundle.getEntries()) {
296 			theEventWriter.beginObject();
297 
298 			boolean deleted = nextEntry.getDeletedAt() != null && nextEntry.getDeletedAt().isEmpty() == false;
299 			if (deleted) {
300 				writeTagWithTextNode(theEventWriter, "deleted", nextEntry.getDeletedAt());
301 			}
302 			writeTagWithTextNode(theEventWriter, "title", nextEntry.getTitle());
303 			writeTagWithTextNode(theEventWriter, "id", nextEntry.getId());
304 
305 			linkStarted = false;
306 			linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "self", nextEntry.getLinkSelf(), linkStarted);
307 			linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "alternate", nextEntry.getLinkAlternate(), linkStarted);
308 			linkStarted = writeAtomLinkInDstu1Format(theEventWriter, "search", nextEntry.getLinkSearch(), linkStarted);
309 			if (linkStarted) {
310 				theEventWriter.endArray();
311 			}
312 
313 			writeOptionalTagWithTextNode(theEventWriter, "updated", nextEntry.getUpdated());
314 			writeOptionalTagWithTextNode(theEventWriter, "published", nextEntry.getPublished());
315 
316 			writeCategories(theEventWriter, nextEntry.getCategories());
317 
318 			writeAuthor(nextEntry, theEventWriter);
319 
320 			IResource resource = nextEntry.getResource();
321 			if (resource != null && !resource.isEmpty() && !deleted) {
322 				RuntimeResourceDefinition resDef = myContext.getResourceDefinition(resource);
323 				encodeResourceToJsonStreamWriter(resDef, resource, theEventWriter, "content", false);
324 			}
325 
326 			if (nextEntry.getSummary().isEmpty() == false) {
327 				write(theEventWriter, "summary", nextEntry.getSummary().getValueAsString());
328 			}
329 
330 			theEventWriter.endObject(); // entry object
331 		}
332 		theEventWriter.endArray(); // entry array
333 
334 		theEventWriter.endObject(); // resource object
335 	}

434 	private void encodeChildElementToStreamWriter(RuntimeResourceDefinition theResDef, IBaseResource theResource, JsonWriter theEventWriter, IBase theNextValue, BaseRuntimeElementDefinition<?> theChildDef, String theChildName, boolean theContainedResource, CompositeChildElement theChildElem,
435 			boolean theForceEmpty) throws IOException {
436 
437 		switch (theChildDef.getChildType()) {
438 		case ID_DATATYPE: {
439 			IIdType value = (IIdType) theNextValue;
440 			String encodedValue = "id".equals(theChildName) ? value.getIdPart() : value.getValue();
441 			if (isBlank(encodedValue)) {
442 				break;
443 			}
444 			if (theChildName != null) {
445 				write(theEventWriter, theChildName, encodedValue);
446 			} else {
447 				theEventWriter.value(encodedValue);
448 			}
449 			break;
450 		}
451 		case PRIMITIVE_DATATYPE: {
452 			final IPrimitiveType<?> value = (IPrimitiveType<?>) theNextValue;
453 			if (isBlank(value.getValueAsString())) {
454 				if (theForceEmpty) {
455 					theEventWriter.nullValue();
456 				}
457 				break;
458 			}
459 
460 			if (value instanceof IBaseIntegerDatatype) {
461 				if (theChildName != null) {
462 					write(theEventWriter, theChildName, ((IBaseIntegerDatatype) value).getValue());
463 				} else {
464 					theEventWriter.value(((IBaseIntegerDatatype) value).getValue());
465 				}
466 			} else if (value instanceof IBaseDecimalDatatype) {
467 				BigDecimal decimalValue = ((IBaseDecimalDatatype) value).getValue();
468 				decimalValue = new BigDecimal(decimalValue.toString()) {
469 					private static final long serialVersionUID = 1L;
470 
471 					@Override
472 					public String toString() {
473 						return value.getValueAsString();
474 					}
475 				};
476 				if (theChildName != null) {
477 					write(theEventWriter, theChildName, decimalValue);
478 				} else {
479 					theEventWriter.value(decimalValue);
480 				}
481 			} else if (value instanceof IBaseBooleanDatatype) {
482 				if (theChildName != null) {
483 					write(theEventWriter, theChildName, ((IBaseBooleanDatatype) value).getValue());
484 				} else {
485 					Boolean booleanValue = ((IBaseBooleanDatatype) value).getValue();
486 					if (booleanValue != null) {
487 						theEventWriter.value(booleanValue.booleanValue());
488 					}
489 				}
490 			} else {
491 				String valueStr = value.getValueAsString();
492 				if (theChildName != null) {
493 					write(theEventWriter, theChildName, valueStr);
494 				} else {
495 					theEventWriter.value(valueStr);
496 				}
497 			}
498 			break;
499 		}
500 		case RESOURCE_BLOCK:
501 		case COMPOSITE_DATATYPE: {
502 			if (theChildName != null) {
503 				theEventWriter.name(theChildName);
504 				theEventWriter.beginObject();
505 			} else {
506 				theEventWriter.beginObject();
507 			}
508 			encodeCompositeElementToStreamWriter(theResDef, theResource, theNextValue, theEventWriter, theContainedResource, theChildElem);
509 			theEventWriter.endObject();
510 			break;
511 		}
512 		case CONTAINED_RESOURCE_LIST:
513 		case CONTAINED_RESOURCES: {
514 			/*
515 			 * Disabled per #103 ContainedDt value = (ContainedDt) theNextValue; for (IResource next :
516 			 * value.getContainedResources()) { if (getContainedResources().getResourceId(next) != null) { continue; }
517 			 * encodeResourceToJsonStreamWriter(theResDef, next, theWriter, null, true,
518 			 * fixContainedResourceId(next.getId().getValue())); }
519 			 */
520 			List<IBaseResource> containedResources = getContainedResources().getContainedResources();
521 			if (containedResources.size() > 0) {
522 				beginArray(theEventWriter, theChildName);
523 
524 				for (IBaseResource next : containedResources) {
525 					IIdType resourceId = getContainedResources().getResourceId(next);

786 						}
787 					}

794 				}

1030 			ISupportsUndeclaredExtensions element = (ISupportsUndeclaredExtensions) theElement;
1031 			List<ExtensionDt> ext = element.getUndeclaredExtensions();
1032 			for (ExtensionDt next : ext) {
1033 				if (next == null || next.isEmpty()) {
1034 					continue;
1035 				}
1036 				extensions.add(new HeldExtension(next, false, theChildElem));
1037 			}
1038 
1039 			ext = element.getUndeclaredModifierExtensions();
1040 			for (ExtensionDt next : ext) {
1041 				if (next == null || next.isEmpty()) {
1042 					continue;
1043 				}
1044 				modifierExtensions.add(new HeldExtension(next, true, theChildElem));
1045 			}
1046 		} else {
1047 			if (theElement instanceof IBaseHasExtensions) {
1048 				IBaseHasExtensions element = (IBaseHasExtensions) theElement;
1049 				List<? extends IBaseExtension<?, ?>> ext = element.getExtension();
1050 				for (IBaseExtension<?, ?> next : ext) {
1051 					if (next == null || (ElementUtil.isEmpty(next.getValue()) && next.getExtension().isEmpty())) {
1052 						continue;
1053 					}
1054 					extensions.add(new HeldExtension(next, false, theChildElem));
1055 				}
1056 			}
1057 			if (theElement instanceof IBaseHasModifierExtensions) {
1058 				IBaseHasModifierExtensions element = (IBaseHasModifierExtensions) theElement;
1059 				List<? extends IBaseExtension<?, ?>> ext = element.getModifierExtension();
1060 				for (IBaseExtension<?, ?> next : ext) {
1061 					if (next == null || next.isEmpty()) {
