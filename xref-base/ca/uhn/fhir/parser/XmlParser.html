145 		XMLStreamWriter eventWriter;
146 		eventWriter = XmlUtil.createXmlStreamWriter(theWriter);
147 		eventWriter = decorateStreamWriter(eventWriter);
148 		return eventWriter;
149 	}

151 	private XMLStreamWriter decorateStreamWriter(XMLStreamWriter eventWriter) {
152 		if (myPrettyPrint) {
153 			PrettyPrintWriterWrapper retVal = new PrettyPrintWriterWrapper(eventWriter);
154 			return retVal;
155 		} else {
156 			NonPrettyPrintWriterWrapper retVal = new NonPrettyPrintWriterWrapper(eventWriter);
157 			return retVal;

161 	private <T> T doXmlLoop(XMLEventReader streamReader, ParserState<T> parserState) {
162 		ourLog.trace("Entering XML parsing loop with state: {}", parserState);
163 
164 		try {
165 			while (streamReader.hasNext()) {
166 				XMLEvent nextEvent = streamReader.nextEvent();
167 				try {
168 					if (nextEvent.isStartElement()) {
169 						StartElement elem = nextEvent.asStartElement();
170 
171 						String namespaceURI = elem.getName().getNamespaceURI();
172 
173 						if ("extension".equals(elem.getName().getLocalPart())) {
174 							Attribute urlAttr = elem.getAttributeByName(new QName("url"));
175 							if (urlAttr == null || isBlank(urlAttr.getValue())) {
176 								throw new DataFormatException("Extension element has no 'url' attribute");
177 							}
178 							parserState.enteringNewElementExtension(elem, urlAttr.getValue(), false);
179 						} else if ("modifierExtension".equals(elem.getName().getLocalPart())) {
180 							Attribute urlAttr = elem.getAttributeByName(new QName("url"));
181 							if (urlAttr == null || isBlank(urlAttr.getValue())) {
182 								throw new DataFormatException("Extension element has no 'url' attribute");
183 							}
184 							parserState.enteringNewElementExtension(elem, urlAttr.getValue(), true);
185 						} else {
186 
187 							String elementName = elem.getName().getLocalPart();
188 							parserState.enteringNewElement(namespaceURI, elementName);
189 
190 						}
191 
192 						@SuppressWarnings("unchecked")
193 						Iterator<Attribute> attributes = elem.getAttributes();
194 						for (Iterator<Attribute> iter = attributes; iter.hasNext();) {
195 							Attribute next = iter.next();
196 							parserState.attributeValue(next.getName().getLocalPart(), next.getValue());
197 						}
198 
199 					} else if (nextEvent.isAttribute()) {
200 						Attribute elem = (Attribute) nextEvent;
201 						String name = (elem.getName().getLocalPart());
202 						parserState.attributeValue(name, elem.getValue());
203 					} else if (nextEvent.isEndElement()) {
204 						parserState.endingElement();
205 						if (parserState.isComplete()) {
206 							return parserState.getObject();
207 						}
208 					} else if (nextEvent.isCharacters()) {
209 						parserState.string(nextEvent.asCharacters().getData());
210 					}
211 
212 					parserState.xmlEvent(nextEvent);
213 
214 				} catch (DataFormatException e) {
215 					throw new DataFormatException("DataFormatException at [" + nextEvent.getLocation().toString() + "]: " + e.getMessage(), e);
216 				}
217 			}
218 			return null;
219 		} catch (XMLStreamException e) {
220 			throw new DataFormatException(e);
221 		}
222 	}

225 	public String encodeBundleToString(Bundle theBundle) throws DataFormatException {
226 		StringWriter stringWriter = new StringWriter();
227 		try {
228 			encodeBundleToWriter(theBundle, stringWriter);
229 		} catch (IOException e) {
230 			throw new InternalErrorException("IOException writing to StringWriter - Should not happen", e);
231 		}
232 
233 		return stringWriter.toString();
234 	}

237 	public void doEncodeBundleToWriter(Bundle theBundle, Writer theWriter) throws DataFormatException {
238 		try {
239 			XMLStreamWriter eventWriter = createXmlWriter(theWriter);
240 			if (myContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
241 				encodeBundleToWriterDstu2(theBundle, eventWriter);
242 			} else {
243 				encodeBundleToWriterDstu1(theBundle, eventWriter);
244 			}
245 		} catch (XMLStreamException e) {
246 			throw new ConfigurationException("Failed to initialize STaX event factory", e);
247 		}
248 	}

250 	private void encodeBundleToWriterDstu1(Bundle theBundle, XMLStreamWriter eventWriter) throws XMLStreamException {
251 		eventWriter.writeStartElement("feed");
252 		eventWriter.writeDefaultNamespace(ATOM_NS);
253 
254 		writeTagWithTextNode(eventWriter, "title", theBundle.getTitle());
255 		writeTagWithTextNode(eventWriter, "id", theBundle.getBundleId());
256 
257 		writeAtomLink(eventWriter, Constants.LINK_SELF, theBundle.getLinkSelf());
258 		writeAtomLink(eventWriter, Constants.LINK_FIRST, theBundle.getLinkFirst());
259 		writeAtomLink(eventWriter, Constants.LINK_PREVIOUS, theBundle.getLinkPrevious());
260 		writeAtomLink(eventWriter, Constants.LINK_NEXT, theBundle.getLinkNext());
261 		writeAtomLink(eventWriter, Constants.LINK_LAST, theBundle.getLinkLast());
262 		writeAtomLink(eventWriter, Constants.LINK_FHIR_BASE, theBundle.getLinkBase());
263 
264 		if (theBundle.getTotalResults().getValue() != null) {
265 			eventWriter.writeStartElement("os", "totalResults", OPENSEARCH_NS);
266 			eventWriter.writeNamespace("os", OPENSEARCH_NS);
267 			eventWriter.writeCharacters(theBundle.getTotalResults().getValue().toString());
268 			eventWriter.writeEndElement();
269 		}
270 
271 		writeOptionalTagWithTextNode(eventWriter, "updated", theBundle.getUpdated());
272 
273 		if (StringUtils.isNotBlank(theBundle.getAuthorName().getValue())) {
274 			eventWriter.writeStartElement("author");
275 			writeTagWithTextNode(eventWriter, "name", theBundle.getAuthorName());
276 			writeOptionalTagWithTextNode(eventWriter, "uri", theBundle.getAuthorUri());
277 			eventWriter.writeEndElement();
278 		}
279 
280 		writeCategories(eventWriter, theBundle.getCategories());
281 
282 		for (BundleEntry nextEntry : theBundle.getEntries()) {
283 			boolean deleted = false;
284 			if (nextEntry.getDeletedAt() != null && nextEntry.getDeletedAt().isEmpty() == false) {
285 				deleted = true;
286 				eventWriter.writeStartElement("at", "deleted-entry", TOMBSTONES_NS);
287 				eventWriter.writeNamespace("at", TOMBSTONES_NS);
288 
289 				if (nextEntry.getDeletedResourceId().isEmpty()) {
290 					writeOptionalAttribute(eventWriter, "ref", nextEntry.getId().getValueAsString());
291 				} else {
292 					writeOptionalAttribute(eventWriter, "ref", nextEntry.getDeletedResourceId().getValueAsString());
293 				}
294 
295 				writeOptionalAttribute(eventWriter, "when", nextEntry.getDeletedAt().getValueAsString());
296 				if (nextEntry.getDeletedByEmail().isEmpty() == false || nextEntry.getDeletedByName().isEmpty() == false) {
297 					eventWriter.writeStartElement(TOMBSTONES_NS, "by");
298 					if (nextEntry.getDeletedByName().isEmpty() == false) {
299 						eventWriter.writeStartElement(TOMBSTONES_NS, "name");
300 						eventWriter.writeCharacters(nextEntry.getDeletedByName().getValue());
301 						eventWriter.writeEndElement();
302 					}
303 					if (nextEntry.getDeletedByEmail().isEmpty() == false) {
304 						eventWriter.writeStartElement(TOMBSTONES_NS, "email");
305 						eventWriter.writeCharacters(nextEntry.getDeletedByEmail().getValue());
306 						eventWriter.writeEndElement();
307 					}
308 					eventWriter.writeEndElement();
309 				}
310 				if (nextEntry.getDeletedComment().isEmpty() == false) {
311 					eventWriter.writeStartElement(TOMBSTONES_NS, "comment");
312 					eventWriter.writeCharacters(nextEntry.getDeletedComment().getValue());
313 					eventWriter.writeEndElement();
314 				}
315 			} else {
316 				eventWriter.writeStartElement("entry");
317 			}
318 
319 			writeOptionalTagWithTextNode(eventWriter, "title", nextEntry.getTitle());
320 			if (!deleted) {
321 				if (nextEntry.getId().isEmpty() == false) {
322 					writeTagWithTextNode(eventWriter, "id", nextEntry.getId());
323 				} else {
324 					writeTagWithTextNode(eventWriter, "id", nextEntry.getResource().getId());
325 				}
326 			}
327 			writeOptionalTagWithTextNode(eventWriter, "updated", nextEntry.getUpdated());
328 			writeOptionalTagWithTextNode(eventWriter, "published", nextEntry.getPublished());
329 
330 			writeCategories(eventWriter, nextEntry.getCategories());
331 
332 			if (!nextEntry.getLinkSelf().isEmpty()) {
333 				writeAtomLink(eventWriter, "self", nextEntry.getLinkSelf());
334 			}
335 
336 			if (!nextEntry.getLinkAlternate().isEmpty()) {
337 				writeAtomLink(eventWriter, "alternate", nextEntry.getLinkAlternate());
338 			}
339 
340 			if (!nextEntry.getLinkSearch().isEmpty()) {
341 				writeAtomLink(eventWriter, "search", nextEntry.getLinkSearch());
342 			}
343 
344 			IResource resource = nextEntry.getResource();
345 			if (resource != null && !resource.isEmpty() && !deleted) {
346 				eventWriter.writeStartElement("content");
347 				eventWriter.writeAttribute("type", "text/xml");
348 				encodeResourceToXmlStreamWriter(resource, eventWriter, false);
349 				eventWriter.writeEndElement(); // content
350 			} else {
351 				ourLog.debug("Bundle entry contains null resource");
352 			}
353 
354 			if (!nextEntry.getSummary().isEmpty()) {
355 				eventWriter.writeStartElement("summary");
356 				eventWriter.writeAttribute("type", "xhtml");
357 				encodeXhtml(nextEntry.getSummary(), eventWriter);
358 				eventWriter.writeEndElement();
359 			}
360 
361 			eventWriter.writeEndElement(); // entry
362 		}
363 
364 		eventWriter.writeEndElement();
365 		eventWriter.close();
366 	}

368 	private void encodeBundleToWriterDstu2(Bundle theBundle, XMLStreamWriter theEventWriter) throws XMLStreamException {
369 		theEventWriter.writeStartElement("Bundle");
370 		theEventWriter.writeDefaultNamespace(FHIR_NS);
371 
372 		writeOptionalTagWithValue(theEventWriter, "id", theBundle.getId().getIdPart());
373 
374 		InstantDt updated = (InstantDt) theBundle.getResourceMetadata().get(ResourceMetadataKeyEnum.UPDATED);
375 		IdDt bundleId = theBundle.getId();
376 		if (bundleId != null && isNotBlank(bundleId.getVersionIdPart()) || (updated != null && !updated.isEmpty())) {
377 			theEventWriter.writeStartElement("meta");
378 			writeOptionalTagWithValue(theEventWriter, "versionId", bundleId.getVersionIdPart());
379 			if (updated != null) {
380 				writeOptionalTagWithValue(theEventWriter, "lastUpdated", updated.getValueAsString());
381 			}
382 			theEventWriter.writeEndElement();
383 		}
384 
385 		String bundleBaseUrl = theBundle.getLinkBase().getValue();
386 
387 		writeOptionalTagWithValue(theEventWriter, "type", theBundle.getType().getValue());
388 		if (!myContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU2_HL7ORG)) {
389 			writeOptionalTagWithValue(theEventWriter, "base", bundleBaseUrl);
390 		}
391 		writeOptionalTagWithValue(theEventWriter, "total", theBundle.getTotalResults().getValueAsString());
392 
393 		writeBundleResourceLink(theEventWriter, "first", theBundle.getLinkFirst());
394 		writeBundleResourceLink(theEventWriter, "previous", theBundle.getLinkPrevious());
395 		writeBundleResourceLink(theEventWriter, "next", theBundle.getLinkNext());
396 		writeBundleResourceLink(theEventWriter, "last", theBundle.getLinkLast());
397 		writeBundleResourceLink(theEventWriter, "self", theBundle.getLinkSelf());
398 
399 		for (BundleEntry nextEntry : theBundle.getEntries()) {
400 			theEventWriter.writeStartElement("entry");
401 
402 			boolean deleted = false;
403 			if (nextEntry.getDeletedAt() != null && nextEntry.getDeletedAt().isEmpty() == false) {
404 				deleted = true;
405 			}
406 
407 			if (!myContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU2_HL7ORG)) {
408 				writeOptionalTagWithValue(theEventWriter, "base", determineResourceBaseUrl(bundleBaseUrl, nextEntry));
409 			}
410 			
411 			writeBundleResourceLink(theEventWriter, "alternate", nextEntry.getLinkAlternate());
412 
413 			if (myContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU2_HL7ORG)) {
414 				if (nextEntry.getResource() != null && nextEntry.getResource().getId().getBaseUrl() != null) {
415 					writeOptionalTagWithValue(theEventWriter, "fullUrl", nextEntry.getResource().getId().getValue());
416 				}
417 			}
418 			
419 			IResource resource = nextEntry.getResource();
420 			if (resource != null && !resource.isEmpty() && !deleted) {
421 				theEventWriter.writeStartElement("resource");
422 				encodeResourceToXmlStreamWriter(resource, theEventWriter, false);
423 				theEventWriter.writeEndElement(); // content
424 			} else {
425 				ourLog.debug("Bundle entry contains null resource");
426 			}
427 
428 			if (nextEntry.getSearchMode().isEmpty() == false || nextEntry.getScore().isEmpty() == false) {
429 				theEventWriter.writeStartElement("search");
430 				writeOptionalTagWithValue(theEventWriter, "mode", nextEntry.getSearchMode().getValueAsString());
431 				writeOptionalTagWithValue(theEventWriter, "score", nextEntry.getScore().getValueAsString());
432 				theEventWriter.writeEndElement();
433 				// IResource nextResource = nextEntry.getResource();
434 			}
435 
436 			if (nextEntry.getTransactionMethod().isEmpty() == false || nextEntry.getLinkSearch().isEmpty() == false) {
437 				theEventWriter.writeStartElement("transaction");
438 				writeOptionalTagWithValue(theEventWriter, "method", nextEntry.getTransactionMethod().getValue());
439 				writeOptionalTagWithValue(theEventWriter, "url", nextEntry.getLinkSearch().getValue());
440 				theEventWriter.writeEndElement();
441 			}
442 
443 			if (deleted) {
444 				theEventWriter.writeStartElement("deleted");
445 				writeOptionalTagWithValue(theEventWriter, "type", nextEntry.getId().getResourceType());
446 				writeOptionalTagWithValue(theEventWriter, "id", nextEntry.getId().getIdPart());
447 				writeOptionalTagWithValue(theEventWriter, "versionId", nextEntry.getId().getVersionIdPart());
448 				writeOptionalTagWithValue(theEventWriter, "instant", nextEntry.getDeletedAt().getValueAsString());
449 				theEventWriter.writeEndElement();
450 			}
451 
452 			theEventWriter.writeEndElement(); // entry
453 		}
454 
455 		theEventWriter.writeEndElement();
456 		theEventWriter.close();
457 	}

459 	private void encodeChildElementToStreamWriter(IBaseResource theResource, XMLStreamWriter theEventWriter, IBase nextValue, String childName, BaseRuntimeElementDefinition<?> childDef, String theExtensionUrl, boolean theIncludedResource) throws XMLStreamException, DataFormatException {
460 		if (nextValue == null || nextValue.isEmpty()) {
461 			if (isChildContained(childDef, theIncludedResource)) {
462 				// We still want to go in..
463 			} else {
464 				return;
465 			}
466 		}
467 
468 		switch (childDef.getChildType()) {
469 		case ID_DATATYPE: {
470 			IIdType value = (IIdType) nextValue;
471 			String encodedValue = "id".equals(childName) ? value.getIdPart() : value.getValue();
472 			if (value != null) {
473 				theEventWriter.writeStartElement(childName);
474 				theEventWriter.writeAttribute("value", encodedValue);
475 				encodeExtensionsIfPresent(theResource, theEventWriter, nextValue, theIncludedResource);
476 				theEventWriter.writeEndElement();
477 			}
478 			break;
479 		}
480 		case PRIMITIVE_DATATYPE: {
481 			IPrimitiveType<?> pd = (IPrimitiveType<?>) nextValue;
482 			String value = pd.getValueAsString();
483 			if (value != null) {
484 				theEventWriter.writeStartElement(childName);
485 				theEventWriter.writeAttribute("value", value);
486 				encodeExtensionsIfPresent(theResource, theEventWriter, nextValue, theIncludedResource);
487 				theEventWriter.writeEndElement();
488 			}
489 			break;
490 		}
491 		case RESOURCE_BLOCK:
492 		case COMPOSITE_DATATYPE: {
493 			theEventWriter.writeStartElement(childName);
494 			if (isNotBlank(theExtensionUrl)) {
495 				theEventWriter.writeAttribute("url", theExtensionUrl);
496 			}
497 			BaseRuntimeElementCompositeDefinition<?> childCompositeDef = (BaseRuntimeElementCompositeDefinition<?>) childDef;
498 			encodeCompositeElementToStreamWriter(theResource, nextValue, theEventWriter, childCompositeDef, theIncludedResource);
499 			theEventWriter.writeEndElement();
500 			break;
501 		}
502 		case RESOURCE_REF: {
503 			IBaseReference ref = (IBaseReference) nextValue;
504 			if (!ref.isEmpty()) {
505 				theEventWriter.writeStartElement(childName);
506 				encodeResourceReferenceToStreamWriter(theEventWriter, ref, theResource, theIncludedResource);
507 				theEventWriter.writeEndElement();
508 			}
509 			break;
510 		}
511 		case CONTAINED_RESOURCE_LIST:
512 		case CONTAINED_RESOURCES: {
513 			/*
514 			 * Disable per #103 for (IResource next : value.getContainedResources()) { if
515 			 * (getContainedResources().getResourceId(next) != null) { continue; }
516 			 * theEventWriter.writeStartElement("contained"); encodeResourceToXmlStreamWriter(next, theEventWriter, true,
517 			 * fixContainedResourceId(next.getId().getValue())); theEventWriter.writeEndElement(); }
518 			 */
519 			for (IBaseResource next : getContainedResources().getContainedResources()) {
520 				IIdType resourceId = getContainedResources().getResourceId(next);
521 				theEventWriter.writeStartElement("contained");
522 				encodeResourceToXmlStreamWriter(next, theEventWriter, true, fixContainedResourceId(resourceId.getValue()));
523 				theEventWriter.writeEndElement();
524 			}
525 			break;
526 		}
527 		case RESOURCE: {
528 			theEventWriter.writeStartElement(childName);
529 			IBaseResource resource = (IBaseResource) nextValue;
530 			encodeResourceToXmlStreamWriter(resource, theEventWriter, false);
531 			theEventWriter.writeEndElement();
532 			break;
533 		}
534 		case PRIMITIVE_XHTML: {
535 			XhtmlDt dt = (XhtmlDt) nextValue;
536 			if (dt.hasContent()) {
537 				encodeXhtml(dt, theEventWriter);
538 			}
539 			break;
540 		}
541 		case PRIMITIVE_XHTML_HL7ORG: {
542 			IBaseXhtml dt = (IBaseXhtml) nextValue;
543 			if (dt.isEmpty()) {
544 				break;
545 			} else {
546 				// TODO: this is probably not as efficient as it could be
547 				XhtmlDt hdt = new XhtmlDt();
548 				hdt.setValueAsString(dt.getValueAsString());
549 				encodeXhtml(hdt, theEventWriter);
550 				break;
551 			}
552 		}
553 		case EXTENSION_DECLARED:
554 		case UNDECL_EXT: {
555 			throw new IllegalStateException("state should not happen: " + childDef.getName());
556 		}
557 		}
558 
559 	}

580 						String childName = nextChild.getChildNameByDatatype(child.getDatatype());
581 						BaseRuntimeElementDefinition<?> type = child.getChildByName(childName);
582 						encodeChildElementToStreamWriter(theResource, theEventWriter, narr, childName, type, null, theIncludedResource);
583 						continue;
584 					}
585 				} else {
586 					// Narrative generation not currently supported for HL7org structures
587 					// INarrative narr1 = ((IDomainResource) theResource).getText();
588 					// BaseNarrativeDt<?> narr2 = null;
589 					// if (gen != null && narr1.isEmpty()) {

591 					// String resourceProfile = myContext.getResourceDefinition(theResource).getResourceProfile();
592 					// gen.generateNarrative(resourceProfile, theResource, null);

599 					// theIncludedResource);
600 					// continue;
601 					// }
602 				}
603 			}
604 
605 			if (nextChild instanceof RuntimeChildContainedResources) {
606 				if (!theIncludedResource) {
607 					encodeChildElementToStreamWriter(theResource, theEventWriter, null, nextChild.getChildNameByDatatype(null), nextChild.getChildElementDefinitionByDatatype(null), null, theIncludedResource);
608 				}
609 			} else {
610 
611 				List<? extends IBase> values = nextChild.getAccessor().getValues(theElement);
612 				if (values == null || values.isEmpty()) {
613 					continue;
614 				}

616 					if ((nextValue == null || nextValue.isEmpty())) {

631 
632 					} else if (extensionUrl != null && childName.equals("extension") == false) {
633 						RuntimeChildDeclaredExtensionDefinition extDef = (RuntimeChildDeclaredExtensionDefinition) nextChild;
634 						if (extDef.isModifier()) {
635 							theEventWriter.writeStartElement("modifierExtension");
636 						} else {
637 							theEventWriter.writeStartElement("extension");
638 						}
639 
640 						theEventWriter.writeAttribute("url", extensionUrl);
641 						encodeChildElementToStreamWriter(theResource, theEventWriter, nextValue, childName, childDef, null, theIncludedResource);
642 						theEventWriter.writeEndElement();
643 					} else if (nextChild instanceof RuntimeChildNarrativeDefinition && theIncludedResource) {
644 						// suppress narratives from contained resources
645 					} else {
646 						encodeChildElementToStreamWriter(theResource, theEventWriter, nextValue, childName, childDef, extensionUrl, theIncludedResource);
647 					}
648 				}
649 			}
650 		}
651 	}

704 		/*
705 		 * DSTU2 requires extensions to come in a specific spot within the encoded content - This is a bit of a messy way
706 		 * to make that happen, but hopefully this won't matter as much once we use the HL7 structures
707 		 */
708 
709 		List<BaseRuntimeChildDefinition> preExtensionChildren = new ArrayList<BaseRuntimeChildDefinition>();
710 		List<BaseRuntimeChildDefinition> postExtensionChildren = new ArrayList<BaseRuntimeChildDefinition>();

712 		for (BaseRuntimeChildDefinition next : children) {

717 			} else {
718 				postExtensionChildren.add(next);

720 		}
721 		encodeCompositeElementChildrenToStreamWriter(theResource, theElement, theEventWriter, preExtensionChildren, theIncludedResource);
722 
723 		encodeExtensionsIfPresent(theResource, theEventWriter, theElement, theIncludedResource);
724 		encodeCompositeElementChildrenToStreamWriter(theResource, theElement, theEventWriter, resDef.getExtensions(), theIncludedResource);

766 	private void encodeResourceToXmlStreamWriter(IBaseResource theResource, XMLStreamWriter theEventWriter, boolean theContainedResource, String theResourceId) throws XMLStreamException {
767 		if (!theContainedResource) {
768 			super.containResourcesForEncoding(theResource);
769 		}
770 
771 		RuntimeResourceDefinition resDef = myContext.getResourceDefinition(theResource);
772 		if (resDef == null) {
773 			throw new ConfigurationException("Unknown resource type: " + theResource.getClass());
774 		}
775 
776 		theEventWriter.writeStartElement(resDef.getName());
777 		theEventWriter.writeDefaultNamespace(FHIR_NS);
778 
779 		if (theResource instanceof IAnyResource) {
780 
781 			// HL7.org Structures
782 			writeOptionalTagWithValue(theEventWriter, "id", theResourceId);
783 			encodeCompositeElementToStreamWriter(theResource, theResource, theEventWriter, resDef, theContainedResource);
784 
785 		} else {
786 
787 			if (myContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
788 
789 				// DSTU2+
790 
791 				IResource resource = (IResource) theResource;
792 				writeOptionalTagWithValue(theEventWriter, "id", theResourceId);
793 
794 				InstantDt updated = (InstantDt) resource.getResourceMetadata().get(ResourceMetadataKeyEnum.UPDATED);
795 				IdDt resourceId = resource.getId();
796 				String versionIdPart = resourceId.getVersionIdPart();
797 				if (isBlank(versionIdPart)) {
798 					versionIdPart = ResourceMetadataKeyEnum.VERSION.get(resource);
799 				}
800 				List<BaseCodingDt> securityLabels = extractMetadataListNotNull(resource, ResourceMetadataKeyEnum.SECURITY_LABELS);
801 				List<IdDt> profiles = extractMetadataListNotNull(resource, ResourceMetadataKeyEnum.PROFILES);
802 				TagList tags = ResourceMetadataKeyEnum.TAG_LIST.get(resource);
803 				if (ElementUtil.isEmpty(versionIdPart, updated, securityLabels, tags, profiles) == false) {
804 					theEventWriter.writeStartElement("meta");
805 					writeOptionalTagWithValue(theEventWriter, "versionId", versionIdPart);
806 					if (updated != null) {
807 						writeOptionalTagWithValue(theEventWriter, "lastUpdated", updated.getValueAsString());
808 					}
809 
810 					for (IdDt profile : profiles) {
811 						theEventWriter.writeStartElement("profile");
812 						theEventWriter.writeAttribute("value", profile.getValue());
813 						theEventWriter.writeEndElement();
814 					}
815 					for (BaseCodingDt securityLabel : securityLabels) {
816 						theEventWriter.writeStartElement("security");
817 						BaseRuntimeElementCompositeDefinition<?> def = (BaseRuntimeElementCompositeDefinition<?>) myContext.getElementDefinition(securityLabel.getClass());
818 						encodeCompositeElementChildrenToStreamWriter(resource, securityLabel, theEventWriter, def.getChildren(), theContainedResource);
819 						theEventWriter.writeEndElement();
820 					}
821 					if (tags != null) {
822 						for (Tag tag : tags) {
823 							theEventWriter.writeStartElement("tag");
824 							writeOptionalTagWithValue(theEventWriter, "system", tag.getScheme());
825 							writeOptionalTagWithValue(theEventWriter, "code", tag.getTerm());
826 							writeOptionalTagWithValue(theEventWriter, "display", tag.getLabel());
827 							theEventWriter.writeEndElement();
828 						}
829 					}
830 					theEventWriter.writeEndElement();
831 				}
832 
833 				if (theResource instanceof IBaseBinary) {
834 					IBaseBinary bin = (IBaseBinary) theResource;
835 					writeOptionalTagWithValue(theEventWriter, "contentType", bin.getContentType());
836 					writeOptionalTagWithValue(theEventWriter, "content", bin.getContentAsBase64());
837 				} else {
838 					encodeResourceToStreamWriterInDstu2Format(resDef, theResource, theResource, theEventWriter, resDef, theContainedResource);
839 				}
840 
841 			} else {
842 
843 				// DSTU1
844 				if (theResourceId != null && theContainedResource) {
845 					theEventWriter.writeAttribute("id", theResourceId);
846 				}
847 
848 				if (theResource instanceof IBaseBinary) {
849 					IBaseBinary bin = (IBaseBinary) theResource;
850 					if (bin.getContentType() != null) {
851 						theEventWriter.writeAttribute("contentType", bin.getContentType());
852 					}
853 					theEventWriter.writeCharacters(bin.getContentAsBase64());
854 				} else {
855 					encodeCompositeElementToStreamWriter(theResource, theResource, theEventWriter, resDef, theContainedResource);
856 				}
857 
858 			}
859 
860 		}
861 
862 		theEventWriter.writeEndElement();
863 	}

896 	private void encodeUndeclaredExtensions(IBaseResource theResource, XMLStreamWriter theWriter, List<? extends IBaseExtension<?, ?>> theExtensions, String tagName, boolean theIncludedResource) throws XMLStreamException, DataFormatException {
897 		for (IBaseExtension<?, ?> next : theExtensions) {
898 			if (next == null || (ElementUtil.isEmpty(next.getValue()) && next.getExtension().isEmpty())) {
899 				continue;
900 			}
901 
902 			theWriter.writeStartElement(tagName);
903 
904 			String url = next.getUrl();
905 			theWriter.writeAttribute("url", url);
906 
907 			if (next.getValue() != null) {
908 				IBaseDatatype value = next.getValue();
909 				RuntimeChildUndeclaredExtensionDefinition extDef = myContext.getRuntimeChildUndeclaredExtensionDefinition();
910 				String childName = extDef.getChildNameByDatatype(value.getClass());
911 				BaseRuntimeElementDefinition<?> childDef;
912 				if (childName == null) {
913 					childDef = myContext.getElementDefinition(value.getClass());
914 					if (childDef == null) {
915 						throw new ConfigurationException("Unable to encode extension, unrecognized child element type: " + value.getClass().getCanonicalName());
916 					} else {
917 						childName = RuntimeChildUndeclaredExtensionDefinition.createExtensionChildName(childDef);
918 					}
919 				} else {
920 					childDef = extDef.getChildElementDefinitionByDatatype(value.getClass());
921 					if (childDef == null) {
922 						throw new ConfigurationException("Unable to encode extension, unrecognized child element type: " + value.getClass().getCanonicalName());
923 					}
924 				}
925 				encodeChildElementToStreamWriter(theResource, theWriter, value, childName, childDef, null, theIncludedResource);
926 			}
927 
928 			// child extensions
929 			encodeExtensionsIfPresent(theResource, theWriter, next, theIncludedResource);
930 
931 			theWriter.writeEndElement();
932 		}
933 	}

935 	private void encodeXhtml(XhtmlDt theDt, XMLStreamWriter theEventWriter) throws XMLStreamException {
936 		if (theDt == null || theDt.getValue() == null || getSuppressNarratives()) {
937 			return;
938 		}
939 
940 		boolean firstElement = true;
941 		for (XMLEvent event : theDt.getValue()) {
942 			switch (event.getEventType()) {
943 			case XMLStreamConstants.ATTRIBUTE:
944 				Attribute attr = (Attribute) event;
945 				if (isBlank(attr.getName().getPrefix())) {
946 					if (isBlank(attr.getName().getNamespaceURI())) {
947 						theEventWriter.writeAttribute(attr.getName().getLocalPart(), attr.getValue());
948 					} else {
949 						theEventWriter.writeAttribute(attr.getName().getNamespaceURI(), attr.getName().getLocalPart(), attr.getValue());
950 					}
951 				} else {
952 					theEventWriter.writeAttribute(attr.getName().getPrefix(), attr.getName().getNamespaceURI(), attr.getName().getLocalPart(), attr.getValue());
953 				}
954 
955 				break;
956 			case XMLStreamConstants.CDATA:
957 				theEventWriter.writeCData(((Characters) event).getData());
958 				break;
959 			case XMLStreamConstants.CHARACTERS:
960 			case XMLStreamConstants.SPACE:
961 				String data = ((Characters) event).getData();
962 				theEventWriter.writeCharacters(data);
963 				break;
964 			case XMLStreamConstants.COMMENT:
965 				theEventWriter.writeComment(((Comment) event).getText());
966 				break;
967 			case XMLStreamConstants.END_ELEMENT:
968 				theEventWriter.writeEndElement();
969 				break;
970 			case XMLStreamConstants.ENTITY_REFERENCE:
971 				EntityReference er = (EntityReference) event;
972 				theEventWriter.writeEntityRef(er.getName());
973 				break;
974 			case XMLStreamConstants.NAMESPACE:
975 				Namespace ns = (Namespace) event;
976 				theEventWriter.writeNamespace(ns.getPrefix(), ns.getNamespaceURI());
977 				break;
978 			case XMLStreamConstants.START_ELEMENT:
979 				StartElement se = event.asStartElement();
980 				if (firstElement) {
981 					if (StringUtils.isBlank(se.getName().getPrefix())) {
982 						String namespaceURI = se.getName().getNamespaceURI();
983 						if (StringUtils.isBlank(namespaceURI)) {
984 							namespaceURI = "http://www.w3.org/1999/xhtml";
985 						}
986 						theEventWriter.writeStartElement(se.getName().getLocalPart());
987 						theEventWriter.writeDefaultNamespace(namespaceURI);
988 					} else {
989 						String prefix = se.getName().getPrefix();
990 						String namespaceURI = se.getName().getNamespaceURI();
991 						theEventWriter.writeStartElement(prefix, se.getName().getLocalPart(), namespaceURI);
992 						theEventWriter.writeNamespace(prefix, namespaceURI);
993 					}
994 					firstElement = false;
995 				} else {
996 					if (isBlank(se.getName().getPrefix())) {
997 						if (isBlank(se.getName().getNamespaceURI())) {
998 							theEventWriter.writeStartElement(se.getName().getLocalPart());
999 						} else {
1000 							if (StringUtils.isBlank(se.getName().getPrefix())) {
1001 								theEventWriter.writeStartElement(se.getName().getLocalPart());
1002 								// theEventWriter.writeDefaultNamespace(se.getName().getNamespaceURI());
1003 							} else {
1004 								theEventWriter.writeStartElement(se.getName().getNamespaceURI(), se.getName().getLocalPart());
1005 							}
1006 						}
1007 					} else {
1008 						theEventWriter.writeStartElement(se.getName().getPrefix(), se.getName().getLocalPart(), se.getName().getNamespaceURI());
1009 					}
1010 					for (Iterator<?> attrIter = se.getAttributes(); attrIter.hasNext();) {
1011 						Attribute next = (Attribute) attrIter.next();
1012 						theEventWriter.writeAttribute(next.getName().getLocalPart(), next.getValue());
1013 					}
1014 				}
1015 				break;
1016 			case XMLStreamConstants.DTD:
1017 			case XMLStreamConstants.END_DOCUMENT:
1018 			case XMLStreamConstants.ENTITY_DECLARATION:
1019 			case XMLStreamConstants.NOTATION_DECLARATION:
1020 			case XMLStreamConstants.PROCESSING_INSTRUCTION:
1021 			case XMLStreamConstants.START_DOCUMENT:
1022 				break;
1023 			}
1024 
1025 		}
1026 	}
