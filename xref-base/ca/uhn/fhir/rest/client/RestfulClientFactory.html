98  	@Override
99  	public synchronized HttpClient getHttpClient() {
100 		if (myHttpClient == null) {
101 
102 			PoolingHttpClientConnectionManager connectionManager = new PoolingHttpClientConnectionManager(5000, TimeUnit.MILLISECONDS);
103 
104 			//@formatter:off
105 			RequestConfig defaultRequestConfig = RequestConfig.custom()
106 				    .setSocketTimeout(mySocketTimeout)
107 				    .setConnectTimeout(myConnectTimeout)
108 				    .setConnectionRequestTimeout(myConnectionRequestTimeout)
109 				    .setStaleConnectionCheckEnabled(true)
110 				    .setProxy(myProxy)
111 				    .build();
112 			
113 			HttpClientBuilder builder = HttpClients.custom()
114 				.setConnectionManager(connectionManager)
115 				.setDefaultRequestConfig(defaultRequestConfig)
116 				.disableCookieManagement();
117 			
118 			if (myProxy != null && StringUtils.isNotBlank(myProxyUsername) && StringUtils.isNotBlank(myProxyPassword)) {
119 				CredentialsProvider credsProvider = new BasicCredentialsProvider();
120 				credsProvider.setCredentials(new AuthScope(myProxy.getHostName(), myProxy.getPort()), new UsernamePasswordCredentials(myProxyUsername, myProxyPassword));
121 				builder.setProxyAuthenticationStrategy(new ProxyAuthenticationStrategy());
122 				builder.setDefaultCredentialsProvider(credsProvider);
123 			}
124 			
125 			myHttpClient = builder.build();
126 			//@formatter:on
127 
128 		}
129 
130 		return myHttpClient;
131 	}

152 	@SuppressWarnings("unchecked")
153 	private <T extends IRestfulClient> T instantiateProxy(Class<T> theClientType, InvocationHandler theInvocationHandler) {
154 		T proxy = (T) Proxy.newProxyInstance(theClientType.getClassLoader(), new Class[] { theClientType }, theInvocationHandler);
155 		return proxy;
156 	}

158 	/**
159 	 * Instantiates a new client instance
160 	 * 
161 	 * @param theClientType
162 	 *            The client type, which is an interface type to be instantiated
163 	 * @param theServerBase
164 	 *            The URL of the base for the restful FHIR server to connect to
165 	 * @return A newly created client
166 	 * @throws ConfigurationException
167 	 *             If the interface type is not an interface
168 	 */
169 	@Override
170 	public synchronized <T extends IRestfulClient> T newClient(Class<T> theClientType, String theServerBase) {
171 		if (!theClientType.isInterface()) {
172 			throw new ConfigurationException(theClientType.getCanonicalName() + " is not an interface");
173 		}
174 
175 		
176 		ClientInvocationHandlerFactory invocationHandler = myInvocationHandlers.get(theClientType);
177 		if (invocationHandler == null) {
178 			HttpClient httpClient = getHttpClient();
179 			invocationHandler = new ClientInvocationHandlerFactory(httpClient, myContext, theServerBase, theClientType);
180 			for (Method nextMethod : theClientType.getMethods()) {
181 				BaseMethodBinding<?> binding = BaseMethodBinding.bindMethod(nextMethod, myContext, null);
182 				invocationHandler.addBinding(nextMethod, binding);
183 			}
184 			myInvocationHandlers.put(theClientType, invocationHandler);
185 		}
186 
187 		T proxy = instantiateProxy(theClientType, invocationHandler.newInvocationHandler(this));
188 
189 		return proxy;
190 	}

199 	 * This method is internal to HAPI - It may change in future versions, use with caution.
200 	 */
201 	public void validateServerBaseIfConfiguredToDoSo(String theServerBase, HttpClient theHttpClient, BaseClient theClient) {
202 		String serverBase = normalizeBaseUrlForMap(theServerBase);
203 
204 		switch (myServerValidationMode) {
205 		case NEVER:
206 			break;
207 		case ONCE:
208 			if (!myValidatedServerBaseUrls.contains(serverBase)) {
209 				validateServerBase(serverBase, theHttpClient, theClient);
210 			}
211 			break;
212 		}
213 
214 	}

216 	private String normalizeBaseUrlForMap(String theServerBase) {
217 		String serverBase = theServerBase;
218 		if (!serverBase.endsWith("/")) {
219 			serverBase = serverBase + "/";
220 		}
221 		return serverBase;
222 	}

279 	@SuppressWarnings("unchecked")
280 	void validateServerBase(String theServerBase, HttpClient theHttpClient, BaseClient theClient) {
281 
282 		GenericClient client = new GenericClient(myContext, theHttpClient, theServerBase, this);
283 		for (IClientInterceptor interceptor : theClient.getInterceptors()) {
284 			client.registerInterceptor(interceptor);
285 		}
286 		client.setDontValidateConformance(true);
287 		
288 		IBaseResource conformance;
289 		try {
290 			@SuppressWarnings("rawtypes")
291 			Class implementingClass = myContext.getResourceDefinition("Conformance").getImplementingClass();
292 			conformance = (IBaseResource) client.fetchConformance().ofType(implementingClass).execute();
293 		} catch (FhirClientConnectionException e) {
294 			throw new FhirClientConnectionException(myContext.getLocalizer().getMessage(RestfulClientFactory.class, "failedToRetrieveConformance", theServerBase + Constants.URL_TOKEN_METADATA), e);
295 		}
296 
297 		FhirTerser t = myContext.newTerser();
298 		String serverFhirVersionString = null;
299 		Object value = t.getSingleValueOrNull(conformance, "fhirVersion");
300 		if (value instanceof IPrimitiveType) {
301 			serverFhirVersionString = ((IPrimitiveType<?>) value).getValueAsString();
302 		}
303 		FhirVersionEnum serverFhirVersionEnum = null;
304 		if (StringUtils.isBlank(serverFhirVersionString)) {
305 			// we'll be lenient and accept this
306 		} else {
307 			if (serverFhirVersionString.startsWith("0.80") || serverFhirVersionString.startsWith("0.0.8")) {
308 				serverFhirVersionEnum = FhirVersionEnum.DSTU1;
309 			} else if (serverFhirVersionString.startsWith("0.4")) {
310 				serverFhirVersionEnum = FhirVersionEnum.DSTU2;
311 			} else if (serverFhirVersionString.startsWith("0.5")) {
312 				serverFhirVersionEnum = FhirVersionEnum.DSTU2;
313 			} else {
314 				// we'll be lenient and accept this
315 				ourLog.debug("Server conformance statement indicates unknown FHIR version: {}", serverFhirVersionString);
316 			}
317 		}
318 
319 		if (serverFhirVersionEnum != null) {
320 			FhirVersionEnum contextFhirVersion = myContext.getVersion().getVersion();
321 			if (!contextFhirVersion.isEquivalentTo(serverFhirVersionEnum)) {
322 				throw new FhirClientInappropriateForServerException(myContext.getLocalizer().getMessage(RestfulClientFactory.class, "wrongVersionInConformance", theServerBase + Constants.URL_TOKEN_METADATA, serverFhirVersionString, serverFhirVersionEnum, contextFhirVersion));
323 			}
324 		}
325 		
326 		myValidatedServerBaseUrls.add(normalizeBaseUrlForMap(theServerBase));
327 
328 	}
