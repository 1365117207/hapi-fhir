169 	@Override
170 	public IHttpRequest asHttpRequest(String theUrlBase, Map<String, List<String>> theExtraParams, EncodingEnum theEncoding, Boolean thePrettyPrint) throws DataFormatException {
171 		StringBuilder url = new StringBuilder();
172 
173 		if (myUrlPath == null) {
174 			url.append(theUrlBase);
175 		} else {
176 			if (!myUrlPath.contains("://")) {
177 				url.append(theUrlBase);
178 				if (!theUrlBase.endsWith("/")) {
179 					url.append('/');
180 				}
181 			}
182 			url.append(myUrlPath);
183 		}
184 
185 		appendExtraParamsWithQuestionMark(theExtraParams, url, url.indexOf("?") == -1);
186 		IHttpClient httpClient = getRestfulClientFactory().getHttpClient(url, myIfNoneExistParams, myIfNoneExistString, getRequestType(), getHeaders());
187 
188 		if (myResource != null && IBaseBinary.class.isAssignableFrom(myResource.getClass())) {
189 			IBaseBinary binary = (IBaseBinary) myResource;
190 			if (isNotBlank(binary.getContentType()) && EncodingEnum.forContentTypeStrict(binary.getContentType()) == null) {
191 				return httpClient.createBinaryRequest(getContext(), binary);
192 			}
193 		}
194 
195 		EncodingEnum encoding = theEncoding;
196 		if (myContents != null) {
197 			encoding = MethodUtil.detectEncoding(myContents);
198 		}
199 
200 
201 		if (myParams != null) {
202 			return httpClient.createParamRequest(getContext(), myParams, encoding);
203 		}
204 		encoding = ObjectUtils.defaultIfNull(encoding,  EncodingEnum.XML);
205 		String contents = encodeContents(thePrettyPrint, encoding);
206 		String contentType = getContentType(encoding);
207 		return httpClient.createByteRequest(getContext(), contents, contentType, encoding);
208 	}

210 	private String getContentType(EncodingEnum encoding) {
211 		if (myBundle != null || (getContext().getVersion().getVersion() == FhirVersionEnum.DSTU1 && ((myContents != null && myContentsIsBundle) || myResources != null))) {
212 			return encoding.getBundleContentType();
213 		} else if (getContext().getVersion().getVersion().isOlderThan(FhirVersionEnum.DSTU3)) {
214 			// application/xml+fhir
215 			return encoding.getResourceContentType();
216 		} else {
217 			// application/fhir+xml
218 			return encoding.getResourceContentTypeNonLegacy();
219 		}
220 	}

227 	private String encodeContents(Boolean thePrettyPrint, EncodingEnum encoding) {
228 		IParser parser;
229 
230 		if (encoding == EncodingEnum.JSON) {
231 			parser = getContext().newJsonParser();
232 		} else {
233 			parser = getContext().newXmlParser();
234 		}
235 
236 		if (thePrettyPrint != null) {
237 			parser.setPrettyPrint(thePrettyPrint);
238 		}
239 
240 		if (myForceResourceId != null) {
241 			parser.setEncodeForceResourceId(myForceResourceId);
242 		}
243 		
244 		parser.setOmitResourceId(myOmitResourceId);
245 		if (myTagList != null) {
246 			return parser.encodeTagListToString(myTagList);
247 		} else if (myBundle != null) {
248 			return parser.encodeBundleToString(myBundle);
249 		} else if (myResources != null) {
250 			IVersionSpecificBundleFactory bundleFactory = getContext().newBundleFactory();
251 			bundleFactory.initializeBundleFromResourceList("", myResources, "", "", myResources.size(), myBundleType);
252 			Bundle bundle = bundleFactory.getDstu1Bundle();
253 			if (bundle != null) {
254 				return parser.encodeBundleToString(bundle);
255 			}
256 			IBaseResource bundleRes = bundleFactory.getResourceBundle();
257 			return parser.encodeResourceToString(bundleRes);
258 		} else if (myContents != null) {
259 			return myContents;
260 		} else {
261 			return parser.encodeResourceToString(myResource);
262 		}
263 	}
