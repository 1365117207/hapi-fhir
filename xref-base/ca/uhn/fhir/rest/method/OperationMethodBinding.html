58  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(OperationMethodBinding.class);
59  	private String myDescription;
60  	private final boolean myIdempotent;
61  	private final Integer myIdParamIndex;
62  	private final String myName;
63  	private final OtherOperationTypeEnum myOtherOperatiopnType;
64  	private final ReturnTypeEnum myReturnType;

66  	public OperationMethodBinding(Class<?> theReturnResourceType, Class<? extends IBaseResource> theReturnTypeFromRp, Method theMethod, FhirContext theContext, Object theProvider, boolean theIdempotent, String theOperationName, Class<? extends IBaseResource> theOperationType) {
67  		super(theReturnResourceType, theMethod, theContext, theProvider);
68  
69  		myIdempotent = theIdempotent;
70  		myIdParamIndex = MethodUtil.findIdParameterIndex(theMethod);
71  		Description description = theMethod.getAnnotation(Description.class);
72  		if (description != null) {
73  			myDescription = description.formalDefinition();
74  			if (isBlank(myDescription)) {
75  				myDescription = description.shortDefinition();
76  			}
77  		}
78  		if (isBlank(myDescription)) {
79  			myDescription = null;
80  		}
81  
82  		if (isBlank(theOperationName)) {
83  			throw new ConfigurationException("Method '" + theMethod.getName() + "' on type " + theMethod.getDeclaringClass().getName() + " is annotated with @" + Operation.class.getSimpleName() + " but this annotation has no name defined");
84  		}
85  		if (theOperationName.startsWith("$") == false) {
86  			theOperationName = "$" + theOperationName;
87  		}
88  		myName = theOperationName;
89  
90  		if (theContext.getVersion().getVersion().isEquivalentTo(FhirVersionEnum.DSTU1)) {
91  			throw new ConfigurationException("@" + Operation.class.getSimpleName() + " methods are not supported on servers for FHIR version " + theContext.getVersion().getVersion().name());
92  		}
93  
94  		if (theReturnTypeFromRp != null) {
95  			setResourceName(theContext.getResourceDefinition(theReturnTypeFromRp).getName());
96  		} else {
97  			if (Modifier.isAbstract(theOperationType.getModifiers()) == false) {
98  				setResourceName(theContext.getResourceDefinition(theOperationType).getName());
99  			} else {
100 				setResourceName(null);
101 			}
102 		}
103 
104 		if (theMethod.getReturnType().isAssignableFrom(Bundle.class)) {
105 			throw new ConfigurationException("Can not return a DSTU1 bundle from an @" + Operation.class.getSimpleName() + " method. Found in method " + theMethod.getName() + " defined in type " + theMethod.getDeclaringClass().getName());
106 		}
107 
108 		if (theMethod.getReturnType().equals(IBundleProvider.class)) {
109 			myReturnType = ReturnTypeEnum.BUNDLE;
110 		} else {
111 			myReturnType = ReturnTypeEnum.RESOURCE;
112 		}
113 
114 		if (getResourceName() == null) {
115 			myOtherOperatiopnType = OtherOperationTypeEnum.EXTENDED_OPERATION_SERVER;
116 		} else if (myIdParamIndex == null) {
117 			myOtherOperatiopnType = OtherOperationTypeEnum.EXTENDED_OPERATION_TYPE;
118 		} else {
119 			myOtherOperatiopnType = OtherOperationTypeEnum.EXTENDED_OPERATION_INSTANCE;
120 		}
121 	}

160 	@Override
161 	public boolean incomingServerRequestMatchesMethod(RequestDetails theRequest) {
162 		if (getResourceName() == null) {
163 			if (isNotBlank(theRequest.getResourceName())) {
164 				return false;
165 			}
166 		} else if (!getResourceName().equals(theRequest.getResourceName())) {
167 			return false;
168 		}
169 
170 		boolean requestHasId = theRequest.getId() != null;
171 		if (requestHasId != (myIdParamIndex != null)) {
172 			return false;
173 		}
174 
175 		return myName.equals(theRequest.getOperation());
176 	}

178 	@Override
179 	public BaseHttpClientInvocation invokeClient(Object[] theArgs) throws InternalErrorException {
180 		String id = null;
181 		if (myIdParamIndex != null) {
182 			IdDt idDt = (IdDt) theArgs[myIdParamIndex];
183 			id = idDt.getValue();
184 		}
185 		IBaseParameters parameters = (IBaseParameters) getContext().getResourceDefinition("Parameters").newInstance();
186 
187 		if (theArgs != null) {
188 			for (int idx = 0; idx < theArgs.length; idx++) {
189 				IParameter nextParam = getParameters().get(idx);
190 				nextParam.translateClientArgumentIntoQueryArgument(getContext(), theArgs[idx], null, parameters);
191 			}
192 		}
193 
194 		return createOperationInvocation(getContext(), getResourceName(), id, myName, parameters, false);
195 	}

197 	@Override
198 	public Object invokeServer(RequestDetails theRequest, Object[] theMethodParams) throws BaseServerResponseException {
199 		if (theRequest.getRequestType() == RequestTypeEnum.POST) {
200 			// always ok
201 		} else if (theRequest.getRequestType() == RequestTypeEnum.GET) {
202 			if (!myIdempotent) {
203 				String message = getContext().getLocalizer().getMessage(OperationMethodBinding.class, "methodNotSupported", theRequest.getRequestType(), RequestTypeEnum.POST.name());
204 				throw new MethodNotAllowedException(message, RequestTypeEnum.POST);
205 			}
206 		} else {
207 			if (!myIdempotent) {
208 				String message = getContext().getLocalizer().getMessage(OperationMethodBinding.class, "methodNotSupported", theRequest.getRequestType(), RequestTypeEnum.POST.name());
209 				throw new MethodNotAllowedException(message, RequestTypeEnum.POST);
210 			} else {
211 				String message = getContext().getLocalizer().getMessage(OperationMethodBinding.class, "methodNotSupported", theRequest.getRequestType(), RequestTypeEnum.GET.name(), RequestTypeEnum.POST.name());
212 				throw new MethodNotAllowedException(message, RequestTypeEnum.GET, RequestTypeEnum.POST);
213 			}
214 		}
215 
216 		if (myIdParamIndex != null) {
217 			theMethodParams[myIdParamIndex] = theRequest.getId();
218 		}
219 
220 		Object response = invokeServerMethod(theMethodParams);
221 		IBundleProvider retVal = toResourceList(response);
222 		return retVal;
223 	}

233 	public static BaseHttpClientInvocation createOperationInvocation(FhirContext theContext, String theResourceName, String theId, String theOperationName, IBaseParameters theInput, boolean theUseHttpGet) {
234 		StringBuilder b = new StringBuilder();
235 		if (theResourceName != null) {
236 			b.append(theResourceName);
237 			if (isNotBlank(theId)) {
238 				b.append('/');
239 				b.append(theId);
240 			}
241 		}
242 		if (b.length() > 0) {
243 			b.append('/');
244 		}
245 		if (!theOperationName.startsWith("$")) {
246 			b.append("$");
247 		}
248 		b.append(theOperationName);
249 
250 		if (!theUseHttpGet) {
251 			return new HttpPostClientInvocation(theContext, theInput, b.toString());
252 		} else {
253 			FhirTerser t = theContext.newTerser();
254 			List<Object> parameters = t.getValues(theInput, "Parameters.parameter");
255 
256 			Map<String, List<String>> params = new HashMap<String, List<String>>();
257 			for (Object nextParameter : parameters) {
258 				IPrimitiveType<?> nextNameDt = (IPrimitiveType<?>) t.getSingleValueOrNull((IBase) nextParameter, "name");
259 				if (nextNameDt == null || nextNameDt.isEmpty()) {
260 					ourLog.warn("Ignoring input parameter with no value in Parameters.parameter.name in operation client invocation");
261 					continue;
262 				}
263 				String nextName = nextNameDt.getValueAsString();
264 				if (!params.containsKey(nextName)) {
265 					params.put(nextName, new ArrayList<String>());
266 				}
267 
268 				IBaseDatatype value = (IBaseDatatype) t.getSingleValueOrNull((IBase) nextParameter, "value[x]");
269 				if (value == null) {
270 					continue;
271 				}
272 				if (!(value instanceof IPrimitiveType)) {
273 					throw new IllegalArgumentException("Can not invoke operation as HTTP GET when it has parameters with a composite (non priitive) datatype as the value. Found value: " + value.getClass().getName());
274 				}
275 				IPrimitiveType<?> primitive = (IPrimitiveType<?>) value;
276 				params.get(nextName).add(primitive.getValueAsString());
277 			}
278 			return new HttpGetClientInvocation(params, b.toString());
279 		}
