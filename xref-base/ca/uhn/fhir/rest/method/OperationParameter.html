94  		if (theInnerCollectionType != null) {
95  			myInnerCollectionType = CollectionBinder.getInstantiableCollectionType(theInnerCollectionType, myName);
96  		} else {

106 	@Override
107 	public void translateClientArgumentIntoQueryArgument(FhirContext theContext, Object theSourceClientArgument, Map<String, List<String>> theTargetQueryArguments, IBaseResource theTargetResource) throws InternalErrorException {
108 		assert theTargetResource != null;
109 		Object sourceClientArgument = theSourceClientArgument;
110 		if (sourceClientArgument == null) {
111 			return;
112 		}
113 
114 		if (myConverter != null) {
115 			sourceClientArgument = myConverter.outgoingClient(sourceClientArgument);
116 		}
117 
118 		ParametersUtil.addParameterToParameters(theContext, theTargetResource, sourceClientArgument, myName);
119 	}

121 	@SuppressWarnings("unchecked")
122 	@Override
123 	public Object translateQueryParametersIntoServerArgument(RequestDetails theRequest, byte[] theRequestContents, BaseMethodBinding<?> theMethodBinding) throws InternalErrorException, InvalidRequestException {
124 		List<Object> matchingParamValues = new ArrayList<Object>();
125 
126 		if (theRequest.getRequestType() == RequestTypeEnum.GET) {
127 			String[] paramValues = theRequest.getParameters().get(myName);
128 			if (paramValues != null && paramValues.length > 0) {
129 				if (IPrimitiveType.class.isAssignableFrom(myParameterType)) {
130 					for (String nextValue : paramValues) {
131 						FhirContext ctx = theRequest.getServer().getFhirContext();
132 						RuntimePrimitiveDatatypeDefinition def = (RuntimePrimitiveDatatypeDefinition) ctx.getElementDefinition((Class<? extends IBase>) myParameterType);
133 						IPrimitiveType<?> instance = def.newInstance();
134 						instance.setValueAsString(nextValue);
135 						matchingParamValues.add(instance);
136 					}
137 				} else {
138 					HapiLocalizer localizer = theRequest.getServer().getFhirContext().getLocalizer();
139 					String msg = localizer.getMessage(OperationParameter.class, "urlParamNotPrimitive", myOperationName, myName);
140 					throw new MethodNotAllowedException(msg, RequestTypeEnum.POST);
141 				}
142 			}
143 		} else {
144 
145 			FhirContext ctx = theRequest.getServer().getFhirContext();
146 
147 			if (theRequest.getRequestType() == RequestTypeEnum.GET) {
148 				return null;
149 			}
150 
151 			Class<? extends IBaseResource> wantedResourceType = theMethodBinding.getContext().getResourceDefinition("Parameters").getImplementingClass();
152 			IBaseResource requestContents = ResourceParameter.loadResourceFromRequest(theRequest, theRequestContents, theMethodBinding, wantedResourceType);
153 
154 			RuntimeResourceDefinition def = ctx.getResourceDefinition(requestContents);
155 
156 			BaseRuntimeChildDefinition paramChild = def.getChildByName("parameter");
157 			BaseRuntimeElementCompositeDefinition<?> paramChildElem = (BaseRuntimeElementCompositeDefinition<?>) paramChild.getChildByName("parameter");
158 
159 			RuntimeChildPrimitiveDatatypeDefinition nameChild = (RuntimeChildPrimitiveDatatypeDefinition) paramChildElem.getChildByName("name");
160 			BaseRuntimeChildDefinition valueChild = paramChildElem.getChildByName("value[x]");
161 			BaseRuntimeChildDefinition resourceChild = paramChildElem.getChildByName("resource");
162 
163 			IAccessor paramChildAccessor = paramChild.getAccessor();
164 			List<IBase> values = paramChildAccessor.getValues(requestContents);
165 			for (IBase nextParameter : values) {
166 				List<IBase> nextNames = nameChild.getAccessor().getValues(nextParameter);
167 				if (nextNames != null && nextNames.size() > 0) {
168 					IPrimitiveType<?> nextName = (IPrimitiveType<?>) nextNames.get(0);
169 					if (myName.equals(nextName.getValueAsString())) {
170 
171 						if (myParameterType.isAssignableFrom(nextParameter.getClass())) {
172 							matchingParamValues.add(nextParameter);
173 						} else {
174 							List<IBase> paramValues = valueChild.getAccessor().getValues(nextParameter);
175 							List<IBase> paramResources = resourceChild.getAccessor().getValues(nextParameter);
176 							if (paramValues != null && paramValues.size() > 0) {
177 								tryToAddValues(paramValues, matchingParamValues);
178 							} else if (paramResources != null && paramResources.size() > 0) {
179 								tryToAddValues(paramResources, matchingParamValues);
180 							}
181 						}
182 
183 					}
184 				}
185 			}
186 
187 		}
188 
189 		if (matchingParamValues.isEmpty()) {
190 			return null;
191 		}
192 
193 		if (myInnerCollectionType == null) {
194 			return matchingParamValues.get(0);
195 		}
196 
197 		try {
198 			@SuppressWarnings("rawtypes")
199 			Collection retVal = myInnerCollectionType.newInstance();
200 			retVal.addAll(matchingParamValues);
201 			return retVal;
202 		} catch (InstantiationException e) {
203 			throw new InternalErrorException("Failed to instantiate " + myInnerCollectionType, e);
204 		} catch (IllegalAccessException e) {
205 			throw new InternalErrorException("Failed to instantiate " + myInnerCollectionType, e);
206 		}
207 	}
208 
209 	private void tryToAddValues(List<IBase> theParamValues, List<Object> theMatchingParamValues) {
210 		for (Object nextValue : theParamValues) {
211 			if (nextValue == null) {
212 				continue;
213 			}
214 			if (myConverter != null) {
215 				nextValue = myConverter.incomingServer(nextValue);
216 			}
217 			if (!myParameterType.isAssignableFrom(nextValue.getClass())) {
218 				throw new InvalidRequestException("Request has parameter " + myName + " of type " + nextValue.getClass().getSimpleName() + " but method expects type " + myParameterType.getSimpleName());
219 			}
220 			theMatchingParamValues.add(nextValue);
221 		}
222 	}
