12  /**
13   * Comparator/qualifier for values used in REST params, such as {@link DateParam}, {@link NumberParam}, and
14   * {@link QuantityParam}
15   * 
16   * @since 1.5
17   */
18  public enum ParamPrefixEnum {
19  
20  	/**
21  	 * Code Value: <b>eq</b>
22  	 *
23  	 * The actual value is equal to the given value
24  	 */
25  	APPROXIMATE("~", "ap"),
26  
27  	/**
28  	 * Code Value: <b>eb</b>
29  	 *
30  	 * The range of the search value does overlap not with the range of the target value, and the range above the search value contains the range of the target value
31  	 */
32  	ENDS_BEFORE("", "eb"),
33  
34  	/**
35  	 * Code Value: <b>eq</b>
36  	 *
37  	 * The actual value is equal to the given value
38  	 */
39  	EQUAL("", "eq"),
40  
41  	/**
42  	 * Code Value: <b>gt</b>
43  	 *
44  	 * The actual value is greater than the given value.
45  	 */
46  	GREATERTHAN(">", "gt"),
47  
48  	/**
49  	 * Code Value: <b>ge</b>
50  	 *
51  	 * The actual value is greater than or equal to the given value.
52  	 */
53  	GREATERTHAN_OR_EQUALS(">=", "ge"),
54  
55  	/**
56  	 * Code Value: <b>lt</b>
57  	 *
58  	 * The actual value is less than the given value.
59  	 */
60  	LESSTHAN("<", "lt"),
61  
62  	/**
63  	 * Code Value: <b>le</b>
64  	 *
65  	 * The actual value is less than or equal to the given value.
66  	 */
67  	LESSTHAN_OR_EQUALS("<=", "le"), 
68  	
69  	/**
70  	 * Code Value: <b>ne</b>
71  	 *
72  	 * The actual value is not equal to the given value
73  	 */
74  	NOT_EQUAL("", "ne"),
75  	
76  	/**
77  	 * Code Value: <b>sa</b>
78  	 *
79  	 * The range of the search value does not overlap with the range of the target value, and the range below the search value contains the range of the target value
80  	 */
81  	STARTS_AFTER("", "sa");
82  	
83  	private static final Map<String, ParamPrefixEnum> DSTU1_TO_PREFIX;
84  	private static final Map<String, ParamPrefixEnum> VALUE_TO_PREFIX;
85  
86  	static {
87  		HashMap<String, ParamPrefixEnum> valueToPrefix = new HashMap<String, ParamPrefixEnum>();
88  		HashMap<String, ParamPrefixEnum> dstu1ToPrefix = new HashMap<String, ParamPrefixEnum>();
89  		for (ParamPrefixEnum next : values()) {
90  			valueToPrefix.put(next.getValue(), next);
91  			if (isNotBlank(next.getDstu1Value())) {
92  				dstu1ToPrefix.put(next.getDstu1Value(), next);
93  			}
94  		}
95  
96  		VALUE_TO_PREFIX = Collections.unmodifiableMap(valueToPrefix);
97  		DSTU1_TO_PREFIX = Collections.unmodifiableMap(dstu1ToPrefix);
98  	}
99  
100 	private final String myDstu1Value;
101 
102 	private final String myValue;
103 
104 	private ParamPrefixEnum(String theDstu1Value, String theValue) {
105 		myDstu1Value = theDstu1Value;
106 		myValue = theValue;
107 	}
108 
109 	/**
110 	 * Returns the DSTU1 value, e.g. <code>&lt;</code> or <code>~</code>
111 	 */
112 	public String getDstu1Value() {
113 		return myDstu1Value;
114 	}
115 
116 	/**
117 	 * Returns the DSTU2+ value, e.g. <code>lt</code> or <code>eq</code>
118 	 */
119 	public String getValue() {
120 		return myValue;
121 	}
122 
123 	/**
124 	 * Returns the appropriate value for the given context, based on the
125 	 * FHIR version supported by the context  
126 	 */
127 	public String getValueForContext(FhirContext theContext) {
128 		if (theContext.getVersion().getVersion() == FhirVersionEnum.DSTU1) {
129 			return getDstu1Value();
130 		} else {
131 			return getValue();
132 		}
133 	}
134 
135 	/**
136 	 * Returns the prefix associated with a given DSTU1 value (e.g. <code>&lt;</code> or <code>~</code>)
137 	 * 
138 	 * @param theValue
139 	 *           e.g. <code>&lt;</code> or <code>~</code>
140 	 * @return The prefix, or <code>null</code> if no prefix matches the value
141 	 */
142 	public static ParamPrefixEnum forDstu1Value(String theValue) {
143 		return DSTU1_TO_PREFIX.get(theValue);
144 	}
145 
146 	/**
147 	 * Returns the prefix associated with a given DSTU2+ value (e.g. <code>lt</code> or <code>eq</code>)
148 	 * 
149 	 * @param theValue
150 	 *           e.g. <code>&lt;</code> or <code>~</code>
151 	 * @return The prefix, or <code>null</code> if no prefix matches the value
152 	 */
153 	public static ParamPrefixEnum forValue(String theValue) {
154 		return VALUE_TO_PREFIX.get(theValue);
155 	}
156 }
