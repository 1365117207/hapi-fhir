80  	public static void configureResponseParser(RequestDetails theRequestDetails, IParser parser) {
81  		// Pretty print
82  		boolean prettyPrint = RestfulServerUtils.prettyPrintResponse(theRequestDetails.getServer(), theRequestDetails);
83  
84  		parser.setPrettyPrint(prettyPrint);
85  		parser.setServerBaseUrl(theRequestDetails.getFhirServerBase());
86  
87  		// Summary mode
88  		Set<SummaryEnum> summaryMode = RestfulServerUtils.determineSummaryMode(theRequestDetails);
89  
90  		// _elements
91  		Set<String> elements = ElementsParameter.getElementsValueOrNull(theRequestDetails);
92  		if (elements != null && summaryMode != null && !summaryMode.equals(Collections.singleton(SummaryEnum.FALSE))) {
93  			throw new InvalidRequestException("Cannot combine the " + Constants.PARAM_SUMMARY + " and " + Constants.PARAM_ELEMENTS + " parameters");
94  		}
95  		Set<String> elementsAppliesTo = null;
96  		if (elements != null && isNotBlank(theRequestDetails.getResourceName())) {
97  			elementsAppliesTo = Collections.singleton(theRequestDetails.getResourceName());
98  		}
99  
100 		if (summaryMode != null) {
101 			if (summaryMode.contains(SummaryEnum.COUNT)) {
102 				parser.setEncodeElements(Collections.singleton("Bundle.total"));
103 			} else if (summaryMode.contains(SummaryEnum.TEXT)) {
104 				parser.setEncodeElements(TEXT_ENCODE_ELEMENTS);
105 			} else {
106 				parser.setSuppressNarratives(summaryMode.contains(SummaryEnum.DATA));
107 				parser.setSummaryMode(summaryMode.contains(SummaryEnum.TRUE));
108 			}
109 		}
110 		if (elements != null && elements.size() > 0) {
111 			Set<String> newElements = new HashSet<String>();
112 			for (String next : elements) {
113 				newElements.add("*." + next);
114 			}
115 			parser.setEncodeElements(newElements);
116 			parser.setEncodeElementsAppliesToResourceTypes(elementsAppliesTo);
117 		}
118 	}

120 	public static String createPagingLink(Set<Include> theIncludes, String theServerBase, String theSearchId, int theOffset, int theCount, EncodingEnum theResponseEncoding, boolean thePrettyPrint,
121 			BundleTypeEnum theBundleType) {
122 		try {
123 			StringBuilder b = new StringBuilder();
124 			b.append(theServerBase);
125 			b.append('?');
126 			b.append(Constants.PARAM_PAGINGACTION);
127 			b.append('=');
128 			b.append(URLEncoder.encode(theSearchId, "UTF-8"));
129 
130 			b.append('&');
131 			b.append(Constants.PARAM_PAGINGOFFSET);
132 			b.append('=');
133 			b.append(theOffset);
134 			b.append('&');
135 			b.append(Constants.PARAM_COUNT);
136 			b.append('=');
137 			b.append(theCount);
138 			if (theResponseEncoding != null) {
139 				b.append('&');
140 				b.append(Constants.PARAM_FORMAT);
141 				b.append('=');
142 				b.append(theResponseEncoding.getRequestContentType());
143 			}
144 			if (thePrettyPrint) {
145 				b.append('&');
146 				b.append(Constants.PARAM_PRETTY);
147 				b.append('=');
148 				b.append(Constants.PARAM_PRETTY_VALUE_TRUE);
149 			}
150 
151 			if (theIncludes != null) {
152 				for (Include nextInclude : theIncludes) {
153 					if (isNotBlank(nextInclude.getValue())) {
154 						b.append('&');
155 						b.append(Constants.PARAM_INCLUDE);
156 						b.append('=');
157 						b.append(URLEncoder.encode(nextInclude.getValue(), "UTF-8"));
158 					}
159 				}
160 			}
161 
162 			if (theBundleType != null) {
163 				b.append('&');
164 				b.append(Constants.PARAM_BUNDLETYPE);
165 				b.append('=');
166 				b.append(theBundleType.getCode());
167 			}
168 
169 			return b.toString();
170 		} catch (UnsupportedEncodingException e) {
171 			throw new Error("UTF-8 not supported", e);// should not happen
172 		}
173 	}

194 	private static ResponseEncoding determineRequestEncodingNoDefaultReturnRE(RequestDetails theReq) {
195 		ResponseEncoding retVal = null;
196 		List<String> headers = theReq.getHeaders(Constants.HEADER_CONTENT_TYPE);
197 		if (headers != null) {
198 			Iterator<String> acceptValues = headers.iterator();
199 			if (acceptValues != null) {
200 				while (acceptValues.hasNext() && retVal == null) {
201 					String nextAcceptHeaderValue = acceptValues.next();
202 					if (nextAcceptHeaderValue != null && isNotBlank(nextAcceptHeaderValue)) {
203 						for (String nextPart : nextAcceptHeaderValue.split(",")) {
204 							int scIdx = nextPart.indexOf(';');
205 							if (scIdx == 0) {
206 								continue;
207 							}
208 							if (scIdx != -1) {
209 								nextPart = nextPart.substring(0, scIdx);
210 							}
211 							nextPart = nextPart.trim();
212 							EncodingEnum encoding = EncodingEnum.forContentType(nextPart);
213 							if (encoding != null) {
214 								retVal = new ResponseEncoding(theReq.getServer().getFhirContext(), encoding, nextPart);
215 								break;
216 							}
217 						}
218 					}
219 				}
220 			}
221 		}
222 		return retVal;
223 	}

225 	/**
226 	 * Returns null if the request doesn't express that it wants FHIR. If it expresses that it wants XML and JSON
227 	 * equally, returns thePrefer.
228 	 */
229 	public static ResponseEncoding determineResponseEncodingNoDefault(RequestDetails theReq, EncodingEnum thePrefer) {
230 		String[] format = theReq.getParameters().get(Constants.PARAM_FORMAT);
231 		if (format != null) {
232 			for (String nextFormat : format) {
233 				EncodingEnum retVal = EncodingEnum.forContentType(nextFormat);
234 				if (retVal != null) {
235 					return new ResponseEncoding(theReq.getServer().getFhirContext(), retVal, nextFormat);
236 				}
237 			}
238 		}
239 
240 		/*
241 		 * Some browsers (e.g. FF) request "application/xml" in their Accept header,
242 		 * and we generally want to treat this as a preference for FHIR XML even if
243 		 * it's not the FHIR version of the CT, which should be "application/xml+fhir".
244 		 * 
245 		 * When we're serving up Binary resources though, we are a bit more strict,
246 		 * since Binary is supposed to use native content types unless the client has
247 		 * explicitly requested FHIR.
248 		 */
249 		boolean strict = false;
250 		if ("Binary".equals(theReq.getResourceName())) {
251 			strict = true;
252 		}
253 
254 		/*
255 		 * The Accept header is kind of ridiculous, e.g.
256 		 */
257 		// text/xml, application/xml, application/xhtml+xml, text/html;q=0.9, text/plain;q=0.8, image/png, */*;q=0.5
258 
259 		List<String> acceptValues = theReq.getHeaders(Constants.HEADER_ACCEPT);
260 		float bestQ = -1f;
261 		ResponseEncoding retVal = null;
262 		if (acceptValues != null) {
263 			for (String nextAcceptHeaderValue : acceptValues) {
264 				StringTokenizer tok = new StringTokenizer(nextAcceptHeaderValue, ",");
265 				while (tok.hasMoreTokens()) {
266 					String nextToken = tok.nextToken();
267 					int startSpaceIndex = -1;
268 					for (int i = 0; i < nextToken.length(); i++) {
269 						if (nextToken.charAt(i) != ' ') {
270 							startSpaceIndex = i;
271 							break;
272 						}
273 					}
274 
275 					if (startSpaceIndex == -1) {
276 						continue;
277 					}
278 
279 					int endSpaceIndex = -1;
280 					for (int i = startSpaceIndex; i < nextToken.length(); i++) {
281 						if (nextToken.charAt(i) == ' ' || nextToken.charAt(i) == ';') {
282 							endSpaceIndex = i;
283 							break;
284 						}
285 					}
286 
287 					float q = 1.0f;
288 					ResponseEncoding encoding;
289 					if (endSpaceIndex == -1) {
290 						if (startSpaceIndex == 0) {
291 							encoding = getEncodingForContentType(theReq.getServer().getFhirContext(), strict, nextToken);
292 						} else {
293 							encoding = getEncodingForContentType(theReq.getServer().getFhirContext(), strict, nextToken.substring(startSpaceIndex));
294 						}
295 					} else {
296 						encoding = getEncodingForContentType(theReq.getServer().getFhirContext(), strict, nextToken.substring(startSpaceIndex, endSpaceIndex));
297 						String remaining = nextToken.substring(endSpaceIndex + 1);
298 						StringTokenizer qualifierTok = new StringTokenizer(remaining, ";");
299 						while (qualifierTok.hasMoreTokens()) {
300 							String nextQualifier = qualifierTok.nextToken();
301 							int equalsIndex = nextQualifier.indexOf('=');
302 							if (equalsIndex != -1) {
303 								String nextQualifierKey = nextQualifier.substring(0, equalsIndex).trim();
304 								String nextQualifierValue = nextQualifier.substring(equalsIndex + 1, nextQualifier.length()).trim();
305 								if (nextQualifierKey.equals("q")) {
306 									try {
307 										q = Float.parseFloat(nextQualifierValue);
308 										q = Math.max(q, 0.0f);
309 									} catch (NumberFormatException e) {
310 										ourLog.debug("Invalid Accept header q value: {}", nextQualifierValue);
311 									}
312 								}
313 							}
314 						}
315 					}
316 
317 					if (encoding != null) {
318 						if (q > bestQ || (q == bestQ && encoding.getEncoding() == thePrefer)) {
319 							retVal = encoding;
320 							bestQ = q;
321 						}
322 					}
323 
324 				}
325 
326 			}
327 
328 		}
329 
330 		/*
331 		 * If the client hasn't given any indication about which response
332 		 * encoding they want, let's try the request encoding in case that
333 		 * is useful (basically this catches the case where the request
334 		 * has a Content-Type header but not an Accept header)
335 		 */
336 		if (retVal == null) {
337 			retVal = determineRequestEncodingNoDefaultReturnRE(theReq);
338 		}
339 
340 		return retVal;
341 	}

355 	public static Set<SummaryEnum> determineSummaryMode(RequestDetails theRequest) {
356 		Map<String, String[]> requestParams = theRequest.getParameters();
357 
358 		Set<SummaryEnum> retVal = SummaryEnumParameter.getSummaryValueOrNull(theRequest);
359 
360 		if (retVal == null) {
361 			/*
362 			 * HAPI originally supported a custom parameter called _narrative, but this has been superceded by an official
363 			 * parameter called _summary
364 			 */
365 			String[] narrative = requestParams.get(Constants.PARAM_NARRATIVE);
366 			if (narrative != null && narrative.length > 0) {
367 				try {
368 					NarrativeModeEnum narrativeMode = NarrativeModeEnum.valueOfCaseInsensitive(narrative[0]);
369 					switch (narrativeMode) {
370 					case NORMAL:
371 						retVal = Collections.singleton(SummaryEnum.FALSE);
372 						break;
373 					case ONLY:
374 						retVal = Collections.singleton(SummaryEnum.TEXT);
375 						break;
376 					case SUPPRESS:
377 						retVal = Collections.singleton(SummaryEnum.DATA);
378 						break;
379 					}
380 				} catch (IllegalArgumentException e) {
381 					ourLog.debug("Invalid {} parameger: {}", Constants.PARAM_NARRATIVE, narrative[0]);
382 				}
383 			}
384 		}
385 		if (retVal == null) {
386 			retVal = Collections.singleton(SummaryEnum.FALSE);
387 		}
388 
389 		return retVal;
390 	}

435 
436 		// Determine response encoding
437 		EncodingEnum responseEncoding = RestfulServerUtils.determineResponseEncodingWithDefault(theRequestDetails).getEncoding();
438 		IParser parser;
439 		switch (responseEncoding) {
440 		case JSON:
441 			parser = theContext.newJsonParser();
442 			break;
443 		case XML:
444 		default:
445 			parser = theContext.newXmlParser();
446 			break;
447 		}
448 
449 		configureResponseParser(theRequestDetails, parser);
450 
451 		return parser;

454 	public static Set<String> parseAcceptHeaderAndReturnHighestRankedOptions(HttpServletRequest theRequest) {
455 		Set<String> retVal = new HashSet<String>();
456 
457 		Enumeration<String> acceptValues = theRequest.getHeaders(Constants.HEADER_ACCEPT);
458 		if (acceptValues != null) {
459 			float bestQ = -1f;
460 			while (acceptValues.hasMoreElements()) {
461 				String nextAcceptHeaderValue = acceptValues.nextElement();
462 				Matcher m = ACCEPT_HEADER_PATTERN.matcher(nextAcceptHeaderValue);
463 				float q = 1.0f;
464 				while (m.find()) {
465 					String contentTypeGroup = m.group(1);
466 					if (isNotBlank(contentTypeGroup)) {
467 
468 						String name = m.group(3);
469 						String value = m.group(4);
470 						if (name != null && value != null) {
471 							if ("q".equals(name)) {
472 								try {
473 									q = Float.parseFloat(value);
474 									q = Math.max(q, 0.0f);
475 								} catch (NumberFormatException e) {
476 									ourLog.debug("Invalid Accept header q value: {}", value);
477 								}
478 							}
479 						}
480 
481 						if (q > bestQ) {
482 							retVal.clear();
483 							bestQ = q;
484 						}
485 
486 						if (q == bestQ) {
487 							retVal.add(contentTypeGroup.trim());
488 						}
489 
490 					}
491 
492 					if (!",".equals(m.group(5))) {
493 						break;
494 					}
495 				}
496 
497 			}
498 		}
499 
500 		return retVal;
501 	}

503 	public static PreferReturnEnum parsePreferHeader(String theValue) {
504 		if (isBlank(theValue)) {
505 			return null;
506 		}
507 
508 		StringTokenizer tok = new StringTokenizer(theValue, ",");
509 		while (tok.hasMoreTokens()) {
510 			String next = tok.nextToken();
511 			int eqIndex = next.indexOf('=');
512 			if (eqIndex == -1 || eqIndex >= next.length() - 2) {
513 				continue;
514 			}
515 
516 			String key = next.substring(0, eqIndex).trim();
517 			if (key.equals(Constants.HEADER_PREFER_RETURN) == false) {
518 				continue;
519 			}
520 
521 			String value = next.substring(eqIndex + 1).trim();
522 			if (value.length() < 2) {
523 				continue;
524 			}
525 			if ('"' == value.charAt(0) && '"' == value.charAt(value.length() - 1)) {
526 				value = value.substring(1, value.length() - 1);
527 			}
528 
529 			return PreferReturnEnum.fromHeaderValue(value);
530 		}
531 
532 		return null;
533 	}

535 	public static boolean prettyPrintResponse(IRestfulServerDefaults theServer, RequestDetails theRequest) {
536 		Map<String, String[]> requestParams = theRequest.getParameters();
537 		String[] pretty = requestParams.get(Constants.PARAM_PRETTY);
538 		boolean prettyPrint;
539 		if (pretty != null && pretty.length > 0) {
540 			if (Constants.PARAM_PRETTY_VALUE_TRUE.equals(pretty[0])) {
541 				prettyPrint = true;
542 			} else {
543 				prettyPrint = false;
544 			}
545 		} else {
546 			prettyPrint = theServer.isDefaultPrettyPrint();
547 			List<String> acceptValues = theRequest.getHeaders(Constants.HEADER_ACCEPT);
548 			if (acceptValues != null) {
549 				for (String nextAcceptHeaderValue : acceptValues) {
550 					if (nextAcceptHeaderValue.contains("pretty=true")) {
551 						prettyPrint = true;
552 					}
553 				}
554 			}
555 		}
556 		return prettyPrint;
557 	}

559 	public static Object streamResponseAsBundle(IRestfulServerDefaults theServer, Bundle bundle, Set<SummaryEnum> theSummaryMode, boolean respondGzip, RequestDetails theRequestDetails)
560 			throws IOException {
561 
562 		int status = 200;
563 
564 		// Determine response encoding
565 		EncodingEnum responseEncoding = RestfulServerUtils.determineResponseEncodingWithDefault(theRequestDetails).getEncoding();
566 
567 		String contentType = responseEncoding.getBundleContentType();
568 
569 		String charset = Constants.CHARSET_NAME_UTF8;
570 		Writer writer = theRequestDetails.getResponse().getResponseWriter(status, null, contentType, charset, respondGzip);
571 
572 		try {
573 			IParser parser = RestfulServerUtils.getNewParser(theServer.getFhirContext(), theRequestDetails);
574 			if (theSummaryMode.contains(SummaryEnum.TEXT)) {
575 				parser.setEncodeElements(TEXT_ENCODE_ELEMENTS);
576 			}
577 			parser.encodeBundleToWriter(bundle, writer);
578 		} catch (Exception e) {
579 			// always send a response, even if the parsing went wrong
580 		}
581 		return theRequestDetails.getResponse().sendWriterResponse(status, contentType, charset, writer);
582 	}

589 	public static Object streamResponseAsResource(IRestfulServerDefaults theServer, IBaseResource theResource, Set<SummaryEnum> theSummaryMode, int theStausCode, String theStatusMessage,
590 			boolean theAddContentLocationHeader, boolean respondGzip, RequestDetails theRequestDetails, IIdType theOperationResourceId, IPrimitiveType<Date> theOperationResourceLastUpdated)
591 			throws IOException {
592 		IRestfulResponse restUtil = theRequestDetails.getResponse();
593 
594 		// Determine response encoding
595 		ResponseEncoding responseEncoding = RestfulServerUtils.determineResponseEncodingNoDefault(theRequestDetails, theServer.getDefaultResponseEncoding());
596 
597 		String serverBase = theRequestDetails.getFhirServerBase();
598 		IIdType fullId = null;
599 		if (theOperationResourceId != null) {
600 			fullId = theOperationResourceId;
601 		} else if (theResource != null) {
602 			if (theResource.getIdElement() != null) {
603 				IIdType resourceId = theResource.getIdElement();
604 				fullId = fullyQualifyResourceIdOrReturnNull(theServer, theResource, serverBase, resourceId);
605 			}
606 		}
607 
608 		if (theAddContentLocationHeader && fullId != null) {
609 			if (theServer.getFhirContext().getVersion().getVersion().isOlderThan(FhirVersionEnum.DSTU3)) {
610 				restUtil.addHeader(Constants.HEADER_CONTENT_LOCATION, fullId.getValue());
611 			}
612 			restUtil.addHeader(Constants.HEADER_LOCATION, fullId.getValue());
613 		}
614 
615 		if (theServer.getETagSupport() == ETagSupportEnum.ENABLED) {
616 			if (fullId != null && fullId.hasVersionIdPart()) {
617 				restUtil.addHeader(Constants.HEADER_ETAG, "W/\"" + fullId.getVersionIdPart() + '"');
618 			}
619 		}
620 
621 		String contentType;
622 		if (theResource instanceof IBaseBinary && responseEncoding == null) {
623 			IBaseBinary bin = (IBaseBinary) theResource;
624 			if (isNotBlank(bin.getContentType())) {
625 				contentType = bin.getContentType();
626 			} else {
627 				contentType = Constants.CT_OCTET_STREAM;
628 			}
629 			// Force binary resources to download - This is a security measure to prevent
630 			// malicious images or HTML blocks being served up as content.
631 			restUtil.addHeader(Constants.HEADER_CONTENT_DISPOSITION, "Attachment;");
632 
633 			return restUtil.sendAttachmentResponse(bin, theStausCode, contentType);
634 		}
635 
636 		// Ok, we're not serving a binary resource, so apply default encoding
637 		if (responseEncoding == null) {
638 			responseEncoding = new ResponseEncoding(theServer.getFhirContext(), theServer.getDefaultResponseEncoding(), null);
639 		}
640 
641 		boolean encodingDomainResourceAsText = theSummaryMode.contains(SummaryEnum.TEXT);
642 		if (encodingDomainResourceAsText) {
643 			/*
644 			 * If the user requests "text" for a bundle, only suppress the non text elements in the Element.entry.resource
645 			 * parts, we're not streaming just the narrative as HTML (since bundles don't even
646 			 * have one)
647 			 */
648 			if ("Bundle".equals(theServer.getFhirContext().getResourceDefinition(theResource).getName())) {
649 				encodingDomainResourceAsText = false;
650 			}
651 		}
652 
653 		/*
654 		 * Last-Modified header
655 		 */
656 
657 		IPrimitiveType<Date> lastUpdated;
658 		if (theOperationResourceLastUpdated != null) {
659 			lastUpdated = theOperationResourceLastUpdated;
660 		} else {
661 			lastUpdated = extractLastUpdatedFromResource(theResource);
662 		}
663 		if (lastUpdated != null && lastUpdated.isEmpty() == false) {
664 			restUtil.addHeader(Constants.HEADER_LAST_MODIFIED, DateUtils.formatDate(lastUpdated.getValue()));
665 		}
666 
667 		/*
668 		 * Category header (DSTU1 only)
669 		 */
670 
671 		if (theResource instanceof IResource && theServer.getFhirContext().getVersion().getVersion() == FhirVersionEnum.DSTU1) {
672 			TagList list = (TagList) ((IResource) theResource).getResourceMetadata().get(ResourceMetadataKeyEnum.TAG_LIST);
673 			if (list != null) {
674 				for (Tag tag : list) {
675 					if (StringUtils.isNotBlank(tag.getTerm())) {
676 						restUtil.addHeader(Constants.HEADER_CATEGORY, tag.toHeaderValue());
677 					}
678 				}
679 			}
680 		}
681 
682 		/*
683 		 * Stream the response body
684 		 */
685 
686 		if (theResource == null) {
687 			contentType = null;
688 		} else if (encodingDomainResourceAsText) {
689 			contentType = Constants.CT_HTML;
690 		} else {
691 			contentType = responseEncoding.getResourceContentType();
692 		}
693 		String charset = Constants.CHARSET_NAME_UTF8;
694 
695 		Writer writer = restUtil.getResponseWriter(theStausCode, theStatusMessage, contentType, charset, respondGzip);
696 		if (theResource == null) {
697 			// No response is being returned
698 		} else if (encodingDomainResourceAsText && theResource instanceof IResource) {
699 			writer.append(((IResource) theResource).getText().getDiv().getValueAsString());
700 		} else {
701 			IParser parser = getNewParser(theServer.getFhirContext(), theRequestDetails);
702 			parser.encodeResourceToWriter(theResource, writer);
703 		}
704 
705 		return restUtil.sendWriterResponse(theStausCode, contentType, charset, writer);
706 	}

716 			ourLog.debug("Failed to parse {} value '{}': {}", new Object[] { theParamName, retVal[0], e });
717 			return null;
718 		}

748 	/**
749 	 * Return type for {@link RestfulServerUtils#determineRequestEncodingNoDefault(RequestDetails)}
750 	 */
751 	public static class ResponseEncoding {
752 		private final EncodingEnum myEncoding;
753 		private final Boolean myNonLegacy;
754 
755 		public ResponseEncoding(FhirContext theCtx, EncodingEnum theEncoding, String theContentType) {
756 			super();
757 			myEncoding = theEncoding;
758 			if (theContentType != null) {
759 				if (theContentType.equals(EncodingEnum.JSON_PLAIN_STRING) || theContentType.equals(EncodingEnum.XML_PLAIN_STRING)) {
760 					FhirVersionEnum ctxtEnum = theCtx.getVersion().getVersion();
761 					myNonLegacy = ctxtEnum.isNewerThan(FhirVersionEnum.DSTU3)
762 							|| (ctxtEnum.isEquivalentTo(FhirVersionEnum.DSTU3) && !"1.4.0".equals(theCtx.getVersion().getVersion().getFhirVersionString()));
763 				} else {
764 					myNonLegacy = EncodingEnum.isNonLegacy(theContentType);
765 				}
766 			} else {
767 				FhirVersionEnum ctxtEnum = theCtx.getVersion().getVersion();
768 				if (ctxtEnum.isOlderThan(FhirVersionEnum.DSTU3)
769 					|| (ctxtEnum.isEquivalentTo(FhirVersionEnum.DSTU3) && "1.4.0".equals(theCtx.getVersion().getVersion().getFhirVersionString()))) {
770 					myNonLegacy = null;
771 				} else {
772 					myNonLegacy = Boolean.TRUE;
773 				}
774 			}
775 		}
776 
777 		public EncodingEnum getEncoding() {
778 			return myEncoding;
779 		}
780 
781 		public String getResourceContentType() {
782 			if (Boolean.TRUE.equals(isNonLegacy())) {
783 				return getEncoding().getResourceContentTypeNonLegacy();
784 			} else {
785 				return getEncoding().getResourceContentType();
786 			}
787 		}
788 
789 		public Boolean isNonLegacy() {
790 			return myNonLegacy;
791 		}
792 	}

794 	public static void addAcceptHeaderToRequest(EncodingEnum theEncoding, IHttpRequest theHttpRequest, FhirContext theContext) {
795 		if (theEncoding == null) {
796 			if (theContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU2) == false) {
797 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.HEADER_ACCEPT_VALUE_XML_OR_JSON_LEGACY);
798 			} else {
799 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.HEADER_ACCEPT_VALUE_XML_OR_JSON_NON_LEGACY);
800 			}
801 		} else if (theEncoding == EncodingEnum.JSON) {
802 			if (theContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU2) == false) {
803 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.CT_FHIR_JSON);
804 			} else {
805 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.HEADER_ACCEPT_VALUE_JSON_NON_LEGACY);
806 			}
807 		} else if (theEncoding == EncodingEnum.XML) {
808 			if (theContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU2) == false) {
809 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.CT_FHIR_XML);
810 			} else {
811 				theHttpRequest.addHeader(Constants.HEADER_ACCEPT, Constants.HEADER_ACCEPT_VALUE_XML_NON_LEGACY);
812 			}
813 		}
