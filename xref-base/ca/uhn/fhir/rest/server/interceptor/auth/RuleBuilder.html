34  public class RuleBuilder implements IAuthRuleBuilder {
35  
36  	private ArrayList<IAuthRule> myRules;
37  
38  	public RuleBuilder() {
39  		myRules = new ArrayList<IAuthRule>();
40  	}
41  	
42  	@Override
43  	public IAuthRuleBuilderRule allow() {
44  		return allow(null);
45  	}
46  
47  	@Override
48  	public IAuthRuleBuilderRule allow(String theRuleName) {
49  		return new RuleBuilderRule(RuleVerdictEnum.ALLOW, theRuleName);
50  	}
51  
52  	@Override
53  	public IAuthRuleBuilderRuleOpClassifierFinished allowAll() {
54  		return allowAll(null);
55  	}
56  
57  	@Override
58  	public IAuthRuleBuilderRuleOpClassifierFinished allowAll(String theRuleName) {
59  		myRules.add(new Rule(theRuleName).setOp(RuleOpEnum.ALLOW_ALL));
60  		return new RuleBuilderFinished();
61  	}
62  
63  	@Override
64  	public List<IAuthRule> build() {
65  		return myRules;
66  	}
67  
68  	@Override
69  	public IAuthRuleBuilderRule deny() {
70  		return deny(null);
71  	}
72  
73  	@Override
74  	public IAuthRuleBuilderRule deny(String theRuleName) {
75  		return new RuleBuilderRule(RuleVerdictEnum.DENY, theRuleName);
76  	}
77  
78  	@Override
79  	public IAuthRuleBuilderRuleOpClassifierFinished denyAll() {
80  		return denyAll(null);
81  	}
82  
83  	@Override
84  	public IAuthRuleBuilderRuleOpClassifierFinished denyAll(String theRuleName) {
85  		myRules.add(new Rule(theRuleName).setOp(RuleOpEnum.DENY_ALL));
86  		return new RuleBuilderFinished();
87  	}
88  
89  	private final class RuleBuilderFinished implements IAuthRuleBuilderRuleOpClassifierFinished {
90  
91  		@Override
92  		public IAuthRuleBuilder andThen() {
93  			return RuleBuilder.this;
94  		}
95  
96  		@Override
97  		public List<IAuthRule> build() {
98  			return myRules;
99  		}
100 	}

102 	private class RuleBuilderRule implements IAuthRuleBuilderRule {
103 
104 		private RuleOpEnum myRuleOp;
105 		private RuleVerdictEnum myRuleMode;
106 		private String myRuleName;
107 
108 		public RuleBuilderRule(RuleVerdictEnum theRuleMode, String theRuleName) {
109 			myRuleMode = theRuleMode;
110 			myRuleName = theRuleName;
111 		}
112 
113 		@Override
114 		public RuleBuilderFinished metadata() {
115 			Rule rule = new Rule(myRuleName);
116 			rule.setOp(RuleOpEnum.METADATA);
117 			rule.setMode(myRuleMode);
118 			myRules.add(rule);
119 			return new RuleBuilderFinished();
120 		}
121 		
122 		@Override
123 		public IAuthRuleBuilderRuleOp read() {
124 			myRuleOp = RuleOpEnum.READ;
125 			return new RuleBuilderRuleOp();
126 		}
127 
128 		@Override
129 		public IAuthRuleBuilderRuleTransaction transaction() {
130 			myRuleOp = RuleOpEnum.TRANSACTION;
131 			return new RuleBuilderRuleTransaction();
132 		}
133 		
134 		@Override
135 		public IAuthRuleBuilderRuleOp write() {
136 			myRuleOp = RuleOpEnum.WRITE;
137 			return new RuleBuilderRuleOp();
138 		}
139 		
140 		private class RuleBuilderRuleOp implements IAuthRuleBuilderRuleOp {
141 
142 			private AppliesTypeEnum myAppliesTo;
143 			private Set<?> myAppliesToTypes;
144 
145 			@Override
146 			public IAuthRuleBuilderRuleOpClassifier allResources() {
147 				myAppliesTo = AppliesTypeEnum.ALL_RESOURCES;
148 				return new RuleBuilderRuleOpClassifier();
149 			}
150 
151 			@Override
152 			public IAuthRuleBuilderRuleOpClassifier resourcesOfType(Class<? extends IBaseResource> theType) {
153 				Validate.notNull(theType, "theType must not be null");
154 				myAppliesTo = AppliesTypeEnum.TYPES;
155 				myAppliesToTypes = Collections.singleton(theType);
156 				return new RuleBuilderRuleOpClassifier();
157 			}
158 
159 			private class RuleBuilderRuleOpClassifier implements IAuthRuleBuilderRuleOpClassifier {
160 
161 				private ClassifierTypeEnum myClassifierType;
162 				private String myInCompartmentName;
163 				private Collection<? extends IIdType> myInCompartmentOwners;
164 
165 				private IAuthRuleBuilderRuleOpClassifierFinished finished() {
166 					
167 					Rule rule = new Rule(myRuleName);
168 					rule.setMode(myRuleMode);
169 					rule.setOp(myRuleOp);
170 					rule.setAppliesTo(myAppliesTo);
171 					rule.setAppliesToTypes(myAppliesToTypes);
172 					rule.setClassifierType(myClassifierType);
173 					rule.setClassifierCompartmentName(myInCompartmentName);
174 					rule.setClassifierCompartmentOwners(myInCompartmentOwners);
175 					myRules.add(rule);
176 					
177 					return new RuleBuilderFinished();
178 				}
179 
180 				@Override
181 				public IAuthRuleBuilderRuleOpClassifierFinished inCompartment(String theCompartmentName, Collection<? extends IIdType> theOwners) {
182 					Validate.notBlank(theCompartmentName, "theCompartmentName must not be null");
183 					Validate.notNull(theOwners, "theOwners must not be null");
184 					Validate.noNullElements(theOwners, "theOwners must not contain any null elements");
185 					for (IIdType next : theOwners) {
186 						validateOwner(next);
187 					}
188 					myClassifierType = ClassifierTypeEnum.IN_COMPARTMENT;
189 					return finished();
190 				}
191 
192 				@Override
193 				public IAuthRuleBuilderRuleOpClassifierFinished inCompartment(String theCompartmentName, IIdType theOwner) {
194 					Validate.notBlank(theCompartmentName, "theCompartmentName must not be null");
195 					Validate.notNull(theOwner, "theOwner must not be null");
196 					validateOwner(theOwner);
197 					myInCompartmentName = theCompartmentName;
198 					myInCompartmentOwners = Collections.singletonList(theOwner);
199 					myClassifierType = ClassifierTypeEnum.IN_COMPARTMENT;
200 					return finished();
201 				}
202 
203 				private void validateOwner(IIdType theOwner) {
204 					Validate.notBlank(theOwner.getIdPart(), "owner.getIdPart() must not be null or empty");
205 					Validate.notBlank(theOwner.getIdPart(), "owner.getResourceType() must not be null or empty");
206 				}
207 
208 				@Override
209 				public IAuthRuleBuilderRuleOpClassifierFinished withAnyId() {
210 					myClassifierType = ClassifierTypeEnum.ANY_ID;
211 					return finished();
212 				}
213 
214 			}
215 
216 		}

218 		private class RuleBuilderRuleTransaction implements IAuthRuleBuilderRuleTransaction {
219 
220 			@Override
221 			public IAuthRuleBuilderRuleTransactionOp withAnyOperation() {
222 				return new RuleBuilderRuleTransactionOp();
223 			}
224 			private class RuleBuilderRuleTransactionOp implements IAuthRuleBuilderRuleTransactionOp {
225 
226 				@Override
227 				public IAuthRuleBuilderRuleOpClassifierFinished andApplyNormalRules() {
228 					Rule rule = new Rule(myRuleName);
229 					rule.setMode(myRuleMode);
230 					rule.setOp(myRuleOp);
231 					rule.setTransactionAppliesToOp(TransactionAppliesToEnum.ANY_OPERATION);
232 					myRules.add(rule);
233 					return new RuleBuilderFinished();
234 				}
235 				
236 			}
237 			
238 		}
