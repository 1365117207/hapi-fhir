25  /**
26   * Base interface for ID datatype. 
27   * 
28   * <p>
29   * <b>Concrete Implementations:</b> This interface is often returned and/or accepted by methods in HAPI's API
30   * where either {@link ca.uhn.fhir.model.primitive.IdDt} (the HAPI structure ID type) or 
31   * <code>org.hl7.fhir.instance.model.IdType</code> (the RI structure ID type) will be used, depending on 
32   * which version of the strctures your application is using.   
33   * </p>
34   */
35  public interface IIdType {
36  
37  	boolean isEmpty();
38  
39  	/**
40  	 * Returns <code>true</code> if the ID is a local reference (in other words, it begins with the '#' character)
41  	 */
42  	boolean isLocal();
43  
44  	/**
45  	 * Returns the value of this ID. Note that this value may be a fully qualified URL, a relative/partial URL, or a simple ID. Use {@link #getIdPart()} to get just the ID portion.
46  	 * 
47  	 * @see #getIdPart()
48  	 */
49  	String getValue();
50  
51  	/**
52  	 * Returns only the logical ID part of this ID. For example, given the ID
53  	 * "http://example,.com/fhir/Patient/123/_history/456", this method would
54  	 * return "123".
55  	 */
56  	String getIdPart();
57  
58  	/**
59  	 * Returns <code>true</code> if this ID contains an actual ID part. For example, the ID part is
60  	 * the '123' in the following ID: <code>http://example.com/fhir/Patient/123/_history/55</code>;
61  	 */
62  	boolean hasIdPart();
63  
64  	/**
65  	 * Returns the server base URL if this ID contains one. For example, the base URL is
66  	 * the 'http://example.com/fhir' in the following ID: <code>http://example.com/fhir/Patient/123/_history/55</code>
67  	 */
68  	String getBaseUrl();
69  
70  	IIdType toUnqualifiedVersionless();
71  
72  	IIdType toVersionless();
73  
74  	IIdType setValue(String theString);
75  
76  	boolean hasVersionIdPart();
77  
78  	String getVersionIdPart();
79  
80  	IIdType toUnqualified();
81  
82  	boolean hasResourceType();
83  
84  	IIdType withResourceType(String theResName);
85  
86  	String getResourceType();
87  
88  	IIdType withServerBase(String theServerBase, String theResourceName);
89  
90  	/**
91  	 * Returns <code>true</code> if this ID contains an absolute URL (in other words, a URL starting with "http://" or "https://"
92  	 */
93  	boolean isAbsolute();
94  
95  	boolean isIdPartValidLong();
96  
97  	Long getIdPartAsLong();
98  
99  	boolean hasBaseUrl();
100 
101 	IIdType withVersion(String theVersion);
102 
103 	void applyTo(IBaseResource theResource);
104 
105 }
