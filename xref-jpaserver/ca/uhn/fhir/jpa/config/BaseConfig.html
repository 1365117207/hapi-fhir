23  import javax.annotation.Resource;
24  
25  import org.springframework.beans.factory.annotation.Autowired;
26  import org.springframework.context.ApplicationContext;
27  import org.springframework.context.annotation.Bean;
28  import org.springframework.context.annotation.Configuration;
29  import org.springframework.context.annotation.Lazy;
30  import org.springframework.context.support.PropertySourcesPlaceholderConfigurer;
31  import org.springframework.core.env.Environment;
32  import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
33  import org.springframework.scheduling.TaskScheduler;
34  import org.springframework.scheduling.annotation.EnableScheduling;
35  import org.springframework.scheduling.annotation.SchedulingConfigurer;
36  import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
37  import org.springframework.scheduling.config.ScheduledTaskRegistrar;
38  
39  import ca.uhn.fhir.context.FhirContext;
40  
41  @Configuration
42  @EnableScheduling
43  @EnableJpaRepositories(basePackages = "ca.uhn.fhir.jpa.dao.data")
44  public class BaseConfig implements SchedulingConfigurer {
45  
46  	private static FhirContext ourFhirContextDstu1;
47  	private static FhirContext ourFhirContextDstu2;
48  	private static FhirContext ourFhirContextDstu2Hl7Org;
49  	private static FhirContext ourFhirContextDstu3;
50  
51  	@Resource
52  	private ApplicationContext myAppCtx;
53  
54  	@Autowired
55  	protected Environment myEnv;
56  
57  	@Override
58  	public void configureTasks(ScheduledTaskRegistrar theTaskRegistrar) {
59  		theTaskRegistrar.setTaskScheduler(taskScheduler());
60  	}
61  
62  	@Bean(name = "myFhirContextDstu1")
63  	@Lazy
64  	public FhirContext fhirContextDstu1() {
65  		if (ourFhirContextDstu1 == null) {
66  			ourFhirContextDstu1 = FhirContext.forDstu1();
67  		}
68  		return ourFhirContextDstu1;
69  	}
70  
71  	@Bean(name = "myFhirContextDstu2")
72  	@Lazy
73  	public FhirContext fhirContextDstu2() {
74  		if (ourFhirContextDstu2 == null) {
75  			ourFhirContextDstu2 = FhirContext.forDstu2();
76  		}
77  		return ourFhirContextDstu2;
78  	}
79  
80  	@Bean(name = "myFhirContextDstu2Hl7Org")
81  	@Lazy
82  	public FhirContext fhirContextDstu2Hl7Org() {
83  		if (ourFhirContextDstu2Hl7Org == null) {
84  			ourFhirContextDstu2Hl7Org = FhirContext.forDstu2Hl7Org();
85  		}
86  		return ourFhirContextDstu2Hl7Org;
87  	}
88  
89  	@Bean(name = "myFhirContextDstu3")
90  	@Lazy
91  	public FhirContext fhirContextDstu3() {
92  		if (ourFhirContextDstu3 == null) {
93  			ourFhirContextDstu3 = FhirContext.forDstu3();
94  		}
95  		return ourFhirContextDstu3;
96  	}
97  
98  	@Bean
99  	public TaskScheduler taskScheduler() {
100 		ThreadPoolTaskScheduler retVal = new ThreadPoolTaskScheduler();
101 		retVal.setPoolSize(5);
102 		return retVal;
103 	}
104 
105 	/**
106 	 * This lets the "@Value" fields reference properties from the properties file
107 	 */
108 	@Bean
109 	public static PropertySourcesPlaceholderConfigurer propertySourcesPlaceholderConfigurer() {
110 		return new PropertySourcesPlaceholderConfigurer();
111 	}
