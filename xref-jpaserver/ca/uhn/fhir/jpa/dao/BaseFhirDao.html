153 	protected List<ResourceLink> extractResourceLinks(ResourceTable theEntity, IResource theResource) {
154 		ArrayList<ResourceLink> retVal = new ArrayList<ResourceLink>();
155 
156 		RuntimeResourceDefinition def = getContext().getResourceDefinition(theResource);
157 		for (RuntimeSearchParam nextSpDef : def.getSearchParams()) {
158 			if (nextSpDef.getParamType() != SearchParamTypeEnum.REFERENCE) {
159 				continue;
160 			}
161 
162 			String nextPathsUnsplit = nextSpDef.getPath();
163 			if (isBlank(nextPathsUnsplit)) {
164 				continue;
165 			}
166 
167 			boolean multiType = false;
168 			if (nextPathsUnsplit.endsWith("[x]")) {
169 				multiType = true;
170 			}
171 
172 			for (Object nextObject : extractValues(nextPathsUnsplit, theResource)) {
173 				if (nextObject == null) {
174 					continue;
175 				}
176 
177 				ResourceLink nextEntity;
178 				if (nextObject instanceof BaseResourceReferenceDt) {
179 					BaseResourceReferenceDt nextValue = (BaseResourceReferenceDt) nextObject;
180 					if (nextValue.isEmpty()) {
181 						continue;
182 					}
183 					if (nextValue.getReference().isEmpty() || nextValue.getReference().getValue().startsWith("#")) {
184 						// This is a blank or contained resource reference
185 						continue;
186 					}
187 
188 					String typeString = nextValue.getReference().getResourceType();
189 					if (isBlank(typeString)) {
190 						throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource type - " + nextValue.getReference().getValue());
191 					}
192 					Class<? extends IBaseResource> type = getContext().getResourceDefinition(typeString).getImplementingClass();
193 					String id = nextValue.getReference().getIdPart();
194 					if (StringUtils.isBlank(id)) {
195 						continue;
196 					}
197 
198 					IFhirResourceDao<?> dao = getDao(type);
199 					if (dao == null) {
200 						StringBuilder b = new StringBuilder();
201 						b.append("This server (version ");
202 						b.append(myContext.getVersion().getVersion());
203 						b.append(") is not able to handle resources of type[");
204 						b.append(nextValue.getReference().getResourceType());
205 						b.append("] - Valid resource types for this server: ");
206 						b.append(myResourceTypeToDao.keySet().toString());
207 
208 						throw new InvalidRequestException(b.toString());
209 					}
210 					Long valueOf;
211 					try {
212 						valueOf = translateForcedIdToPid(nextValue.getReference());
213 					} catch (Exception e) {
214 						String resName = getContext().getResourceDefinition(type).getName();
215 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit + " (this is an invalid ID, must be numeric on this server)");
216 					}
217 					ResourceTable target = myEntityManager.find(ResourceTable.class, valueOf);
218 					if (target == null) {
219 						String resName = getContext().getResourceDefinition(type).getName();
220 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
221 					}
222 					nextEntity = new ResourceLink(nextPathsUnsplit, theEntity, target);
223 				} else {
224 					if (!multiType) {
225 						throw new ConfigurationException("Search param " + nextSpDef.getName() + " is of unexpected datatype: " + nextObject.getClass());
226 					} else {
227 						continue;
228 					}
229 				}
230 				if (nextEntity != null) {
231 					retVal.add(nextEntity);
232 				}
233 			}
234 		}
235 
236 		theEntity.setHasLinks(retVal.size() > 0);
237 
238 		return retVal;
239 	}

261 	private List<Object> extractValues(String thePaths, IResource theResource) {
262 		List<Object> values = new ArrayList<Object>();
263 		String[] nextPathsSplit = thePaths.split("\\|");
264 		FhirTerser t = getContext().newTerser();
265 		for (String nextPath : nextPathsSplit) {
266 			String nextPathTrimmed = nextPath.trim();
267 			try {
268 				values.addAll(t.getValues(theResource, nextPathTrimmed));
269 			} catch (Exception e) {
270 				RuntimeResourceDefinition def = myContext.getResourceDefinition(theResource);
271 				ourLog.warn("Failed to index values from path[{}] in resource type[{}]: ", nextPathTrimmed, def.getName(), e.toString());
272 			}
273 		}
274 		return values;
275 	}

277 	private void findMatchingTagIds(String theResourceName, IdDt theResourceId, Set<Long> tagIds, Class<? extends BaseTag> entityClass) {
278 		{
279 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
280 			CriteriaQuery<Tuple> cq = builder.createTupleQuery();
281 			Root<? extends BaseTag> from = cq.from(entityClass);
282 			cq.multiselect(from.get("myTagId").as(Long.class)).distinct(true);
283 
284 			if (theResourceName != null) {
285 				Predicate typePredicate = builder.equal(from.get("myResourceType"), theResourceName);
286 				if (theResourceId != null) {
287 					cq.where(typePredicate, builder.equal(from.get("myResourceId"), translateForcedIdToPid(theResourceId)));
288 				} else {
289 					cq.where(typePredicate);
290 				}
291 			}
292 
293 			TypedQuery<Tuple> query = myEntityManager.createQuery(cq);
294 			for (Tuple next : query.getResultList()) {
295 				tagIds.add(next.get(0, Long.class));
296 			}
297 		}

352 	protected TagList getTags(Class<? extends IResource> theResourceType, IdDt theResourceId) {
353 		String resourceName = null;
354 		if (theResourceType != null) {
355 			resourceName = toResourceName(theResourceType);
356 			if (theResourceId != null && theResourceId.hasVersionIdPart()) {
357 				IFhirResourceDao<? extends IResource> dao = getDao(theResourceType);
358 				BaseHasResource entity = dao.readEntity(theResourceId);
359 				TagList retVal = new TagList();
360 				for (BaseTag next : entity.getTags()) {
361 					retVal.add(next.getTag().toTag());
362 				}
363 				return retVal;
364 			}
365 		}
366 
367 		Set<Long> tagIds = new HashSet<Long>();
368 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceTag.class);
369 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceHistoryTag.class);
370 		if (tagIds.isEmpty()) {
371 			return new TagList();
372 		}
373 		{
374 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
375 			CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
376 			Root<TagDefinition> from = cq.from(TagDefinition.class);
377 			cq.where(from.get("myId").in(tagIds));
378 			cq.orderBy(builder.asc(from.get("myScheme")), builder.asc(from.get("myTerm")));
379 			TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
380 			q.setMaxResults(getConfig().getHardTagListLimit());
381 
382 			TagList retVal = new TagList();
383 			for (TagDefinition next : q.getResultList()) {
384 				retVal.add(next.toTag());
385 			}
386 
387 			return retVal;
388 		}

391 	protected IBundleProvider history(String theResourceName, Long theId, Date theSince) {
392 		final List<HistoryTuple> tuples = new ArrayList<HistoryTuple>();
393 
394 		final InstantDt end = createHistoryToTimestamp();
395 
396 		StopWatch timer = new StopWatch();
397 
398 		int limit = 10000;
399 
400 		// Get list of IDs
401 		searchHistoryCurrentVersion(theResourceName, theId, theSince, end.getValue(), limit, tuples);
402 		assert tuples.size() < 2 || !tuples.get(tuples.size() - 2).getUpdated().before(tuples.get(tuples.size() - 1).getUpdated());
403 		ourLog.info("Retrieved {} history IDs from current versions in {} ms", tuples.size(), timer.getMillisAndRestart());
404 
405 		searchHistoryHistory(theResourceName, theId, theSince, end.getValue(), limit, tuples);
406 		assert tuples.size() < 2 || !tuples.get(tuples.size() - 2).getUpdated().before(tuples.get(tuples.size() - 1).getUpdated());
407 		ourLog.info("Retrieved {} history IDs from previous versions in {} ms", tuples.size(), timer.getMillisAndRestart());
408 
409 		// Sort merged list
410 		Collections.sort(tuples, Collections.reverseOrder());
411 		assert tuples.size() < 2 || !tuples.get(tuples.size() - 2).getUpdated().before(tuples.get(tuples.size() - 1).getUpdated());
412 
413 		return new IBundleProvider() {
414 
415 			@Override
416 			public InstantDt getPublished() {
417 				return end;
418 			}
419 
420 			@Override
421 			public List<IResource> getResources(final int theFromIndex, final int theToIndex) {
422 				final StopWatch timer = new StopWatch();
423 				TransactionTemplate template = new TransactionTemplate(myPlatformTransactionManager);
424 				return template.execute(new TransactionCallback<List<IResource>>() {
425 					@Override
426 					public List<IResource> doInTransaction(TransactionStatus theStatus) {
427 						List<BaseHasResource> resEntities = Lists.newArrayList();
428 
429 						List<HistoryTuple> tupleSubList = tuples.subList(theFromIndex, theToIndex);
430 						searchHistoryCurrentVersion(tupleSubList, resEntities);
431 						ourLog.info("Loaded history from current versions in {} ms", timer.getMillisAndRestart());
432 
433 						searchHistoryHistory(tupleSubList, resEntities);
434 						ourLog.info("Loaded history from previous versions in {} ms", timer.getMillisAndRestart());
435 
436 						Collections.sort(resEntities, new Comparator<BaseHasResource>() {
437 							@Override
438 							public int compare(BaseHasResource theO1, BaseHasResource theO2) {
439 								return theO2.getUpdated().getValue().compareTo(theO1.getUpdated().getValue());
440 							}
441 						});
442 
443 						int limit = theToIndex - theFromIndex;
444 						if (resEntities.size() > limit) {
445 							resEntities = resEntities.subList(0, limit);
446 						}
447 
448 						ArrayList<IResource> retVal = new ArrayList<IResource>();
449 						for (BaseHasResource next : resEntities) {
450 							RuntimeResourceDefinition type;
451 							try {
452 								type = myContext.getResourceDefinition(next.getResourceType());
453 							} catch (DataFormatException e) {
454 								if (next.getFhirVersion() != getContext().getVersion().getVersion()) {
455 									ourLog.info("Ignoring history resource of type[{}] because it is not compatible with version[{}]", next.getResourceType(), getContext().getVersion().getVersion());
456 									continue;
457 								}
458 								throw e;
459 							}
460 							IResource resource = (IResource) toResource(type.getImplementingClass(), next);
461 							retVal.add(resource);
462 						}
463 						return retVal;
464 					}
465 				});
466 			}
467 
468 			@Override
469 			public int size() {
470 				return tuples.size();
471 			}
472 		};
473 	}

475 	protected boolean isValidPid(IdDt theId) {
476 		String idPart = theId.getIdPart();
477 		for (int i = 0; i < idPart.length(); i++) {
478 			char nextChar = idPart.charAt(i);
479 			if (nextChar < '0' || nextChar > '9') {
480 				return false;
481 			}
482 		}
483 		return true;
484 	}

525 	protected void populateResourceIntoEntity(IResource theResource, ResourceTable theEntity) {
526 
527 		if (theEntity.getPublished().isEmpty()) {
528 			theEntity.setPublished(new Date());
529 		}
530 		theEntity.setUpdated(new Date());
531 
532 		theEntity.setResourceType(toResourceName(theResource));
533 
534 		List<BaseResourceReferenceDt> refs = myContext.newTerser().getAllPopulatedChildElementsOfType(theResource, BaseResourceReferenceDt.class);
535 		for (BaseResourceReferenceDt nextRef : refs) {
536 			if (nextRef.getReference().isEmpty() == false) {
537 				if (nextRef.getReference().hasVersionIdPart()) {
538 					nextRef.setReference(nextRef.getReference().toUnqualifiedVersionless());
539 				}
540 			}
541 		}
542 
543 		String encoded = myConfig.getResourceEncoding().newParser(myContext).encodeResourceToString(theResource);
544 		ResourceEncodingEnum encoding = myConfig.getResourceEncoding();
545 		theEntity.setEncoding(encoding);
546 		theEntity.setFhirVersion(myContext.getVersion().getVersion());
547 		try {
548 			switch (encoding) {
549 			case JSON:
550 				theEntity.setResource(encoded.getBytes("UTF-8"));
551 				break;
552 			case JSONC:
553 				theEntity.setResource(GZipUtil.compress(encoded));

558 		}

569 		String title = ResourceMetadataKeyEnum.TITLE.get(theResource);
570 		if (title != null && title.length() > BaseHasResource.MAX_TITLE_LENGTH) {

588 	protected SearchParameterMap translateMatchUrl(String theMatchUrl, RuntimeResourceDefinition resourceDef) {
589 		SearchParameterMap paramMap = new SearchParameterMap();
590 		List<NameValuePair> parameters;
591 		try {
592 			parameters = URLEncodedUtils.parse(new URI(theMatchUrl), "UTF-8");
593 		} catch (URISyntaxException e) {
594 			throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Error was: " + e.toString());
595 		}
596 
597 		ArrayListMultimap<String, QualifiedParamList> nameToParamLists = ArrayListMultimap.create();
598 		for (NameValuePair next : parameters) {
599 			String paramName = next.getName();
600 			String qualifier = null;
601 			for (int i = 0; i < paramMap.size(); i++) {
602 				switch (paramName.charAt(i)) {
603 					case '.':
604 					case ':':
605 						qualifier = paramName.substring(i);
606 						paramName = paramName.substring(0, i);
607 						i = Integer.MAX_VALUE;
608 						break;
609 				}
610 			}
611 
612 			QualifiedParamList paramList = QualifiedParamList.splitQueryStringByCommasIgnoreEscape(qualifier, next.getValue());
613 			nameToParamLists.put(paramName, paramList);
614 		}
615 
616 		for (String nextParamName : nameToParamLists.keySet()) {
617 			RuntimeSearchParam paramDef = resourceDef.getSearchParam(nextParamName);
618 			if (paramDef == null) {
619 				throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Resource type " + resourceDef.getName() + " does not have a parameter with name: " + nextParamName);
620 			}
621 
622 			List<QualifiedParamList> paramList = nameToParamLists.get(nextParamName);
623 			IQueryParameterAnd<?> param = MethodUtil.parseQueryParams(paramDef, nextParamName, paramList);
624 			paramMap.add(nextParamName, param);
625 		}
626 		return paramMap;
627 	}

798 	protected <T extends IBaseResource> T toResource(Class<T> theResourceType, BaseHasResource theEntity) {
799 		String resourceText = null;
800 		switch (theEntity.getEncoding()) {
801 		case JSON:
802 			try {
803 				resourceText = new String(theEntity.getResource(), "UTF-8");
804 			} catch (UnsupportedEncodingException e) {
805 				throw new Error("Should not happen", e);
806 			}
807 			break;
808 		case JSONC:
809 			resourceText = GZipUtil.decompress(theEntity.getResource());
810 			break;
811 		}
812 
813 		IParser parser = theEntity.getEncoding().newParser(getContext(theEntity.getFhirVersion()));
814 		T retVal;
815 		try {
816 			retVal = parser.parseResource(theResourceType, resourceText);
817 		} catch (Exception e) {
818 			StringBuilder b = new StringBuilder();
819 			b.append("Failed to parse database resource[");
820 			b.append(theResourceType);
821 			b.append("/");
822 			b.append(theEntity.getIdDt().getIdPart());
823 			b.append(" (pid ");
824 			b.append(theEntity.getId());
825 			b.append(", version ");
826 			b.append(myContext.getVersion().getVersion());
827 			b.append("): ");
828 			b.append(e.getMessage());
829 			String msg = b.toString();
830 			ourLog.error(msg, e);
831 			throw new DataFormatException(msg, e);
832 		}
833 		IResource res = (IResource) retVal;
834 		res.setId(theEntity.getIdDt());
835 
836 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.VERSION_ID, theEntity.getVersion());
837 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.PUBLISHED, theEntity.getPublished());
838 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.UPDATED, theEntity.getUpdated());
839 
840 		if (theEntity.getTitle() != null) {
841 			ResourceMetadataKeyEnum.TITLE.put(res, theEntity.getTitle());
842 		}
843 
844 		if (theEntity.getDeleted() != null) {
845 			ResourceMetadataKeyEnum.DELETED_AT.put(res, new InstantDt(theEntity.getDeleted()));
846 		}
847 
848 		Collection<? extends BaseTag> tags = theEntity.getTags();
849 		if (theEntity.isHasTags()) {
850 			TagList tagList = new TagList();
851 			for (BaseTag next : tags) {
852 				tagList.add(new Tag(next.getTag().getScheme(), next.getTag().getTerm(), next.getTag().getLabel()));
853 			}
854 			res.getResourceMetadata().put(ResourceMetadataKeyEnum.TAG_LIST, tagList);
855 		}
856 
857 		return retVal;
858 	}

898 		}
899 
900 		if (theResource != null) {

1060 	protected static String normalizeString(String theString) {
1061 		char[] out = new char[theString.length()];
1062 		theString = Normalizer.normalize(theString, Normalizer.Form.NFD);
1063 		int j = 0;
1064 		for (int i = 0, n = theString.length(); i < n; ++i) {
1065 			char c = theString.charAt(i);
1066 			if (c <= '\u007F') {
1067 				out[j++] = c;
1068 			}
1069 		}
1070 		return new String(out).toUpperCase();
1071 	}
