144 		if (id.isEmpty() == false && id.hasIdPart()) {

149 			fid.setForcedId(id.getIdPart());
150 			fid.setResource(entity);
151 			entity.setForcedId(fid);
152 		}
153 	}

161 	protected List<ResourceLink> extractResourceLinks(ResourceTable theEntity, IResource theResource) {
162 		ArrayList<ResourceLink> retVal = new ArrayList<ResourceLink>();
163 
164 		RuntimeResourceDefinition def = getContext().getResourceDefinition(theResource);
165 		for (RuntimeSearchParam nextSpDef : def.getSearchParams()) {
166 			if (nextSpDef.getParamType() != RestSearchParameterTypeEnum.REFERENCE) {
167 				continue;
168 			}
169 
170 			String nextPathsUnsplit = nextSpDef.getPath();
171 			if (isBlank(nextPathsUnsplit)) {
172 				continue;
173 			}
174 
175 			boolean multiType = false;
176 			if (nextPathsUnsplit.endsWith("[x]")) {
177 				multiType = true;
178 			}
179 
180 			for (Object nextObject : extractValues(nextPathsUnsplit, theResource)) {
181 				if (nextObject == null) {
182 					continue;
183 				}
184 
185 				ResourceLink nextEntity;
186 				if (nextObject instanceof BaseResourceReferenceDt) {
187 					BaseResourceReferenceDt nextValue = (BaseResourceReferenceDt) nextObject;
188 					if (nextValue.isEmpty()) {
189 						continue;
190 					}
191 					if (nextValue.getReference().isEmpty() || nextValue.getReference().getValue().startsWith("#")) {
192 						// This is a blank or contained resource reference
193 						continue;
194 					}
195 
196 					String typeString = nextValue.getReference().getResourceType();
197 					if (isBlank(typeString)) {
198 						throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource type - " + nextValue.getReference().getValue());
199 					}
200 					Class<? extends IBaseResource> type = getContext().getResourceDefinition(typeString).getImplementingClass();
201 					String id = nextValue.getReference().getIdPart();
202 					if (StringUtils.isBlank(id)) {
203 						continue;
204 					}
205 
206 					IFhirResourceDao<?> dao = getDao(type);
207 					if (dao == null) {
208 						StringBuilder b = new StringBuilder();
209 						b.append("This server (version ");
210 						b.append(myContext.getVersion().getVersion());
211 						b.append(") is not able to handle resources of type[");
212 						b.append(nextValue.getReference().getResourceType());
213 						b.append("] - Valid resource types for this server: ");
214 						b.append(myResourceTypeToDao.keySet().toString());
215 
216 						throw new InvalidRequestException(b.toString());
217 					}
218 					Long valueOf;
219 					try {
220 						valueOf = translateForcedIdToPid(nextValue.getReference());
221 					} catch (ResourceNotFoundException e) {
222 						String resName = getContext().getResourceDefinition(type).getName();
223 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
224 					}
225 					ResourceTable target = myEntityManager.find(ResourceTable.class, valueOf);
226 					if (target == null) {
227 						String resName = getContext().getResourceDefinition(type).getName();
228 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
229 					}
230 					nextEntity = new ResourceLink(nextPathsUnsplit, theEntity, target);
231 				} else {
232 					if (!multiType) {
233 						throw new ConfigurationException("Search param " + nextSpDef.getName() + " is of unexpected datatype: " + nextObject.getClass());
234 					} else {
235 						continue;
236 					}
237 				}
238 				if (nextEntity != null) {
239 					retVal.add(nextEntity);
240 				}
241 			}
242 		}
243 
244 		theEntity.setHasLinks(retVal.size() > 0);
245 
246 		return retVal;
247 	}

269 	private List<Object> extractValues(String thePaths, IResource theResource) {
270 		List<Object> values = new ArrayList<Object>();
271 		String[] nextPathsSplit = thePaths.split("\\|");
272 		FhirTerser t = getContext().newTerser();
273 		for (String nextPath : nextPathsSplit) {
274 			String nextPathTrimmed = nextPath.trim();
275 			try {
276 				values.addAll(t.getValues(theResource, nextPathTrimmed));
277 			} catch (Exception e) {
278 				RuntimeResourceDefinition def = myContext.getResourceDefinition(theResource);
279 				ourLog.warn("Failed to index values from path[{}] in resource type[{}]: ", new Object[] { nextPathTrimmed, def.getName(), e.toString() } );
280 			}
281 		}
282 		return values;
283 	}

285 	private void findMatchingTagIds(String theResourceName, IIdType theResourceId, Set<Long> tagIds, Class<? extends BaseTag> entityClass) {
286 		{
287 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
288 			CriteriaQuery<Tuple> cq = builder.createTupleQuery();
289 			Root<? extends BaseTag> from = cq.from(entityClass);
290 			cq.multiselect(from.get("myTagId").as(Long.class)).distinct(true);
291 
292 			if (theResourceName != null) {
293 				Predicate typePredicate = builder.equal(from.get("myResourceType"), theResourceName);
294 				if (theResourceId != null) {
295 					cq.where(typePredicate, builder.equal(from.get("myResourceId"), translateForcedIdToPid(theResourceId)));
296 				} else {
297 					cq.where(typePredicate);
298 				}
299 			}
300 
301 			TypedQuery<Tuple> query = myEntityManager.createQuery(cq);
302 			for (Tuple next : query.getResultList()) {
303 				tagIds.add(next.get(0, Long.class));
304 			}
305 		}
306 	}

346 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
347 		CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
348 		Root<TagDefinition> from = cq.from(TagDefinition.class);
349 
350 		//@formatter:off
351 		if (isNotBlank(theScheme)) {
352 			cq.where(
353 				builder.and(
354 					builder.equal(from.get("myTagType"), theTagType), 
355 					builder.equal(from.get("mySystem"), theScheme), 
356 					builder.equal(from.get("myCode"), theTerm))
357 				);
358 		} else {
359 			cq.where(
360 				builder.and(
361 					builder.equal(from.get("myTagType"), theTagType), 
362 					builder.isNull(from.get("mySystem")), 
363 					builder.equal(from.get("myCode"), theTerm))
364 				);
365 		}
366 		//@formatter:on
367 
368 		TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
369 		try {
370 			return q.getSingleResult();
371 		} catch (NoResultException e) {
372 			TagDefinition retVal = new TagDefinition(theTagType, theScheme, theTerm, theLabel);
373 			myEntityManager.persist(retVal);
374 			return retVal;
375 		}

378 	protected TagList getTags(Class<? extends IResource> theResourceType, IIdType theResourceId) {
379 		String resourceName = null;
380 		if (theResourceType != null) {
381 			resourceName = toResourceName(theResourceType);
382 			if (theResourceId != null && theResourceId.hasVersionIdPart()) {
383 				IFhirResourceDao<? extends IResource> dao = getDao(theResourceType);
384 				BaseHasResource entity = dao.readEntity(theResourceId);
385 				TagList retVal = new TagList();
386 				for (BaseTag next : entity.getTags()) {
387 					retVal.add(next.getTag().toTag());
388 				}
389 				return retVal;
390 			}
391 		}
392 
393 		Set<Long> tagIds = new HashSet<Long>();
394 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceTag.class);
395 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceHistoryTag.class);
396 		if (tagIds.isEmpty()) {
397 			return new TagList();
398 		}
399 		{
400 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
401 			CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
402 			Root<TagDefinition> from = cq.from(TagDefinition.class);
403 			cq.where(from.get("myId").in(tagIds));
404 			cq.orderBy(builder.asc(from.get("mySystem")), builder.asc(from.get("myCode")));
405 			TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
406 			q.setMaxResults(getConfig().getHardTagListLimit());
407 
408 			TagList retVal = new TagList();
409 			for (TagDefinition next : q.getResultList()) {
410 				retVal.add(next.toTag());
411 			}
412 
413 			return retVal;
414 		}

417 	protected IBundleProvider history(String theResourceName, Long theId, Date theSince) {
418 		final List<HistoryTuple> tuples = new ArrayList<HistoryTuple>();
419 
420 		final InstantDt end = createHistoryToTimestamp();
421 
422 		StopWatch timer = new StopWatch();
423 
424 		int limit = 10000;
425 
426 		// Get list of IDs
427 		searchHistoryCurrentVersion(theResourceName, theId, theSince, end.getValue(), limit, tuples);
428 		ourLog.info("Retrieved {} history IDs from current versions in {} ms", tuples.size(), timer.getMillisAndRestart());
429 
430 		searchHistoryHistory(theResourceName, theId, theSince, end.getValue(), limit, tuples);
431 		ourLog.info("Retrieved {} history IDs from previous versions in {} ms", tuples.size(), timer.getMillisAndRestart());
432 
433 		// Sort merged list
434 		Collections.sort(tuples, Collections.reverseOrder());
435 		assert tuples.size() < 2 || !tuples.get(tuples.size() - 2).getUpdated().before(tuples.get(tuples.size() - 1).getUpdated()) : tuples.toString();
436 
437 		return new IBundleProvider() {
438 
439 			@Override
440 			public InstantDt getPublished() {
441 				return end;
442 			}
443 
444 			@Override
445 			public List<IBaseResource> getResources(final int theFromIndex, final int theToIndex) {
446 				final StopWatch timer = new StopWatch();
447 				TransactionTemplate template = new TransactionTemplate(myPlatformTransactionManager);
448 				return template.execute(new TransactionCallback<List<IBaseResource>>() {
449 					@Override
450 					public List<IBaseResource> doInTransaction(TransactionStatus theStatus) {
451 						List<BaseHasResource> resEntities = Lists.newArrayList();
452 
453 						List<HistoryTuple> tupleSubList = tuples.subList(theFromIndex, theToIndex);
454 						searchHistoryCurrentVersion(tupleSubList, resEntities);
455 						ourLog.info("Loaded history from current versions in {} ms", timer.getMillisAndRestart());
456 
457 						searchHistoryHistory(tupleSubList, resEntities);
458 						ourLog.info("Loaded history from previous versions in {} ms", timer.getMillisAndRestart());
459 
460 						Collections.sort(resEntities, new Comparator<BaseHasResource>() {
461 							@Override
462 							public int compare(BaseHasResource theO1, BaseHasResource theO2) {
463 								return theO2.getUpdated().getValue().compareTo(theO1.getUpdated().getValue());
464 							}
465 						});
466 
467 						int limit = theToIndex - theFromIndex;
468 						if (resEntities.size() > limit) {
469 							resEntities = resEntities.subList(0, limit);
470 						}
471 
472 						ArrayList<IBaseResource> retVal = new ArrayList<IBaseResource>();
473 						for (BaseHasResource next : resEntities) {
474 							RuntimeResourceDefinition type;
475 							try {
476 								type = myContext.getResourceDefinition(next.getResourceType());
477 							} catch (DataFormatException e) {
478 								if (next.getFhirVersion() != getContext().getVersion().getVersion()) {
479 									ourLog.info("Ignoring history resource of type[{}] because it is not compatible with version[{}]", next.getResourceType(), getContext().getVersion().getVersion());
480 									continue;
481 								}
482 								throw e;
483 							}
484 							IResource resource = (IResource) toResource(type.getImplementingClass(), next);
485 							retVal.add(resource);
486 						}
487 						return retVal;
488 					}
489 				});
490 			}
491 
492 			@Override
493 			public int size() {
494 				return tuples.size();
495 			}
496 
497 			@Override
498 			public Integer preferredPageSize() {
499 				return null;
500 			}
501 		};
502 	}

504 	protected boolean isValidPid(IIdType theId) {
505 		String idPart = theId.getIdPart();
506 		for (int i = 0; i < idPart.length(); i++) {
507 			char nextChar = idPart.charAt(i);
508 			if (nextChar < '0' || nextChar > '9') {
509 				return false;
510 			}
511 		}
512 		return true;
513 	}

516 		Set<Long> pids = new HashSet<Long>();
517 		for (IIdType next : theIncludePids) {
518 			if (next.isIdPartValidLong()) {
519 				pids.add(next.getIdPartAsLong());
520 			} else {
521 				try {
522 					pids.add(translateForcedIdToPid(next));
523 				} catch (ResourceNotFoundException e) {
524 					ourLog.warn("Failed to translate forced ID [{}] to PID", next.getValue());
525 				}
526 			}
527 		}
528 
529 		if (pids.isEmpty()) {
530 			return new ArrayList<IBaseResource>();
531 		}
532 
533 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
534 		CriteriaQuery<ResourceTable> cq = builder.createQuery(ResourceTable.class);
535 		Root<ResourceTable> from = cq.from(ResourceTable.class);
536 		// cq.where(builder.equal(from.get("myResourceType"),
537 		// getContext().getResourceDefinition(myResourceType).getName()));
538 		// if (theIncludePids != null) {
539 		cq.where(from.get("myId").in(pids));
540 		// }
541 		TypedQuery<ResourceTable> q = myEntityManager.createQuery(cq);
542 
543 		ArrayList<IBaseResource> retVal = new ArrayList<IBaseResource>();
544 		for (ResourceTable next : q.getResultList()) {
545 			IResource resource = (IResource) toResource(next);
546 			retVal.add(resource);
547 		}
548 
549 		return retVal;
550 	}

639 	static SearchParameterMap translateMatchUrl(String theMatchUrl, RuntimeResourceDefinition resourceDef) {
640 		SearchParameterMap paramMap = new SearchParameterMap();
641 		List<NameValuePair> parameters;
642 		try {
643 			String matchUrl = theMatchUrl;
644 			if (matchUrl.indexOf('?') == -1) {
645 				throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Error was: URL does not contain any parameters ('?' not detected)");
646 			}
647 			matchUrl = matchUrl.replace("|", "%7C");
648 			matchUrl = matchUrl.replace("=>=", "=%3E%3D");
649 			matchUrl = matchUrl.replace("=<=", "=%3C%3D");
650 			matchUrl = matchUrl.replace("=>", "=%3E");
651 			matchUrl = matchUrl.replace("=<", "=%3C");
652 			parameters = URLEncodedUtils.parse(new URI(matchUrl), "UTF-8");
653 		} catch (URISyntaxException e) {
654 			throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Error was: " + e.toString());
655 		}
656 
657 		ArrayListMultimap<String, QualifiedParamList> nameToParamLists = ArrayListMultimap.create();
658 		for (NameValuePair next : parameters) {
659 			String paramName = next.getName();
660 			String qualifier = null;
661 			for (int i = 0; i < paramMap.size(); i++) {
662 				switch (paramName.charAt(i)) {
663 				case '.':
664 				case ':':
665 					qualifier = paramName.substring(i);
666 					paramName = paramName.substring(0, i);
667 					i = Integer.MAX_VALUE;
668 					break;
669 				}
670 			}
671 
672 			QualifiedParamList paramList = QualifiedParamList.splitQueryStringByCommasIgnoreEscape(qualifier, next.getValue());
673 			nameToParamLists.put(paramName, paramList);
674 		}
675 
676 		for (String nextParamName : nameToParamLists.keySet()) {
677 			List<QualifiedParamList> paramList = nameToParamLists.get(nextParamName);
678 			if (Constants.PARAM_LASTUPDATED.equals(nextParamName)) {
679 				if (paramList != null && paramList.size() > 0) {
680 					if (paramList.size() > 2) {
681 						throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Can not have more than 2 " + Constants.PARAM_LASTUPDATED + " parameter repetitions");
682 					} else {
683 						DateRangeParam p1 = new DateRangeParam();
684 						p1.setValuesAsQueryTokens(paramList);
685 						paramMap.setLastUpdated(p1);
686 					}
687 				}
688 				continue;
689 			}
690 
691 			if (Constants.PARAM_COUNT.equals(nextParamName)) {
692 				if (paramList.size() > 0 && paramList.get(0).size() > 0) {
693 					String intString = paramList.get(0).get(0);
694 					try {
695 						paramMap.setCount(Integer.parseInt(intString));
696 					} catch (NumberFormatException e) {
697 						throw new InvalidRequestException("Invalid " + Constants.PARAM_COUNT + " value: " + intString);
698 					}
699 				}
700 				continue;
701 			}
702 
703 			RuntimeSearchParam paramDef = resourceDef.getSearchParam(nextParamName);
704 			if (paramDef == null) {
705 				throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Resource type " + resourceDef.getName() + " does not have a parameter with name: " + nextParamName);
706 			}
707 
708 			IQueryParameterAnd<?> param = MethodUtil.parseQueryParams(paramDef, nextParamName, paramList);
709 			paramMap.add(nextParamName, param);
710 		}
711 		return paramMap;
712 	}

901 	protected <T extends IBaseResource> T toResource(Class<T> theResourceType, BaseHasResource theEntity) {
902 		String resourceText = null;
903 		switch (theEntity.getEncoding()) {
904 		case JSON:
905 			try {
906 				resourceText = new String(theEntity.getResource(), "UTF-8");
907 			} catch (UnsupportedEncodingException e) {
908 				throw new Error("Should not happen", e);
909 			}
910 			break;
911 		case JSONC:
912 			resourceText = GZipUtil.decompress(theEntity.getResource());
913 			break;
914 		}
915 
916 		IParser parser = theEntity.getEncoding().newParser(getContext(theEntity.getFhirVersion()));
917 		T retVal;
918 		try {
919 			retVal = parser.parseResource(theResourceType, resourceText);
920 		} catch (Exception e) {
921 			StringBuilder b = new StringBuilder();
922 			b.append("Failed to parse database resource[");
923 			b.append(theResourceType);
924 			b.append("/");
925 			b.append(theEntity.getIdDt().getIdPart());
926 			b.append(" (pid ");
927 			b.append(theEntity.getId());
928 			b.append(", version ");
929 			b.append(myContext.getVersion().getVersion());
930 			b.append("): ");
931 			b.append(e.getMessage());
932 			String msg = b.toString();
933 			ourLog.error(msg, e);
934 			throw new DataFormatException(msg, e);
935 		}
936 		IResource res = (IResource) retVal;
937 		res.setId(theEntity.getIdDt());
938 
939 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.VERSION_ID, theEntity.getVersion());
940 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.PUBLISHED, theEntity.getPublished());
941 		res.getResourceMetadata().put(ResourceMetadataKeyEnum.UPDATED, theEntity.getUpdated());
942 
943 		if (theEntity.getTitle() != null) {
944 			ResourceMetadataKeyEnum.TITLE.put(res, theEntity.getTitle());
945 		}
946 
947 		if (theEntity.getDeleted() != null) {
948 			ResourceMetadataKeyEnum.DELETED_AT.put(res, new InstantDt(theEntity.getDeleted()));
949 		}
950 
951 		Collection<? extends BaseTag> tags = theEntity.getTags();
952 		if (theEntity.isHasTags()) {
953 			TagList tagList = new TagList();
954 			List<BaseCodingDt> securityLabels = new ArrayList<BaseCodingDt>();
955 			List<IdDt> profiles = new ArrayList<IdDt>();
956 			for (BaseTag next : tags) {
957 				switch (next.getTag().getTagType()) {
958 				case PROFILE:
959 					profiles.add(new IdDt(next.getTag().getCode()));
960 					break;
961 				case SECURITY_LABEL:
962 					BaseCodingDt secLabel = myContext.getVersion().newCodingDt();
963 					secLabel.setSystem(next.getTag().getSystem());
964 					secLabel.setCode(next.getTag().getCode());
965 					secLabel.setDisplay(next.getTag().getDisplay());
966 					securityLabels.add(secLabel);
967 					break;
968 				case TAG:
969 					tagList.add(new Tag(next.getTag().getSystem(), next.getTag().getCode(), next.getTag().getDisplay()));
970 					break;
971 				}
972 			}
973 			if (tagList.size() > 0) {
974 				ResourceMetadataKeyEnum.TAG_LIST.put(res, tagList);
975 			}
976 			if (securityLabels.size() > 0) {
977 				ResourceMetadataKeyEnum.SECURITY_LABELS.put(res, securityLabels);
978 			}
979 			if (profiles.size() > 0) {
980 				ResourceMetadataKeyEnum.PROFILES.put(res, profiles);
981 			}
982 		}
983 
984 		return retVal;
985 	}

1022 	protected ResourceTable updateEntity(final IResource theResource, ResourceTable entity, boolean theUpdateHistory, Date theDeletedTimestampOrNull, boolean thePerformIndexing,
1023 			boolean theUpdateVersion) {
1024 		if (entity.getPublished() == null) {
1025 			entity.setPublished(new Date());
1026 		}
1027 
1028 		if (theResource != null) {
1029 			String resourceType = myContext.getResourceDefinition(theResource).getName();
1030 			if (isNotBlank(entity.getResourceType()) && !entity.getResourceType().equals(resourceType)) {
1031 				throw new UnprocessableEntityException("Existing resource ID[" + entity.getIdDt().toUnqualifiedVersionless() + "] is of type[" + entity.getResourceType() + "] - Cannot update with ["
1032 						+ resourceType + "]");
1033 			}
1034 		}
1035 
1036 		if (theUpdateHistory) {
1037 			final ResourceHistoryTable historyEntry = entity.toHistory();
1038 			myEntityManager.persist(historyEntry);
1039 		}
1040 
1041 		if (theUpdateVersion) {
1042 			entity.setVersion(entity.getVersion() + 1);
1043 		}
1044 
1045 		Collection<ResourceIndexedSearchParamString> paramsString = new ArrayList<ResourceIndexedSearchParamString>(entity.getParamsString());
1046 		Collection<ResourceIndexedSearchParamToken> paramsToken = new ArrayList<ResourceIndexedSearchParamToken>(entity.getParamsToken());
1047 		Collection<ResourceIndexedSearchParamNumber> paramsNumber = new ArrayList<ResourceIndexedSearchParamNumber>(entity.getParamsNumber());
1048 		Collection<ResourceIndexedSearchParamQuantity> paramsQuantity = new ArrayList<ResourceIndexedSearchParamQuantity>(entity.getParamsQuantity());
1049 		Collection<ResourceIndexedSearchParamDate> paramsDate = new ArrayList<ResourceIndexedSearchParamDate>(entity.getParamsDate());
1050 		Collection<ResourceLink> resourceLinks = new ArrayList<ResourceLink>(entity.getResourceLinks());
1051 
1052 		List<ResourceIndexedSearchParamString> stringParams = null;
1053 		List<ResourceIndexedSearchParamToken> tokenParams = null;
1054 		List<ResourceIndexedSearchParamNumber> numberParams = null;
1055 		List<ResourceIndexedSearchParamQuantity> quantityParams = null;
1056 		List<ResourceIndexedSearchParamDate> dateParams = null;
1057 		List<ResourceLink> links = null;
1058 
1059 		if (theDeletedTimestampOrNull != null) {
1060 
1061 			stringParams = Collections.emptyList();
1062 			tokenParams = Collections.emptyList();
1063 			numberParams = Collections.emptyList();
1064 			quantityParams = Collections.emptyList();
1065 			dateParams = Collections.emptyList();
1066 			links = Collections.emptyList();
1067 			entity.setDeleted(theDeletedTimestampOrNull);
1068 			entity.setUpdated(theDeletedTimestampOrNull);
1069 
1070 		} else {
1071 
1072 			entity.setDeleted(null);
1073 
1074 			if (thePerformIndexing) {
1075 
1076 				stringParams = extractSearchParamStrings(entity, theResource);
1077 				numberParams = extractSearchParamNumber(entity, theResource);
1078 				quantityParams = extractSearchParamQuantity(entity, theResource);
1079 				dateParams = extractSearchParamDates(entity, theResource);
1080 
1081 				// ourLog.info("Indexing resource: {}", entity.getId());
1082 				ourLog.trace("Storing string indexes: {}", stringParams);
1083 
1084 				tokenParams = new ArrayList<ResourceIndexedSearchParamToken>();
1085 				for (BaseResourceIndexedSearchParam next : extractSearchParamTokens(entity, theResource)) {
1086 					if (next instanceof ResourceIndexedSearchParamToken) {
1087 						tokenParams.add((ResourceIndexedSearchParamToken) next);
1088 					} else {
1089 						stringParams.add((ResourceIndexedSearchParamString) next);
1090 					}
1091 				}
1092 
1093 				links = extractResourceLinks(entity, theResource);
1094 				populateResourceIntoEntity(theResource, entity);
1095 				entity.setUpdated(new Date());
1096 				entity.setLanguage(theResource.getLanguage().getValue());
