169 	protected List<ResourceLink> extractResourceLinks(ResourceTable theEntity, IResource theResource) {
170 		ArrayList<ResourceLink> retVal = new ArrayList<ResourceLink>();
171 
172 		RuntimeResourceDefinition def = getContext().getResourceDefinition(theResource);
173 		for (RuntimeSearchParam nextSpDef : def.getSearchParams()) {
174 			if (nextSpDef.getParamType() != RestSearchParameterTypeEnum.REFERENCE) {
175 				continue;
176 			}
177 
178 			String nextPathsUnsplit = nextSpDef.getPath();
179 			if (isBlank(nextPathsUnsplit)) {
180 				continue;
181 			}
182 
183 			boolean multiType = false;
184 			if (nextPathsUnsplit.endsWith("[x]")) {
185 				multiType = true;
186 			}
187 
188 			for (Object nextObject : extractValues(nextPathsUnsplit, theResource)) {
189 				if (nextObject == null) {
190 					continue;
191 				}
192 
193 				ResourceLink nextEntity;
194 				if (nextObject instanceof BaseResourceReferenceDt) {
195 					BaseResourceReferenceDt nextValue = (BaseResourceReferenceDt) nextObject;
196 					if (nextValue.isEmpty()) {
197 						continue;
198 					}
199 					if (nextValue.getReference().isEmpty() || nextValue.getReference().getValue().startsWith("#")) {
200 						// This is a blank or contained resource reference
201 						continue;
202 					}
203 
204 					String typeString = nextValue.getReference().getResourceType();
205 					if (isBlank(typeString)) {
206 						throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource type - " + nextValue.getReference().getValue());
207 					}
208 					RuntimeResourceDefinition resourceDefinition;
209 					try {
210 						resourceDefinition = getContext().getResourceDefinition(typeString);
211 					} catch (DataFormatException e) {
212 						throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Resource type is unknown or not supported on this server - " + nextValue.getReference().getValue());
213 					}
214 
215 					Class<? extends IBaseResource> type = resourceDefinition.getImplementingClass();
216 					String id = nextValue.getReference().getIdPart();
217 					if (StringUtils.isBlank(id)) {
218 						throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource ID - " + nextValue.getReference().getValue());
219 					}
220 
221 					IFhirResourceDao<?> dao = getDao(type);
222 					if (dao == null) {
223 						StringBuilder b = new StringBuilder();
224 						b.append("This server (version ");
225 						b.append(myContext.getVersion().getVersion());
226 						b.append(") is not able to handle resources of type[");
227 						b.append(nextValue.getReference().getResourceType());
228 						b.append("] - Valid resource types for this server: ");
229 						b.append(myResourceTypeToDao.keySet().toString());
230 
231 						throw new InvalidRequestException(b.toString());
232 					}
233 					Long valueOf;
234 					try {
235 						valueOf = translateForcedIdToPid(nextValue.getReference());
236 					} catch (ResourceNotFoundException e) {
237 						String resName = getContext().getResourceDefinition(type).getName();
238 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
239 					}
240 					ResourceTable target = myEntityManager.find(ResourceTable.class, valueOf);
241 					if (target == null) {
242 						String resName = getContext().getResourceDefinition(type).getName();
243 						throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
244 					}
245 					nextEntity = new ResourceLink(nextPathsUnsplit, theEntity, target);
246 				} else {
247 					if (!multiType) {
248 						throw new ConfigurationException("Search param " + nextSpDef.getName() + " is of unexpected datatype: " + nextObject.getClass());
249 					} else {
250 						continue;
251 					}
252 				}
253 				if (nextEntity != null) {
254 					retVal.add(nextEntity);
255 				}
256 			}
257 		}
258 
259 		theEntity.setHasLinks(retVal.size() > 0);
260 
261 		return retVal;
262 	}

292 	private List<Object> extractValues(String thePaths, IResource theResource) {
293 		List<Object> values = new ArrayList<Object>();
294 		String[] nextPathsSplit = thePaths.split("\\|");
295 		FhirTerser t = getContext().newTerser();
296 		for (String nextPath : nextPathsSplit) {
297 			String nextPathTrimmed = nextPath.trim();
298 			try {
299 				values.addAll(t.getValues(theResource, nextPathTrimmed));
300 			} catch (Exception e) {
301 				RuntimeResourceDefinition def = myContext.getResourceDefinition(theResource);
302 				ourLog.warn("Failed to index values from path[{}] in resource type[{}]: ", new Object[] { nextPathTrimmed, def.getName(), e.toString() });
303 			}
304 		}
305 		return values;
306 	}

308 	private void findMatchingTagIds(String theResourceName, IIdType theResourceId, Set<Long> tagIds, Class<? extends BaseTag> entityClass) {
309 		{
310 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
311 			CriteriaQuery<Tuple> cq = builder.createTupleQuery();
312 			Root<? extends BaseTag> from = cq.from(entityClass);
313 			cq.multiselect(from.get("myTagId").as(Long.class)).distinct(true);
314 
315 			if (theResourceName != null) {
316 				Predicate typePredicate = builder.equal(from.get("myResourceType"), theResourceName);
317 				if (theResourceId != null) {
318 					cq.where(typePredicate, builder.equal(from.get("myResourceId"), translateForcedIdToPid(theResourceId)));
319 				} else {
320 					cq.where(typePredicate);
321 				}
322 			}
323 
324 			TypedQuery<Tuple> query = myEntityManager.createQuery(cq);
325 			for (Tuple next : query.getResultList()) {
326 				tagIds.add(next.get(0, Long.class));
327 			}
328 		}
329 	}

331 	protected void notifyInterceptors(RestOperationTypeEnum operationType, ActionRequestDetails requestDetails) {
332 		if (requestDetails.getId() != null && requestDetails.getId().hasResourceType() && isNotBlank(requestDetails.getResourceType())) {
333 			if (requestDetails.getId().getResourceType().equals(requestDetails.getResourceType()) == false) {
334 				throw new InternalErrorException("Inconsistent server state - Resource types don't match: " + requestDetails.getId().getResourceType() + " / " + requestDetails.getResourceType());
335 			}
336 		}
337 		List<IServerInterceptor> interceptors = getConfig().getInterceptors();
338 		if (interceptors == null) {
339 			return;
340 		}
341 		for (IServerInterceptor next : interceptors) {
342 			next.incomingRequestPreHandled(operationType, requestDetails);
343 		}

385 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
386 		CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
387 		Root<TagDefinition> from = cq.from(TagDefinition.class);
388 
389 		//@formatter:off
390 		if (isNotBlank(theScheme)) {
391 			cq.where(
392 				builder.and(
393 					builder.equal(from.get("myTagType"), theTagType), 
394 					builder.equal(from.get("mySystem"), theScheme), 
395 					builder.equal(from.get("myCode"), theTerm))
396 				);
397 		} else {
398 			cq.where(
399 				builder.and(
400 					builder.equal(from.get("myTagType"), theTagType), 
401 					builder.isNull(from.get("mySystem")), 
402 					builder.equal(from.get("myCode"), theTerm))
403 				);
404 		}
405 		//@formatter:on
406 
407 		TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
408 		try {
409 			return q.getSingleResult();
410 		} catch (NoResultException e) {
411 			TagDefinition retVal = new TagDefinition(theTagType, theScheme, theTerm, theLabel);
412 			myEntityManager.persist(retVal);
413 			return retVal;
414 		}

417 	protected TagList getTags(Class<? extends IResource> theResourceType, IIdType theResourceId) {
418 		String resourceName = null;
419 		if (theResourceType != null) {
420 			resourceName = toResourceName(theResourceType);
421 			if (theResourceId != null && theResourceId.hasVersionIdPart()) {
422 				IFhirResourceDao<? extends IResource> dao = getDao(theResourceType);
423 				BaseHasResource entity = dao.readEntity(theResourceId);
424 				TagList retVal = new TagList();
425 				for (BaseTag next : entity.getTags()) {
426 					retVal.add(next.getTag().toTag());
427 				}
428 				return retVal;
429 			}
430 		}
431 
432 		Set<Long> tagIds = new HashSet<Long>();
433 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceTag.class);
434 		findMatchingTagIds(resourceName, theResourceId, tagIds, ResourceHistoryTag.class);
435 		if (tagIds.isEmpty()) {
436 			return new TagList();
437 		}
438 		{
439 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
440 			CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
441 			Root<TagDefinition> from = cq.from(TagDefinition.class);
442 			cq.where(from.get("myId").in(tagIds));
443 			cq.orderBy(builder.asc(from.get("mySystem")), builder.asc(from.get("myCode")));
444 			TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
445 			q.setMaxResults(getConfig().getHardTagListLimit());
446 
447 			TagList retVal = new TagList();
448 			for (TagDefinition next : q.getResultList()) {
449 				retVal.add(next.toTag());
450 			}
451 
452 			return retVal;
453 		}

456 	protected IBundleProvider history(String theResourceName, Long theId, Date theSince) {
457 		final List<HistoryTuple> tuples = new ArrayList<HistoryTuple>();
458 
459 		final InstantDt end = createHistoryToTimestamp();
460 
461 		StopWatch timer = new StopWatch();
462 
463 		int limit = 10000;
464 
465 		// Get list of IDs
466 		searchHistoryCurrentVersion(theResourceName, theId, theSince, end.getValue(), limit, tuples);
467 		ourLog.info("Retrieved {} history IDs from current versions in {} ms", tuples.size(), timer.getMillisAndRestart());
468 
469 		searchHistoryHistory(theResourceName, theId, theSince, end.getValue(), limit, tuples);
470 		ourLog.info("Retrieved {} history IDs from previous versions in {} ms", tuples.size(), timer.getMillisAndRestart());
471 
472 		// Sort merged list
473 		Collections.sort(tuples, Collections.reverseOrder());
474 		assert tuples.size() < 2 || !tuples.get(tuples.size() - 2).getUpdated().before(tuples.get(tuples.size() - 1).getUpdated()) : tuples.toString();
475 
476 		return new IBundleProvider() {
477 
478 			@Override
479 			public InstantDt getPublished() {
480 				return end;
481 			}
482 
483 			@Override
484 			public List<IBaseResource> getResources(final int theFromIndex, final int theToIndex) {
485 				final StopWatch timer = new StopWatch();
486 				TransactionTemplate template = new TransactionTemplate(myPlatformTransactionManager);
487 				return template.execute(new TransactionCallback<List<IBaseResource>>() {
488 					@Override
489 					public List<IBaseResource> doInTransaction(TransactionStatus theStatus) {
490 						List<BaseHasResource> resEntities = Lists.newArrayList();
491 
492 						List<HistoryTuple> tupleSubList = tuples.subList(theFromIndex, theToIndex);
493 						searchHistoryCurrentVersion(tupleSubList, resEntities);
494 						ourLog.info("Loaded history from current versions in {} ms", timer.getMillisAndRestart());
495 
496 						searchHistoryHistory(tupleSubList, resEntities);
497 						ourLog.info("Loaded history from previous versions in {} ms", timer.getMillisAndRestart());
498 
499 						Collections.sort(resEntities, new Comparator<BaseHasResource>() {
500 							@Override
501 							public int compare(BaseHasResource theO1, BaseHasResource theO2) {
502 								return theO2.getUpdated().getValue().compareTo(theO1.getUpdated().getValue());
503 							}
504 						});
505 
506 						int limit = theToIndex - theFromIndex;
507 						if (resEntities.size() > limit) {
508 							resEntities = resEntities.subList(0, limit);
509 						}
510 
511 						ArrayList<IBaseResource> retVal = new ArrayList<IBaseResource>();
512 						for (BaseHasResource next : resEntities) {
513 							RuntimeResourceDefinition type;
514 							try {
515 								type = myContext.getResourceDefinition(next.getResourceType());
516 							} catch (DataFormatException e) {
517 								if (next.getFhirVersion() != getContext().getVersion().getVersion()) {
518 									ourLog.info("Ignoring history resource of type[{}] because it is not compatible with version[{}]", next.getResourceType(), getContext().getVersion().getVersion());
519 									continue;
520 								}
521 								throw e;
522 							}
523 							IResource resource = (IResource) toResource(type.getImplementingClass(), next);
524 							retVal.add(resource);
525 						}
526 						return retVal;
527 					}
528 				});
529 			}
530 
531 			@Override
532 			public int size() {
533 				return tuples.size();
534 			}

681 	static SearchParameterMap translateMatchUrl(String theMatchUrl, RuntimeResourceDefinition resourceDef) {
682 		SearchParameterMap paramMap = new SearchParameterMap();
683 		List<NameValuePair> parameters;
684 		try {
685 			String matchUrl = theMatchUrl;
686 			if (matchUrl.indexOf('?') == -1) {
687 				throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Error was: URL does not contain any parameters ('?' not detected)");
688 			}
689 			matchUrl = matchUrl.replace("|", "%7C");
690 			matchUrl = matchUrl.replace("=>=", "=%3E%3D");
691 			matchUrl = matchUrl.replace("=<=", "=%3C%3D");
692 			matchUrl = matchUrl.replace("=>", "=%3E");
693 			matchUrl = matchUrl.replace("=<", "=%3C");
694 			parameters = URLEncodedUtils.parse(new URI(matchUrl), "UTF-8");
695 		} catch (URISyntaxException e) {
696 			throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Error was: " + e.toString());
697 		}
698 
699 		ArrayListMultimap<String, QualifiedParamList> nameToParamLists = ArrayListMultimap.create();
700 		for (NameValuePair next : parameters) {
701 			String paramName = next.getName();
702 			String qualifier = null;
703 			for (int i = 0; i < paramMap.size(); i++) {
704 				switch (paramName.charAt(i)) {
705 				case '.':
706 				case ':':
707 					qualifier = paramName.substring(i);
708 					paramName = paramName.substring(0, i);
709 					i = Integer.MAX_VALUE;
710 					break;
711 				}
712 			}
713 
714 			QualifiedParamList paramList = QualifiedParamList.splitQueryStringByCommasIgnoreEscape(qualifier, next.getValue());
715 			nameToParamLists.put(paramName, paramList);
716 		}
717 
718 		for (String nextParamName : nameToParamLists.keySet()) {
719 			List<QualifiedParamList> paramList = nameToParamLists.get(nextParamName);
720 			if (Constants.PARAM_LASTUPDATED.equals(nextParamName)) {
721 				if (paramList != null && paramList.size() > 0) {
722 					if (paramList.size() > 2) {
723 						throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Can not have more than 2 " + Constants.PARAM_LASTUPDATED + " parameter repetitions");
724 					} else {
725 						DateRangeParam p1 = new DateRangeParam();
726 						p1.setValuesAsQueryTokens(paramList);
727 						paramMap.setLastUpdated(p1);
728 					}
729 				}
730 				continue;
731 			}
732 
733 			if (Constants.PARAM_COUNT.equals(nextParamName)) {
734 				if (paramList.size() > 0 && paramList.get(0).size() > 0) {
735 					String intString = paramList.get(0).get(0);
736 					try {
737 						paramMap.setCount(Integer.parseInt(intString));
738 					} catch (NumberFormatException e) {
739 						throw new InvalidRequestException("Invalid " + Constants.PARAM_COUNT + " value: " + intString);
740 					}
741 				}
742 				continue;
743 			}
744 
745 			RuntimeSearchParam paramDef = resourceDef.getSearchParam(nextParamName);
746 			if (paramDef == null) {
747 				throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Resource type " + resourceDef.getName() + " does not have a parameter with name: " + nextParamName);
748 			}
749 
750 			IQueryParameterAnd<?> param = MethodUtil.parseQueryParams(paramDef, nextParamName, paramList);
751 			paramMap.add(nextParamName, param);
752 		}
753 		return paramMap;
754 	}
