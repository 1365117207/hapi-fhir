192 	public ResourceTable delete(IIdType theId, List<DeleteConflict> deleteConflicts) {
193 		if (theId == null || !theId.hasIdPart()) {
194 			throw new InvalidRequestException("Can not perform delete, no ID provided");
195 		}
196 		final ResourceTable entity = readEntityLatestVersion(theId);
197 		if (theId.hasVersionIdPart() && Long.parseLong(theId.getVersionIdPart()) != entity.getVersion()) {
198 			throw new InvalidRequestException("Trying to delete " + theId + " but this is not the current version");
199 		}
200 
201 		validateOkToDelete(deleteConflicts, entity);
202 
203 		// Notify interceptors
204 		ActionRequestDetails requestDetails = new ActionRequestDetails(theId, theId.getResourceType(), getContext());
205 		notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
206 
207 		Date updateTime = new Date();
208 		ResourceTable savedEntity = updateEntity(null, entity, true, updateTime, updateTime);
209 
210 		// Notify JPA interceptors
211 		for (IServerInterceptor next : getConfig().getInterceptors()) {
212 			if (next instanceof IJpaServerInterceptor) {
213 				((IJpaServerInterceptor) next).resourceDeleted(requestDetails, entity);
214 			}
215 		}
216 		return savedEntity;
217 	}

237 	public List<ResourceTable> deleteByUrl(String theUrl, List<DeleteConflict> deleteConflicts) {
238 		Set<Long> resource = processMatchUrl(theUrl, myResourceType);
239 		if (resource.size() > 1) {
240 			if (myDaoConfig.isAllowMultipleDelete() == false) {
241 				throw new PreconditionFailedException(getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "DELETE", theUrl, resource.size()));
242 			}
243 		}
244 
245 		List<ResourceTable> retVal = new ArrayList<ResourceTable>();
246 		for (Long pid : resource) {
247 			ResourceTable entity = myEntityManager.find(ResourceTable.class, pid);
248 			retVal.add(entity);
249 
250 			validateOkToDelete(deleteConflicts, entity);
251 
252 			// Notify interceptors
253 			IdDt idToDelete = entity.getIdDt();
254 			ActionRequestDetails requestDetails = new ActionRequestDetails(idToDelete, idToDelete.getResourceType(), getContext());
255 			notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
256 
257 			// Perform delete
258 			Date updateTime = new Date();
259 			updateEntity(null, entity, true, updateTime, updateTime);
260 
261 			// Notify JPA interceptors
262 			for (IServerInterceptor next : getConfig().getInterceptors()) {
263 				if (next instanceof IJpaServerInterceptor) {
264 					((IJpaServerInterceptor) next).resourceDeleted(requestDetails, entity);
265 				}
266 			}
267 
268 		}
269 
270 		return retVal;
271 	}

273 	private DaoMethodOutcome doCreate(T theResource, String theIfNoneExist, boolean thePerformIndexing, Date theUpdateTime) {
274 		StopWatch w = new StopWatch();
275 
276 		preProcessResourceForStorage(theResource);
277 
278 		ResourceTable entity = new ResourceTable();
279 		entity.setResourceType(toResourceName(theResource));
280 
281 		if (isNotBlank(theIfNoneExist)) {
282 			Set<Long> match = processMatchUrl(theIfNoneExist, myResourceType);
283 			if (match.size() > 1) {
284 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "CREATE", theIfNoneExist, match.size());
285 				throw new PreconditionFailedException(msg);
286 			} else if (match.size() == 1) {
287 				Long pid = match.iterator().next();
288 				entity = myEntityManager.find(ResourceTable.class, pid);
289 				return toMethodOutcome(entity, theResource).setCreated(false);
290 			}
291 		}
292 
293 		if (isNotBlank(theResource.getIdElement().getIdPart())) {
294 			if (isValidPid(theResource.getIdElement())) {
295 				throw new UnprocessableEntityException("This server cannot create an entity with a user-specified numeric ID - Client should not specify an ID when creating a new resource, or should include at least one letter in the ID to force a client-defined ID");
296 			}
297 			createForcedIdIfNeeded(entity, theResource.getIdElement());
298 
299 			if (entity.getForcedId() != null) {
300 				try {
301 					translateForcedIdToPid(theResource.getIdElement());
302 					throw new UnprocessableEntityException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "duplicateCreateForcedId", theResource.getIdElement().getIdPart()));
303 				} catch (ResourceNotFoundException e) {
304 					// good, this ID doesn't exist so we can create it
305 				}
306 			}
307 
308 		}
309 
310 		// Notify interceptors
311 		ActionRequestDetails requestDetails = new ActionRequestDetails(theResource.getIdElement(), toResourceName(theResource), theResource, getContext());
312 		notifyInterceptors(RestOperationTypeEnum.CREATE, requestDetails);
313 
314 		// Perform actual DB update
315 		updateEntity(theResource, entity, false, null, thePerformIndexing, true, theUpdateTime);
316 
317 		// Notify JPA interceptors
318 		for (IServerInterceptor next : getConfig().getInterceptors()) {
319 			if (next instanceof IJpaServerInterceptor) {
320 				((IJpaServerInterceptor) next).resourceCreated(requestDetails, entity);
321 			}
322 		}
323 
324 		DaoMethodOutcome outcome = toMethodOutcome(entity, theResource).setCreated(true);
325 
326 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
327 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
328 
329 		ourLog.info(msg);
330 		return outcome;
331 	}

381 	public IBundleProvider history(final IIdType theId, final Date theSince) {
382 		// Notify interceptors
383 		ActionRequestDetails requestDetails = new ActionRequestDetails(theId, getResourceName(), getContext());
384 		notifyInterceptors(RestOperationTypeEnum.HISTORY_INSTANCE, requestDetails);
385 
386 		final InstantDt end = createHistoryToTimestamp();
387 		final String resourceType = getContext().getResourceDefinition(myResourceType).getName();
388 
389 		T currentTmp;
390 		try {
391 			BaseHasResource entity = readEntity(theId.toVersionless(), false);
392 			validateResourceType(entity);
393 			currentTmp = toResource(myResourceType, entity, true);
394 			Date lastUpdated;
395 			if (currentTmp instanceof IResource) {
396 				lastUpdated = ResourceMetadataKeyEnum.UPDATED.get((IResource) currentTmp).getValue();
397 			} else {
398 				lastUpdated = ((IAnyResource) currentTmp).getMeta().getLastUpdated();
399 			}
400 			if (lastUpdated.after(end.getValue())) {
401 				currentTmp = null;
402 			}
403 		} catch (ResourceNotFoundException e) {
404 			currentTmp = null;
405 		}
406 
407 		final T current = currentTmp;
408 
409 		StringBuilder B = new StringBuilder();
410 		B.append("SELECT count(h) FROM ResourceHistoryTable h ");
411 		B.append("WHERE h.myResourceId = :PID AND h.myResourceType = :RESTYPE");
412 		B.append(" AND h.myUpdated < :END");
413 		B.append((theSince != null ? " AND h.myUpdated >= :SINCE" : ""));
414 		String querySring = B.toString();
415 
416 		TypedQuery<Long> countQuery = myEntityManager.createQuery(querySring, Long.class);
417 		countQuery.setParameter("PID", translateForcedIdToPid(theId));
418 		countQuery.setParameter("RESTYPE", resourceType);
419 		countQuery.setParameter("END", end.getValue(), TemporalType.TIMESTAMP);
420 		if (theSince != null) {
421 			countQuery.setParameter("SINCE", theSince, TemporalType.TIMESTAMP);
422 		}
423 		int historyCount = countQuery.getSingleResult().intValue();
424 
425 		final int offset;
426 		final int count;
427 		if (current != null) {
428 			count = historyCount + 1;
429 			offset = 1;
430 		} else {
431 			offset = 0;
432 			count = historyCount;
433 		}
434 
435 		if (count == 0) {
436 			throw new ResourceNotFoundException(theId);
437 		}
438 
439 		return new IBundleProvider() {
440

442 			public InstantDt getPublished() {
443 				return end;
444 			}

447 			public List<IBaseResource> getResources(int theFromIndex, int theToIndex) {
448 				List<IBaseResource> retVal = new ArrayList<IBaseResource>();
449 				if (theFromIndex == 0 && current != null) {
450 					retVal.add(current);
451 				}
452 
453 				StringBuilder b = new StringBuilder();
454 				b.append("SELECT h FROM ResourceHistoryTable h WHERE h.myResourceId = :PID AND h.myResourceType = :RESTYPE AND h.myUpdated < :END ");
455 				b.append((theSince != null ? " AND h.myUpdated >= :SINCE" : ""));
456 				b.append(" ORDER BY h.myUpdated DESC");
457 				TypedQuery<ResourceHistoryTable> q = myEntityManager.createQuery(b.toString(), ResourceHistoryTable.class);
458 				q.setParameter("PID", translateForcedIdToPid(theId));
459 				q.setParameter("RESTYPE", resourceType);
460 				q.setParameter("END", end.getValue(), TemporalType.TIMESTAMP);
461 				if (theSince != null) {
462 					q.setParameter("SINCE", theSince, TemporalType.TIMESTAMP);
463 				}
464 
465 				int firstResult = Math.max(0, theFromIndex - offset);
466 				q.setFirstResult(firstResult);
467 
468 				int maxResults = (theToIndex - theFromIndex) + 1;
469 				q.setMaxResults(maxResults);
470 
471 				List<ResourceHistoryTable> results = q.getResultList();
472 				for (ResourceHistoryTable next : results) {
473 					if (retVal.size() == maxResults) {
474 						break;
475 					}
476 					retVal.add(toResource(myResourceType, next, true));
477 				}
478 
479 				return retVal;
480 			}

508 	public <MT extends IBaseMetaType> MT metaAddOperation(IIdType theResourceId, MT theMetaAdd) {
509 		// Notify interceptors
510 		ActionRequestDetails requestDetails = new ActionRequestDetails(theResourceId, getResourceName(), getContext());
511 		notifyInterceptors(RestOperationTypeEnum.META_ADD, requestDetails);
512 
513 		StopWatch w = new StopWatch();
514 		BaseHasResource entity = readEntity(theResourceId);
515 		if (entity == null) {
516 			throw new ResourceNotFoundException(theResourceId);
517 		}
518 
519 		List<TagDefinition> tags = toTagList(theMetaAdd);
520 
521 		//@formatter:off
522 		for (TagDefinition nextDef : tags) {
523 			
524 			boolean hasTag = false;
525 			for (BaseTag next : new ArrayList<BaseTag>(entity.getTags())) {
526 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) && 
527 						ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) && 
528 						ObjectUtil.equals(next.getTag().getCode(), nextDef.getCode())) {
529 					hasTag = true;
530 					break;
531 				}
532 			}
533 
534 			if (!hasTag) {
535 				entity.setHasTags(true);
536 				
537 				TagDefinition def = getTag(nextDef.getTagType(), nextDef.getSystem(), nextDef.getCode(), nextDef.getDisplay());
538 				BaseTag newEntity = entity.addTag(def);
539 				myEntityManager.persist(newEntity);
540 			}
541 		}
542 		//@formatter:on
543 
544 		myEntityManager.merge(entity);
545 		ourLog.info("Processed metaAddOperation on {} in {}ms", new Object[] { theResourceId, w.getMillisAndRestart() });
546 
547 		@SuppressWarnings("unchecked")
548 		MT retVal = (MT) metaGetOperation(theMetaAdd.getClass(), theResourceId);
549 		return retVal;
550 	}

813 	public BaseHasResource readEntity(IIdType theId, boolean theCheckForForcedId) {
814 		validateResourceTypeAndThrowIllegalArgumentException(theId);
815 
816 		Long pid = translateForcedIdToPid(theId);
817 		BaseHasResource entity = myEntityManager.find(ResourceTable.class, pid);
818 
819 		if (entity == null) {
820 			throw new ResourceNotFoundException(theId);
821 		}
822 
823 		if (theId.hasVersionIdPart()) {
824 			if (theId.isVersionIdPartValidLong() == false) {
825 				throw new ResourceNotFoundException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "invalidVersion", theId.getVersionIdPart(), theId.toUnqualifiedVersionless()));
826 			}
827 			if (entity.getVersion() != theId.getVersionIdPartAsLong().longValue()) {
828 				entity = null;
829 			}
830 		}
831 
832 		if (entity == null) {
833 			if (theId.hasVersionIdPart()) {
834 				TypedQuery<ResourceHistoryTable> q = myEntityManager.createQuery("SELECT t from ResourceHistoryTable t WHERE t.myResourceId = :RID AND t.myResourceType = :RTYP AND t.myResourceVersion = :RVER", ResourceHistoryTable.class);
835 				q.setParameter("RID", pid);
836 				q.setParameter("RTYP", myResourceName);
837 				q.setParameter("RVER", theId.getVersionIdPartAsLong());
838 				try {
839 					entity = q.getSingleResult();
840 				} catch (NoResultException e) {
841 					throw new ResourceNotFoundException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "invalidVersion", theId.getVersionIdPart(), theId.toUnqualifiedVersionless()));
842 				}
843 			}
844 		}
845 
846 		validateResourceType(entity);
847 
848 		if (theCheckForForcedId) {
849 			validateGivenIdIsAppropriateToRetrieveResource(theId, entity);
850 		}
851 		return entity;
852 	}

1014 	@Override
1015 	public DaoMethodOutcome update(T theResource, String theMatchUrl, boolean thePerformIndexing) {
1016 		StopWatch w = new StopWatch();
1017 
1018 		preProcessResourceForStorage(theResource);
1019 
1020 		final ResourceTable entity;
1021 
1022 		IIdType resourceId;
1023 		if (isNotBlank(theMatchUrl)) {
1024 			Set<Long> match = processMatchUrl(theMatchUrl, myResourceType);
1025 			if (match.size() > 1) {
1026 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "UPDATE", theMatchUrl, match.size());
1027 				throw new PreconditionFailedException(msg);
1028 			} else if (match.size() == 1) {
1029 				Long pid = match.iterator().next();
1030 				entity = myEntityManager.find(ResourceTable.class, pid);
1031 				resourceId = entity.getIdDt();
1032 			} else {
1033 				return create(theResource, null, thePerformIndexing);
1034 			}
1035 		} else {
1036 			resourceId = theResource.getIdElement();
1037 			if (resourceId == null || isBlank(resourceId.getIdPart())) {
1038 				throw new InvalidRequestException("Can not update a resource with no ID");
1039 			}
1040 			try {
1041 				entity = readEntityLatestVersion(resourceId);
1042 			} catch (ResourceNotFoundException e) {
1043 				if (resourceId.isIdPartValidLong()) {
1044 					throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedNumericId", theResource.getIdElement().getIdPart()));
1045 				}
1046 				return doCreate(theResource, null, thePerformIndexing, new Date());
1047 			}
1048 		}
1049 
1050 		if (resourceId.hasVersionIdPart() && Long.parseLong(resourceId.getVersionIdPart()) != entity.getVersion()) {
1051 			throw new InvalidRequestException("Trying to update " + resourceId + " but this is not the current version");
1052 		}
1053 
1054 		if (resourceId.hasResourceType() && !resourceId.getResourceType().equals(getResourceName())) {
1055 			throw new UnprocessableEntityException("Invalid resource ID[" + entity.getIdDt().toUnqualifiedVersionless() + "] of type[" + entity.getResourceType() + "] - Does not match expected [" + getResourceName() + "]");
1056 		}
1057 
1058 		// Notify interceptors
1059 		ActionRequestDetails requestDetails = new ActionRequestDetails(resourceId, getResourceName(), theResource, getContext());
1060 		notifyInterceptors(RestOperationTypeEnum.UPDATE, requestDetails);
1061 
1062 		// Perform update
1063 		ResourceTable savedEntity = updateEntity(theResource, entity, true, null, thePerformIndexing, true, new Date());
1064 
1065 		// Notify JPA interceptors
1066 		for (IServerInterceptor next : getConfig().getInterceptors()) {
1067 			if (next instanceof IJpaServerInterceptor) {
1068 				((IJpaServerInterceptor) next).resourceUpdated(requestDetails, entity);
1069 			}
1070 		}
1071 
1072 		DaoMethodOutcome outcome = toMethodOutcome(savedEntity, theResource).setCreated(false);
1073 
1074 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
1075 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
1076 
1077 		ourLog.info(msg);
1078 		return outcome;
1079 	}
