295 		if (theList == null || theList.isEmpty()) {

304 		CriteriaQuery<Long> cq = builder.createQuery(Long.class);
305 		Root<ResourceIndexedSearchParamNumber> from = cq.from(ResourceIndexedSearchParamNumber.class);
306 		cq.select(from.get("myResourcePid").as(Long.class));
307 
308 		List<Predicate> codePredicates = new ArrayList<Predicate>();
309 		for (IQueryParameterType nextOr : theList) {
310 			IQueryParameterType params = nextOr;
311 
312 			if (addPredicateMissingFalseIfPresent(builder, theParamName, from, codePredicates, nextOr)) {
313 				continue;
314 			}
315 
316 			if (params instanceof NumberParam) {
317 				NumberParam param = (NumberParam) params;
318 
319 				BigDecimal value = param.getValue();
320 				if (value == null) {
321 					return thePids;
322 				}
323 
324 				Path<Object> fromObj = from.get("myValue");
325 				if (param.getComparator() == null) {
326 					double mul = value.doubleValue() * 1.01;
327 					double low = value.doubleValue() - mul;
328 					double high = value.doubleValue() + mul;
329 					Predicate lowPred = builder.ge(fromObj.as(Long.class), low);
330 					Predicate highPred = builder.le(fromObj.as(Long.class), high);
331 					codePredicates.add(builder.and(lowPred, highPred));
332 				} else {
333 					switch (param.getComparator()) {
334 					case GREATERTHAN:
335 						codePredicates.add(builder.greaterThan(fromObj.as(BigDecimal.class), value));
336 						break;
337 					case GREATERTHAN_OR_EQUALS:
338 						codePredicates.add(builder.ge(fromObj.as(BigDecimal.class), value));
339 						break;
340 					case LESSTHAN:
341 						codePredicates.add(builder.lessThan(fromObj.as(BigDecimal.class), value));
342 						break;
343 					case LESSTHAN_OR_EQUALS:
344 						codePredicates.add(builder.le(fromObj.as(BigDecimal.class), value));
345 						break;
346 					}
347 				}
348 			} else {
349 				throw new IllegalArgumentException("Invalid token type: " + params.getClass());
350 			}
351 
352 		}
353 
354 		Predicate masterCodePredicate = builder.or(codePredicates.toArray(new Predicate[0]));
355 
356 		Predicate type = builder.equal(from.get("myResourceType"), myResourceName);
357 		Predicate name = builder.equal(from.get("myParamName"), theParamName);
358 		if (thePids.size() > 0) {
359 			Predicate inPids = (from.get("myResourcePid").in(thePids));
360 			cq.where(builder.and(type, name, masterCodePredicate, inPids));
361 		} else {
362 			cq.where(builder.and(type, name, masterCodePredicate));
363 		}
364 
365 		TypedQuery<Long> q = myEntityManager.createQuery(cq);
366 		return new HashSet<Long>(q.getResultList());
367 	}

434 		if (theList == null || theList.isEmpty()) {
435 			return thePids;
436 		}
437 
438 		if (Boolean.TRUE.equals(theList.get(0).getMissing())) {
439 			return addPredicateParamMissing(thePids, "myParamsQuantity", theParamName, ResourceIndexedSearchParamQuantity.class);
440 		}
441 
442 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
443 		CriteriaQuery<Long> cq = builder.createQuery(Long.class);
444 		Root<ResourceIndexedSearchParamQuantity> from = cq.from(ResourceIndexedSearchParamQuantity.class);
445 		cq.select(from.get("myResourcePid").as(Long.class));
446 
447 		List<Predicate> codePredicates = new ArrayList<Predicate>();
448 		for (IQueryParameterType nextOr : theList) {
449 			IQueryParameterType params = nextOr;
450 
451 			if (addPredicateMissingFalseIfPresent(builder, theParamName, from, codePredicates, nextOr)) {
452 				continue;
453 			}
454 
455 			String systemValue;
456 			String unitsValue;
457 			QuantityCompararatorEnum cmpValue;
458 			BigDecimal valueValue;
459 			boolean approx = false;
460 
461 			if (params instanceof BaseQuantityDt) {
462 				BaseQuantityDt param = (BaseQuantityDt) params;
463 				systemValue = param.getSystemElement().getValueAsString();
464 				unitsValue = param.getUnitsElement().getValueAsString();
465 				cmpValue = QuantityCompararatorEnum.VALUESET_BINDER.fromCodeString(param.getComparatorElement().getValueAsString());
466 				valueValue = param.getValueElement().getValue();
467 			} else if (params instanceof QuantityParam) {
468 				QuantityParam param = (QuantityParam) params;
469 				systemValue = param.getSystem().getValueAsString();
470 				unitsValue = param.getUnits();
471 				cmpValue = param.getComparator();
472 				valueValue = param.getValue().getValue();
473 				approx = param.isApproximate();
474 			} else {
475 				throw new IllegalArgumentException("Invalid quantity type: " + params.getClass());
476 			}
477 
478 			Predicate system = null;
479 			if (!isBlank(systemValue)) {
480 				system = builder.equal(from.get("mySystem"), systemValue);
481 			}
482 
483 			Predicate code = null;
484 			if (!isBlank(unitsValue)) {
485 				code = builder.equal(from.get("myUnits"), unitsValue);
486 			}
487 
488 			Predicate num;
489 			if (cmpValue == null) {
490 				BigDecimal mul = approx ? new BigDecimal(0.1) : new BigDecimal(0.01);
491 				BigDecimal low = valueValue.subtract(valueValue.multiply(mul));
492 				BigDecimal high = valueValue.add(valueValue.multiply(mul));
493 				Predicate lowPred = builder.gt(from.get("myValue").as(BigDecimal.class), low);
494 				Predicate highPred = builder.lt(from.get("myValue").as(BigDecimal.class), high);
495 				num = builder.and(lowPred, highPred);
496 			} else {
497 				switch (cmpValue) {
498 				case GREATERTHAN:
499 					Expression<Number> path = from.get("myValue");
500 					num = builder.gt(path, valueValue);
501 					break;
502 				case GREATERTHAN_OR_EQUALS:
503 					path = from.get("myValue");
504 					num = builder.ge(path, valueValue);
505 					break;
506 				case LESSTHAN:
