93  	public void addTag(IIdType theId, TagTypeEnum theTagType, String theScheme, String theTerm, String theLabel) {
94  		StopWatch w = new StopWatch();
95  		BaseHasResource entity = readEntity(theId);
96  		if (entity == null) {
97  			throw new ResourceNotFoundException(theId);
98  		}
99  
100 		//@formatter:off
101 		for (BaseTag next : new ArrayList<BaseTag>(entity.getTags())) {
102 			if (ObjectUtil.equals(next.getTag().getTagType(), theTagType) && 
103 					ObjectUtil.equals(next.getTag().getSystem(), theScheme) && 
104 					ObjectUtil.equals(next.getTag().getCode(), theTerm)) {
105 				return;
106 			}
107 		}
108 		//@formatter:on
109 
110 		entity.setHasTags(true);
111 
112 		TagDefinition def = getTag(TagTypeEnum.TAG, theScheme, theTerm, theLabel);
113 		BaseTag newEntity = entity.addTag(def);
114 
115 		myEntityManager.persist(newEntity);
116 		myEntityManager.merge(entity);
117 
118 		ourLog.info("Processed addTag {}/{} on {} in {}ms", new Object[] { theScheme, theTerm, theId, w.getMillisAndRestart() });
119 	}

137 	public DaoMethodOutcome create(T theResource, String theIfNoneExist, boolean thePerformIndexing, RequestDetails theRequestDetails) {
138 		if (isNotBlank(theResource.getIdElement().getIdPart())) {
139 			if (getContext().getVersion().getVersion().equals(FhirVersionEnum.DSTU1)) {
140 				if (theResource.getIdElement().isIdPartValidLong()) {
141 					String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedNumericId", theResource.getIdElement().getIdPart());
142 					throw new InvalidRequestException(message, createErrorOperationOutcome(message, "processing"));
143 				}
144 			} else if (getContext().getVersion().getVersion().isOlderThan(FhirVersionEnum.DSTU3)) {
145 				String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedId", theResource.getIdElement().getIdPart());
146 				throw new InvalidRequestException(message, createErrorOperationOutcome(message, "processing"));
147 			} else {
148 				// As of DSTU3, ID and version in the body should be ignored for a create/update
149 				theResource.setId("");
150 			}
151 		}
152 
153 		return doCreate(theResource, theIfNoneExist, thePerformIndexing, new Date(), theRequestDetails);
154 	}

176 	@Override
177 	public ResourceTable delete(IIdType theId, List<DeleteConflict> deleteConflicts, RequestDetails theRequestDetails) {
178 		if (theId == null || !theId.hasIdPart()) {
179 			throw new InvalidRequestException("Can not perform delete, no ID provided");
180 		}
181 		final ResourceTable entity = readEntityLatestVersion(theId);
182 		if (theId.hasVersionIdPart() && Long.parseLong(theId.getVersionIdPart()) != entity.getVersion()) {
183 			throw new ResourceVersionConflictException("Trying to delete " + theId + " but this is not the current version");
184 		}
185 
186 		validateOkToDelete(deleteConflicts, entity);
187 
188 		// Notify interceptors
189 		if (theRequestDetails != null) {
190 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, getContext(), theId.getResourceType(), theId);
191 			notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
192 		}
193 
194 		Date updateTime = new Date();
195 		ResourceTable savedEntity = updateEntity(null, entity, updateTime, updateTime);
196 
197 		// Notify JPA interceptors
198 		if (theRequestDetails != null) {
199 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, getContext(), theId.getResourceType(), theId);
200 			T resourceToDelete = toResource(myResourceType, entity, false);
201 			theRequestDetails.getRequestOperationCallback().resourceDeleted(resourceToDelete);
202 			for (IServerInterceptor next : getConfig().getInterceptors()) {
203 				if (next instanceof IJpaServerInterceptor) {
204 					((IJpaServerInterceptor) next).resourceDeleted(requestDetails, entity);
205 				}
206 			}
207 		}
208 
209 		return savedEntity;
210 	}

214 		List<DeleteConflict> deleteConflicts = new ArrayList<DeleteConflict>();

226 	public List<ResourceTable> deleteByUrl(String theUrl, List<DeleteConflict> deleteConflicts, RequestDetails theRequestDetails) {
227 		Set<Long> resource = processMatchUrl(theUrl, myResourceType);
228 		if (resource.size() > 1) {
229 			if (myDaoConfig.isAllowMultipleDelete() == false) {
230 				throw new PreconditionFailedException(getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "DELETE", theUrl, resource.size()));
231 			}
232 		}
233 
234 		List<ResourceTable> retVal = new ArrayList<ResourceTable>();
235 		for (Long pid : resource) {
236 			ResourceTable entity = myEntityManager.find(ResourceTable.class, pid);
237 			retVal.add(entity);
238 
239 			validateOkToDelete(deleteConflicts, entity);
240 
241 			// Notify interceptors
242 			IdDt idToDelete = entity.getIdDt();
243 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, idToDelete.getResourceType(), idToDelete);
244 			notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
245 
246 			// Perform delete
247 			Date updateTime = new Date();
248 			updateEntity(null, entity, updateTime, updateTime);
249 
250 			// Notify JPA interceptors
251 			T resourceToDelete = toResource(myResourceType, entity, false);
252 			theRequestDetails.getRequestOperationCallback().resourceDeleted(resourceToDelete);
253 			for (IServerInterceptor next : getConfig().getInterceptors()) {
254 				if (next instanceof IJpaServerInterceptor) {
255 					((IJpaServerInterceptor) next).resourceDeleted(requestDetails, entity);
256 				}
257 			}
258 
259 		}
260 
261 		return retVal;
262 	}

264 	@Override
265 	public DaoMethodOutcome deleteByUrl(String theUrl, RequestDetails theRequestDetails) {
266 		StopWatch w = new StopWatch();
267 		List<DeleteConflict> deleteConflicts = new ArrayList<DeleteConflict>();
268 
269 		List<ResourceTable> deletedResources = deleteByUrl(theUrl, deleteConflicts, theRequestDetails);
270 
271 		validateDeleteConflictsEmptyOrThrowException(deleteConflicts);
272 		
273 		IBaseOperationOutcome oo;
274 		if (deletedResources.isEmpty()) {
275 			oo = OperationOutcomeUtil.newInstance(getContext());
276 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "unableToDeleteNotFound", theUrl);
277 			String severity = "warning";
278 			String code = "not-found";
279 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
280 		} else {
281 			oo = OperationOutcomeUtil.newInstance(getContext());
282 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", theUrl, deletedResources.size(), w.getMillis());
283 			String severity = "information";
284 			String code = "informational";
285 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
286 		}
287 
288 		ourLog.info("Processed delete on {} (matched {} resource(s)) in {}ms", new Object[] { theUrl, deletedResources.size(), w.getMillis() });
289 		DaoMethodOutcome retVal = new DaoMethodOutcome();
290 		retVal.setOperationOutcome(oo);
291 		return retVal;
292 	}

301 	private DaoMethodOutcome doCreate(T theResource, String theIfNoneExist, boolean thePerformIndexing, Date theUpdateTime, RequestDetails theRequestDetails) {
302 		StopWatch w = new StopWatch();
303 
304 		preProcessResourceForStorage(theResource);
305 
306 		ResourceTable entity = new ResourceTable();
307 		entity.setResourceType(toResourceName(theResource));
308 
309 		if (isNotBlank(theIfNoneExist)) {
310 			Set<Long> match = processMatchUrl(theIfNoneExist, myResourceType);
311 			if (match.size() > 1) {
312 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "CREATE", theIfNoneExist, match.size());
313 				throw new PreconditionFailedException(msg);
314 			} else if (match.size() == 1) {
315 				Long pid = match.iterator().next();
316 				entity = myEntityManager.find(ResourceTable.class, pid);
317 				return toMethodOutcome(entity, theResource).setCreated(false);
318 			}
319 		}
320 
321 		if (isNotBlank(theResource.getIdElement().getIdPart())) {
322 			if (isValidPid(theResource.getIdElement())) {
323 				throw new UnprocessableEntityException(
324 						"This server cannot create an entity with a user-specified numeric ID - Client should not specify an ID when creating a new resource, or should include at least one letter in the ID to force a client-defined ID");
325 			}
326 			createForcedIdIfNeeded(entity, theResource.getIdElement());
327 
328 			if (entity.getForcedId() != null) {
329 				try {
330 					translateForcedIdToPid(getResourceName(), theResource.getIdElement().getIdPart());
331 					throw new UnprocessableEntityException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "duplicateCreateForcedId", theResource.getIdElement().getIdPart()));
332 				} catch (ResourceNotFoundException e) {
333 					// good, this ID doesn't exist so we can create it
334 				}
335 			}
336 
337 		}
338 
339 		// Notify interceptors
340 		if (theRequestDetails != null) {
341 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, getContext(), theResource);
342 			notifyInterceptors(RestOperationTypeEnum.CREATE, requestDetails);
343 		}
344 
345 		// Perform actual DB update
346 		updateEntity(theResource, entity, null, thePerformIndexing, thePerformIndexing, theUpdateTime);
347 		theResource.setId(entity.getIdDt());
348 
349 		// Notify JPA interceptors
350 		if (theRequestDetails != null) {
351 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, getContext(), theResource);
352 			theRequestDetails.getRequestOperationCallback().resourceCreated(theResource);
353 			for (IServerInterceptor next : getConfig().getInterceptors()) {
354 				if (next instanceof IJpaServerInterceptor) {
355 					((IJpaServerInterceptor) next).resourceCreated(requestDetails, entity);
356 				}
357 			}
358 		}
359 
360 		DaoMethodOutcome outcome = toMethodOutcome(entity, theResource).setCreated(true);
361 
362 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
363 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
364 
365 		ourLog.info(msg);
366 		return outcome;
367 	}

369 	private <MT extends IBaseMetaType> void doMetaAdd(MT theMetaAdd, BaseHasResource entity) {
370 		List<TagDefinition> tags = toTagList(theMetaAdd);
371 
372 		//@formatter:off
373 		for (TagDefinition nextDef : tags) {
374 			
375 			boolean hasTag = false;
376 			for (BaseTag next : new ArrayList<BaseTag>(entity.getTags())) {
377 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) && 
378 						ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) && 
379 						ObjectUtil.equals(next.getTag().getCode(), nextDef.getCode())) {
380 					hasTag = true;
381 					break;
382 				}
383 			}
384 
385 			if (!hasTag) {
386 				entity.setHasTags(true);
387 				
388 				TagDefinition def = getTag(nextDef.getTagType(), nextDef.getSystem(), nextDef.getCode(), nextDef.getDisplay());
389 				BaseTag newEntity = entity.addTag(def);
390 				myEntityManager.persist(newEntity);
391 			}
392 		}
393 		//@formatter:on
394 
395 		myEntityManager.merge(entity);
396 	}

398 	private <MT extends IBaseMetaType> void doMetaDelete(MT theMetaDel, BaseHasResource entity) {
399 		List<TagDefinition> tags = toTagList(theMetaDel);
400 
401 		//@formatter:off
402 		for (TagDefinition nextDef : tags) {
403 			for (BaseTag next : new ArrayList<BaseTag>(entity.getTags())) {
404 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) && 
405 						ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) && 
406 						ObjectUtil.equals(next.getTag().getCode(), nextDef.getCode())) {
407 					myEntityManager.remove(next);
408 					entity.getTags().remove(next);
409 				}
410 			}
411 		}
412 		//@formatter:on
413 
414 		if (entity.getTags().isEmpty()) {
415 			entity.setHasTags(false);
416 		}
417 
418 		myEntityManager.merge(entity);
419 	}

486 	public <MT extends IBaseMetaType> MT metaAddOperation(IIdType theResourceId, MT theMetaAdd, RequestDetails theRequestDetails) {
487 		// Notify interceptors
488 		ActionRequestDetails requestDetails = new ActionRequestDetails(theRequestDetails, getResourceName(), theResourceId);
489 		notifyInterceptors(RestOperationTypeEnum.META_ADD, requestDetails);
490 
491 		StopWatch w = new StopWatch();
492 		BaseHasResource entity = readEntity(theResourceId);
493 		if (entity == null) {
494 			throw new ResourceNotFoundException(theResourceId);
495 		}
496 
497 		ResourceTable latestVersion = readEntityLatestVersion(theResourceId);
498 		if (latestVersion.getVersion() != entity.getVersion()) {
499 			doMetaAdd(theMetaAdd, entity);
500 		} else {
501 			doMetaAdd(theMetaAdd, latestVersion);
502 
503 			// Also update history entry
504 			ResourceHistoryTable history = myResourceHistoryTableDao.findForIdAndVersion(entity.getId(), entity.getVersion());
505 			doMetaAdd(theMetaAdd, history);
506 		}
507 
508 		ourLog.info("Processed metaAddOperation on {} in {}ms", new Object[] { theResourceId, w.getMillisAndRestart() });
509 
510 		@SuppressWarnings("unchecked")
511 		MT retVal = (MT) metaGetOperation(theMetaAdd.getClass(), theResourceId, theRequestDetails);
512 		return retVal;
513 	}

768 	public BaseHasResource readEntity(IIdType theId, boolean theCheckForForcedId) {
769 		validateResourceTypeAndThrowIllegalArgumentException(theId);
770 
771 		Long pid = translateForcedIdToPid(getResourceName(), theId.getIdPart());
772 		BaseHasResource entity = myEntityManager.find(ResourceTable.class, pid);
773 
774 		if (entity == null) {
775 			throw new ResourceNotFoundException(theId);
776 		}
777 
778 		if (theId.hasVersionIdPart()) {
779 			if (theId.isVersionIdPartValidLong() == false) {
780 				throw new ResourceNotFoundException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "invalidVersion", theId.getVersionIdPart(), theId.toUnqualifiedVersionless()));
781 			}
782 			if (entity.getVersion() != theId.getVersionIdPartAsLong().longValue()) {
783 				entity = null;
784 			}
785 		}
786 
787 		if (entity == null) {
788 			if (theId.hasVersionIdPart()) {
789 				TypedQuery<ResourceHistoryTable> q = myEntityManager
790 						.createQuery("SELECT t from ResourceHistoryTable t WHERE t.myResourceId = :RID AND t.myResourceType = :RTYP AND t.myResourceVersion = :RVER", ResourceHistoryTable.class);
791 				q.setParameter("RID", pid);
792 				q.setParameter("RTYP", myResourceName);
793 				q.setParameter("RVER", theId.getVersionIdPartAsLong());
794 				try {
795 					entity = q.getSingleResult();
796 				} catch (NoResultException e) {
797 					throw new ResourceNotFoundException(getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "invalidVersion", theId.getVersionIdPart(), theId.toUnqualifiedVersionless()));
798 				}
799 			}
800 		}
801 
802 		validateResourceType(entity);
803 
804 		if (theCheckForForcedId) {
805 			validateGivenIdIsAppropriateToRetrieveResource(theId, entity);
806 		}
807 		return entity;
808 	}

1002 	@Override
1003 	public DaoMethodOutcome update(T theResource, String theMatchUrl, boolean thePerformIndexing, RequestDetails theRequestDetails) {
1004 		StopWatch w = new StopWatch();
1005 
1006 		preProcessResourceForStorage(theResource);
1007 
1008 		final ResourceTable entity;
1009 
1010 		IIdType resourceId;
1011 		if (isNotBlank(theMatchUrl)) {
1012 			Set<Long> match = processMatchUrl(theMatchUrl, myResourceType);
1013 			if (match.size() > 1) {
1014 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "UPDATE", theMatchUrl, match.size());
1015 				throw new PreconditionFailedException(msg);
1016 			} else if (match.size() == 1) {
1017 				Long pid = match.iterator().next();
1018 				entity = myEntityManager.find(ResourceTable.class, pid);
1019 				resourceId = entity.getIdDt();
1020 			} else {
1021 				return create(theResource, null, thePerformIndexing, theRequestDetails);
1022 			}
1023 		} else {
1024 			/*
1025 			 * Note: resourcdeId will not be null or empty here, because we check it and reject requests in BaseOutcomeReturningMethodBindingWithResourceParam
1026 			 */
1027 			resourceId = theResource.getIdElement();
1028 
1029 			try {
1030 				entity = readEntityLatestVersion(resourceId);
1031 			} catch (ResourceNotFoundException e) {
1032 				if (resourceId.isIdPartValidLong()) {
1033 					throw new InvalidRequestException(
1034 							getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedNumericId", theResource.getIdElement().getIdPart()));
1035 				}
1036 				return doCreate(theResource, null, thePerformIndexing, new Date(), theRequestDetails);
1037 			}
1038 		}
1039 
1040 		if (resourceId.hasVersionIdPart() && Long.parseLong(resourceId.getVersionIdPart()) != entity.getVersion()) {
1041 			throw new ResourceVersionConflictException("Trying to update " + resourceId + " but this is not the current version");
1042 		}
1043 
1044 		if (resourceId.hasResourceType() && !resourceId.getResourceType().equals(getResourceName())) {
1045 			throw new UnprocessableEntityException(
1046 					"Invalid resource ID[" + entity.getIdDt().toUnqualifiedVersionless() + "] of type[" + entity.getResourceType() + "] - Does not match expected [" + getResourceName() + "]");
1047 		}
1048 
1049 		// Notify interceptors
1050 		ActionRequestDetails requestDetails = null;
1051 		if (theRequestDetails != null) {
1052 			requestDetails = new ActionRequestDetails(theRequestDetails, theResource, getResourceName(), resourceId);
1053 			notifyInterceptors(RestOperationTypeEnum.UPDATE, requestDetails);
1054 		}
1055 
1056 		// Perform update
1057 		ResourceTable savedEntity = updateEntity(theResource, entity, null, thePerformIndexing, thePerformIndexing, new Date());
1058 
1059 		// Notify interceptors
1060 		if (theRequestDetails != null) {
1061 			theRequestDetails.getRequestOperationCallback().resourceUpdated(theResource);
1062 			for (IServerInterceptor next : getConfig().getInterceptors()) {
1063 				if (next instanceof IJpaServerInterceptor) {
1064 					((IJpaServerInterceptor) next).resourceUpdated(requestDetails, entity);
1065 				}
1066 			}
1067 		}
1068 
1069 		DaoMethodOutcome outcome = toMethodOutcome(savedEntity, theResource).setCreated(false);
1070 
1071 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
1072 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
1073 
1074 		ourLog.info(msg);
1075 		return outcome;
1076 	}
