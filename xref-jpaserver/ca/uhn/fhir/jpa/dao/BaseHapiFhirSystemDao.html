73  	private int doPerformReindexingPass(final Integer theCount) {
74  		TransactionTemplate txTemplate = new TransactionTemplate(myTxManager);
75  		txTemplate.setPropagationBehavior(TransactionTemplate.PROPAGATION_REQUIRED);
76  		return txTemplate.execute(new TransactionCallback<Integer>() {
77  			@SuppressWarnings("unchecked")
78  			@Override
79  			public Integer doInTransaction(TransactionStatus theStatus) {
80  				TypedQuery<ResourceTable> q = myEntityManager.createQuery("SELECT t FROM " + ResourceTable.class.getSimpleName() + " t WHERE t.myIndexStatus IS null", ResourceTable.class);
81  
82  				int maxResult = 500;
83  				if (theCount != null) {
84  					maxResult = Math.min(theCount, 2000);
85  				}
86  
87  				q.setMaxResults(maxResult);
88  				List<ResourceTable> resources = q.getResultList();
89  				if (resources.isEmpty()) {
90  					return 0;
91  				}
92  
93  				ourLog.info("Indexing {} resources", resources.size());
94  
95  				int count = 0;
96  				long start = System.currentTimeMillis();
97  
98  				for (ResourceTable resourceTable : resources) {
99  					try {
100 						final IBaseResource resource = toResource(resourceTable, false);
101 
102 						@SuppressWarnings("rawtypes")
103 						final IFhirResourceDao dao = getDao(resource.getClass());
104 
105 						dao.reindex(resource, resourceTable);
106 					} catch (Exception e) {
107 						ourLog.error("Failed to index resource {}: {}", new Object[] { resourceTable.getIdDt(), e.toString(), e });
108 						throw new ReindexFailureException(resourceTable.getId());
109 					}
110 					count++;
111 				}
112 
113 				long delay = System.currentTimeMillis() - start;
114 				long avg = (delay / resources.size());
115 				ourLog.info("Indexed {} / {} resources in {}ms - Avg {}ms / resource", new Object[] { count, resources.size(), delay, avg });
116 
117 				return resources.size();
118 			}
119 		});
120 	}

126 		notifyInterceptors(RestOperationTypeEnum.GET_TAGS, requestDetails);

134 	@Override
135 	public Map<String, Long> getResourceCounts() {
136 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
137 		CriteriaQuery<Tuple> cq = builder.createTupleQuery();
138 		Root<?> from = cq.from(ResourceTable.class);
139 		cq.multiselect(from.get("myResourceType").as(String.class), builder.count(from.get("myResourceType")).as(Long.class));
140 		cq.groupBy(from.get("myResourceType"));
141 
142 		TypedQuery<Tuple> q = myEntityManager.createQuery(cq);
143 
144 		Map<String, Long> retVal = new HashMap<String, Long>();
145 		for (Tuple next : q.getResultList()) {
146 			String resourceName = next.get(0, String.class);
147 			Long count = next.get(1, Long.class);
148 			retVal.put(resourceName, count);
149 		}
150 		return retVal;
151 	}

179 	private void markResourceAsIndexingFailed(final long theId) {
180 		TransactionTemplate txTemplate = new TransactionTemplate(myTxManager);
181 		txTemplate.setPropagationBehavior(TransactionTemplate.PROPAGATION_REQUIRES_NEW);
182 		txTemplate.execute(new TransactionCallback<Void>() {
183 			@Override
184 			public Void doInTransaction(TransactionStatus theStatus) {
185 				ourLog.info("Marking resource with PID {} as indexing_failed", new Object[] { theId });
186 				Query q = myEntityManager.createQuery("UPDATE ResourceTable t SET t.myIndexStatus = :status WHERE t.myId = :id");
187 				q.setParameter("status", INDEX_STATUS_INDEXING_FAILED);
188 				q.setParameter("id", theId);
189 				q.executeUpdate();
190 				return null;
191 			}
192 		});
193 	}
194 
195 	@Override
196 	@Transactional(propagation = Propagation.NOT_SUPPORTED)
197 	public int performReindexingPass(final Integer theCount) {
198 		try {
199 			return doPerformReindexingPass(theCount);
200 		} catch (ReindexFailureException e) {
201 			ourLog.warn("Reindexing failed for resource {}", e.getResourceId());
202 			markResourceAsIndexingFailed(e.getResourceId());
203 			return -1;
204 		}
205 	}

211 	protected ResourceTable tryToLoadEntity(IdDt nextId) {
212 		ResourceTable entity;
213 		try {
214 			Long pid = translateForcedIdToPid(nextId);
215 			entity = myEntityManager.find(ResourceTable.class, pid);
216 		} catch (ResourceNotFoundException e) {
217 			entity = null;
218 		}
219 		return entity;
220 	}
