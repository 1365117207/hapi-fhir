51  public class SearchParamRegistryDstu3 extends BaseSearchParamRegistry {
52  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(SearchParamRegistryDstu3.class);
53  
54  	private volatile Map<String, Map<String, RuntimeSearchParam>> myActiveSearchParams;
55  
56  	@Autowired
57  	private DaoConfig myDaoConfig;
58  
59  	private long myLastRefresh;
60  
61  	@Autowired
62  	private IFhirResourceDao<SearchParameter> mySpDao;
63  
64  	@Override
65  	public void forceRefresh() {
66  		myLastRefresh = 0;
67  	}
68  
69  	@Override
70  	public Map<String, Map<String, RuntimeSearchParam>> getActiveSearchParams() {
71  		refreshCacheIfNeccesary();
72  		return myActiveSearchParams;
73  	}
74  
75  	@Override
76  	public Map<String, RuntimeSearchParam> getActiveSearchParams(String theResourceName) {
77  		refreshCacheIfNeccesary();
78  		return myActiveSearchParams.get(theResourceName);
79  	}

81  	private Map<String, RuntimeSearchParam> getSearchParamMap(Map<String, Map<String, RuntimeSearchParam>> searchParams, String theResourceName) {
82  		Map<String, RuntimeSearchParam> retVal = searchParams.get(theResourceName);
83  		if (retVal == null) {
84  			retVal = new HashMap<String, RuntimeSearchParam>();
85  			searchParams.put(theResourceName, retVal);
86  		}
87  		return retVal;
88  	}

90  	private void refreshCacheIfNeccesary() {
91  		long refreshInterval = 60 * DateUtils.MILLIS_PER_MINUTE;
92  		if (System.currentTimeMillis() - refreshInterval > myLastRefresh) {
93  			StopWatch sw = new StopWatch();
94  
95  			Map<String, Map<String, RuntimeSearchParam>> searchParams = new HashMap<String, Map<String, RuntimeSearchParam>>();
96  			for (Entry<String, Map<String, RuntimeSearchParam>> nextBuiltInEntry : getBuiltInSearchParams().entrySet()) {
97  				for (RuntimeSearchParam nextParam : nextBuiltInEntry.getValue().values()) {
98  					String nextResourceName = nextBuiltInEntry.getKey();
99  					getSearchParamMap(searchParams, nextResourceName).put(nextParam.getName(), nextParam);
100 				}
101 			}
102 
103 			IBundleProvider allSearchParamsBp = mySpDao.search(new SearchParameterMap());
104 			int size = allSearchParamsBp.size();
105 
106 			// Just in case..
107 			if (size > 10000) {
108 				ourLog.warn("Unable to support >10000 search params!");
109 				size = 10000;
110 			}
111 
112 			List<IBaseResource> allSearchParams = allSearchParamsBp.getResources(0, size);
113 			for (IBaseResource nextResource : allSearchParams) {
114 				SearchParameter nextSp = (SearchParameter) nextResource;
115 				RuntimeSearchParam runtimeSp = toRuntimeSp(nextSp);
116 				if (runtimeSp == null) {
117 					continue;
118 				}
119 
120 				int dotIdx = runtimeSp.getPath().indexOf('.');
121 				if (dotIdx == -1) {
122 					ourLog.warn("Can not determine resource type of {}", runtimeSp.getPath());
123 					continue;
124 				}
125 				String resourceType = runtimeSp.getPath().substring(0, dotIdx);
126 
127 				Map<String, RuntimeSearchParam> searchParamMap = getSearchParamMap(searchParams, resourceType);
128 				String name = runtimeSp.getName();
129 				if (myDaoConfig.isDefaultSearchParamsCanBeOverridden() || !searchParamMap.containsKey(name)) {
130 					searchParamMap.put(name, runtimeSp);
131 				}
132 			}
133 
134 			Map<String, Map<String, RuntimeSearchParam>> activeSearchParams = new HashMap<String, Map<String, RuntimeSearchParam>>();
135 			for (Entry<String, Map<String, RuntimeSearchParam>> nextEntry : searchParams.entrySet()) {
136 				for (RuntimeSearchParam nextSp : nextEntry.getValue().values()) {
137 					String nextName = nextSp.getName();
138 					if (nextSp.getStatus() != RuntimeSearchParamStatusEnum.ACTIVE) {
139 						nextSp = null;
140 					}
141 
142 					if (!activeSearchParams.containsKey(nextEntry.getKey())) {
143 						activeSearchParams.put(nextEntry.getKey(), new HashMap<String, RuntimeSearchParam>());
144 					}
145 					if (activeSearchParams.containsKey(nextEntry.getKey())) {
146 						ourLog.debug("Replacing existing/built in search param {}:{} with new one", nextEntry.getKey(), nextName);
147 					}
148 
149 					if (nextSp != null) {
150 						activeSearchParams.get(nextEntry.getKey()).put(nextName, nextSp);
151 					} else {
152 						activeSearchParams.get(nextEntry.getKey()).remove(nextName);
153 					}
154 				}
155 			}
156 
157 			myActiveSearchParams = activeSearchParams;
158 
159 			myLastRefresh = System.currentTimeMillis();
160 			ourLog.info("Refreshed search parameter cache in {}ms", sw.getMillis());
161 		}
162 	}

164 	private RuntimeSearchParam toRuntimeSp(SearchParameter theNextSp) {
165 		String name = theNextSp.getCode();
166 		String description = theNextSp.getDescription();
167 		String path = theNextSp.getExpression();
168 		RestSearchParameterTypeEnum paramType = null;
169 		RuntimeSearchParamStatusEnum status = null;
170 		switch (theNextSp.getType()) {
171 		case COMPOSITE:
172 			paramType = RestSearchParameterTypeEnum.COMPOSITE;
173 			break;
174 		case DATE:
175 			paramType = RestSearchParameterTypeEnum.DATE;
176 			break;
177 		case NUMBER:
178 			paramType = RestSearchParameterTypeEnum.NUMBER;
179 			break;
180 		case QUANTITY:
181 			paramType = RestSearchParameterTypeEnum.QUANTITY;
182 			break;
183 		case REFERENCE:
184 			paramType = RestSearchParameterTypeEnum.REFERENCE;
185 			break;
186 		case STRING:
187 			paramType = RestSearchParameterTypeEnum.STRING;
188 			break;
189 		case TOKEN:
190 			paramType = RestSearchParameterTypeEnum.TOKEN;
191 			break;
192 		case URI:
193 			paramType = RestSearchParameterTypeEnum.URI;
194 			break;
195 		case NULL:
196 			break;
197 		}
198 		if (theNextSp.getStatus() != null) {
199 			switch (theNextSp.getStatus()) {
200 			case ACTIVE:
201 				status = RuntimeSearchParamStatusEnum.ACTIVE;
202 				break;
203 			case DRAFT:
204 				status = RuntimeSearchParamStatusEnum.DRAFT;
205 				break;
206 			case RETIRED:
207 				status = RuntimeSearchParamStatusEnum.RETIRED;
208 				break;
209 			case NULL:
210 				break;
211 			}
212 		}
213 		Set<String> providesMembershipInCompartments = Collections.emptySet();
214 		Set<String> targets = toStrings(theNextSp.getTarget());
215 
216 		if (isBlank(name) || isBlank(path) || paramType == null) {
217 			return null;
218 		}
219 
220 		IIdType id = theNextSp.getIdElement();
221 		String uri = "";
222 		RuntimeSearchParam retVal = new RuntimeSearchParam(id, uri, name, description, path, paramType, null, providesMembershipInCompartments, targets, status);
223 		return retVal;
224 	}

226 	private Set<String> toStrings(List<CodeType> theTarget) {
227 		HashSet<String> retVal = new HashSet<String>();
228 		for (CodeType next : theTarget) {
229 			if (isNotBlank(next.getValue())) {
230 				retVal.add(next.getValue());
231 			}
232 		}
233 		return retVal;
234 	}
