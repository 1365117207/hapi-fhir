100 public class ResourceIndexedSearchParamString extends BaseResourceIndexedSearchParam {
101 
102 	/*
103 	 * Note that MYSQL chokes on unique indexes for lengths > 255 so be careful here 
104 	 */
105 	public static final int MAX_LENGTH = 200;
106 
107 	private static final long serialVersionUID = 1L;
108 
109 	@Id
110 	@SequenceGenerator(name="SEQ_SPIDX_STRING", sequenceName="SEQ_SPIDX_STRING")
111 	@GeneratedValue(strategy = GenerationType.AUTO, generator = "SEQ_SPIDX_STRING")
112 	@Column(name = "SP_ID")
113 	private Long myId;
114 
115 	@ManyToOne(optional = false)
116 	@JoinColumn(name = "RES_ID", referencedColumnName="RES_ID", insertable=false, updatable=false)
117 	@ContainedIn
118 	private ResourceTable myResourceTable;
119 
120 	@Column(name = "SP_VALUE_EXACT", length = MAX_LENGTH, nullable = true)
121 	@Fields({
122 		@Field(name = "myValueText", index = org.hibernate.search.annotations.Index.YES, store = Store.YES, analyze = Analyze.YES, analyzer = @Analyzer(definition = "standardAnalyzer")),
123 		@Field(name = "myValueTextEdgeNGram", index = org.hibernate.search.annotations.Index.YES, store = Store.NO, analyze = Analyze.YES, analyzer = @Analyzer(definition = "autocompleteEdgeAnalyzer")),
124 		@Field(name = "myValueTextNGram", index = org.hibernate.search.annotations.Index.YES, store = Store.NO, analyze = Analyze.YES, analyzer = @Analyzer(definition = "autocompleteNGramAnalyzer")),
125 		@Field(name = "myValueTextPhonetic", index = org.hibernate.search.annotations.Index.YES, store = Store.NO, analyze = Analyze.YES, analyzer = @Analyzer(definition = "autocompletePhoneticAnalyzer"))
126 	})
127 	private String myValueExact;
128 
129 	@Column(name = "SP_VALUE_NORMALIZED", length = MAX_LENGTH, nullable = true)
130 	private String myValueNormalized;
131 
132 	public ResourceIndexedSearchParamString() {
133 	}
134 
135 
136 	public ResourceIndexedSearchParamString(String theName, String theValueNormalized, String theValueExact) {
137 		setParamName(theName);
138 		setValueNormalized(theValueNormalized);
139 		setValueExact(theValueExact);
140 	}
141 
142 	@Override
143 	public boolean equals(Object theObj) {
144 		if (this == theObj) {
145 			return true;
146 		}
147 		if (theObj == null) {
148 			return false;
149 		}
150 		if (!(theObj instanceof ResourceIndexedSearchParamString)) {
151 			return false;
152 		}
153 		ResourceIndexedSearchParamString obj = (ResourceIndexedSearchParamString) theObj;
154 		EqualsBuilder b = new EqualsBuilder();
155 		b.append(getParamName(), obj.getParamName());
156 		b.append(getResource(), obj.getResource());
157 		b.append(getValueExact(), obj.getValueExact());
158 		return b.isEquals();
159 	}
160 
161 	@Override
162 	protected Long getId() {
163 		return myId;
164 	}
165 
166 	public String getValueExact() {
167 		return myValueExact;
168 	}
169 
170 	public String getValueNormalized() {
171 		return myValueNormalized;
172 	}
173 
174 	@Override
175 	public int hashCode() {
176 		HashCodeBuilder b = new HashCodeBuilder();
177 		b.append(getParamName());
178 		b.append(getResource());
179 		b.append(getValueExact());
180 		return b.toHashCode();
181 	}
182 
183 	public void setValueExact(String theValueExact) {
184 		if (StringUtils.defaultString(theValueExact).length() > MAX_LENGTH) {
185 			throw new IllegalArgumentException("Value is too long: " + theValueExact.length());
186 		}
187 		myValueExact = theValueExact;
188 	}
189 
190 	public void setValueNormalized(String theValueNormalized) {
191 		if (StringUtils.defaultString(theValueNormalized).length() > MAX_LENGTH) {
192 			throw new IllegalArgumentException("Value is too long: " + theValueNormalized.length());
193 		}
194 		myValueNormalized = theValueNormalized;
195 	}
196 
197 	@Override
198 	public String toString() {
199 		ToStringBuilder b = new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE);
200 		b.append("paramName", getParamName());
201 		b.append("resourceId", getResource().getId()); // TODO: add a field so we don't need to resolve this
202 		b.append("value", getValueNormalized());
203 		return b.build();
204 	}
