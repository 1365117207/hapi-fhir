535 	public ResourceHistoryTable toHistory(ResourceHistoryTable theResourceHistoryTable) {
536 		ResourceHistoryTable retVal = theResourceHistoryTable != null ? theResourceHistoryTable : new ResourceHistoryTable();
537 
538 		retVal.setResourceId(myId);
539 		retVal.setResourceType(myResourceType);
540 		retVal.setVersion(myVersion);
541 
542 		retVal.setTitle(getTitle());
543 		retVal.setPublished(getPublished());
544 		retVal.setUpdated(getUpdated());
545 		retVal.setEncoding(getEncoding());
546 		retVal.setFhirVersion(getFhirVersion());
547 		retVal.setResource(getResource());
548 		retVal.setDeleted(getDeleted());
549 		retVal.setForcedId(getForcedId());
550 
551 		retVal.getTags().clear();
552 
553 		retVal.setHasTags(isHasTags());
554 		if (isHasTags()) {
555 			for (ResourceTag next : getTags()) {
556 				retVal.addTag(next);
557 			}
558 		}
559 
560 		return retVal;
561 	}
