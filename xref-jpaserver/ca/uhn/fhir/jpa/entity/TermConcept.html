143 		TermConceptParentChildLink link = new TermConceptParentChildLink();

222 	private void parentPids(TermConcept theNextConcept, Set<Long> theParentPids) {
223 		for (TermConceptParentChildLink nextParentLink : theNextConcept.getParents()) {
224 			TermConcept parent = nextParentLink.getParent();
225 			Long parentConceptId = parent.getId();
226 			Validate.notNull(parentConceptId);
227 			if (parent != null && theParentPids.add(parentConceptId)) {
228 				parentPids(parent, theParentPids);
229 			}
230 		}
231 	}

233 	@PreUpdate
234 	@PrePersist
235 	public void prePersist() {
236 		Set<Long> parentPids = new HashSet<Long>();
237 		TermConcept entity = this;
238 		parentPids(entity, parentPids);
239 		entity.setParentPids(parentPids);
240 
241 		ourLog.trace("Code {}/{} has parents {}", entity.getId(), entity.getCode(), entity.getParentPidsAsString());
242 	}

268 		StringBuilder b = new StringBuilder();
269 		for (Long next : theParentPids) {

271 				b.append(' ');

273 			b.append(next);
274 		}

276 		if (b.length() == 0) {
277 			b.append("NONE");
278 		}

280 		myParentPids = b.toString();
281 	}
