23  import static org.apache.commons.lang3.StringUtils.isNotBlank;
24  
25  import javax.servlet.http.HttpServletRequest;
26  
27  import org.hl7.fhir.dstu3.model.BooleanType;
28  import org.hl7.fhir.dstu3.model.CodeSystem;
29  import org.hl7.fhir.dstu3.model.CodeType;
30  import org.hl7.fhir.dstu3.model.CodeableConcept;
31  import org.hl7.fhir.dstu3.model.Coding;
32  import org.hl7.fhir.dstu3.model.Parameters;
33  import org.hl7.fhir.dstu3.model.StringType;
34  import org.hl7.fhir.dstu3.model.UriType;
35  
36  import ca.uhn.fhir.jpa.dao.IFhirResourceDaoCodeSystem;
37  import ca.uhn.fhir.jpa.dao.IFhirResourceDaoCodeSystem.LookupCodeResult;
38  import ca.uhn.fhir.rest.annotation.Operation;
39  import ca.uhn.fhir.rest.annotation.OperationParam;
40  import ca.uhn.fhir.rest.method.RequestDetails;
41  import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException;
42  
43  public class BaseJpaResourceProviderCodeSystemDstu3 extends JpaResourceProviderDstu3<CodeSystem> {
44  
45  	//@formatter:off
46  	@SuppressWarnings("unchecked")
47  	@Operation(name = "$lookup", idempotent = true, returnParameters= {
48  		@OperationParam(name="name", type=StringType.class, min=1),
49  		@OperationParam(name="version", type=StringType.class, min=0),
50  		@OperationParam(name="display", type=StringType.class, min=1),
51  		@OperationParam(name="abstract", type=BooleanType.class, min=1),
52  	})
53  	public Parameters lookup(
54  			HttpServletRequest theServletRequest,
55  			@OperationParam(name="code", min=0, max=1) CodeType theCode, 
56  			@OperationParam(name="system", min=0, max=1) UriType theSystem,
57  			@OperationParam(name="coding", min=0, max=1) Coding theCoding, 
58  			RequestDetails theRequestDetails 
59  			) {
60  		//@formatter:on
61  		
62  		startRequest(theServletRequest);
63  		try {
64  			IFhirResourceDaoCodeSystem<CodeSystem, Coding, CodeableConcept> dao = (IFhirResourceDaoCodeSystem<CodeSystem, Coding, CodeableConcept>) getDao();
65  			LookupCodeResult result = dao.lookupCode(theCode, theSystem, theCoding, theRequestDetails);
66  			if (result.isFound()==false) {
67  				throw new ResourceNotFoundException("Unable to find code[" + result.getSearchedForCode() + "] in system[" + result.getSearchedForSystem() + "]");
68  			}
69  			Parameters retVal = new Parameters();
70  			retVal.addParameter().setName("name").setValue(new StringType(result.getCodeSystemDisplayName()));
71  			if (isNotBlank(result.getCodeSystemVersion())) {
72  				retVal.addParameter().setName("version").setValue(new StringType(result.getCodeSystemVersion()));
73  			}
74  			retVal.addParameter().setName("display").setValue(new StringType(result.getCodeDisplay()));
75  			retVal.addParameter().setName("abstract").setValue(new BooleanType(result.isCodeIsAbstract()));			
76  			return retVal;
77  		} finally {
78  			endRequest(theServletRequest);
79  		}
80  	}
81
