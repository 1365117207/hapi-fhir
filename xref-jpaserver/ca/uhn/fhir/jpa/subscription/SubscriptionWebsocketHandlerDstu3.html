54  public class SubscriptionWebsocketHandlerDstu3 extends TextWebSocketHandler implements ISubscriptionWebsocketHandler, Runnable {
55  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(SubscriptionWebsocketHandlerDstu3.class);
56  
57  	@Autowired
58  	@Qualifier("myFhirContextDstu3")
59  	private FhirContext myCtx;
60  
61  	private ScheduledFuture<?> myScheduleFuture;
62  
63  	private IState myState = new InitialState();
64  
65  	@Autowired
66  	private IFhirResourceDaoSubscription<Subscription> mySubscriptionDao;
67  
68  	private IIdType mySubscriptionId;
69  	private Long mySubscriptionPid;
70  
71  	@Autowired
72  	@Qualifier("websocketTaskScheduler")
73  	private TaskScheduler myTaskScheduler;

75  	@Override
76  	public void afterConnectionClosed(WebSocketSession theSession, CloseStatus theStatus) throws Exception {
77  		super.afterConnectionClosed(theSession, theStatus);
78  		ourLog.info("Closing WebSocket connection from {}", theSession.getRemoteAddress());
79  	}

81  	@Override
82  	public void afterConnectionEstablished(WebSocketSession theSession) throws Exception {
83  		super.afterConnectionEstablished(theSession);
84  		ourLog.info("Incoming WebSocket connection from {}", theSession.getRemoteAddress());
85  	}

87  	protected void handleFailure(Exception theE) {
88  		ourLog.error("Failure during communication", theE);
89  	}

91  	@Override
92  	protected void handleTextMessage(WebSocketSession theSession, TextMessage theMessage) throws Exception {
93  		ourLog.info("Textmessage: " + theMessage.getPayload());
94  
95  		myState.handleTextMessage(theSession, theMessage);
96  	}
97  
98  	@Override
99  	public void handleTransportError(WebSocketSession theSession, Throwable theException) throws Exception {
100 		super.handleTransportError(theSession, theException);
101 		ourLog.error("Transport error", theException);
102 	}

104 	@PostConstruct
105 	public void postConstruct() {
106 		ourLog.info("Creating scheduled task for subscription websocket connection");
107 		myScheduleFuture = myTaskScheduler.scheduleWithFixedDelay(this, 1000);
108 	}
109 
110 	@PreDestroy
111 	public void preDescroy() {
112 		ourLog.info("Cancelling scheduled task for subscription websocket connection");
113 		myScheduleFuture.cancel(true);
114 		IState state = myState;
115 		if (state != null) {
116 			state.closing();
117 		}
118 	}
119 
120 	@Override
121 	public void run() {
122 		Long subscriptionPid = mySubscriptionPid;
123 		if (subscriptionPid == null) {
124 			return;
125 		}
126 
127 		ourLog.debug("Subscription {} websocket handler polling", subscriptionPid);
128 
129 		List<IBaseResource> results = mySubscriptionDao.getUndeliveredResourcesAndPurge(subscriptionPid);
130 		if (results.isEmpty() == false) {
131 			myState.deliver(results);
132 		}
133 	}

135 	private class BoundDynamicSubscriptionState implements IState {
136 
137 		private EncodingEnum myEncoding;
138 		private WebSocketSession mySession;
139 
140 		public BoundDynamicSubscriptionState(WebSocketSession theSession, EncodingEnum theEncoding) {
141 			mySession = theSession;
142 			myEncoding = theEncoding;
143 		}
144 
145 		@Override
146 		public void closing() {
147 			ourLog.info("Deleting subscription {}", mySubscriptionId);
148 			try {
149 				mySubscriptionDao.delete(mySubscriptionId, new ServletRequestDetails());
150 			} catch (Exception e) {
151 				handleFailure(e);
152 			}
153 		}
154 
155 		@Override
156 		public void deliver(List<IBaseResource> theResults) {
157 			try {
158 			for (IBaseResource nextResource : theResults) {
159 				ourLog.info("Sending WebSocket message for resource: {}", nextResource.getIdElement());
160 				String encoded = myEncoding.newParser(myCtx).encodeResourceToString(nextResource);
161 				String payload = "add " + mySubscriptionId.getIdPart() + '\n' + encoded;
162 				mySession.sendMessage(new TextMessage(payload));
163 			}
164 			} catch (IOException e) {
165 				handleFailure(e);
166 			}
167 		}
168 
169 		@Override
170 		public void handleTextMessage(WebSocketSession theSession, TextMessage theMessage) {
171 			try {
172 				theSession.sendMessage(new TextMessage("Unexpected client message: " + theMessage.getPayload()));
173 			} catch (IOException e) {
174 				handleFailure(e);
175 			}
176 		}
177 
178 	}

215 	private class InitialState implements IState {
216 
217 		private IIdType bindSimple(WebSocketSession theSession, String theBindString) {
218 			IdType id = new IdType(theBindString);
219 
220 			if (!id.hasIdPart() || !id.isIdPartValid()) {
221 				try {
222 					String message = "Invalid bind request - No ID included";
223 					ourLog.warn(message);
224 					theSession.close(new CloseStatus(CloseStatus.PROTOCOL_ERROR.getCode(), message));
225 				} catch (IOException e) {
226 					handleFailure(e);
227 				}
228 				return null;
229 			}
230 
231 			if (id.hasResourceType() == false) {
232 				id = id.withResourceType("Subscription");
233 			}
234 
235 			try {
236 				Subscription subscription = mySubscriptionDao.read(id, new ServletRequestDetails());
237 				mySubscriptionPid = mySubscriptionDao.getSubscriptionTablePidForSubscriptionResource(id);
238 				mySubscriptionId = subscription.getIdElement();
239 				myState = new BoundStaticSubscipriptionState(theSession);
240 			} catch (ResourceNotFoundException e) {
241 				try {
242 					String message = "Invalid bind request - Unknown subscription: " + id.getValue();
243 					ourLog.warn(message);
244 					theSession.close(new CloseStatus(CloseStatus.PROTOCOL_ERROR.getCode(), message));
245 				} catch (IOException e1) {
246 					handleFailure(e);
247 				}
248 				return null;
249 			}
250 
251 			return id;
252 		}
253 
254 		private IIdType bingSearch(WebSocketSession theSession, String theRemaining) {
255 			Subscription subscription = new Subscription();
256 			subscription.getChannel().setType(SubscriptionChannelType.WEBSOCKET);
257 			subscription.setStatus(SubscriptionStatus.ACTIVE);
258 			subscription.setCriteria(theRemaining);
259 
260 			try {
261 				String params = theRemaining.substring(theRemaining.indexOf('?')+1);
262 				List<NameValuePair> paramValues = URLEncodedUtils.parse(params, Constants.CHARSET_UTF8, '&');
263 				EncodingEnum encoding = EncodingEnum.JSON;
264 				for (NameValuePair nameValuePair : paramValues) {
265 					if (Constants.PARAM_FORMAT.equals(nameValuePair.getName())) {
266 						EncodingEnum nextEncoding = Constants.FORMAT_VAL_TO_ENCODING.get(nameValuePair.getValue());
267 						if (nextEncoding != null) {
268 							encoding = nextEncoding;
269 						}
270 					}
271 				}
272 				
273 				IIdType id = mySubscriptionDao.create(subscription, new ServletRequestDetails()).getId();
274 
275 				mySubscriptionPid = mySubscriptionDao.getSubscriptionTablePidForSubscriptionResource(id);
276 				mySubscriptionId = subscription.getIdElement();
277 				myState = new BoundDynamicSubscriptionState(theSession, encoding);
278 
279 				return id;
280 			} catch (UnprocessableEntityException e) {
281 				ourLog.warn("Failed to bind subscription: " + e.getMessage());
282 				try {
283 					theSession.close(new CloseStatus(CloseStatus.PROTOCOL_ERROR.getCode(), "Invalid bind request - " + e.getMessage()));
284 				} catch (IOException e2) {
285 					handleFailure(e2);
286 				}
287 			} catch (Exception e) {
288 				handleFailure(e);
289 				try {
290 					theSession.close(new CloseStatus(CloseStatus.PROTOCOL_ERROR.getCode(), "Invalid bind request - No ID included"));
291 				} catch (IOException e2) {
292 					handleFailure(e2);
293 				}
294 			}
295 			return null;
296 		}
297 
298 		@Override
299 		public void closing() {
300 			// nothing
301 		}
302 
303 		@Override
304 		public void deliver(List<IBaseResource> theResults) {
305 			throw new IllegalStateException();
306 		}
307 
308 		@Override
309 		public void handleTextMessage(WebSocketSession theSession, TextMessage theMessage) {
310 			String message = theMessage.getPayload();
311 			if (message.startsWith("bind ")) {
312 				String remaining = message.substring("bind ".length());
313 
314 				IIdType subscriptionId;
315 				if (remaining.contains("?")) {
316 					subscriptionId = bingSearch(theSession, remaining);
317 				} else {
318 					subscriptionId = bindSimple(theSession, remaining);
319 					if (subscriptionId == null) {
320 						return;
321 					}
322 				}
323 
324 				try {
325 					theSession.sendMessage(new TextMessage("bound " + subscriptionId.getIdPart()));
326 				} catch (IOException e) {
327 					handleFailure(e);
328 				}
329 
330 			}
331 		}
