100 	private boolean addToSet(Set<TermConcept> theSetToPopulate, TermConcept theConcept) {
101 		boolean retVal = theSetToPopulate.add(theConcept);
102 		if (retVal) {
103 			if (theSetToPopulate.size() >= myDaoConfig.getMaximumExpansionSize()) {
104 				String msg = myContext.getLocalizer().getMessage(BaseHapiTerminologySvc.class, "expansionTooLarge", myDaoConfig.getMaximumExpansionSize());
105 				throw new InvalidRequestException(msg);
106 			}
107 		}
108 		return retVal;
109 	}

111 	private void fetchChildren(TermConcept theConcept, Set<TermConcept> theSetToPopulate) {
112 		for (TermConceptParentChildLink nextChildLink : theConcept.getChildren()) {
113 			TermConcept nextChild = nextChildLink.getChild();
114 			if (addToSet(theSetToPopulate, nextChild)) {
115 				fetchChildren(nextChild, theSetToPopulate);
116 			}
117 		}
118 	}

126 	private void fetchParents(TermConcept theConcept, Set<TermConcept> theSetToPopulate) {
127 		for (TermConceptParentChildLink nextChildLink : theConcept.getParents()) {
128 			TermConcept nextChild = nextChildLink.getParent();
129 			if (addToSet(theSetToPopulate, nextChild)) {
130 				fetchParents(nextChild, theSetToPopulate);
131 			}
132 		}
133 	}

149 		Stopwatch stopwatch = Stopwatch.createStarted();
150 
151 		TermConcept concept = fetchLoadedCode(theCodeSystemResourcePid, theCodeSystemVersionPid, theCode);
152 		if (concept == null) {
153 			return Collections.emptySet();
154 		}
155 
156 		Set<TermConcept> retVal = new HashSet<TermConcept>();
157 		retVal.add(concept);
158 
159 		fetchParents(concept, retVal);
160 
161 		ourLog.info("Fetched {} codes above code {} in {}ms", new Object[] { retVal.size(), theCode, stopwatch.elapsed(TimeUnit.MILLISECONDS) });
162 		return retVal;
163 	}

241 	private void persistChildren(TermConcept theConcept, TermCodeSystemVersion theCodeSystem, IdentityHashMap<TermConcept, Object> theConceptsStack, int theTotalConcepts) {
242 		if (theConceptsStack.put(theConcept, PLACEHOLDER_OBJECT) != null) {
243 			return;
244 		}
245 
246 		if (theConceptsStack.size() == 1 || theConceptsStack.size() % 10000 == 0) {
247 			float pct = (float) theConceptsStack.size() / (float) theTotalConcepts;
248 			ourLog.info("Have processed {}/{} concepts ({}%)", theConceptsStack.size(), theTotalConcepts, (int)( pct*100.0f));
249 		}
250 
251 		theConcept.setCodeSystem(theCodeSystem);
252 		theConcept.setIndexStatus(BaseHapiFhirDao.INDEX_STATUS_INDEXED);
253 
254 		if (theConceptsStack.size() <= myDaoConfig.getDeferIndexingForCodesystemsOfSize()) {
255 			saveConcept(theConcept);
256 		} else {
257 			myConceptsToSaveLater.add(theConcept);
258 		}
259 		
260 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
261 			persistChildren(next.getChild(), theCodeSystem, theConceptsStack, theTotalConcepts);
262 		}
263 
264 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
265 			if (theConceptsStack.size() <= myDaoConfig.getDeferIndexingForCodesystemsOfSize()) {
266 				saveConceptLink(next);
267 			} else {
268 				myConceptLinksToSaveLater.add(next);
269 			}
270 		}
271 		
272 	}

280 	private int saveConcept(TermConcept theConcept) {
281 		int retVal = 0;
282 		retVal += ensureParentsSaved(theConcept.getParents());
283 		if (theConcept.getId() == null || theConcept.getIndexStatus() == null) {
284 			retVal++;
285 			theConcept.setIndexStatus(BaseHapiFhirDao.INDEX_STATUS_INDEXED);
286 			myConceptDao.saveAndFlush(theConcept);
287 		}
288 		
289 		ourLog.trace("Saved {} and got PID {}", theConcept.getCode(), theConcept.getId());
290 		return retVal;
291 	}

293 	private int ensureParentsSaved(Collection<TermConceptParentChildLink> theParents) {
294 		ourLog.trace("Checking {} parents", theParents.size());
295 		int retVal = 0;
296 		
297 		for (TermConceptParentChildLink nextLink : theParents) {
298 			if (nextLink.getRelationshipType() == RelationshipTypeEnum.ISA) {
299 				TermConcept nextParent = nextLink.getParent();
300 				retVal += ensureParentsSaved(nextParent.getParents());
301 				if (nextParent.getId() == null) {
302 					myConceptDao.saveAndFlush(nextParent);
303 					retVal++;
304 					ourLog.debug("Saved parent code {} and got id {}", nextParent.getCode(), nextParent.getId());
305 				}
306 			}
307 		}
308 		
309 		return retVal;
310 	}

312 	private void populateVersion(TermConcept theNext, TermCodeSystemVersion theCodeSystemVersion) {
313 		if (theNext.getCodeSystem() != null) {
314 			return;
315 		}
316 		theNext.setCodeSystem(theCodeSystemVersion);
317 		for (TermConceptParentChildLink next : theNext.getChildren()) {
318 			populateVersion(next.getChild(), theCodeSystemVersion);
319 		}
320 	}

322 	@Scheduled(fixedRate=5000)
323 	@Transactional(propagation=Propagation.REQUIRED)
324 	@Override
325 	public synchronized void saveDeferred() {
326 		if (!myProcessDeferred) {
327 			return;
328 		} else if (myConceptsToSaveLater.isEmpty() && myConceptLinksToSaveLater.isEmpty()) {
329 			processReindexing();
330 			return;
331 		}
332 		
333 		int codeCount = 0, relCount = 0;
334 		StopWatch stopwatch = new StopWatch();
335 		
336 		int count = Math.min(myDaoConfig.getDeferIndexingForCodesystemsOfSize(), myConceptsToSaveLater.size());
337 		ourLog.info("Saving {} deferred concepts...", count);
338 		while (codeCount < count && myConceptsToSaveLater.size() > 0) {
339 			TermConcept next = myConceptsToSaveLater.remove(0);
340 			codeCount += saveConcept(next);
341 		}
342 
343 		if (codeCount > 0) {
344 			ourLog.info("Saved {} deferred concepts ({} codes remain and {} relationships remain) in {}ms ({}ms / code)", new Object[] {codeCount, myConceptsToSaveLater.size(), myConceptLinksToSaveLater.size(), stopwatch.getMillis(), stopwatch.getMillisPerOperation(codeCount)});
345 		}
346 		
347 		if (codeCount == 0) {
348 			count = Math.min(myDaoConfig.getDeferIndexingForCodesystemsOfSize(), myConceptLinksToSaveLater.size());
349 			ourLog.info("Saving {} deferred concept relationships...", count);
350 			while (relCount < count && myConceptLinksToSaveLater.size() > 0) {
351 				TermConceptParentChildLink next = myConceptLinksToSaveLater.remove(0);
352 				
353 				if (myConceptDao.findOne(next.getChild().getId()) == null || myConceptDao.findOne(next.getParent().getId()) == null) {
354 					ourLog.warn("Not inserting link from child {} to parent {} because it appears to have been deleted", next.getParent().getCode(), next.getChild().getCode());
355 					continue;
356 				}
357 				
358 				saveConceptLink(next);
359 				relCount++;
360 			}
361 		}
362 		
363 		if (relCount > 0) {
364 			ourLog.info("Saved {} deferred relationships ({} remain) in {}ms ({}ms / code)", new Object[] {relCount, myConceptLinksToSaveLater.size(), stopwatch.getMillis(), stopwatch.getMillisPerOperation(codeCount)});
365 		}
366 		
367 		if ((myConceptsToSaveLater.size() + myConceptLinksToSaveLater.size()) == 0) {
368 			ourLog.info("All deferred concepts and relationships have now been synchronized to the database");
369 		}
370 	}

372 	@Autowired
373 	private PlatformTransactionManager myTransactionMgr;
374 	
375 	private void processReindexing() {
376 		if (System.currentTimeMillis() < myNextReindexPass) {
377 			return;
378 		}
379 		
380 		TransactionTemplate tt = new TransactionTemplate(myTransactionMgr);
381 		tt.setPropagationBehavior(TransactionTemplate.PROPAGATION_REQUIRES_NEW);
382 		tt.execute(new TransactionCallbackWithoutResult() {
383 			@Override
384 			protected void doInTransactionWithoutResult(TransactionStatus theArg0) {
385 				int maxResult = 1000;
386 				Page<TermConcept> resources = myConceptDao.findResourcesRequiringReindexing(new PageRequest(0, maxResult));
387 				if (resources.hasContent() == false) {
388 					myNextReindexPass = System.currentTimeMillis() + DateUtils.MILLIS_PER_MINUTE;
389 					return;
390 				}
391 
392 				ourLog.info("Indexing {} / {} concepts", resources.getContent().size(), resources.getTotalElements());
393 
394 				int count = 0;
395 				StopWatch stopwatch = new StopWatch();
396 
397 				for (TermConcept resourceTable : resources) {
398 					saveConcept(resourceTable);
399 					count++;
400 				}
401 				
402 				ourLog.info("Indexed {} / {} concepts in {}ms - Avg {}ms / resource", new Object[] { count, resources.getContent().size(), stopwatch.getMillis(), stopwatch.getMillisPerOperation(count) });
403 			}
404 		});
405 
406 	}

413 	@Override
414 	@Transactional(propagation = Propagation.REQUIRED)
415 	public void storeNewCodeSystemVersion(Long theCodeSystemResourcePid, String theSystemUri, TermCodeSystemVersion theCodeSystemVersion) {
416 		ourLog.info("Storing code system");
417 
418 		ValidateUtil.isTrueOrThrowInvalidRequest(theCodeSystemVersion.getResource() != null, "No resource supplied");
419 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theSystemUri, "No system URI supplied");
420 
421 		// Grab the existing versions so we can delete them later
422 		List<TermCodeSystemVersion> existing = myCodeSystemVersionDao.findByCodeSystemResource(theCodeSystemResourcePid);
423 
424 		/*
425 		 * For now we always delete old versions.. At some point it would be nice to allow configuration to keep old versions
426 		 */
427 
428 		ourLog.info("Deleting old code system versions");
429 		for (TermCodeSystemVersion next : existing) {
430 			ourLog.info(" * Deleting code system version {}", next.getPid());
431 			myConceptParentChildLinkDao.deleteByCodeSystemVersion(next.getPid());
432 			myConceptDao.deleteByCodeSystemVersion(next.getPid());
433 		}
434 
435 		ourLog.info("Flushing...");
436 		
437 		myConceptParentChildLinkDao.flush();
438 		myConceptDao.flush();
439 
440 		ourLog.info("Done flushing");
441 
442 		/*
443 		 * Do the upload
444 		 */
445 
446 		TermCodeSystem codeSystem = getCodeSystem(theSystemUri);
447 		if (codeSystem == null) {
448 			codeSystem = myCodeSystemDao.findByResourcePid(theCodeSystemResourcePid);
449 			if (codeSystem == null) {
450 				codeSystem = new TermCodeSystem();
451 			}
452 			codeSystem.setResource(theCodeSystemVersion.getResource());
453 			codeSystem.setCodeSystemUri(theSystemUri);
454 			myCodeSystemDao.save(codeSystem);
455 		} else {
456 			if (!ObjectUtil.equals(codeSystem.getResource().getId(), theCodeSystemVersion.getResource().getId())) {
457 				String msg = myContext.getLocalizer().getMessage(BaseHapiTerminologySvc.class, "cannotCreateDuplicateCodeSystemUri", theSystemUri,
458 						codeSystem.getResource().getIdDt().toUnqualifiedVersionless().getValue());
459 				throw new UnprocessableEntityException(msg);
460 			}
461 		}
462 
463 		ourLog.info("Validating all codes in CodeSystem for storage (this can take some time for large sets)");
464 
465 		// Validate the code system
466 		ArrayList<String> conceptsStack = new ArrayList<String>();
467 		IdentityHashMap<TermConcept, Object> allConcepts = new IdentityHashMap<TermConcept, Object>();
468 		int totalCodeCount = 0;
469 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
470 			totalCodeCount += validateConceptForStorage(next, theCodeSystemVersion, conceptsStack, allConcepts);
471 		}
472 
473 		ourLog.info("Saving version");
474 
475 		TermCodeSystemVersion codeSystemVersion = myCodeSystemVersionDao.saveAndFlush(theCodeSystemVersion);
476 
477 		ourLog.info("Saving code system");
478 
479 		codeSystem.setCurrentVersion(theCodeSystemVersion);
480 		codeSystem = myCodeSystemDao.saveAndFlush(codeSystem);
481 
482 		ourLog.info("Setting codesystemversion on {} concepts...", totalCodeCount);
483 
484 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
485 			populateVersion(next, codeSystemVersion);
486 		}
487 
488 		ourLog.info("Saving {} concepts...", totalCodeCount);
489 		
490 		IdentityHashMap<TermConcept, Object> conceptsStack2 = new IdentityHashMap<TermConcept, Object>();
491 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
492 			persistChildren(next, codeSystemVersion, conceptsStack2, totalCodeCount);
493 		}
494 
495 		ourLog.info("Done saving concepts, flushing to database");
496 
497 		myConceptDao.flush();
498 		myConceptParentChildLinkDao.flush();
499 
500 		ourLog.info("Done deleting old code system versions");
501 		
502 		if (myConceptsToSaveLater.size() > 0 || myConceptLinksToSaveLater.size() > 0) {
503 			ourLog.info("Note that some concept saving was deferred - still have {} concepts and {} relationships", myConceptsToSaveLater.size(), myConceptLinksToSaveLater.size());
504 		}
505 	}

521 	private int validateConceptForStorage(TermConcept theConcept, TermCodeSystemVersion theCodeSystem, ArrayList<String> theConceptsStack,
522 			IdentityHashMap<TermConcept, Object> theAllConcepts) {
523 		ValidateUtil.isTrueOrThrowInvalidRequest(theConcept.getCodeSystem() != null, "CodesystemValue is null");
524 		ValidateUtil.isTrueOrThrowInvalidRequest(theConcept.getCodeSystem() == theCodeSystem, "CodeSystems are not equal");
525 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theConcept.getCode(), "Codesystem contains a code with no code value");
526 
527 		if (theConceptsStack.contains(theConcept.getCode())) {
528 			throw new InvalidRequestException("CodeSystem contains circular reference around code " + theConcept.getCode());
529 		}
530 		theConceptsStack.add(theConcept.getCode());
531 		
532 		int retVal = 0;
533 		if (theAllConcepts.put(theConcept, theAllConcepts) == null) {
534 			if (theAllConcepts.size() % 1000 == 0) {
535 				ourLog.info("Have validated {} concepts", theAllConcepts.size());
536 			}
537 			retVal = 1;
538 		}
539 
540 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
541 			next.setCodeSystem(theCodeSystem);
542 			retVal += validateConceptForStorage(next.getChild(), theCodeSystem, theConceptsStack, theAllConcepts);
543 		}
544 
545 		theConceptsStack.remove(theConceptsStack.size() - 1);
546 
547 		return retVal;
548 	}
