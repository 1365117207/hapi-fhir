134 	private Class<?> determineElementType(Field next) {
135 		Class<?> nextElementType = next.getType();
136 		if (List.class.equals(nextElementType)) {
137 			nextElementType = ReflectionUtil.getGenericCollectionTypeOfField(next);
138 		} else if (Collection.class.isAssignableFrom(nextElementType)) {
139 			throw new ConfigurationException("Field '" + next.getName() + "' in type '" + next.getClass().getCanonicalName() + "' is a Collection - Only java.util.List curently supported");
140 		}
141 		return nextElementType;
142 	}

145 	private IValueSetEnumBinder<Enum<?>> getBoundCodeBinder(Field theNext) {
146 		Class<?> bound = getGenericCollectionTypeOfCodedField(theNext);
147 		if (bound == null) {
148 			throw new ConfigurationException("Field '" + theNext + "' has no parameter for " + BoundCodeDt.class.getSimpleName() + " to determine enum type");
149 		}
150 
151 		try {
152 			Field bindingField = bound.getField("VALUESET_BINDER");
153 			return (IValueSetEnumBinder<Enum<?>>) bindingField.get(null);
154 		} catch (IllegalArgumentException e) {
155 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
156 		} catch (IllegalAccessException e) {
157 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
158 		} catch (NoSuchFieldException e) {
159 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
160 		} catch (SecurityException e) {
161 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
162 		}
163 	}

165 	private void init(Map<Class<? extends IElement>, BaseRuntimeElementDefinition<?>> theExistingDefinitions, Set<Class<? extends IElement>> toScan) {
166 		if (theExistingDefinitions != null) {
167 			myClassToElementDefinitions.putAll(theExistingDefinitions);
168 		}
169 
170 		int startSize = myClassToElementDefinitions.size();
171 		long start = System.currentTimeMillis();
172 
173 		InputStream str = ModelScanner.class.getResourceAsStream("/ca/uhn/fhir/model/dstu/fhirversion.properties");
174 		if (str == null) {
175 			str = ModelScanner.class.getResourceAsStream("ca/uhn/fhir/model/dstu/fhirversion.properties");
176 		}
177 		if (str == null) {
178 			throw new ConfigurationException("Can not find model property file on classpath: " + "/ca/uhn/fhir/model/dstu/model.properties");
179 		}
180 		Properties prop = new Properties();
181 		try {
182 			prop.load(str);
183 			for (Entry<Object, Object> nextEntry : prop.entrySet()) {
184 				String nextKey = nextEntry.getKey().toString();
185 				String nextValue = nextEntry.getValue().toString();
186 
187 				if (!nextKey.startsWith("datatype.")) {
188 					if (nextKey.startsWith("resource.")) {
189 						String resName = nextKey.substring("resource.".length()).toLowerCase();
190 						myNameToResourceType.put(resName, nextValue);
191 					}
192 					continue;
193 				}
194 
195 				try {
196 					@SuppressWarnings("unchecked")
197 					Class<? extends IElement> nextClass = (Class<? extends IElement>) Class.forName((String) nextValue);
198 					if (!IElement.class.isAssignableFrom(nextClass)) {
199 						ourLog.warn("Class is not assignable from " + IElement.class.getSimpleName() + ": " + nextValue);
200 						continue;
201 					}
202 
203 					toScan.add(nextClass);
204 				} catch (ClassNotFoundException e) {
205 					ourLog.warn("Unknown class exception: " + nextValue, e);
206 				}
207 			}
208 		} catch (IOException e) {
209 			throw new ConfigurationException("Failed to load model property file from classpath: " + "/ca/uhn/fhir/model/dstu/model.properties");
210 		}
211 
212 		// toScan.add(DateDt.class);
213 		// toScan.add(CodeDt.class);
214 		// toScan.add(DecimalDt.class);
215 		// toScan.add(AttachmentDt.class);
216 		// toScan.add(ResourceReferenceDt.class);
217 		// toScan.add(QuantityDt.class);
218 
219 		do {
220 			for (Class<? extends IElement> nextClass : toScan) {
221 				scan(nextClass);
222 			}
223 			for (Iterator<Class<? extends IElement>> iter = myScanAlso.iterator(); iter.hasNext();) {
224 				if (myClassToElementDefinitions.containsKey(iter.next())) {
225 					iter.remove();
226 				}
227 			}
228 			toScan.clear();
229 			toScan.addAll(myScanAlso);
230 			myScanAlso.clear();
231 		} while (!toScan.isEmpty());
232 
233 		for (Entry<Class<? extends IElement>, BaseRuntimeElementDefinition<?>> nextEntry : myClassToElementDefinitions.entrySet()) {
234 			if (theExistingDefinitions != null && theExistingDefinitions.containsKey(nextEntry.getKey())) {
235 				continue;
236 			}
237 			BaseRuntimeElementDefinition<?> next = nextEntry.getValue();
238 			next.sealAndInitialize(myClassToElementDefinitions);
239 		}
240 
241 		myRuntimeChildUndeclaredExtensionDefinition = new RuntimeChildUndeclaredExtensionDefinition();
242 		myRuntimeChildUndeclaredExtensionDefinition.sealAndInitialize(myClassToElementDefinitions);
243 
244 		long time = System.currentTimeMillis() - start;
245 		int size = myClassToElementDefinitions.size() - startSize;
246 		ourLog.info("Done scanning FHIR library, found {} model entries in {}ms", size, time);
247 	}

249 	private void scan(Class<? extends IElement> theClass) throws ConfigurationException {
250 		BaseRuntimeElementDefinition<?> existingDef = myClassToElementDefinitions.get(theClass);
251 		if (existingDef != null) {
252 			return;
253 		}
254 
255 		ResourceDef resourceDefinition = theClass.getAnnotation(ResourceDef.class);
256 		if (resourceDefinition != null) {
257 			if (!IResource.class.isAssignableFrom(theClass)) {
258 				throw new ConfigurationException("Resource type contains a @" + ResourceDef.class.getSimpleName() + " annotation but does not implement " + IResource.class.getCanonicalName() + ": " + theClass.getCanonicalName());
259 			}
260 			@SuppressWarnings("unchecked")
261 			Class<? extends IResource> resClass = (Class<? extends IResource>) theClass;
262 			scanResource(resClass, resourceDefinition);
263 		}
264 
265 		DatatypeDef datatypeDefinition = theClass.getAnnotation(DatatypeDef.class);
266 		if (datatypeDefinition != null) {
267 			if (ICompositeDatatype.class.isAssignableFrom(theClass)) {
268 				@SuppressWarnings("unchecked")
269 				Class<? extends ICompositeDatatype> resClass = (Class<? extends ICompositeDatatype>) theClass;
270 				scanCompositeDatatype(resClass, datatypeDefinition);
271 			} else if (IPrimitiveDatatype.class.isAssignableFrom(theClass)) {
272 				@SuppressWarnings({ "unchecked" })
273 				Class<? extends IPrimitiveDatatype<?>> resClass = (Class<? extends IPrimitiveDatatype<?>>) theClass;
274 				scanPrimitiveDatatype(resClass, datatypeDefinition);
275 			} else {
276 				throw new ConfigurationException("Resource type contains a @" + DatatypeDef.class.getSimpleName() + " annotation but does not implement " + IDatatype.class.getCanonicalName() + ": " + theClass.getCanonicalName());
277 			}
278 		}
279 
280 		Block blockDefinition = theClass.getAnnotation(Block.class);
281 		if (blockDefinition != null) {
282 			if (IResourceBlock.class.isAssignableFrom(theClass)) {
283 				@SuppressWarnings("unchecked")
284 				Class<? extends IResourceBlock> blockClass = (Class<? extends IResourceBlock>) theClass;
285 				scanBlock(blockClass, blockDefinition);
286 			} else {
287 				throw new ConfigurationException("Type contains a @" + Block.class.getSimpleName() + " annotation but does not implement " + IResourceBlock.class.getCanonicalName() + ": " + theClass.getCanonicalName());
288 			}
289 		}
290 
291 		if (blockDefinition == null && datatypeDefinition == null && resourceDefinition == null) {
292 			throw new ConfigurationException("Resource type does not contain any valid HAPI-FHIR annotations: " + theClass.getCanonicalName());
293 		}
294 	}

296 	private void scanBlock(Class<? extends IResourceBlock> theClass, Block theBlockDefinition) {
297 		ourLog.debug("Scanning resource block class: {}", theClass.getName());
298 
299 		String resourceName = theClass.getCanonicalName();
300 		if (isBlank(resourceName)) {
301 			throw new ConfigurationException("Block type @" + Block.class.getSimpleName() + " annotation contains no name: " + theClass.getCanonicalName());
302 		}
303 
304 		RuntimeResourceBlockDefinition resourceDef = new RuntimeResourceBlockDefinition(resourceName, theClass);
305 		myClassToElementDefinitions.put(theClass, resourceDef);
306 
307 		scanCompositeElementForChildren(theClass, resourceDef);
308 	}

310 	private void scanCompositeDatatype(Class<? extends ICompositeDatatype> theClass, DatatypeDef theDatatypeDefinition) {
311 		ourLog.debug("Scanning resource class: {}", theClass.getName());
312 
313 		RuntimeCompositeDatatypeDefinition resourceDef;
314 		if (theClass.equals(ExtensionDt.class)) {
315 			resourceDef = new RuntimeExtensionDtDefinition(theDatatypeDefinition, theClass);
316 		} else {
317 			resourceDef = new RuntimeCompositeDatatypeDefinition(theDatatypeDefinition, theClass);
318 		}
319 		myClassToElementDefinitions.put(theClass, resourceDef);
320 		scanCompositeElementForChildren(theClass, resourceDef);
321 	}

323 	@SuppressWarnings("unchecked")
324 	private void scanCompositeElementForChildren(Class<? extends ICompositeElement> theClass, BaseRuntimeElementCompositeDefinition<?> theDefinition) {
325 		Set<String> elementNames = new HashSet<String>();
326 		TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderToElementDef = new TreeMap<Integer, BaseRuntimeDeclaredChildDefinition>();
327 		TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderToExtensionDef = new TreeMap<Integer, BaseRuntimeDeclaredChildDefinition>();
328 
329 		LinkedList<Class<? extends ICompositeElement>> classes = new LinkedList<Class<? extends ICompositeElement>>();
330 
331 		/*
332 		 * We scan classes for annotated fields in the class but also all of its superclasses
333 		 */
334 		Class<? extends ICompositeElement> current = theClass;
335 		do {
336 			classes.push(current);
337 			if (ICompositeElement.class.isAssignableFrom(current.getSuperclass())) {
338 				current = (Class<? extends ICompositeElement>) current.getSuperclass();
339 			} else {
340 				current = null;
341 			}
342 		} while (current != null);
343 
344 		for (Class<? extends ICompositeElement> next : classes) {
345 			scanCompositeElementForChildren(next, elementNames, orderToElementDef, orderToExtensionDef);
346 		}
347 
348 		// while (orderToElementDef.size() > 0 && orderToElementDef.firstKey() <
349 		// 0) {
350 		// BaseRuntimeDeclaredChildDefinition elementDef =
351 		// orderToElementDef.remove(orderToElementDef.firstKey());
352 		// if (elementDef.getElementName().equals("identifier")) {
353 		// orderToElementDef.put(theIdentifierOrder, elementDef);
354 		// } else {
355 		// throw new ConfigurationException("Don't know how to handle element: "
356 		// + elementDef.getElementName());
357 		// }
358 		// }
359 
360 		TreeSet<Integer> orders = new TreeSet<Integer>();
361 		orders.addAll(orderToElementDef.keySet());
362 		orders.addAll(orderToExtensionDef.keySet());
363 
364 		for (Integer i : orders) {
365 			BaseRuntimeChildDefinition nextChild = orderToElementDef.get(i);
366 			if (nextChild != null) {
367 				theDefinition.addChild(nextChild);
368 			}
369 			BaseRuntimeDeclaredChildDefinition nextExt = orderToExtensionDef.get(i);
370 			if (nextExt != null) {
371 				theDefinition.addExtension((RuntimeChildDeclaredExtensionDefinition) nextExt);
372 			}
373 		}
374 
375 	}

377 	@SuppressWarnings("unchecked")
378 	private void scanCompositeElementForChildren(Class<? extends ICompositeElement> theClass, Set<String> elementNames, TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> theOrderToElementDef, TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> theOrderToExtensionDef) {
379 		int baseElementOrder = theOrderToElementDef.isEmpty() ? 0 : theOrderToElementDef.lastEntry().getKey() + 1;
380 
381 		for (Field next : theClass.getDeclaredFields()) {
382 
383 			Child childAnnotation = next.getAnnotation(Child.class);
384 			if (childAnnotation == null) {
385 				ourLog.debug("Ignoring non-type field '" + next.getName() + "' on target type: " + theClass);
386 				continue;
387 			}
388 
389 			Description descriptionAnnotation = next.getAnnotation(Description.class);
390 
391 			TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderMap = theOrderToElementDef;
392 			Extension extensionAttr = next.getAnnotation(Extension.class);
393 			if (extensionAttr != null) {
394 				orderMap = theOrderToExtensionDef;
395 			}
396 
397 			String elementName = childAnnotation.name();
398 			int order = childAnnotation.order();
399 			if (order == Child.REPLACE_PARENT) {
400 
401 				if (extensionAttr != null) {
402 
403 					for (Entry<Integer, BaseRuntimeDeclaredChildDefinition> nextEntry : orderMap.entrySet()) {
404 						BaseRuntimeDeclaredChildDefinition nextDef = nextEntry.getValue();
405 						if (nextDef instanceof RuntimeChildDeclaredExtensionDefinition) {
406 							if (nextDef.getExtensionUrl().equals(extensionAttr.url())) {
407 								order = nextEntry.getKey();
408 								orderMap.remove(nextEntry.getKey());
409 								elementNames.remove(elementName);
410 								break;
411 							}
412 						}
413 					}
414 					if (order == Child.REPLACE_PARENT) {
415 						throw new ConfigurationException("Field " + next.getName() + "' on target type " + theClass.getSimpleName() + " has order() of REPLACE_PARENT (" + Child.REPLACE_PARENT + ") but no parent element with extension URL " + extensionAttr.url()
416 								+ " could be found on type " + next.getDeclaringClass().getSimpleName());
417 					}
418 
419 				} else {
420 
421 					for (Entry<Integer, BaseRuntimeDeclaredChildDefinition> nextEntry : orderMap.entrySet()) {
422 						BaseRuntimeDeclaredChildDefinition nextDef = nextEntry.getValue();
423 						if (elementName.equals(nextDef.getElementName())) {
424 							order = nextEntry.getKey();
425 							orderMap.remove(nextEntry.getKey());
426 							elementNames.remove(elementName);
427 							break;
428 						}
429 					}
430 					if (order == Child.REPLACE_PARENT) {
431 						throw new ConfigurationException("Field " + next.getName() + "' on target type " + theClass.getSimpleName() + " has order() of REPLACE_PARENT (" + Child.REPLACE_PARENT + ") but no parent element with name " + elementName + " could be found on type "
432 								+ next.getDeclaringClass().getSimpleName());
433 					}
434 
435 				}
436 
437 			}
438 			if (order < 0 && order != Child.ORDER_UNKNOWN) {
439 				throw new ConfigurationException("Invalid order '" + order + "' on @Child for field '" + next.getName() + "' on target type: " + theClass);
440 			}
441 			if (order != Child.ORDER_UNKNOWN) {
442 				order = order + baseElementOrder;
443 			}
444 			int min = childAnnotation.min();
445 			int max = childAnnotation.max();
446 			/*
447 			 * Anything that's marked as unknown is given a new ID that is <0 so that it doesn't conflict wityh any
448 			 * given IDs and can be figured out later
449 			 */
450 			while (order == Child.ORDER_UNKNOWN && orderMap.containsKey(order)) {
451 				order--;
452 			}
453 
454 			List<Class<? extends IElement>> choiceTypes = new ArrayList<Class<? extends IElement>>();
455 			for (Class<? extends IElement> nextChoiceType : childAnnotation.type()) {
456 				choiceTypes.add(nextChoiceType);
457 			}
458 
459 			if (orderMap.containsKey(order)) {
460 				throw new ConfigurationException("Detected duplicate field order '" + childAnnotation.order() + "' for element named '" + elementName + "' in type '" + theClass.getCanonicalName() + "'");
461 			}
462 
463 			if (elementNames.contains(elementName)) {
464 				throw new ConfigurationException("Detected duplicate field name '" + elementName + "' in type '" + theClass.getCanonicalName() + "'");
465 			}
466 
467 			Class<?> nextElementType = determineElementType(next);
468 
469 			if (nextElementType.equals(ContainedDt.class)) {
470 				/*
471 				 * Child is contained resources
472 				 */
473 				RuntimeChildContainedResources def = new RuntimeChildContainedResources(next, childAnnotation, descriptionAnnotation, elementName);
474 				orderMap.put(order, def);
475 
476 			} else if (choiceTypes.size() > 1 && !ResourceReferenceDt.class.isAssignableFrom(nextElementType)) {
477 				/*
478 				 * Child is a choice element
479 				 */
480 				for (Class<? extends IElement> nextType : choiceTypes) {
481 					addScanAlso(nextType);
482 				}
483 				RuntimeChildChoiceDefinition def = new RuntimeChildChoiceDefinition(next, elementName, childAnnotation, descriptionAnnotation, choiceTypes);
484 				orderMap.put(order, def);
485 
486 			} else if (next.getType().equals(ExtensionDt.class)) {
487 
488 				RuntimeChildExtensionDt def = new RuntimeChildExtensionDt(next, elementName, childAnnotation, descriptionAnnotation);
489 				orderMap.put(order, def);
490 				if (IElement.class.isAssignableFrom(nextElementType)) {
491 					addScanAlso((Class<? extends IElement>) nextElementType);
492 				}
493 
494 			} else if (extensionAttr != null) {
495 				/*
496 				 * Child is an extension
497 				 */
498 				Class<? extends IElement> et = (Class<? extends IElement>) nextElementType;
499 				RuntimeChildDeclaredExtensionDefinition def = new RuntimeChildDeclaredExtensionDefinition(next, childAnnotation, descriptionAnnotation, extensionAttr, elementName, extensionAttr.url(), et);
500 				orderMap.put(order, def);
501 				if (IElement.class.isAssignableFrom(nextElementType)) {
502 					addScanAlso((Class<? extends IElement>) nextElementType);
503 				}
504 			} else if (BaseResourceReferenceDt.class.isAssignableFrom(nextElementType)) {
505 				/*
506 				 * Child is a resource reference
507 				 */
508 				List<Class<? extends IResource>> refTypesList = new ArrayList<Class<? extends IResource>>();
509 				for (Class<? extends IElement> nextType : childAnnotation.type()) {
510 					if (IResource.class.isAssignableFrom(nextType) == false) {
511 						throw new ConfigurationException("Field '" + next.getName() + "' in class '" + next.getDeclaringClass().getCanonicalName() + "' is of type " + ResourceReferenceDt.class + " but contains a non-resource type: " + nextType.getCanonicalName());
512 					}
513 					refTypesList.add((Class<? extends IResource>) nextType);
514 					addScanAlso(nextType);
515 				}
516 				RuntimeChildResourceDefinition def = new RuntimeChildResourceDefinition(next, elementName, childAnnotation, descriptionAnnotation, refTypesList);
517 				orderMap.put(order, def);
518 
519 			} else if (IResourceBlock.class.isAssignableFrom(nextElementType)) {
520 				/*
521 				 * Child is a resource block (i.e. a sub-tag within a resource) TODO: do these have a better name
522 				 * according to HL7?
523 				 */
524 
525 				Class<? extends IResourceBlock> blockDef = (Class<? extends IResourceBlock>) nextElementType;
526 				addScanAlso(blockDef);
527 				RuntimeChildResourceBlockDefinition def = new RuntimeChildResourceBlockDefinition(next, childAnnotation, descriptionAnnotation, elementName, blockDef);
528 				orderMap.put(order, def);
529 
530 			} else if (IDatatype.class.equals(nextElementType) || IElement.class.equals(nextElementType)) {
531 
532 				RuntimeChildAny def = new RuntimeChildAny(next, elementName, childAnnotation, descriptionAnnotation);
533 				orderMap.put(order, def);
534 
535 			} else if (IDatatype.class.isAssignableFrom(nextElementType)) {
536 				Class<? extends IDatatype> nextDatatype = (Class<? extends IDatatype>) nextElementType;
537 
538 				addScanAlso(nextDatatype);
539 				BaseRuntimeChildDatatypeDefinition def;
540 				if (IPrimitiveDatatype.class.isAssignableFrom(nextElementType)) {
541 					if (nextElementType.equals(BoundCodeDt.class)) {
542 						IValueSetEnumBinder<Enum<?>> binder = getBoundCodeBinder(next);
543 						def = new RuntimeChildPrimitiveBoundCodeDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype, binder);
544 					} else {
545 						def = new RuntimeChildPrimitiveDatatypeDefinition(next, elementName, descriptionAnnotation, childAnnotation, nextDatatype);
546 					}
547 				} else {
548 					if (nextElementType.equals(BoundCodeableConceptDt.class)) {
549 						IValueSetEnumBinder<Enum<?>> binder = getBoundCodeBinder(next);
550 						def = new RuntimeChildCompositeBoundDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype, binder);
551 					} else if (NarrativeDt.class.getSimpleName().equals(nextElementType.getSimpleName())) {
552 						def = new RuntimeChildNarrativeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype);
553 					} else {
554 						def = new RuntimeChildCompositeDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype);
555 					}
556 				}
557 
558 				CodeableConceptElement concept = next.getAnnotation(CodeableConceptElement.class);
559 				if (concept != null) {
560 					if (!ICodedDatatype.class.isAssignableFrom(nextDatatype)) {
561 						throw new ConfigurationException("Field '" + elementName + "' in type '" + theClass.getCanonicalName() + "' is marked as @" + CodeableConceptElement.class.getCanonicalName() + " but type is not a subtype of " + ICodedDatatype.class.getName());
562 					} else {
563 						Class<? extends ICodeEnum> type = concept.type();
564 						myScanAlsoCodeTable.add(type);
565 						def.setCodeType(type);
566 					}
567 				}
568 
569 				orderMap.put(order, def);
570 
571 			} else {
572 				throw new ConfigurationException("Field '" + elementName + "' in type '" + theClass.getCanonicalName() + "' is not a valid child type: " + nextElementType);
573 			}
574 
575 			elementNames.add(elementName);
576 		}
577 	}

580 		ourLog.debug("Scanning resource class: {}", theClass.getName());
581 
582 		String resourceName = theDatatypeDefinition.name();
583 		if (isBlank(resourceName)) {
584 			throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name: " + theClass.getCanonicalName());
585 		}
586 
587 		BaseRuntimeElementDefinition<?> resourceDef;
588 		if (theClass.equals(XhtmlDt.class)) {
589 			@SuppressWarnings("unchecked")
590 			Class<XhtmlDt> clazz = (Class<XhtmlDt>) theClass;
591 			resourceDef = new RuntimePrimitiveDatatypeNarrativeDefinition(resourceName, clazz);
592 		} else {
593 			resourceDef = new RuntimePrimitiveDatatypeDefinition(theDatatypeDefinition, theClass);
594 		}
595 		myClassToElementDefinitions.put(theClass, resourceDef);
596 
597 		return resourceName;
598 	}

600 	private String scanResource(Class<? extends IResource> theClass, ResourceDef resourceDefinition) {
601 		ourLog.debug("Scanning resource class: {}", theClass.getName());
602 
603 		boolean primaryNameProvider = true;
604 		String resourceName = resourceDefinition.name();
605 		if (isBlank(resourceName)) {
606 			Class<?> parent = theClass.getSuperclass();
607 			primaryNameProvider = false;
608 			while (parent.equals(Object.class) == false && isBlank(resourceName)) {
609 				ResourceDef nextDef = parent.getAnnotation(ResourceDef.class);
610 				if (nextDef != null) {
611 					resourceName = nextDef.name();
612 				}
613 				parent = parent.getSuperclass();
614 			}
615 			if (isBlank(resourceName)) {
616 				throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name(): " + theClass.getCanonicalName() + " - This is only allowed for types that extend other resource types ");
617 			}
618 		}
619 
620 		// if (myNameToResourceDefinitions.containsKey(resourceName)) {
621 		// if (!myNameToResourceDefinitions.get(resourceName).getImplementingClass().equals(theClass)) {
622 		// // throw new
623 		// // ConfigurationException("Detected duplicate element name '" +
624 		// // resourceName + "' in types '" + theClass.getCanonicalName() +
625 		// // "' and '"
626 		// // +
627 		// // myNameToResourceDefinitions.get(resourceName).getImplementingClass()
628 		// // + "'");
629 		// } else {
630 		// return resourceName;
631 		// }
632 		// }
633 
634 		String resourceId = resourceDefinition.id();
635 		if (isBlank(resourceId)) {
636 			// throw new ConfigurationException("Resource type @" +
637 			// ResourceDef.class.getSimpleName() +
638 			// " annotation contains no resource ID: " +
639 			// theClass.getCanonicalName());
640 		} else {
641 			if (myIdToResourceDefinition.containsKey(resourceId)) {
642 				throw new ConfigurationException("The following resource types have the same ID of '" + resourceId + "' - " + theClass.getCanonicalName() + " and " + myIdToResourceDefinition.get(resourceId).getImplementingClass().getCanonicalName());
643 			}
644 		}
645 
646 		RuntimeResourceDefinition resourceDef = new RuntimeResourceDefinition(resourceName, theClass, resourceDefinition);
647 		myClassToElementDefinitions.put(theClass, resourceDef);
648 		if (primaryNameProvider) {
649 			myNameToResourceDefinitions.put(resourceName, resourceDef);
650 		}
651 		scanCompositeElementForChildren(theClass, resourceDef);
652 
653 		myIdToResourceDefinition.put(resourceId, resourceDef);
654 
655 		scanResourceForSearchParams(theClass, resourceDef);
656 
657 		return resourceName;
658 	}

660 	private void scanResourceForSearchParams(Class<? extends IResource> theClass, RuntimeResourceDefinition theResourceDef) {
661 
662 		Map<String, RuntimeSearchParam> nameToParam = new HashMap<String, RuntimeSearchParam>();
663 		Map<Field, SearchParamDefinition> compositeFields = new LinkedHashMap<Field, SearchParamDefinition>();
664 
665 		for (Field nextField : theClass.getFields()) {
666 			SearchParamDefinition searchParam = nextField.getAnnotation(SearchParamDefinition.class);
667 			if (searchParam != null) {
668 				SearchParamTypeEnum paramType = SearchParamTypeEnum.valueOf(searchParam.type().toUpperCase());
669 				if (paramType == null) {
670 					throw new ConfigurationException("Searc param " + searchParam.name() + " has an invalid type: " + searchParam.type());
671 				}
672 				if (paramType == SearchParamTypeEnum.COMPOSITE) {
673 					compositeFields.put(nextField, searchParam);
674 					continue;
675 				}
676 				RuntimeSearchParam param = new RuntimeSearchParam(searchParam.name(), searchParam.description(), searchParam.path(), paramType);
677 				theResourceDef.addSearchParam(param);
678 				nameToParam.put(param.getName(), param);
679 			}
680 		}
681 
682 		for (Entry<Field, SearchParamDefinition> nextEntry : compositeFields.entrySet()) {
683 			SearchParamDefinition searchParam = nextEntry.getValue();
684 
685 			List<RuntimeSearchParam> compositeOf = new ArrayList<RuntimeSearchParam>();
686 			for (String nextName : searchParam.compositeOf()) {
687 				RuntimeSearchParam param = nameToParam.get(nextName);
688 				if (param == null) {
689 					ourLog.warn("Search parameter {}.{} declares that it is a composite with compositeOf value '{}' but that is not a valid parametr name itself. Valid values are: {}", new Object[] { theResourceDef.getName(), searchParam.name(), nextName, nameToParam.keySet() });
690 					continue;
691 				}
692 				compositeOf.add(param);
693 			}
694 
695 			RuntimeSearchParam param = new RuntimeSearchParam(searchParam.name(), searchParam.description(), searchParam.path(), SearchParamTypeEnum.COMPOSITE, compositeOf);
696 			theResourceDef.addSearchParam(param);
697 		}
698 	}

700 	private static Class<?> getGenericCollectionTypeOfCodedField(Field next) {
701 		Class<?> type;
702 		ParameterizedType collectionType = (ParameterizedType) next.getGenericType();
703 		Type firstArg = collectionType.getActualTypeArguments()[0];
704 		if (ParameterizedType.class.isAssignableFrom(firstArg.getClass())) {
705 			ParameterizedType pt = ((ParameterizedType) firstArg);
706 			firstArg = pt.getActualTypeArguments()[0];
707 			type = (Class<?>) firstArg;
708 		} else {
709 			type = (Class<?>) firstArg;
710 		}
711 		return type;
712 	}
