73  	private void fillBasics(StructureElement theElement, BaseRuntimeElementDefinition<?> def, LinkedList<String> path, BaseRuntimeDeclaredChildDefinition theChild) {
74  		if (path.isEmpty()) {
75  			path.add(def.getName());
76  			theElement.setName(def.getName());
77  		} else {
78  			path.add(WordUtils.uncapitalize(theChild.getElementName()));
79  			theElement.setName(theChild.getElementName());
80  		}
81  		theElement.setPath(StringUtils.join(path, '.'));
82  	}

84  	private void fillExtensions(Structure theStruct, LinkedList<String> path, List<RuntimeChildDeclaredExtensionDefinition> extList, String elementName, boolean theIsModifier) {
85  		if (extList.size() > 0) {
86  			StructureElement extSlice = theStruct.addElement();
87  			extSlice.setName(elementName);
88  			extSlice.setPath(join(path, '.') + '.' + elementName);
89  			extSlice.getSlicing().getDiscriminator().setValue("url");
90  			extSlice.getSlicing().setOrdered(false);
91  			extSlice.getSlicing().setRules(SlicingRulesEnum.OPEN);
92  			extSlice.getDefinition().addType().setCode(DataTypeEnum.EXTENSION);
93  
94  			for (RuntimeChildDeclaredExtensionDefinition nextExt : extList) {
95  				StructureElement nextProfileExt = theStruct.addElement();
96  				nextProfileExt.getDefinition().setIsModifier(theIsModifier);
97  				nextProfileExt.setName(extSlice.getName());
98  				nextProfileExt.setPath(extSlice.getPath());
99  				fillMinAndMaxAndDefinitions(nextExt, nextProfileExt);
100 				StructureElementDefinitionType type = nextProfileExt.getDefinition().addType();
101 				type.setCode(DataTypeEnum.EXTENSION);
102 				if (nextExt.isDefinedLocally()) {
103 					type.setProfile(nextExt.getExtensionUrl().substring(nextExt.getExtensionUrl().indexOf('#')));
104 				} else {
105 					type.setProfile(nextExt.getExtensionUrl());
106 				}
107 			}
108 		} else {
109 			StructureElement extSlice = theStruct.addElement();
110 			extSlice.setName(elementName);
111 			extSlice.setPath(join(path, '.') + '.' + elementName);
112 			extSlice.getDefinition().setIsModifier(theIsModifier);
113 			extSlice.getDefinition().addType().setCode(DataTypeEnum.EXTENSION);
114 			extSlice.getDefinition().setMin(0);
115 			extSlice.getDefinition().setMax("*");
116 		}
117 	}

119 	private void fillMinAndMaxAndDefinitions(BaseRuntimeDeclaredChildDefinition child, StructureElement elem) {
120 		elem.getDefinition().setMin(child.getMin());
121 		if (child.getMax() == Child.MAX_UNLIMITED) {
122 			elem.getDefinition().setMax("*");
123 		} else {
124 			elem.getDefinition().setMax(Integer.toString(child.getMax()));
125 		}
126 
127 		if (isNotBlank(child.getShortDefinition())) {
128 			elem.getDefinition().getShort().setValue(child.getShortDefinition());
129 		}
130 		if (isNotBlank(child.getFormalDefinition())) {
131 			elem.getDefinition().getFormal().setValue(child.getFormalDefinition());
132 		}
133 	}

135 	private void fillName(StructureElement elem, BaseRuntimeElementDefinition<?> nextDef) {
136 		if (nextDef instanceof RuntimeResourceReferenceDefinition) {
137 			RuntimeResourceReferenceDefinition rr = (RuntimeResourceReferenceDefinition) nextDef;
138 			for (Class<? extends IResource> next : rr.getResourceTypes()) {
139 				StructureElementDefinitionType type = elem.getDefinition().addType();
140 				type.getCode().setValue("ResourceReference");
141 
142 				if (next != IResource.class) {
143 					RuntimeResourceDefinition resDef = rr.getDefinitionForResourceType(next);
144 					type.getProfile().setValueAsString(resDef.getResourceProfile());
145 				}
146 			}
147 
148 			return;
149 		}
150 
151 		StructureElementDefinitionType type = elem.getDefinition().addType();
152 		String name = nextDef.getName();
153 		DataTypeEnum fromCodeString = DataTypeEnum.VALUESET_BINDER.fromCodeString(name);
154 		if (fromCodeString == null) {
155 			throw new ConfigurationException("Unknown type: " + name);
156 		}
157 		type.setCode(fromCodeString);
158 	}

160 	private void fillProfile(Structure theStruct, StructureElement theElement, BaseRuntimeElementDefinition<?> def, LinkedList<String> path, BaseRuntimeDeclaredChildDefinition theChild) {
161 
162 		fillBasics(theElement, def, path, theChild);
163 
164 		String expectedPath = StringUtils.join(path, '.');
165 
166 		ourLog.info("Filling profile for: {} - Path: {}", expectedPath);
167 		String name = def.getName();
168 		if (!expectedPath.equals(name)) {
169 			path.pollLast();
170 			theElement.getDefinition().getNameReference().setValue(def.getName());
171 			return;
172 		}
173 
174 		fillExtensions(theStruct, path, def.getExtensionsNonModifier(), "extension", false);
175 		fillExtensions(theStruct, path, def.getExtensionsModifier(), "modifierExtension", true);
176 
177 		if (def.getChildType() == ChildTypeEnum.RESOURCE) {
178 			StructureElement narrative = theStruct.addElement();
179 			narrative.setName("text");
180 			narrative.setPath(join(path, '.') + ".text");
181 			narrative.getDefinition().addType().setCode(DataTypeEnum.NARRATIVE);
182 			narrative.getDefinition().setIsModifier(false);
183 			narrative.getDefinition().setMin(0);
184 			narrative.getDefinition().setMax("1");
185 
186 			StructureElement contained = theStruct.addElement();
187 			contained.setName("contained");
188 			contained.setPath(join(path, '.') + ".contained");
189 			contained.getDefinition().addType().getCode().setValue("Resource");
190 			contained.getDefinition().setIsModifier(false);
191 			contained.getDefinition().setMin(0);
192 			contained.getDefinition().setMax("1");
193 		}
194 
195 		if (def instanceof BaseRuntimeElementCompositeDefinition) {
196 			BaseRuntimeElementCompositeDefinition<?> cdef = ((BaseRuntimeElementCompositeDefinition<?>) def);
197 			for (BaseRuntimeChildDefinition nextChild : cdef.getChildren()) {
198 				if (nextChild instanceof RuntimeChildUndeclaredExtensionDefinition) {
199 					continue;
200 				}
201 
202 				BaseRuntimeDeclaredChildDefinition child = (BaseRuntimeDeclaredChildDefinition) nextChild;
203 				StructureElement elem = theStruct.addElement();
204 				fillMinAndMaxAndDefinitions(child, elem);
205 
206 				if (child instanceof RuntimeChildResourceBlockDefinition) {
207 					RuntimeResourceBlockDefinition nextDef = (RuntimeResourceBlockDefinition) child.getSingleChildOrThrow();
208 					fillProfile(theStruct, elem, nextDef, path, child);
209 				} else if (child instanceof RuntimeChildContainedResources) {
210 					// ignore
211 				} else if (child instanceof RuntimeChildDeclaredExtensionDefinition) {
212 					throw new IllegalStateException("Unexpected child type: " + child.getClass().getCanonicalName());
213 				} else if (child instanceof RuntimeChildCompositeDatatypeDefinition || child instanceof RuntimeChildPrimitiveDatatypeDefinition || child instanceof RuntimeChildChoiceDefinition
214 						|| child instanceof RuntimeChildResourceDefinition) {
215 					Iterator<String> childNamesIter = child.getValidChildNames().iterator();
216 					String nextName = childNamesIter.next();
217 					BaseRuntimeElementDefinition<?> nextDef = child.getChildByName(nextName);
218 					fillBasics(elem, nextDef, path, child);
219 					fillName(elem, nextDef);
220 					while (childNamesIter.hasNext()) {
221 						nextDef = child.getChildByName(childNamesIter.next());
222 						fillName(elem, nextDef);
223 					}
224 					path.pollLast();
225 				} else {
226 					throw new IllegalStateException("Unexpected child type: " + child.getClass().getCanonicalName());
227 				}
228 
229 			}
230 		} else {
231 			throw new IllegalStateException("Unexpected child type: " + def.getClass().getCanonicalName());
232 		}
233 
234 		path.pollLast();
235 	}

254 	private void scanForExtensions(Profile theProfile, BaseRuntimeElementDefinition<?> def) {
255 		BaseRuntimeElementCompositeDefinition<?> cdef = ((BaseRuntimeElementCompositeDefinition<?>) def);
256 
257 		for (RuntimeChildDeclaredExtensionDefinition nextChild : cdef.getExtensions()) {
258 			if (myExtensionDefToCode.containsKey(nextChild)) {
259 				continue;
260 			}
261 
262 			if (nextChild.isDefinedLocally() == false) {
263 				continue;
264 			}
265 
266 			ExtensionDefn defn = theProfile.addExtensionDefn();
267 			String code = null;
268 			if (nextChild.getExtensionUrl().contains("#") && !nextChild.getExtensionUrl().endsWith("#")) {
269 				code = nextChild.getExtensionUrl().substring(nextChild.getExtensionUrl().indexOf('#') + 1);
270 			} else {
271 				throw new ConfigurationException("Locally defined extension has no '#[code]' part in extension URL: " + nextChild.getExtensionUrl());
272 			}
273 
274 			defn.setCode(code);
275 			if (myExtensionDefToCode.values().contains(code)) {
276 				throw new IllegalStateException("Duplicate extension code: " + code);
277 			}
278 			myExtensionDefToCode.put(nextChild, code);
279 
280 			if (nextChild.getChildType() != null && IPrimitiveDatatype.class.isAssignableFrom(nextChild.getChildType())) {
281 				RuntimePrimitiveDatatypeDefinition pdef = (RuntimePrimitiveDatatypeDefinition) nextChild.getSingleChildOrThrow();
282 				defn.getDefinition().addType().setCode(DataTypeEnum.VALUESET_BINDER.fromCodeString(pdef.getName()));
283 			} else {
284 				RuntimeResourceBlockDefinition pdef = (RuntimeResourceBlockDefinition) nextChild.getSingleChildOrThrow();
285 				scanForExtensions(theProfile, pdef);
286 
287 				for (RuntimeChildDeclaredExtensionDefinition nextChildExt : pdef.getExtensions()) {
288 					StructureElementDefinitionType type = defn.getDefinition().addType();
289 					type.setCode(DataTypeEnum.EXTENSION);
290 					type.setProfile("#" + myExtensionDefToCode.get(nextChildExt));
291 				}
292 
293 			}
294 		}
295 	}

297 	@Override
298 	public void sealAndInitialize(Map<Class<? extends IElement>, BaseRuntimeElementDefinition<?>> theClassToElementDefinitions) {
299 		super.sealAndInitialize(theClassToElementDefinitions);
300 
301 		myNameToSearchParam = Collections.unmodifiableMap(myNameToSearchParam);
302 
303 		ArrayList<RuntimeSearchParam> searchParams = new ArrayList<RuntimeSearchParam>(myNameToSearchParam.values());
304 		Collections.sort(searchParams, new Comparator<RuntimeSearchParam>() {
305 			@Override
306 			public int compare(RuntimeSearchParam theArg0, RuntimeSearchParam theArg1) {
307 				return theArg0.getName().compareTo(theArg1.getName());
308 			}
309 		});
310 		mySearchParams = Collections.unmodifiableList(searchParams);
311 	}

313 	public synchronized Profile toProfile() {
314 		if (myProfileDef != null) {
315 			return myProfileDef;
316 		}
317 
318 		Profile retVal = new Profile();
319 		
320 		RuntimeResourceDefinition def = this;
321 
322 		if (StringUtils.isBlank(myId)) {
323 			myId = getName().toLowerCase();
324 		}
325 		
326 		retVal.setId(new IdDt(myId));
327 		
328 		// Scan for extensions
329 		scanForExtensions(retVal, def);
330 		Collections.sort(retVal.getExtensionDefn(), new Comparator<ExtensionDefn>() {
331 			@Override
332 			public int compare(ExtensionDefn theO1, ExtensionDefn theO2) {
333 				return theO1.getCode().compareTo(theO2.getCode());
334 			}
335 		});
336 
337 		// Scan for children
338 		retVal.setName(getName());
339 		Structure struct = retVal.addStructure();
340 		LinkedList<String> path = new LinkedList<String>();
341 
342 		StructureElement element = struct.addElement();
343 		element.getDefinition().setMin(1);
344 		element.getDefinition().setMax("1");
345 
346 		fillProfile(struct, element, def, path, null);
347 
348 		retVal.getStructure().get(0).getElement().get(0).getDefinition().addType().getCode().setValue("Resource");
349 
350 		myProfileDef = retVal;
351 
352 		return retVal;
353 	}
