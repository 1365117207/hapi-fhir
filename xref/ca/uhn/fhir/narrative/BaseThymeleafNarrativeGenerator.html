83  	@Override
84  	public NarrativeDt generateNarrative(String theProfile, IResource theResource) {
85  		if (!myInitialized) {
86  			initialize();
87  		}
88  
89  		String name = null;
90  		if (StringUtils.isNotBlank(theProfile)) {
91  			name = myProfileToNarrativeName.get(theProfile);
92  		}
93  		if (name == null) {
94  			name = myClassToNarrativeName.get(theResource.getClass());
95  		}
96  
97  		if (name == null) {
98  			if (myIgnoreMissingTemplates) {
99  				ourLog.debug("No narrative template available for profile: {}", theProfile);
100 				return new NarrativeDt(new XhtmlDt("<div>No narrative template available for resource profile: " + theProfile + "</div>"), NarrativeStatusEnum.EMPTY);
101 			} else {
102 				throw new DataFormatException("No narrative template for class " + theResource.getClass().getCanonicalName());
103 			}
104 		}
105 
106 		try {
107 			Context context = new Context();
108 			context.setVariable("resource", theResource);
109 			
110 			String result = myProfileTemplateEngine.process(name, context);
111 
112 			if (myCleanWhitespace) {
113 				ourLog.trace("Pre-whitespace cleaning: ", result);
114 				result = cleanWhitespace(result);
115 				ourLog.trace("Post-whitespace cleaning: ", result);
116 			}
117 
118 			XhtmlDt div = new XhtmlDt(result);
119 			return new NarrativeDt(div, NarrativeStatusEnum.GENERATED);
120 		} catch (Exception e) {
121 			if (myIgnoreFailures) {
122 				ourLog.error("Failed to generate narrative", e);
123 				return new NarrativeDt(new XhtmlDt("<div>No narrative available - Error: " + e.getMessage() + "</div>"), NarrativeStatusEnum.EMPTY);
124 			} else {
125 				throw new DataFormatException(e);
126 			}
127 		}
128 	}

130 	private synchronized void initialize() {
131 		if (myInitialized) {
132 			return;
133 		}
134 		myProfileToNarrativeName = new HashMap<String, String>();
135 		myClassToNarrativeName = new HashMap<Class<?>, String>();
136 		myNameToNarrativeTemplate = new HashMap<String, String>();
137 
138 		List<String> propFileName = getPropertyFile();
139 
140 		try {
141 			if (myApplyDefaultDatatypeTemplates) {
142 				loadProperties(DefaultThymeleafNarrativeGenerator.NARRATIVES_PROPERTIES);
143 			}
144 			for (String next : propFileName) {
145 				loadProperties(next);
146 			}
147 		} catch (IOException e) {
148 			throw new ConfigurationException("Can not load property file " + propFileName, e);
149 		}
150 
151 		{
152 			myProfileTemplateEngine = new TemplateEngine();
153 			TemplateResolver resolver = new TemplateResolver();
154 			resolver.setResourceResolver(new ProfileResourceResolver());
155 			myProfileTemplateEngine.setTemplateResolver(resolver);
156 			StandardDialect dialect = new StandardDialect();
157 			HashSet<IProcessor> additionalProcessors = new HashSet<IProcessor>();
158 			additionalProcessors.add(new NarrativeAttributeProcessor());
159 			dialect.setAdditionalProcessors(additionalProcessors);
160 			myProfileTemplateEngine.setDialect(dialect);
161 			myProfileTemplateEngine.initialize();
162 		}
163 
164 		myInitialized = true;
165 	}

231 	private void loadProperties(String propFileName) throws IOException {
232 		ourLog.debug("Loading narrative properties file: {}", propFileName);
233 		
234 		Properties file = new Properties();
235 
236 		InputStream resource = loadResource(propFileName);
237 		file.load(resource);
238 		for (Object nextKeyObj : file.keySet()) {
239 			String nextKey = (String) nextKeyObj;
240 			if (nextKey.endsWith(".profile")) {
241 				String name = nextKey.substring(0, nextKey.indexOf(".profile"));
242 				if (isBlank(name)) {
243 					continue;
244 				}
245 
246 				String narrativePropName = name + ".narrative";
247 				String narrativeName = file.getProperty(narrativePropName);
248 				if (isBlank(narrativeName)) {
249 					throw new ConfigurationException("Found property '" + nextKey + "' but no corresponding property '" + narrativePropName + "' in file " + propFileName);
250 				}
251 
252 				String narrative = IOUtils.toString(loadResource(narrativeName));
253 				myProfileToNarrativeName.put(file.getProperty(nextKey), name);
254 				myNameToNarrativeTemplate.put(name, narrative);
255 
256 			} else if (nextKey.endsWith(".class")) {
257 
258 				String name = nextKey.substring(0, nextKey.indexOf(".class"));
259 				if (isBlank(name)) {
260 					continue;
261 				}
262 
263 				String className = file.getProperty(nextKey);
264 
265 				Class<?> dtClass;
266 				try {
267 					dtClass = Class.forName(className);
268 				} catch (ClassNotFoundException e) {
269 					ourLog.warn("Unknown datatype class '{}' identified in narrative file {}", name, propFileName);
270 					continue;
271 				}
272 
273 				String narrativePropName = name + ".narrative";
274 				String narrativeName = file.getProperty(narrativePropName);
275 				if (isBlank(narrativeName)) {
276 					throw new ConfigurationException("Found property '" + nextKey + "' but no corresponding property '" + narrativePropName + "' in file " + propFileName);
277 				}
278 
279 				String narrative = IOUtils.toString(loadResource(narrativeName));
280 				myClassToNarrativeName.put(dtClass, name);
281 				myNameToNarrativeTemplate.put(name, narrative);
282 
283 			} else if (nextKey.endsWith(".narrative")) {
284 				continue;
285 			} else {
286 				throw new ConfigurationException("Invalid property name: " + nextKey);
287 			}
288 
289 		}
290 	}

292 	private InputStream loadResource(String name) throws IOException {
293 		if (name.startsWith("classpath:")) {
294 			String cpName = name.substring("classpath:".length());
295 			InputStream resource = DefaultThymeleafNarrativeGenerator.class.getResourceAsStream(cpName);
296 			if (resource == null) {
297 				resource = DefaultThymeleafNarrativeGenerator.class.getResourceAsStream("/" + cpName);
298 				if (resource == null) {
299 					throw new IOException("Can not find '" + cpName + "' on classpath");
300 				}
301 			}
302 			return resource;
303 		} else if (name.startsWith("file:")) {
304 			File file = new File(name.substring("file:".length()));
305 			if (file.exists() == false) {
306 				throw new IOException("File not found: " + file.getAbsolutePath());
307 			}
308 			return new FileInputStream(file);
309 		} else {
310 			throw new IOException("Invalid resource name: '" + name + "' (must start with classpath: or file: )");
311 		}
312 	}

314 	static String cleanWhitespace(String theResult) {
315 		StringBuilder b = new StringBuilder();
316 		boolean inWhitespace = false;
317 		boolean betweenTags = false;
318 		boolean lastNonWhitespaceCharWasTagEnd = false;
319 		for (int i = 0; i < theResult.length(); i++) {
320 			char nextChar = theResult.charAt(i);
321 			if (nextChar == '>') {
322 				b.append(nextChar);
323 				betweenTags = true;
324 				lastNonWhitespaceCharWasTagEnd = true;
325 				continue;
326 			} else if (nextChar == '\n' || nextChar == '\r') {
327 				// if (inWhitespace) {
328 				// b.append(' ');
329 				// inWhitespace = false;
330 				// }
331 				continue;
332 			}
333 
334 			if (betweenTags) {
335 				if (Character.isWhitespace(nextChar)) {
336 					inWhitespace = true;
337 				} else if (nextChar == '<') {
338 					if (inWhitespace && !lastNonWhitespaceCharWasTagEnd) {
339 						b.append(' ');
340 					}
341 					inWhitespace = false;
342 					b.append(nextChar);
343 					inWhitespace = false;
344 					betweenTags = false;
345 					lastNonWhitespaceCharWasTagEnd = false;
346 				} else {
347 					lastNonWhitespaceCharWasTagEnd = false;
348 					if (inWhitespace) {
349 						b.append(' ');
350 						inWhitespace = false;
351 					}
352 					b.append(nextChar);
353 				}
354 			} else {
355 				b.append(nextChar);
356 			}
357 		}
358 		return b.toString();
359 	}

361 	public class NarrativeAttributeProcessor extends AbstractAttrProcessor {
362 
363 		protected NarrativeAttributeProcessor() {
364 			super("narrative");
365 		}
366 
367 		@Override
368 		public int getPrecedence() {
369 			return 0;
370 		}
371 
372 		@Override
373 		protected ProcessorResult processAttribute(Arguments theArguments, Element theElement, String theAttributeName) {
374 			final String attributeValue = theElement.getAttributeValue(theAttributeName);
375 
376 			final Configuration configuration = theArguments.getConfiguration();
377 			final IStandardExpressionParser expressionParser = StandardExpressions.getExpressionParser(configuration);
378 
379 			final IStandardExpression expression = expressionParser.parseExpression(configuration, theArguments, attributeValue);
380 			final Object value = expression.execute(configuration, theArguments);
381 
382 
383 			theElement.removeAttribute(theAttributeName);
384 			theElement.clearChildren();
385 
386 			Context context = new Context();
387 			context.setVariable("resource", value);
388 			
389 			String name = myClassToNarrativeName.get(value.getClass());
390 			if (name == null) {
391 				if (myIgnoreMissingTemplates) {
392 					ourLog.debug("No narrative template available for type: {}", value.getClass());
393 					return ProcessorResult.ok();
394 				} else {
395 					throw new DataFormatException("No narrative template for class " + value.getClass());
396 				}
397 			}
398 
399 			String result = myProfileTemplateEngine.process(name, context);
400 			Document dom = DOMUtils.getXhtmlDOMFor(new StringReader(result));
401 
402 			Element firstChild = (Element) dom.getFirstChild();
403 			for (Node next : firstChild.getChildren()) {
404 				theElement.addChild(next);
405 			}
406 
407 			return ProcessorResult.ok();
408 		}
409 
410 	}

425 	private final class ProfileResourceResolver implements IResourceResolver {
426 		@Override
427 		public String getName() {
428 			return getClass().getCanonicalName();
429 		}
430 
431 		@Override
432 		public InputStream getResourceAsStream(TemplateProcessingParameters theTemplateProcessingParameters, String theName) {
433 			String template = myNameToNarrativeTemplate.get(theName);
434 			if (template == null) {
435 				ourLog.info("No narative template for resource profile: {}", theName);
436 				return new ReaderInputStream(new StringReader(""));
437 			}
438 			return new ReaderInputStream(new StringReader(template));
439 		}
