23  import java.io.IOException;
24  import java.io.Reader;
25  import java.io.Writer;
26  
27  import ca.uhn.fhir.context.ConfigurationException;
28  import ca.uhn.fhir.model.api.Bundle;
29  import ca.uhn.fhir.model.api.IResource;
30  import ca.uhn.fhir.model.api.TagList;
31  
32  public interface IParser {
33  
34  	String encodeBundleToString(Bundle theBundle) throws DataFormatException;
35  
36  	void encodeBundleToWriter(Bundle theBundle, Writer theWriter) throws IOException, DataFormatException;
37  
38  	String encodeResourceToString(IResource theResource) throws DataFormatException;
39  
40  	void encodeResourceToWriter(IResource theResource, Writer theWriter) throws IOException, DataFormatException;
41  
42  	/**
43  	 * Encodes a tag list, as defined in the <a
44  	 * href="http://hl7.org/implement/standards/fhir/http.html#tags">FHIR
45  	 * Specification</a>.
46  	 * 
47  	 * @param theTagList The tag list to encode. Must not be null.
48  	 * @return An encoded tag list
49  	 */
50  	String encodeTagListToString(TagList theTagList);
51  
52  	/**
53  	 * Encodes a tag list, as defined in the <a
54  	 * href="http://hl7.org/implement/standards/fhir/http.html#tags">FHIR
55  	 * Specification</a>.
56  	 * 
57  	 * @param theTagList The tag list to encode. Must not be null.
58  	 * @param theWriter The writer to encode to
59  	 */
60  	void encodeTagListToWriter(TagList theTagList, Writer theWriter) throws IOException;
61  
62  	
63  	<T extends IResource> Bundle parseBundle(Class<T> theResourceType, Reader theReader);
64  
65  	Bundle parseBundle(Reader theReader);
66  
67  	Bundle parseBundle(String theMessageString) throws ConfigurationException, DataFormatException;
68  
69  	/**
70  	 * Parses a resource
71  	 * 
72  	 * @param theResourceType
73  	 *            The resource type to use. This can be used to explicitly
74  	 *            specify a class which extends a built-in type (e.g. a custom
75  	 *            type extending the default Patient class)
76  	 * @param theReader
77  	 *            The reader to parse inpou from
78  	 * @return A parsed resource
79  	 * @throws DataFormatException
80  	 *             If the resource can not be parsed because the data is not
81  	 *             recognized or invalid for any reason
82  	 */
83  	<T extends IResource> T parseResource(Class<T> theResourceType, Reader theReader) throws DataFormatException;
84  
85  	/**
86  	 * Parses a resource
87  	 * 
88  	 * @param theResourceType
89  	 *            The resource type to use. This can be used to explicitly
90  	 *            specify a class which extends a built-in type (e.g. a custom
91  	 *            type extending the default Patient class)
92  	 * @param theString
93  	 *            The string to parse
94  	 * @return A parsed resource
95  	 * @throws DataFormatException
96  	 *             If the resource can not be parsed because the data is not
97  	 *             recognized or invalid for any reason
98  	 */
99  	<T extends IResource> T parseResource(Class<T> theResourceType, String theString) throws DataFormatException;
100 
101 	IResource parseResource(Reader theReader) throws ConfigurationException, DataFormatException;
102 
103 	IResource parseResource(String theMessageString) throws ConfigurationException, DataFormatException;
104 
105 	/**
106 	 * Parses a tag list, as defined in the <a
107 	 * href="http://hl7.org/implement/standards/fhir/http.html#tags">FHIR
108 	 * Specification</a>.
109 	 * 
110 	 * @param theReader
111 	 *            A reader which will supply a tag list
112 	 * @return A parsed tag list
113 	 */
114 	TagList parseTagList(Reader theReader);
115 
116 	/**
117 	 * Parses a tag list, as defined in the <a
118 	 * href="http://hl7.org/implement/standards/fhir/http.html#tags">FHIR
119 	 * Specification</a>.
120 	 * 
121 	 * @param theString
122 	 *            A string containing a tag list
123 	 * @return A parsed tag list
124 	 */
125 	TagList parseTagList(String theString);
126 
127 	/**
128 	 * Sets the "pretty print" flag, meaning that the parser will encode
129 	 * resources with human-readable spacing and newlines between elements
130 	 * instead of condensing output as much as possible.
131 	 * 
132 	 * @param thePrettyPrint
133 	 *            The flag
134 	 * @return Returns an instance of <code>this</code> parser so that method
135 	 *         calls can be conveniently chained
136 	 */
137 	IParser setPrettyPrint(boolean thePrettyPrint);
138 
139 	/**
140 	 * If set to <code>true</code> (default is <code>false</code>), narratives
141 	 * will not be included in the encoded values.
142 	 */
143 	IParser setSuppressNarratives(boolean theSuppressNarratives);
144 
145 }
