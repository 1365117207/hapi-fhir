110 	private void addToHeldExtensions(int valueIdx, List<ExtensionDt> ext, ArrayList<ArrayList<HeldExtension>> list) {
111 		if (ext.size() > 0) {
112 			list.ensureCapacity(valueIdx);
113 			while (list.size() <= valueIdx) {
114 				list.add(null);
115 			}
116 			if (list.get(valueIdx) == null) {
117 				list.set(valueIdx, new ArrayList<JsonParser.HeldExtension>());
118 			}
119 			for (ExtensionDt next : ext) {
120 				list.get(valueIdx).add(new HeldExtension(next));
121 			}
122 		}
123 	}

125 	private void assertObjectOfType(JsonValue theResourceTypeObj, ValueType theValueType, String thePosition) {
126 		if (theResourceTypeObj == null) {
127 			throw new DataFormatException("Invalid JSON content detected, missing required element: '" + thePosition + "'");
128 		}
129 		
130 		if (theResourceTypeObj.getValueType() != theValueType) {
131 			throw new DataFormatException("Invalid content of element " + thePosition + ", expected " + theValueType);
132 		}
133 	}

135 	private JsonGenerator createJsonGenerator(Writer theWriter) {
136 		Map<String, Object> properties = new HashMap<String, Object>(1);
137 		if (myPrettyPrint) {
138 			properties.put(JsonGenerator.PRETTY_PRINTING, myPrettyPrint);
139 		}
140 		JsonGeneratorFactory jgf = Json.createGeneratorFactory(properties);
141 		JsonGenerator eventWriter = jgf.createGenerator(theWriter);
142 		return eventWriter;

146 	public void encodeBundleToWriter(Bundle theBundle, Writer theWriter) throws IOException {
147 		JsonGenerator eventWriter = createJsonGenerator(theWriter);
148 		eventWriter.writeStartObject();
149 
150 		eventWriter.write("resourceType", "Bundle");
151 
152 		writeTagWithTextNode(eventWriter, "title", theBundle.getTitle());
153 		writeTagWithTextNode(eventWriter, "id", theBundle.getBundleId());
154 		writeOptionalTagWithTextNode(eventWriter, "updated", theBundle.getUpdated());
155 		writeOptionalTagWithTextNode(eventWriter, "published", theBundle.getPublished());
156 
157 		boolean linkStarted = false;
158 		linkStarted = writeAtomLink(eventWriter, "self", theBundle.getLinkSelf(), linkStarted);
159 		linkStarted = writeAtomLink(eventWriter, "first", theBundle.getLinkFirst(), linkStarted);
160 		linkStarted = writeAtomLink(eventWriter, "previous", theBundle.getLinkPrevious(), linkStarted);
161 		linkStarted = writeAtomLink(eventWriter, "next", theBundle.getLinkNext(), linkStarted);
162 		linkStarted = writeAtomLink(eventWriter, "last", theBundle.getLinkLast(), linkStarted);
163 		linkStarted = writeAtomLink(eventWriter, "fhir-base", theBundle.getLinkBase(), linkStarted);
164 		if (linkStarted) {
165 			eventWriter.writeEnd();
166 		}
167 		
168 		writeCategories(eventWriter, theBundle.getCategories());
169 
170 		writeOptionalTagWithTextNode(eventWriter, "totalResults", theBundle.getTotalResults());
171 
172 		writeAuthor(theBundle, eventWriter);
173 
174 		eventWriter.writeStartArray("entry");
175 		for (BundleEntry nextEntry : theBundle.getEntries()) {
176 			eventWriter.writeStartObject();
177 
178 			boolean deleted = nextEntry.getDeletedAt() != null && nextEntry.getDeletedAt().isEmpty() == false;
179 			if (deleted) {
180 				writeTagWithTextNode(eventWriter, "deleted", nextEntry.getDeletedAt());
181 			}
182 			writeTagWithTextNode(eventWriter, "title", nextEntry.getTitle());
183 			writeTagWithTextNode(eventWriter, "id", nextEntry.getId());
184 
185 			linkStarted = false;
186 			linkStarted = writeAtomLink(eventWriter, "self", nextEntry.getLinkSelf(), linkStarted);
187 			linkStarted = writeAtomLink(eventWriter, "alternate", nextEntry.getLinkAlternate(), linkStarted);
188 			linkStarted = writeAtomLink(eventWriter, "search", nextEntry.getLinkSearch(), linkStarted);
189 			if (linkStarted) {
190 				eventWriter.writeEnd();
191 			}
192 
193 			writeOptionalTagWithTextNode(eventWriter, "updated", nextEntry.getUpdated());
194 			writeOptionalTagWithTextNode(eventWriter, "published", nextEntry.getPublished());
195 
196 			writeCategories(eventWriter, nextEntry.getCategories());
197 
198 			writeAuthor(nextEntry, eventWriter);
199 
200 			IResource resource = nextEntry.getResource();
201 			if (resource != null && !resource.isEmpty() && !deleted) {
202 				RuntimeResourceDefinition resDef = myContext.getResourceDefinition(resource);
203 				encodeResourceToJsonStreamWriter(resDef, resource, eventWriter, "content", false);
204 			}
205 
206 			eventWriter.writeEnd(); // entry object
207 		}
208 		eventWriter.writeEnd(); // entry array
209 
210 		eventWriter.writeEnd();
211 		eventWriter.flush();
212 	}

214 	private void writeCategories(JsonGenerator eventWriter, TagList categories) {
215 		if (categories != null && categories.size() > 0) {
216 			eventWriter.writeStartArray("category");
217 			for (Tag next : categories) {
218 				eventWriter.writeStartObject();
219 				eventWriter.write("term", defaultString(next.getTerm()));
220 				eventWriter.write("label", defaultString(next.getLabel()));
221 				eventWriter.write("scheme", defaultString(next.getScheme()));
222 				eventWriter.writeEnd();
223 			}
224 			eventWriter.writeEnd();
225 		}

228 	private void encodeChildElementToStreamWriter(RuntimeResourceDefinition theResDef, IResource theResource, JsonGenerator theWriter, IElement theValue, BaseRuntimeElementDefinition<?> theChildDef,
229 			String theChildName) throws IOException {
230 
231 		switch (theChildDef.getChildType()) {
232 		case PRIMITIVE_DATATYPE: {
233 			IPrimitiveDatatype<?> value = (IPrimitiveDatatype<?>) theValue;
234 			if (isBlank(value.getValueAsString())) {
235 				break;
236 			}
237 			
238 			if (value instanceof IntegerDt) {
239 				if (theChildName != null) {
240 					theWriter.write(theChildName, ((IntegerDt) value).getValue());
241 				} else {
242 					theWriter.write(((IntegerDt) value).getValue());
243 				}
244 			} else if (value instanceof DecimalDt) {
245 				if (theChildName != null) {
246 					theWriter.write(theChildName, ((DecimalDt) value).getValue());
247 				} else {
248 					theWriter.write(((DecimalDt) value).getValue());
249 				}
250 			} else if (value instanceof BooleanDt) {
251 				if (theChildName != null) {
252 					theWriter.write(theChildName, ((BooleanDt) value).getValue());
253 				} else {
254 					theWriter.write(((BooleanDt) value).getValue());
255 				}
256 			} else {
257 				String valueStr = value.getValueAsString();
258 				if (theChildName != null) {
259 					theWriter.write(theChildName, valueStr);
260 				} else {
261 					theWriter.write(valueStr);
262 				}
263 			}
264 			break;
265 		}
266 		case RESOURCE_BLOCK:
267 		case COMPOSITE_DATATYPE: {
268 			BaseRuntimeElementCompositeDefinition<?> childCompositeDef = (BaseRuntimeElementCompositeDefinition<?>) theChildDef;
269 			if (theChildName != null) {
270 				theWriter.writeStartObject(theChildName);
271 			} else {
272 				theWriter.writeStartObject();
273 			}
274 			if (theValue instanceof ExtensionDt) {
275 				theWriter.write("url", ((ExtensionDt) theValue).getUrlAsString());
276 			}
277 			encodeCompositeElementToStreamWriter(theResDef, theResource, theValue, theWriter, childCompositeDef);
278 			theWriter.writeEnd();
279 			break;
280 		}
281 		case RESOURCE_REF: {
282 			ResourceReferenceDt referenceDt = (ResourceReferenceDt) theValue;
283 			IdDt value = referenceDt.getReference();
284 			if (theChildName != null) {
285 				theWriter.writeStartObject(theChildName);
286 			} else {
287 				theWriter.writeStartObject();
288 			}
289 
290 			String reference = value.getValue();
291 			if (StringUtils.isBlank(reference)) {
292 				if (value.getResourceType() != null && StringUtils.isNotBlank(value.getIdPart())) {
293 					reference = myContext.getResourceDefinition(value.getResourceType()).getName() + '/' + value.getIdPart();
294 				}
295 			}
296 
297 			if (StringUtils.isNotBlank(reference)) {
298 				theWriter.write(XmlParser.RESREF_REFERENCE, reference);
299 			}
300 			if (referenceDt.getDisplay().isEmpty() == false) {
301 				theWriter.write(XmlParser.RESREF_DISPLAY, referenceDt.getDisplay().getValueAsString());
302 			}
303 			theWriter.writeEnd();
304 			break;
305 		}
306 		case CONTAINED_RESOURCES: {
307 			theWriter.writeStartArray(theChildName);
308 			ContainedDt value = (ContainedDt) theValue;
309 			for (IResource next : value.getContainedResources()) {
310 				encodeResourceToJsonStreamWriter(theResDef, next, theWriter, null, true);
311 			}
312 			theWriter.writeEnd();
313 			break;
314 		}
315 		case PRIMITIVE_XHTML: {
316 			if (!getSuppressNarratives()) {
317 				XhtmlDt dt = (XhtmlDt) theValue;
318 				if (theChildName != null) {
319 					theWriter.write(theChildName, dt.getValueAsString());
320 				} else {
321 					theWriter.write(dt.getValueAsString());
322 				}
323 			} else {
324 				if (theChildName != null) {
325 					// do nothing
326 				} else {
327 					theWriter.writeNull();
328 				}
329 			}
330 			break;
331 		}
332 		case UNDECL_EXT:
333 		default:
334 			throw new IllegalStateException("Should not have this state here: " + theChildDef.getChildType().name());
335 		}
336 
337 	}

339 	private void encodeCompositeElementChildrenToStreamWriter(RuntimeResourceDefinition theResDef, IResource theResource, IElement theElement, JsonGenerator theEventWriter,
340 			List<? extends BaseRuntimeChildDefinition> theChildren) throws IOException {
341 		for (BaseRuntimeChildDefinition nextChild : theChildren) {
342 			if (nextChild instanceof RuntimeChildNarrativeDefinition) {
343 				INarrativeGenerator gen = myContext.getNarrativeGenerator();
344 				if (gen != null) {
345 					NarrativeDt narr = gen.generateNarrative(theResDef.getResourceProfile(), theResource);
346 					if (narr != null) {
347 						RuntimeChildNarrativeDefinition child = (RuntimeChildNarrativeDefinition) nextChild;
348 						String childName = nextChild.getChildNameByDatatype(child.getDatatype());
349 						BaseRuntimeElementDefinition<?> type = child.getChildByName(childName);
350 						encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, narr, type, childName);
351 						continue;
352 					}
353 				}
354 			}
355 
356 			List<? extends IElement> values = nextChild.getAccessor().getValues(theElement);
357 			if (values == null || values.isEmpty()) {
358 				continue;
359 			}
360 
361 			String currentChildName = null;
362 			boolean inArray = false;
363 
364 			ArrayList<ArrayList<HeldExtension>> extensions = new ArrayList<ArrayList<HeldExtension>>(0);
365 			ArrayList<ArrayList<HeldExtension>> modifierExtensions = new ArrayList<ArrayList<HeldExtension>>(0);
366 
367 			int valueIdx = 0;
368 			for (IElement nextValue : values) {
369 				if (nextValue == null || nextValue.isEmpty()) {
370 					continue;
371 				}
372 
373 				Class<? extends IElement> type = nextValue.getClass();
374 				String childName = nextChild.getChildNameByDatatype(type);
375 				BaseRuntimeElementDefinition<?> childDef = nextChild.getChildElementDefinitionByDatatype(type);
376 				if (childDef == null) {
377 					super.throwExceptionForUnknownChildType(nextChild, type);
378 				}
379 				boolean primitive = childDef.getChildType() == ChildTypeEnum.PRIMITIVE_DATATYPE;
380 				
381 				if (nextChild instanceof RuntimeChildDeclaredExtensionDefinition) {
382 					// Don't encode extensions
383 //					RuntimeChildDeclaredExtensionDefinition extDef = (RuntimeChildDeclaredExtensionDefinition) nextChild;
384 //					if (extDef.isModifier()) {
385 //						addToHeldExtensions(valueIdx, modifierExtensions, extDef, nextValue);
386 //					} else {
387 //						addToHeldExtensions(valueIdx, extensions, extDef, nextValue);
388 //					}
389 				} else {
390 
391 					if (currentChildName == null || !currentChildName.equals(childName)) {
392 						if (inArray) {
393 							theEventWriter.writeEnd();
394 						}
395 						if (nextChild.getMax() > 1 || nextChild.getMax() == Child.MAX_UNLIMITED) {
396 							theEventWriter.writeStartArray(childName);
397 							inArray = true;
398 							encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, nextValue, childDef, null);
399 						} else {
400 							encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, nextValue, childDef, childName);
401 						}
402 						currentChildName = childName;
403 					} else {
404 						encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, nextValue, childDef, null);
405 					}
406 
407 					if (nextValue instanceof ISupportsUndeclaredExtensions && primitive) {
408 						List<ExtensionDt> ext = ((ISupportsUndeclaredExtensions) nextValue).getUndeclaredExtensions();
409 						addToHeldExtensions(valueIdx, ext, extensions);
410 
411 						ext = ((ISupportsUndeclaredExtensions) nextValue).getUndeclaredModifierExtensions();
412 						addToHeldExtensions(valueIdx, ext, modifierExtensions);
413 					}
414 
415 				}
416 
417 				valueIdx++;
418 			}
419 
420 			if (inArray) {
421 				theEventWriter.writeEnd();
422 			}
423 
424 			if (extensions.size() > 0 || modifierExtensions.size() > 0) {
425 				theEventWriter.writeStartArray('_' + currentChildName);
426 
427 				for (int i = 0; i < valueIdx; i++) {
428 					boolean haveContent = false;
429 					if (extensions.size() > i && extensions.get(i) != null && extensions.get(i).isEmpty() == false) {
430 						haveContent = true;
431 						theEventWriter.writeStartObject();
432 						theEventWriter.writeStartArray("extension");
433 						for (HeldExtension nextExt : extensions.get(i)) {
434 							nextExt.write(theResDef, theResource, theEventWriter);
435 						}
436 						theEventWriter.writeEnd();
437 						theEventWriter.writeEnd();
438 					}
439 
440 					if (!haveContent) {
441 						// theEventWriter.writeEnd();
442 						theEventWriter.writeNull();
443 					}
444 				}
445 
446 				theEventWriter.writeEnd();
447 			}
448 		}
449 	}

458 	private void encodeResourceToJsonStreamWriter(RuntimeResourceDefinition theResDef, IResource theResource, JsonGenerator theEventWriter, String theObjectNameOrNull,
459 			boolean theIsSubElementWithinResource) throws IOException {
460 		if (!theIsSubElementWithinResource) {
461 			super.containResourcesForEncoding(theResource);
462 		}
463 
464 		RuntimeResourceDefinition resDef = myContext.getResourceDefinition(theResource);
465 
466 		if (theObjectNameOrNull == null) {
467 			theEventWriter.writeStartObject();
468 		} else {
469 			theEventWriter.writeStartObject(theObjectNameOrNull);
470 		}
471 
472 		theEventWriter.write("resourceType", resDef.getName());
473 		if (theIsSubElementWithinResource && theResource.getId() != null && isNotBlank(theResource.getId().getValue())) {
474 			theEventWriter.write("id", theResource.getId().getValue());
475 		}
476 
477 		if (theResource instanceof Binary) {
478 			Binary bin = (Binary) theResource;
479 			theEventWriter.write("contentType", bin.getContentType());
480 			theEventWriter.write("content", bin.getContentAsBase64());
481 		} else {
482 			encodeCompositeElementToStreamWriter(theResDef, theResource, theResource, theEventWriter, resDef);
483 		}
484 		theEventWriter.writeEnd();
485 	}

500 		JsonGenerator eventWriter = createJsonGenerator(theWriter);
501 
502 		eventWriter.writeStartObject();
503 
504 		eventWriter.write("resourceType", TagList.ELEMENT_NAME);
505 
506 		eventWriter.writeStartArray(TagList.ATTR_CATEGORY);
507 		for (Tag next : theTagList) {
508 			eventWriter.writeStartObject();
509 
510 			if (isNotBlank(next.getTerm())) {
511 				eventWriter.write(Tag.ATTR_TERM, next.getTerm());
512 			}
513 			if (isNotBlank(next.getLabel())) {
514 				eventWriter.write(Tag.ATTR_LABEL, next.getLabel());
515 			}
516 			if (isNotBlank(next.getScheme())) {
517 				eventWriter.write(Tag.ATTR_SCHEME, next.getScheme());
518 			}
519 
520 			eventWriter.writeEnd();
521 		}
522 		eventWriter.writeEnd();
523 
524 		eventWriter.writeEnd();
525 		eventWriter.flush();
526 	}

529 	 * This is useful only for the two cases where extensions are encoded as direct children (e.g. not in some object called _name): resource extensions, and extension extensions

531 	private void extractAndWriteExtensionsAsDirectChild(IElement theElement, JsonGenerator theEventWriter, BaseRuntimeElementDefinition<?> theElementDef, RuntimeResourceDefinition theResDef,
532 			IResource theResource) throws IOException {
533 		List<HeldExtension> extensions = new ArrayList<HeldExtension>(0);
534 		List<HeldExtension> modifierExtensions = new ArrayList<HeldExtension>(0);
535 
536 		// Undeclared extensions
537 		extractUndeclaredExtensions(theElement, extensions, modifierExtensions);
538 
539 		// Declared extensions
540 		extractDeclaredExtensions(theElement, theElementDef, extensions, modifierExtensions);
541 
542 		// Write the extensions
543 		writeExtensionsAsDirectChild(theResource, theEventWriter, theResDef, extensions, modifierExtensions);
544 	}

546 	private void extractDeclaredExtensions(IElement theResource, BaseRuntimeElementDefinition<?> resDef, List<HeldExtension> extensions, List<HeldExtension> modifierExtensions) {
547 		for (RuntimeChildDeclaredExtensionDefinition nextDef : resDef.getExtensionsNonModifier()) {
548 			for (IElement nextValue : nextDef.getAccessor().getValues(theResource)) {
549 				if (nextValue != null) {
550 					if (nextValue == null || nextValue.isEmpty()) {
551 						continue;
552 					}
553 					extensions.add(new HeldExtension(nextDef, nextValue));
554 				}
555 			}
556 		}
557 		for (RuntimeChildDeclaredExtensionDefinition nextDef : resDef.getExtensionsModifier()) {
558 			for (IElement nextValue : nextDef.getAccessor().getValues(theResource)) {
559 				if (nextValue != null) {
560 					if (nextValue == null || nextValue.isEmpty()) {
561 						continue;
562 					}
563 					modifierExtensions.add(new HeldExtension(nextDef, nextValue));
564 				}
565 			}
566 		}

570 		if (theResource instanceof ISupportsUndeclaredExtensions) {

577 			}

579 			ext = ((ISupportsUndeclaredExtensions) theResource).getUndeclaredModifierExtensions();

581 				if (next == null || next.isEmpty()) {
582 					continue;
583 				}
584 				modifierExtensions.add(new HeldExtension(next));
585 			}
586 		}

589 	private void parseAlternates(JsonValue theAlternateVal, ParserState<?> theState) {
590 		if (theAlternateVal == null || theAlternateVal.getValueType() == ValueType.NULL) {
591 			return;
592 		}
593 		JsonObject alternate = (JsonObject) theAlternateVal;
594 		for (Entry<String, JsonValue> nextEntry : alternate.entrySet()) {
595 			String nextKey = nextEntry.getKey();
596 			JsonValue nextVal = nextEntry.getValue();
597 			if ("extension".equals(nextKey)) {
598 				boolean isModifier = false;
599 				JsonArray array = (JsonArray) nextEntry.getValue();
600 				parseExtension(theState, array, isModifier);
601 			} else if ("modifierExtension".equals(nextKey)) {
602 				boolean isModifier = true;
603 				JsonArray array = (JsonArray) nextEntry.getValue();
604 				parseExtension(theState, array, isModifier);
605 			} else if ("id".equals(nextKey)) {
606 				switch (nextVal.getValueType()) {
607 				case STRING:
608 					theState.attributeValue("id", ((JsonString) nextVal).getString());
609 					break;
610 				case NULL:
611 					break;
612 				default:
613 					break;
614 				}
615 			}
616 		}
617 	}

621 		JsonReader reader;		
622 		JsonObject object;
623

625 			reader = Json.createReader(theReader);

629 				throw new DataFormatException("Failed to parse JSON encoded FHIR content: " + e.getMessage() + " - This may indicate that single quotes are being used as JSON escapes where double quotes are required", e);

636 		if (!"Bundle".equals(resourceType)) {

638 		}

641 		state.enteringNewElement(null, "feed");
642 
643 		parseBundleChildren(object, state);
644 
645 		state.endingElement();
646 
647 		Bundle retVal = state.getObject();
648

652 	private void parseBundleChildren(JsonObject theObject, ParserState<?> theState) {
653 		for (String nextName : theObject.keySet()) {
654 			if ("resourceType".equals(nextName)) {
655 				continue;
656 			} else if ("link".equals(nextName)) {
657 				JsonArray entries = theObject.getJsonArray(nextName);
658 				for (JsonValue jsonValue : entries) {
659 					theState.enteringNewElement(null, "link");
660 					JsonObject linkObj = (JsonObject) jsonValue;
661 					String rel = linkObj.getString("rel", null);
662 					String href = linkObj.getString("href", null);
663 					theState.attributeValue("rel", rel);
664 					theState.attributeValue("href", href);
665 					theState.endingElement();
666 				}
667 				continue;
668 			} else if ("entry".equals(nextName)) {
669 				JsonArray entries = theObject.getJsonArray(nextName);
670 				for (JsonValue jsonValue : entries) {
671 					theState.enteringNewElement(null, "entry");
672 					parseBundleChildren((JsonObject) jsonValue, theState);
673 					theState.endingElement();
674 				}
675 				continue;
676 			} else if (BUNDLE_TEXTNODE_CHILDREN.contains(nextName)) {
677 				theState.enteringNewElement(null, nextName);
678 				theState.string(theObject.getString(nextName, null));
679 				theState.endingElement();
680 				continue;
681 			}
682 
683 			JsonValue nextVal = theObject.get(nextName);
684 			parseChildren(theState, nextName, nextVal, null);
685 
686 		}
687 	}

689 	private void parseChildren(JsonObject theObject, ParserState<?> theState) {
690 		String elementId = null;
691 		for (String nextName : theObject.keySet()) {
692 			if ("resourceType".equals(nextName)) {
693 				continue;
694 			} else if ("id".equals(nextName)) {
695 				elementId = theObject.getString(nextName);
696 				continue;
697 			} else if ("extension".equals(nextName)) {
698 				JsonArray array = theObject.getJsonArray(nextName);
699 				parseExtension(theState, array, false);
700 				continue;
701 			} else if ("modifierExtension".equals(nextName)) {
702 				JsonArray array = theObject.getJsonArray(nextName);
703 				parseExtension(theState, array, true);
704 				continue;
705 			} else if (nextName.charAt(0) == '_') {
706 				continue;
707 			}
708 
709 			JsonValue nextVal = theObject.get(nextName);
710 			JsonValue alternateVal = theObject.get('_' + nextName);
711 
712 			parseChildren(theState, nextName, nextVal, alternateVal);
713 
714 		}
715 
716 		if (elementId != null) {
717 			IElement object = (IElement) theState.getObject();
718 			if (object instanceof IIdentifiableElement) {
719 				((IIdentifiableElement) object).setElementSpecificId(elementId);
720 			} else if (object instanceof IResource) {
721 				((IResource) object).setId(new IdDt(elementId));
722 			}
723 		}
724 	}

726 	private void parseChildren(ParserState<?> theState, String theName, JsonValue theJsonVal, JsonValue theAlternateVal) {
727 		switch (theJsonVal.getValueType()) {
728 		case ARRAY: {
729 			JsonArray nextArray = (JsonArray) theJsonVal;
730 			JsonArray nextAlternateArray = (JsonArray) theAlternateVal;
731 			for (int i = 0; i < nextArray.size(); i++) {
732 				JsonValue nextObject = nextArray.get(i);
733 				JsonValue nextAlternate = null;
734 				if (nextAlternateArray != null) {
735 					nextAlternate = nextAlternateArray.get(i);
736 				}
737 				parseChildren(theState, theName, nextObject, nextAlternate);
738 			}
739 			break;
740 		}
741 		case OBJECT: {
742 			theState.enteringNewElement(null, theName);
743 			parseAlternates(theAlternateVal, theState);
744 			JsonObject nextObject = (JsonObject) theJsonVal;
745 			boolean preResource = false;
746 			if (theState.isPreResource()) {
747 				String resType = nextObject.getString("resourceType");
748 				if (isBlank(resType)) {
749 					throw new DataFormatException("Missing 'resourceType' from resource");
750 				}
751 				theState.enteringNewElement(null, resType);
752 				preResource = true;
753 			}
754 			parseChildren(nextObject, theState);
755 			if (preResource) {
756 				theState.endingElement();
757 			}
758 			theState.endingElement();
759 			break;
760 		}
761 		case STRING: {
762 			JsonString nextValStr = (JsonString) theJsonVal;
763 			theState.enteringNewElement(null, theName);
764 			theState.attributeValue("value", nextValStr.getString());
765 			parseAlternates(theAlternateVal, theState);
766 			theState.endingElement();
767 			break;
768 		}
769 		case NUMBER:
770 		case FALSE:
771 		case TRUE:
772 			theState.enteringNewElement(null, theName);
773 			theState.attributeValue("value", theJsonVal.toString());
774 			parseAlternates(theAlternateVal, theState);
775 			theState.endingElement();
776 			break;
777 		case NULL:
778 			break;
779 		}
780 	}

782 	private void parseExtension(ParserState<?> theState, JsonArray array, boolean theIsModifier) {
783 		for (int i = 0; i < array.size(); i++) {
784 			JsonObject nextExtObj = array.getJsonObject(i);
785 			String url = nextExtObj.getString("url");
786 			theState.enteringNewElementExtension(null, url, theIsModifier);
787 			for (Iterator<String> iter = nextExtObj.keySet().iterator(); iter.hasNext();) {
788 				String next = iter.next();
789 				if ("url".equals(next)) {
790 					continue;
791 				} else if ("extension".equals(next)) {
792 					JsonArray jsonVal = (JsonArray) nextExtObj.get(next);
793 					parseExtension(theState, jsonVal, false);
794 				} else if ("modifierExtension".equals(next)) {
795 					JsonArray jsonVal = (JsonArray) nextExtObj.get(next);
796 					parseExtension(theState, jsonVal, true);
797 				} else {
798 					JsonValue jsonVal = nextExtObj.get(next);
799 					parseChildren(theState, next, jsonVal, null);
800 				}
801 			}
802 			theState.endingElement();
803 		}
804 	}

934 	private class HeldExtension {
935 
936 		private RuntimeChildDeclaredExtensionDefinition myDef;
937 		private ExtensionDt myUndeclaredExtension;
938 		private IElement myValue;
939 
940 		public HeldExtension(ExtensionDt theUndeclaredExtension) {
941 			assert theUndeclaredExtension != null;
942 			myUndeclaredExtension = theUndeclaredExtension;
943 		}
944 
945 		public HeldExtension(RuntimeChildDeclaredExtensionDefinition theDef, IElement theValue) {
946 			assert theDef != null;
947 			assert theValue != null;
948 			myDef = theDef;
949 			myValue = theValue;
950 		}
951 
952 		public void write(RuntimeResourceDefinition theResDef, IResource theResource, JsonGenerator theEventWriter) throws IOException {
953 			if (myUndeclaredExtension != null) {
954 				writeUndeclaredExt(theResDef, theResource, theEventWriter, myUndeclaredExtension);
955 			} else {
956 				theEventWriter.writeStartObject();
957 				theEventWriter.write("url", myDef.getExtensionUrl());
958 
959 				BaseRuntimeElementDefinition<?> def = myDef.getChildElementDefinitionByDatatype(myValue.getClass());
960 				if (def.getChildType() == ChildTypeEnum.RESOURCE_BLOCK) {
961 					extractAndWriteExtensionsAsDirectChild(myValue, theEventWriter, def, theResDef, theResource);
962 				} else {
963 					// encodeChildElementToStreamWriter(theResDef, theResource,
964 					// theEventWriter, myValue, def, "value" +
965 					// WordUtils.capitalize(def.getName()));
966 					String childName = myDef.getChildNameByDatatype(myValue.getClass());
967 					encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, myValue, def, childName);
968 				}
969 
970 				// theEventWriter.name(myUndeclaredExtension.get);
971 
972 				theEventWriter.writeEnd();
973 			}
974 		}
975 
976 		private void writeUndeclaredExt(RuntimeResourceDefinition theResDef, IResource theResource, JsonGenerator theEventWriter, ExtensionDt ext) throws IOException {
977 			IElement value = ext.getValue();
978 
979 			theEventWriter.writeStartObject();
980 			theEventWriter.write("url", ext.getUrl().getValue());
981 
982 			boolean noValue = value == null || value.isEmpty();
983 			if (noValue && ext.getAllUndeclaredExtensions().isEmpty()) {
984 				ourLog.debug("Extension with URL[{}] has no value", ext.getUrl().getValue());
985 			} else if (noValue) {
986 				theEventWriter.writeStartArray("extension");
987 				for (ExtensionDt next : ext.getUndeclaredExtensions()) {
988 					writeUndeclaredExt(theResDef, theResource, theEventWriter, next);
989 				}
990 				theEventWriter.writeEnd();
991 			} else {
992 				RuntimeChildUndeclaredExtensionDefinition extDef = myContext.getRuntimeChildUndeclaredExtensionDefinition();
993 				String childName = extDef.getChildNameByDatatype(value.getClass());
994 				if (childName == null) {
995 					throw new ConfigurationException("Unable to encode extension, unregognized child element type: " + value.getClass().getCanonicalName());
996 				}
997 				BaseRuntimeElementDefinition<?> childDef = extDef.getChildElementDefinitionByDatatype(value.getClass());
998 				encodeChildElementToStreamWriter(theResDef, theResource, theEventWriter, value, childDef, childName);
999 			}
1000 
1001 			// theEventWriter.name(myUndeclaredExtension.get);
1002 
1003 			theEventWriter.writeEnd();
1004 		}
