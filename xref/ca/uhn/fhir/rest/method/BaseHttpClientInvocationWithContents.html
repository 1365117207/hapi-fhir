42  import ca.uhn.fhir.rest.server.EncodingEnum;

44  
45  public abstract class BaseHttpClientInvocationWithContents extends BaseHttpClientInvocation {
46  
47  	private final FhirContext myContext;
48  	private final IResource myResource;
49  	private final String myUrlExtension;
50  	private final TagList myTagList;
51  	private final List<IResource> myResources;
52  
53  	public BaseHttpClientInvocationWithContents(FhirContext theContext, IResource theResource, String theUrlExtension) {
54  		super();
55  		myContext = theContext;
56  		myResource = theResource;
57  		myUrlExtension = theUrlExtension;
58  		myTagList = null;
59  		myResources = null;
60  	}
61  
62  	public BaseHttpClientInvocationWithContents(FhirContext theContext, TagList theTagList, String... theUrlExtension) {
63  		super();
64  		if (theTagList == null) {
65  			throw new NullPointerException("Tag list must not be null");
66  		}
67  
68  		myResource = null;
69  		myContext = theContext;
70  		myTagList = theTagList;
71  		myResources = null;
72  
73  		myUrlExtension = StringUtils.join(theUrlExtension, '/');
74  	}
75  
76  	public BaseHttpClientInvocationWithContents(FhirContext theContext, List<IResource> theResources) {
77  		myContext = theContext;
78  		myResource = null;
79  		myTagList = null;
80  		myUrlExtension = null;
81  		myResources = theResources;
82  	}
83  
84  	@Override
85  	public HttpRequestBase asHttpRequest(String theUrlBase, Map<String, List<String>> theExtraParams, EncodingEnum theEncoding) throws DataFormatException {
86  		StringBuilder b = new StringBuilder();
87  		b.append(theUrlBase);
88  		if (!theUrlBase.endsWith("/")) {
89  			b.append('/');
90  		}
91  		b.append(StringUtils.defaultString(myUrlExtension));
92  
93  		appendExtraParamsWithQuestionMark(theExtraParams, b, b.indexOf("?") == -1);
94  		String url = b.toString();
95  
96  		if (myResource != null && Binary.class.isAssignableFrom(myResource.getClass())) {
97  			Binary binary = (Binary) myResource;
98  			ByteArrayEntity entity = new ByteArrayEntity(binary.getContent(), ContentType.parse(binary.getContentType()));
99  			HttpRequestBase retVal = createRequest(url, entity);
100 			return retVal;
101 		}
102 
103 		IParser parser;
104 		String contentType;
105 		if (theEncoding == EncodingEnum.JSON) {
106 			parser = myContext.newJsonParser();
107 			contentType = Constants.CT_FHIR_JSON;
108 		} else {
109 			parser = myContext.newXmlParser();
110 			contentType = Constants.CT_FHIR_XML;
111 		}
112 
113 		String contents;
114 		if (myTagList != null) {
115 			contents = parser.encodeTagListToString(myTagList);
116 		} else if (myResources != null) {
117 			Bundle bundle = RestfulServer.createBundleFromResourceList(myContext, "", myResources, "", "", myResources.size());
118 			contents = parser.encodeBundleToString(bundle);
119 		} else {
120 			contents = parser.encodeResourceToString(myResource);
121 		}
122 
123 		StringEntity entity = new StringEntity(contents, ContentType.create(contentType, "UTF-8"));
124 
125 		HttpRequestBase retVal = createRequest(url, entity);
126 		super.addHeadersToRequest(retVal);
127 		return retVal;
128 	}
129 
130 	protected abstract HttpRequestBase createRequest(String url, AbstractHttpEntity theEntity);
131 
132 }
