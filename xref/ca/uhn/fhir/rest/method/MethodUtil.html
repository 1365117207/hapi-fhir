90  	public static HttpPutClientInvocation createUpdateInvocation(IResource theResource, String theResourceBody, IdDt theId, FhirContext theContext) {
91  		String resourceName = theContext.getResourceDefinition(theResource).getName();
92  		StringBuilder urlBuilder = new StringBuilder();
93  		urlBuilder.append(resourceName);
94  		urlBuilder.append('/');
95  		urlBuilder.append(theId.getIdPart());
96  	
97  		HttpPutClientInvocation retVal;
98  		String urlExtension = urlBuilder.toString();
99  		if (StringUtils.isBlank(theResourceBody)) {
100 			retVal = new HttpPutClientInvocation(theContext, theResource, urlExtension);
101 		} else {
102 			retVal = new HttpPutClientInvocation(theContext, theResourceBody, false, urlExtension);
103 		}
104 
105 		if (theId.hasVersionIdPart()) {
106 			String versionId = theId.getVersionIdPart();
107 			if (StringUtils.isNotBlank(versionId)) {
108 				urlBuilder.append('/');
109 				urlBuilder.append(Constants.PARAM_HISTORY);
110 				urlBuilder.append('/');
111 				urlBuilder.append(versionId);
112 				retVal.addHeader(Constants.HEADER_CONTENT_LOCATION, urlBuilder.toString());
113 			}
114 		} 
115 		
116 		addTagsToPostOrPut(theResource, retVal);
117 	
118 		return retVal;
119 	}

121 	public static void parseClientRequestResourceHeaders(Map<String, List<String>> theHeaders, IResource resource) {
122 		List<String> lmHeaders = theHeaders.get(Constants.HEADER_LAST_MODIFIED_LOWERCASE);
123 		if (lmHeaders != null && lmHeaders.size() > 0 && StringUtils.isNotBlank(lmHeaders.get(0))) {
124 			String headerValue = lmHeaders.get(0);
125 			Date headerDateValue;
126 			try {
127 				headerDateValue = DateUtils.parseDate(headerValue);
128 				InstantDt lmValue = new InstantDt(headerDateValue);
129 				resource.getResourceMetadata().put(ResourceMetadataKeyEnum.UPDATED, lmValue);
130 			} catch (Exception e) {
131 				ourLog.warn("Unable to parse date string '{}'. Error is: {}", headerValue, e.toString());
132 			}
133 		}
134 
135 		List<String> clHeaders = theHeaders.get(Constants.HEADER_CONTENT_LOCATION_LC);
136 		if (clHeaders != null && clHeaders.size() > 0 && StringUtils.isNotBlank(clHeaders.get(0))) {
137 			String headerValue = clHeaders.get(0);
138 			resource.getId().setValue(headerValue);
139 		}
140 		
141 		List<String> categoryHeaders = theHeaders.get(Constants.HEADER_CATEGORY_LC);
142 		if (categoryHeaders != null && categoryHeaders.size() > 0 && StringUtils.isNotBlank(categoryHeaders.get(0))) {
143 			TagList tagList = new TagList();
144 			for (String header : categoryHeaders) {
145 				parseTagValue(tagList, header);
146 			}
147 			ResourceMetadataKeyEnum.TAG_LIST.put(resource, tagList);
148 		}
149 	}

156 	private static void parseTagValue(TagList theTagList, String theCompleteHeaderValue, StringBuilder theBuffer) {
157 		int firstSemicolon = theBuffer.indexOf(";");
158 		int deleteTo;
159 		if (firstSemicolon == -1) {
160 			firstSemicolon = theBuffer.indexOf(",");
161 			if (firstSemicolon == -1) {
162 				firstSemicolon = theBuffer.length();
163 				deleteTo = theBuffer.length();
164 			} else {
165 				deleteTo = firstSemicolon;
166 			}
167 		} else {
168 			deleteTo = firstSemicolon + 1;
169 		}
170 
171 		String term = theBuffer.substring(0, firstSemicolon);
172 		String scheme = null;
173 		String label = null;
174 		if (isBlank(term)) {
175 			return;
176 		}
177 
178 		theBuffer.delete(0, deleteTo);
179 		while (theBuffer.length() > 0 && theBuffer.charAt(0) == ' ') {
180 			theBuffer.deleteCharAt(0);
181 		}
182 
183 		while (theBuffer.length() > 0) {
184 			boolean foundSomething = false;
185 			if (theBuffer.length() > SCHEME.length() && theBuffer.substring(0, SCHEME.length()).equals(SCHEME)) {
186 				int closeIdx = theBuffer.indexOf("\"", SCHEME.length());
187 				scheme = theBuffer.substring(SCHEME.length(), closeIdx);
188 				theBuffer.delete(0, closeIdx + 1);
189 				foundSomething = true;
190 			}
191 			if (theBuffer.length() > LABEL.length() && theBuffer.substring(0, LABEL.length()).equals(LABEL)) {
192 				int closeIdx = theBuffer.indexOf("\"", LABEL.length());
193 				label = theBuffer.substring(LABEL.length(), closeIdx);
194 				theBuffer.delete(0, closeIdx + 1);
195 				foundSomething = true;
196 			}
197 			// TODO: support enc2231-string as described in
198 			// http://tools.ietf.org/html/draft-johnston-http-category-header-02
199 			// TODO: support multiple tags in one header as described in
200 			// http://hl7.org/implement/standards/fhir/http.html#tags
201 
202 			while (theBuffer.length() > 0 && (theBuffer.charAt(0) == ' ' || theBuffer.charAt(0) == ';')) {
203 				theBuffer.deleteCharAt(0);
204 			}
205 
206 			if (!foundSomething) {
207 				break;
208 			}
209 		}
210 
211 		if (theBuffer.length() > 0 && theBuffer.charAt(0) == ',') {
212 			theBuffer.deleteCharAt(0);
213 			while (theBuffer.length() > 0 && theBuffer.charAt(0) == ' ') {
214 				theBuffer.deleteCharAt(0);
215 			}
216 			theTagList.add(new Tag(scheme, term, label));
217 			parseTagValue(theTagList, theCompleteHeaderValue, theBuffer);
218 		} else {
219 			theTagList.add(new Tag(scheme, term, label));
220 		}
221 
222 		if (theBuffer.length() > 0) {
223 			ourLog.warn("Ignoring extra text at the end of " + Constants.HEADER_CATEGORY + " tag '" + theBuffer.toString() + "' - Complete tag value was: " + theCompleteHeaderValue);
224 		}
225 
226 	}

229 	static void addTagsToPostOrPut(IResource resource, BaseHttpClientInvocation retVal) {
230 		TagList list = (TagList) resource.getResourceMetadata().get(ResourceMetadataKeyEnum.TAG_LIST);
231 		if (list != null) {
232 			for (Tag tag : list) {
233 				if (StringUtils.isNotBlank(tag.getTerm())) {
234 					retVal.addHeader(Constants.HEADER_CATEGORY, tag.toHeaderValue());
235 				}
236 			}
237 		}

245 		RuntimeResourceDefinition def = theContext.getResourceDefinition(theResource);
246 		String resourceName = def.getName();
247 
248 		StringBuilder urlExtension = new StringBuilder();
249 		urlExtension.append(resourceName);
250 		if (StringUtils.isNotBlank(theId)) {
251 			urlExtension.append('/');
252 			urlExtension.append(theId);
253 		}
254 
255 		HttpPostClientInvocation retVal;
256 		if (StringUtils.isBlank(theResourceBody)) {
257 			retVal = new HttpPostClientInvocation(theContext, theResource, urlExtension.toString());
258 		} else {
259 			retVal = new HttpPostClientInvocation(theContext, theResourceBody, false, urlExtension.toString());
260 		}
261 		addTagsToPostOrPut(theResource, retVal);
262 
263 		return retVal;
264 	}

270 	public static MethodOutcome process2xxResponse(FhirContext theContext, String theResourceName, int theResponseStatusCode, String theResponseMimeType, Reader theResponseReader,
271 			Map<String, List<String>> theHeaders) {
272 		List<String> locationHeaders = new ArrayList<String>();
273 		List<String> lh = theHeaders.get(Constants.HEADER_LOCATION_LC);
274 		if (lh != null) {
275 			locationHeaders.addAll(lh);
276 		}
277 		List<String> clh = theHeaders.get(Constants.HEADER_CONTENT_LOCATION_LC);
278 		if (clh != null) {
279 			locationHeaders.addAll(clh);
280 		}
281 
282 		MethodOutcome retVal = new MethodOutcome();
283 		if (locationHeaders != null && locationHeaders.size() > 0) {
284 			String locationHeader = locationHeaders.get(0);
285 			BaseOutcomeReturningMethodBinding.parseContentLocation(retVal, theResourceName, locationHeader);
286 		}
287 		if (theResponseStatusCode != Constants.STATUS_HTTP_204_NO_CONTENT) {
288 			EncodingEnum ct = EncodingEnum.forContentType(theResponseMimeType);
289 			if (ct != null) {
290 				PushbackReader reader = new PushbackReader(theResponseReader);
291 
292 				try {
293 					int firstByte = reader.read();
294 					if (firstByte == -1) {
295 						BaseOutcomeReturningMethodBinding.ourLog.debug("No content in response, not going to read");
296 						reader = null;
297 					} else {
298 						reader.unread(firstByte);
299 					}
300 				} catch (IOException e) {
301 					BaseOutcomeReturningMethodBinding.ourLog.debug("No content in response, not going to read", e);
302 					reader = null;
303 				}
304 
305 				if (reader != null) {
306 					IParser parser = ct.newParser(theContext);
307 					IResource outcome = parser.parseResource(reader);
308 					if (outcome instanceof OperationOutcome) {
309 						retVal.setOperationOutcome((OperationOutcome) outcome);
310 					}
311 				}
312 
313 			} else {
314 				BaseOutcomeReturningMethodBinding.ourLog.debug("Ignoring response content of type: {}", theResponseMimeType);
315 			}
316 		}
317 		return retVal;
318 	}

320 	public static Integer findParamAnnotationIndex(Method theMethod, Class<?> toFind) {
321 		int paramIndex = 0;
322 		for (Annotation[] annotations : theMethod.getParameterAnnotations()) {
323 			for (int annotationIndex = 0; annotationIndex < annotations.length; annotationIndex++) {
324 				Annotation nextAnnotation = annotations[annotationIndex];
325 				Class<? extends Annotation> class1 = nextAnnotation.getClass();
326 				if (toFind.isAssignableFrom(class1)) {
327 					return paramIndex;
328 				}
329 			}
330 			paramIndex++;
331 		}
332 		return null;
333 	}

335 	public static void extractDescription(SearchParameter theParameter, Annotation[] theAnnotations) {
336 		for (Annotation annotation : theAnnotations) {
337 			if (annotation instanceof Description) {
338 				Description desc = (Description) annotation;
339 				if (isNotBlank(desc.formalDefinition())) {
340 					theParameter.setDescription(desc.formalDefinition());
341 				} else {
342 					theParameter.setDescription(desc.shortDefinition());
343 				}
344 			}
345 		}

382 	@SuppressWarnings("unchecked")
383 	public static List<IParameter> getResourceParameters(Method theMethod) {
384 		List<IParameter> parameters = new ArrayList<IParameter>();
385 
386 		Class<?>[] parameterTypes = theMethod.getParameterTypes();
387 		int paramIndex = 0;
388 		for (Annotation[] annotations : theMethod.getParameterAnnotations()) {
389 
390 			IParameter param = null;
391 			Class<?> parameterType = parameterTypes[paramIndex];
392 			Class<? extends java.util.Collection<?>> outerCollectionType = null;
393 			Class<? extends java.util.Collection<?>> innerCollectionType = null;
394 			if (TagList.class.isAssignableFrom(parameterType)) {
395 				// TagList is handled directly within the method bindings
396 				param = new NullParameter();
397 			} else {
398 				if (Collection.class.isAssignableFrom(parameterType)) {
399 					innerCollectionType = (Class<? extends java.util.Collection<?>>) parameterType;
400 					parameterType = ReflectionUtil.getGenericCollectionTypeOfMethodParameter(theMethod, paramIndex);
401 				}
402 				if (Collection.class.isAssignableFrom(parameterType)) {
403 					outerCollectionType = innerCollectionType;
404 					innerCollectionType = (Class<? extends java.util.Collection<?>>) parameterType;
405 					parameterType = ReflectionUtil.getGenericCollectionTypeOfMethodParameter(theMethod, paramIndex);
406 				}
407 				if (Collection.class.isAssignableFrom(parameterType)) {
408 					throw new ConfigurationException("Argument #" + paramIndex + " of Method '" + theMethod.getName() + "' in type '" + theMethod.getDeclaringClass().getCanonicalName()
409 							+ "' is of an invalid generic type (can not be a collection of a collection of a collection)");
410 				}
411 			}
412 			if (parameterType.equals(HttpServletRequest.class) || parameterType.equals(ServletRequest.class)) {
413 				param = new ServletRequestParameter();
414 			} else if (parameterType.equals(HttpServletResponse.class) || parameterType.equals(ServletResponse.class)) {
415 				param = new ServletResponseParameter();
416 			} else {
417 				for (int i = 0; i < annotations.length && param == null; i++) {
418 					Annotation nextAnnotation = annotations[i];
419 
420 					if (nextAnnotation instanceof RequiredParam) {
421 						SearchParameter parameter = new SearchParameter();
422 						parameter.setName(((RequiredParam) nextAnnotation).name());
423 						parameter.setRequired(true);
424 						parameter.setDeclaredTypes(((RequiredParam) nextAnnotation).targetTypes());
425 						parameter.setCompositeTypes(((RequiredParam) nextAnnotation).compositeTypes());
426 						parameter.setChainlists(((RequiredParam) nextAnnotation).chainWhitelist(), ((RequiredParam) nextAnnotation).chainBlacklist());
427 						parameter.setType(parameterType, innerCollectionType, outerCollectionType);
428 						MethodUtil.extractDescription(parameter, annotations);
429 						param = parameter;
430 					} else if (nextAnnotation instanceof OptionalParam) {
431 						SearchParameter parameter = new SearchParameter();
432 						parameter.setName(((OptionalParam) nextAnnotation).name());
433 						parameter.setRequired(false);
434 						parameter.setDeclaredTypes(((OptionalParam) nextAnnotation).targetTypes());
435 						parameter.setCompositeTypes(((OptionalParam) nextAnnotation).compositeTypes());
436 						parameter.setChainlists(((OptionalParam) nextAnnotation).chainWhitelist(), ((OptionalParam) nextAnnotation).chainBlacklist());
437 						parameter.setType(parameterType, innerCollectionType, outerCollectionType);
438 						MethodUtil.extractDescription(parameter, annotations);
439 						param = parameter;
440 					} else if (nextAnnotation instanceof IncludeParam) {
441 						Class<? extends Collection<Include>> instantiableCollectionType;
442 						Class<?> specType;
443 
444 						if (parameterType == String.class) {
445 							instantiableCollectionType = null;
446 							specType = String.class;
447 						} else if ((parameterType != Include.class && parameterType != PathSpecification.class) || innerCollectionType == null || outerCollectionType != null) {
448 							throw new ConfigurationException("Method '" + theMethod.getName() + "' is annotated with @" + IncludeParam.class.getSimpleName() + " but has a type other than Collection<"
449 									+ Include.class.getSimpleName() + ">");
450 						} else {
451 							instantiableCollectionType = (Class<? extends Collection<Include>>) CollectionBinder.getInstantiableCollectionType(innerCollectionType, "Method '" + theMethod.getName()
452 									+ "'");
453 							specType = parameterType;
454 						}
455 
456 						param = new IncludeParameter((IncludeParam) nextAnnotation, instantiableCollectionType, specType);
457 					} else if (nextAnnotation instanceof ResourceParam) {
458 						if (!IResource.class.isAssignableFrom(parameterType)) {
459 							throw new ConfigurationException("Method '" + theMethod.getName() + "' is annotated with @" + ResourceParam.class.getSimpleName()
460 									+ " but has a type that is not an implemtation of " + IResource.class.getCanonicalName());
461 						}
462 						param = new ResourceParameter((Class<? extends IResource>) parameterType);
463 					} else if (nextAnnotation instanceof IdParam || nextAnnotation instanceof VersionIdParam) {
464 						param = new NullParameter();
465 					} else if (nextAnnotation instanceof ServerBase) {
466 						param = new ServerBaseParamBinder();
467 					} else if (nextAnnotation instanceof Since) {
468 						param = new SinceParameter();
469 					} else if (nextAnnotation instanceof Count) {
470 						param = new CountParameter();
471 					} else if (nextAnnotation instanceof Sort) {
472 						param = new SortParameter();
473 					} else if (nextAnnotation instanceof TransactionParam) {
474 						param = new TransactionParamBinder();
475 					} else {
476 						continue;
477 					}
478 
479 				}
480 
481 			}
482 
483 			if (param == null) {
484 				throw new ConfigurationException("Parameter #" + ((paramIndex + 1)) + "/" + (parameterTypes.length) + " of method '" + theMethod.getName() + "' on type '"
485 						+ theMethod.getDeclaringClass().getCanonicalName() + "' has no recognized FHIR interface parameter annotations. Don't know how to handle this parameter");
486 			}
487 
488 			param.initializeTypes(theMethod, outerCollectionType, innerCollectionType, parameterType);
489 			parameters.add(param);
490 
491 			paramIndex++;
492 		}
493 		return parameters;
494 	}
