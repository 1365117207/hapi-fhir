1   package ca.uhn.fhir.rest.server;
2   
3   import java.util.Collections;
4   import java.util.List;
5   
6   import ca.uhn.fhir.model.api.IResource;
7   import ca.uhn.fhir.model.primitive.InstantDt;
8   
9   public class SimpleBundleProvider implements IBundleProvider {
10  
11  	private List<IResource> myList;
12  	
13  	public SimpleBundleProvider(List<IResource> theList) {
14  		myList = theList;
15  	}
16  
17  	public SimpleBundleProvider(IResource theResource) {
18  		myList = Collections.singletonList(theResource);
19  	}
20  
21  	/**
22  	 * Create an empty bundle
23  	 */
24  	public SimpleBundleProvider() {
25  		myList = Collections.emptyList();
26  	}
27  
28  	@Override
29  	public List<IResource> getResources(int theFromIndex, int theToIndex) {
30  		return myList.subList(theFromIndex, Math.min(theToIndex, myList.size()));
31  	}
32  
33  	@Override
34  	public int size() {
35  		return myList.size();
36  	}
37  
38  	@Override
39  	public InstantDt getPublished() {
40  		return InstantDt.withCurrentTime();
41  	}
42  	
43  }
