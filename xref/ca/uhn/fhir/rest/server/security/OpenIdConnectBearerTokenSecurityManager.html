23  import java.text.ParseException;
24  import java.util.Date;
25  
26  import javax.servlet.http.HttpServletRequest;
27  
28  import org.mitre.jwt.signer.service.JwtSigningAndValidationService;
29  import org.mitre.jwt.signer.service.impl.JWKSetCacheService;
30  import org.mitre.jwt.signer.service.impl.SymmetricCacheService;
31  import org.mitre.oauth2.model.RegisteredClient;
32  import org.mitre.openid.connect.client.service.ClientConfigurationService;
33  import org.mitre.openid.connect.client.service.ServerConfigurationService;
34  import org.mitre.openid.connect.config.ServerConfiguration;
35  import org.springframework.beans.factory.annotation.Autowired;
36  
37  import ca.uhn.fhir.rest.server.Constants;
38  import ca.uhn.fhir.rest.server.exceptions.AuthenticationException;
39  
40  import com.nimbusds.jose.JWSAlgorithm;
41  import com.nimbusds.jwt.ReadOnlyJWTClaimsSet;
42  import com.nimbusds.jwt.SignedJWT;
43  
44  public class OpenIdConnectBearerTokenSecurityManager implements IResourceSecurity {
45  
46  	private static final String BEARER_PREFIX = "Bearer ";
47  
48  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(OpenIdConnectBearerTokenSecurityManager.class);
49  	@Autowired
50  	private ClientConfigurationService myClientConfigurationService;
51  
52  	@Autowired
53  	private ServerConfigurationService myServerConfigurationService;
54  
55  	private int myTimeSkewAllowance = 300;
56  
57  	private SymmetricCacheService symmetricCacheService;
58  	private JWKSetCacheService validationServices;
59  
60  	public OpenIdConnectBearerTokenSecurityManager() {
61  		symmetricCacheService = new SymmetricCacheService();
62  		validationServices = new JWKSetCacheService();
63  	}
64  
65  	@Override
66  	public ISecurityOutcome authenticate(HttpServletRequest theRequest) throws AuthenticationException {
67  		String token = theRequest.getHeader(Constants.HEADER_AUTHORIZATION);
68  		if (token == null) {
69  			throw new AuthenticationException("Not authorized (no authorization header found in request)");
70  		}
71  		if (!token.startsWith(BEARER_PREFIX)) {
72  			throw new AuthenticationException("Not authorized (authorization header does not contain a bearer token)");
73  		}
74  
75  		token = token.substring(BEARER_PREFIX.length());
76  
77  		SignedJWT idToken;
78  		try {
79  			idToken = SignedJWT.parse(token);
80  		} catch (ParseException e) {
81  			throw new AuthenticationException("Not authorized (bearer token could not be validated)", e);
82  		}
83  
84  		// validate our ID Token over a number of tests
85  		ReadOnlyJWTClaimsSet idClaims;
86  		try {
87  			idClaims = idToken.getJWTClaimsSet();
88  		} catch (ParseException e) {
89  			throw new AuthenticationException("Not authorized (bearer token could not be validated)", e);
90  		}
91  
92  		String issuer = idClaims.getIssuer();
93  
94  		ServerConfiguration serverConfig = myServerConfigurationService.getServerConfiguration(issuer);
95  		if (serverConfig == null) {
96  			ourLog.error("No server configuration found for issuer: " + issuer);
97  			throw new AuthenticationException("Not authorized (no server configuration found for issuer " + issuer + ")");
98  		}
99  
100 		RegisteredClient clientConfig = myClientConfigurationService.getClientConfiguration(serverConfig);
101 		if (clientConfig == null) {
102 			ourLog.error("No client configuration found for issuer: " + issuer);
103 			throw new AuthenticationException("Not authorized (no client configuration found for issuer " + issuer + ")");
104 		}
105 
106 		// check the signature
107 		JwtSigningAndValidationService jwtValidator = null;
108 
109 		JWSAlgorithm alg = idToken.getHeader().getAlgorithm();
110 		if (alg.equals(JWSAlgorithm.HS256) || alg.equals(JWSAlgorithm.HS384) || alg.equals(JWSAlgorithm.HS512)) {
111 
112 			// generate one based on client secret
113 			jwtValidator = symmetricCacheService.getSymmetricValidtor(clientConfig.getClient());
114 		} else {
115 			// otherwise load from the server's public key
116 			jwtValidator = validationServices.getValidator(serverConfig.getJwksUri());
117 		}
118 
119 		if (jwtValidator != null) {
120 			if (!jwtValidator.validateSignature(idToken)) {
121 				throw new AuthenticationException("Not authorized (signature validation failed)");
122 			}
123 		} else {
124 			ourLog.error("No validation service found. Skipping signature validation");
125 			throw new AuthenticationException("Not authorized (can't determine signature validator)");
126 		}
127 
128 		// check expiration
129 		if (idClaims.getExpirationTime() == null) {
130 			throw new AuthenticationException("Id Token does not have required expiration claim");
131 		} else {
132 			// it's not null, see if it's expired
133 			Date now = new Date(System.currentTimeMillis() - (myTimeSkewAllowance * 1000));
134 			if (now.after(idClaims.getExpirationTime())) {
135 				throw new AuthenticationException("Id Token is expired: " + idClaims.getExpirationTime());
136 			}
137 		}
138 
139 		// check not before
140 		if (idClaims.getNotBeforeTime() != null) {
141 			Date now = new Date(System.currentTimeMillis() + (myTimeSkewAllowance * 1000));
142 			if (now.before(idClaims.getNotBeforeTime())) {
143 				throw new AuthenticationException("Id Token not valid untill: " + idClaims.getNotBeforeTime());
144 			}
145 		}
146 
147 		// check issued at
148 		if (idClaims.getIssueTime() == null) {
149 			throw new AuthenticationException("Id Token does not have required issued-at claim");
150 		} else {
151 			// since it's not null, see if it was issued in the future
152 			Date now = new Date(System.currentTimeMillis() + (myTimeSkewAllowance * 1000));
153 			if (now.before(idClaims.getIssueTime())) {
154 				throw new AuthenticationException("Id Token was issued in the future: " + idClaims.getIssueTime());
155 			}
156 		}
157 
158 		return new ISecurityOutcome() {
159 		};
160 	}
161 
162 	public int getTimeSkewAllowance() {
163 		return myTimeSkewAllowance;
164 	}
165

168 	}

171 		myServerConfigurationService = theServerConfigurationService;
172 	}
173

175 		myTimeSkewAllowance = theTimeSkewAllowance;
176 	}
177

179 		validationServices = theValidationServices;
180 	}
