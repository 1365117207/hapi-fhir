198 	private RuntimeResourceDefinition getResourceType(HttpServletRequest theReq) throws ServletException {
199 		String resourceName = StringUtils.defaultString(theReq.getParameter(PARAM_RESOURCE));
200 		RuntimeResourceDefinition def = myCtx.getResourceDefinition(resourceName);
201 		if (def == null) {
202 			throw new ServletException("Invalid resourceName: " + resourceName);
203 		}
204 		return def;
205 	}

217 	private void processAction(HttpServletRequest theReq, WebContext theContext) {
218 
219 		GenericClient client = (GenericClient) myCtx.newRestfulGenericClient(myServerBase);
220 		client.setKeepResponses(true);
221 		ResultType returnsResource;
222 		long latency = 0;
223 
224 		String outcomeDescription = null;
225 		try {
226 			String method = theReq.getParameter("action");
227 
228 			String prettyParam = theReq.getParameter("pretty");
229 			if ("true".equals(prettyParam)) {
230 				client.setPrettyPrint(true);
231 			} else if ("false".equals(prettyParam)) {
232 				client.setPrettyPrint(false);
233 			}
234 			EncodingEnum encoding = getRequestEncoding(theReq);
235 			client.setEncoding(encoding);
236 
237 			long start = System.currentTimeMillis();
238 			if ("home".equals(method)) {
239 				return;
240 			} else if ("conformance".equals(method)) {
241 				returnsResource = ResultType.RESOURCE;
242 				client.conformance();
243 			} else if ("read".equals(method)) {
244 				RuntimeResourceDefinition def = getResourceType(theReq);
245 				String id = StringUtils.defaultString(theReq.getParameter("id"));
246 				if (StringUtils.isBlank(id)) {
247 					theContext.getVariables().put("errorMsg", "No ID specified");
248 					return;
249 				}
250 				returnsResource = ResultType.RESOURCE;
251 
252 				String versionId = StringUtils.defaultString(theReq.getParameter("vid"));
253 				if (StringUtils.isBlank(versionId)) {
254 					versionId = null;
255 					outcomeDescription = "Read Resource";
256 				} else {
257 					outcomeDescription = "VRead Resource";
258 				}
259 
260 				client.read(def.getImplementingClass(), new IdDt(def.getName(), id, versionId));
261 
262 			} else if ("get-tags".equals(method)) {
263 
264 				Class<? extends IResource> resType = null;
265 				if (isNotBlank(theReq.getParameter(PARAM_RESOURCE))) {
266 					RuntimeResourceDefinition def = getResourceType(theReq);
267 					resType = def.getImplementingClass();
268 					String id = theReq.getParameter("resource-tags-id");
269 					if (isNotBlank(id)) {
270 						String vid = theReq.getParameter("resource-tags-vid");
271 						if (isNotBlank(vid)) {
272 							client.getTags().forResource(resType, id, vid).execute();
273 						} else {
274 							client.getTags().forResource(resType, id).execute();
275 						}
276 					} else {
277 						client.getTags().forResource(resType).execute();
278 					}
279 				} else {
280 					client.getTags().execute();
281 				}
282 				returnsResource = ResultType.TAGLIST;
283 				outcomeDescription = "Tag List";
284 
285 			} else if ("page".equals(method)) {
286 
287 				String url = defaultString(theReq.getParameter("page-url"));
288 				if (!url.startsWith(myServerBase)) {
289 					theContext.getVariables().put("errorMsg", "Invalid page URL: " + url);
290 					return;
291 				}
292 				
293 				returnsResource = ResultType.TAGLIST;
294 				outcomeDescription = "Tag List";
295 
296 			} else if ("delete".equals(method)) {
297 				RuntimeResourceDefinition def = getResourceType(theReq);
298 				String id = StringUtils.defaultString(theReq.getParameter("resource-delete-id"));
299 				if (StringUtils.isBlank(id)) {
300 					theContext.getVariables().put("errorMsg", "No ID specified");
301 					return;
302 				}
303 
304 				returnsResource = ResultType.BUNDLE;
305 				outcomeDescription = "Delete Resource";
306 
307 				client.delete(def.getImplementingClass(), new IdDt(id));
308 
309 			} else if ("history-resource".equals(method) || "history-server".equals(method)) {
310 				String id = null;
311 				Class<? extends IResource> type = null; // def.getImplementingClass();
312 				if (!"history-server".equals(method)) {
313 					RuntimeResourceDefinition def = getResourceType(theReq);
314 					type = def.getImplementingClass();
315 					id = StringUtils.defaultString(theReq.getParameter("resource-history-id"));
316 					if (StringUtils.isBlank(id)) {
317 						if ("history-instance".equals(method)) {
318 							theContext.getVariables().put("errorMsg", "No ID specified");
319 							return;
320 						} else {
321 							id = null;
322 						}
323 					}
324 				}
325 
326 				DateTimeDt since = null;
327 				String sinceStr = theReq.getParameter("since");
328 				if (isNotBlank(sinceStr)) {
329 					since = new DateTimeDt(sinceStr);
330 				}
331 
332 				Integer limit = null;
333 				String limitStr = theReq.getParameter("limit");
334 				if (isNotBlank(limitStr)) {
335 					limit = Integer.parseInt(limitStr);
336 				}
337 
338 				returnsResource = ResultType.BUNDLE;
339 				outcomeDescription = "Resource History";
340 
341 				client.history(type, id, since, limit);
342 
343 			} else if ("create".equals(method) || "validate".equals(method)) {
344 				boolean validate = "validate".equals(method);
345 
346 				String body = validate ? theReq.getParameter("resource-validate-body") : theReq.getParameter("resource-create-body");
347 				if (isBlank(body)) {
348 					theContext.getVariables().put("errorMsg", "No message body specified");
349 					return;
350 				}
351 
352 				body = body.trim();
353 				IResource resource;
354 				try {
355 					if (body.startsWith("{")) {
356 						resource = myCtx.newJsonParser().parseResource(body);
357 					} else if (body.startsWith("<")) {
358 						resource = myCtx.newXmlParser().parseResource(body);
359 					} else {
360 						theContext.getVariables().put("errorMsg", "Message body does not appear to be a valid FHIR resource instance document. Body should start with '<' (for XML encoding) or '{' (for JSON encoding).");
361 						return;
362 					}
363 				} catch (DataFormatException e) {
364 					ourLog.warn("Failed to parse resource", e);
365 					theContext.getVariables().put("errorMsg", "Failed to parse message body. Error was: " + e.getMessage());
366 					return;
367 				}
368 
369 				if (validate) {
370 					client.validate(resource);
371 					outcomeDescription = "Validate Resource";
372 				} else {
373 					String id = theReq.getParameter("resource-create-id");
374 					if (isNotBlank(id)) {
375 						outcomeDescription = "Update Resource";
376 						client.update(id, resource);
377 					} else {
378 						outcomeDescription = "Create Resource";
379 						client.create(resource);
380 					}
381 				}
382 				returnsResource = ResultType.RESOURCE;
383 
384 			} else if ("search".equals(method)) {
385 				IUntypedQuery search = client.search();
386 				IQuery query;
387 				if (isNotBlank(theReq.getParameter("resource"))) {
388 					query = search.forResource(getResourceType(theReq).getImplementingClass());
389 				} else {
390 					query = search.forAllResources();
391 				}
392 
393 				outcomeDescription = "Search for Resources";
394 
395 				int paramIdx = -1;
396 				while (true) {
397 					paramIdx++;
398 
399 					String nextName = theReq.getParameter("param." + paramIdx + ".name");
400 					if (isBlank(nextName)) {
401 						break;
402 					}
403 					String nextType = theReq.getParameter("param." + paramIdx + ".type");
404 
405 					StringBuilder b = new StringBuilder();
406 					for (int i = 0; i < 100; i++) {
407 						b.append(defaultString(theReq.getParameter("param." + paramIdx + "." + i)));
408 					}
409 
410 					String paramValue = b.toString();
411 					if (isBlank(paramValue)) {
412 						continue;
413 					}
414 
415 					if ("token".equals(nextType)) {
416 						if (paramValue.length() < 2) {
417 							continue;
418 						}
419 					}
420 
421 					// if ("xml".equals(theReq.getParameter("encoding"))) {
422 					// query.encodedXml();
423 					// }else if ("json".equals(theReq.getParameter("encoding"))) {
424 					// query.encodedJson();
425 					// }
426 
427 					query.where(new StringParam(nextName).matches().value(paramValue));
428 
429 				}
430 
431 				String[] incValues = theReq.getParameterValues(Constants.PARAM_INCLUDE);
432 				if (incValues != null) {
433 					for (String next : incValues) {
434 						if (isNotBlank(next)) {
435 							query.include(new Include(next));
436 						}
437 					}
438 				}
439 
440 				String limit = theReq.getParameter("resource-search-limit");
441 				if (isNotBlank(limit)) {
442 					if (!limit.matches("[0-9]+")) {
443 						theContext.getVariables().put("errorMsg", "Search limit must be a numeric value.");
444 						return;
445 					}
446 					query.limitTo(Integer.parseInt(limit));
447 				}
448 
449 				query.execute();
450 				returnsResource = ResultType.BUNDLE;
451 
452 			} else {
453 				theContext.getVariables().put("errorMsg", "Invalid action: " + method);
454 				return;
455 			}
456 
457 			latency = System.currentTimeMillis() - start;
458 		} catch (DataFormatException e) {
459 			ourLog.error("Failed to invoke method", e);
460 			returnsResource = ResultType.NONE;
461 		} catch (BaseServerResponseException e) {
462 			ourLog.error("Failed to invoke method", e);
463 			returnsResource = ResultType.NONE;
464 		} catch (Exception e) {
465 			ourLog.error("Failure during processing", e);
466 			returnsResource = ResultType.NONE;
467 		}
468 
469 		try {
470 			HttpRequestBase lastRequest = client.getLastRequest();
471 			String requestBody = null;
472 			String requestSyntaxHighlighterClass = null;
473 
474 			if (lastRequest instanceof HttpEntityEnclosingRequest) {
475 				HttpEntityEnclosingRequest lastEERequest = (HttpEntityEnclosingRequest) lastRequest;
476 				HttpEntity lastEE = lastEERequest.getEntity();
477 				if (lastEE.isRepeatable()) {
478 					StringWriter requestCapture = new StringWriter();
479 					lastEE.writeTo(new WriterOutputStream(requestCapture, "UTF-8"));
480 					requestBody = requestCapture.toString();
481 					ContentType ct = ContentType.get(lastEE);
482 					String mimeType = ct.getMimeType();
483 					EncodingEnum ctEnum = EncodingEnum.forContentType(mimeType);
484 					if (ctEnum == null) {
485 						requestSyntaxHighlighterClass = "brush: plain";
486 					} else {
487 						switch (ctEnum) {
488 						case JSON:
489 							requestSyntaxHighlighterClass = "brush: jscript";
490 							break;
491 						case XML:
492 						default:
493 							requestSyntaxHighlighterClass = "brush: xml";
494 							break;
495 						}
496 					}
497 				}
498 			}
499 			String resultSyntaxHighlighterClass;
500 			String requestUrl = lastRequest != null ? lastRequest.getURI().toASCIIString() : null;
501 			String action = client.getLastRequest() != null ? client.getLastRequest().getMethod() : null;
502 			String resultStatus = client.getLastResponse() != null ? client.getLastResponse().getStatusLine().toString() : null;
503 			String resultBody = client.getLastResponseBody();
504 
505 			HttpResponse lastResponse = client.getLastResponse();
506 			ContentType ct = lastResponse != null ? ContentType.get(lastResponse.getEntity()) : null;
507 			String mimeType = ct != null ? ct.getMimeType() : null;
508 			EncodingEnum ctEnum = EncodingEnum.forContentType(mimeType);
509 			String narrativeString = "";
510 
511 			StringBuilder resultDescription = new StringBuilder();
512 			Bundle bundle = null;
513 
514 			if (ctEnum == null) {
515 				resultSyntaxHighlighterClass = "brush: plain";
516 				resultDescription.append("Non-FHIR response");
517 			} else {
518 				switch (ctEnum) {
519 				case JSON:
520 					resultSyntaxHighlighterClass = "brush: jscript";
521 					if (returnsResource == ResultType.RESOURCE) {
522 						narrativeString = parseNarrative(ctEnum, resultBody);
523 						resultDescription.append("JSON resource");
524 					} else if (returnsResource == ResultType.BUNDLE) {
525 						resultDescription.append("JSON bundle");
526 						bundle = myCtx.newJsonParser().parseBundle(resultBody);
527 					}
528 					break;
529 				case XML:
530 				default:
531 					resultSyntaxHighlighterClass = "brush: xml";
532 					if (returnsResource == ResultType.RESOURCE) {
533 						narrativeString = parseNarrative(ctEnum, resultBody);
534 						resultDescription.append("XML resource");
535 					} else if (returnsResource == ResultType.BUNDLE) {
536 						resultDescription.append("XML bundle");
537 						bundle = myCtx.newXmlParser().parseBundle(resultBody);
538 					}
539 					break;
540 				}
541 			}
542 
543 			resultDescription.append(" (").append(resultBody.length() + " bytes)");
544 
545 			Header[] requestHeaders = lastRequest != null ? applyHeaderFilters(lastRequest.getAllHeaders()) : new Header[0];
546 			Header[] responseHeaders = lastResponse != null ? applyHeaderFilters(lastResponse.getAllHeaders()) : new Header[0];
547 
548 			theContext.setVariable("outcomeDescription", outcomeDescription);
549 			theContext.setVariable("resultDescription", resultDescription.toString());
550 			theContext.setVariable("action", action);
551 			theContext.setVariable("bundle", bundle);
552 			theContext.setVariable("resultStatus", resultStatus);
553 			theContext.setVariable("requestUrl", requestUrl);
554 			String requestBodyText = StringEscapeUtils.escapeHtml4(requestBody);
555 			theContext.setVariable("requestBody", requestBody);
556 			theContext.setVariable("requestSyntaxHighlighterClass", requestSyntaxHighlighterClass);
557 			String resultBodyText = StringEscapeUtils.escapeHtml4(resultBody);
558 			theContext.setVariable("resultBody", resultBody);
559 			theContext.setVariable("resultBodyIsLong", resultBodyText.length() > 1000);
560 			theContext.setVariable("resultSyntaxHighlighterClass", resultSyntaxHighlighterClass);
561 			theContext.setVariable("requestHeaders", requestHeaders);
562 			theContext.setVariable("responseHeaders", responseHeaders);
563 			theContext.setVariable("narrative", narrativeString);
564 			theContext.setVariable("latencyMs", latency);
565 
566 		} catch (Exception e) {
567 			ourLog.error("Failure during processing", e);
568 			theContext.getVariables().put("errorMsg", "Error during processing: " + e.getMessage());
569 		}
570 	}

584 	@Override
585 	protected void doGet(HttpServletRequest theReq, HttpServletResponse theResp) throws ServletException, IOException {
586 		if (DEBUGMODE) {
587 			myTemplateEngine.getCacheManager().clearAllCaches();
588 		}
589 
590 		try {
591 			ourLog.trace("RequestURI: {}", theReq.getPathInfo());
592 
593 			String resName = theReq.getPathInfo().substring(1);
594 			if (myStaticResources.containsKey(resName)) {
595 				streamResponse(resName, myStaticResources.get(resName), theResp);
596 				return;
597 			}
598 
599 			IGenericClient client = myCtx.newRestfulGenericClient(myServerBase);
600 			Conformance conformance = client.conformance();
601 
602 			WebContext ctx = new WebContext(theReq, theResp, theReq.getServletContext(), theReq.getLocale());
603 
604 			Map<String, Number> resourceCounts = new HashMap<String, Number>();
605 			long total = 0;
606 			for (Rest nextRest : conformance.getRest()) {
607 				for (RestResource nextResource : nextRest.getResource()) {
608 					List<ExtensionDt> exts = nextResource.getUndeclaredExtensionsByUrl(RESOURCE_COUNT_EXT_URL);
609 					if (exts != null && exts.size() > 0) {
610 						Number nextCount = ((DecimalDt) (exts.get(0).getValue())).getValueAsNumber();
611 						resourceCounts.put(nextResource.getType().getValue(), nextCount);
612 						total += nextCount.longValue();
613 					}
614 				}
615 			}
616 			if (total > 0) {
617 				for (Rest nextRest : conformance.getRest()) {
618 					Collections.sort(nextRest.getResource(), new Comparator<RestResource>() {
619 						@Override
620 						public int compare(RestResource theO1, RestResource theO2) {
621 							DecimalDt count1 = new DecimalDt();
622 							List<ExtensionDt> count1exts = theO1.getUndeclaredExtensionsByUrl(RESOURCE_COUNT_EXT_URL);
623 							if (count1exts != null && count1exts.size() > 0) {
624 								count1 = (DecimalDt) count1exts.get(0).getValue();
625 							}
626 							DecimalDt count2 = new DecimalDt();
627 							List<ExtensionDt> count2exts = theO2.getUndeclaredExtensionsByUrl(RESOURCE_COUNT_EXT_URL);
628 							if (count2exts != null && count2exts.size() > 0) {
629 								count2 = (DecimalDt) count2exts.get(0).getValue();
630 							}
631 							int retVal = count2.compareTo(count1);
632 							if (retVal == 0) {
633 								retVal = theO1.getType().getValue().compareTo(theO2.getType().getValue());
634 							}
635 							return retVal;
636 						}
637 					});
638 				}
639 			}
640 		
641 			
642 			ctx.setVariable("resourceCounts", resourceCounts);
643 			ctx.setVariable("conf", conformance);
644 			ctx.setVariable("base", myServerBase);
645 			String resourceName = defaultString(theReq.getParameter(PARAM_RESOURCE));
646 			ctx.setVariable("resourceName", resourceName);
647 			ctx.setVariable("jsonEncodedConf", myCtx.newJsonParser().encodeResourceToString(conformance));
648 			addStandardVariables(ctx, theReq.getParameterMap());
649 
650 			if (isNotBlank(theReq.getParameter("action"))) {
651 				processAction(theReq, ctx);
652 			}
653 
654 			if (isNotBlank(resourceName)) {
655 				RuntimeResourceDefinition def = myCtx.getResourceDefinition(resourceName);
656 				TreeSet<String> includes = new TreeSet<String>();
657 				for (RuntimeSearchParam nextSpDef : def.getSearchParams()) {
658 					if (nextSpDef.getParamType() != SearchParamTypeEnum.REFERENCE) {
659 						continue;
660 					}
661 
662 					String nextPath = nextSpDef.getPath();
663 					includes.add(nextPath);
664 				}
665 				ctx.setVariable("includes", includes);
666 				
667 				if (isNotBlank(theReq.getParameter("update-id"))) {
668 					String updateId = theReq.getParameter("update-id");
669 					String updateVid = defaultIfEmpty(theReq.getParameter("update-vid"),null);
670 					IResource updateResource = client.read(def.getImplementingClass(), new IdDt(resourceName, updateId, updateVid));
671 					EncodingEnum encoding = getRequestEncoding(theReq);
672 					if (encoding==null) {
673 						encoding=EncodingEnum.XML;
674 					}
675 					String updateResourceString = encoding.newParser(myCtx).setPrettyPrint(true).encodeResourceToString(updateResource);
676 					ctx.setVariable("updateResource", updateResourceString);
677 					ctx.setVariable("updateResourceId", updateId);
678 				}
679 
680 			}
681 
682 			theResp.setContentType("text/html");
683 			theResp.setCharacterEncoding("UTF-8");
684 
685 			myTemplateEngine.process(theReq.getPathInfo(), ctx, theResp.getWriter());
686 		} catch (Exception e) {
687 			ourLog.error("Failed to respond", e);
688 			theResp.sendError(500, e.getMessage());
689 		}
690 	}

703 	private Header[] applyHeaderFilters(Header[] theAllHeaders) {
704 		if (myFilterHeaders == null || myFilterHeaders.isEmpty()) {
705 			return theAllHeaders;
706 		}
707 		ArrayList<Header> retVal = new ArrayList<Header>();
708 		for (Header next : theAllHeaders) {
709 			if (!myFilterHeaders.contains(next.getName().toLowerCase())) {
710 				retVal.add(next);
711 			}
712 		}
713 		return retVal.toArray(new Header[retVal.size()]);
714 	}
