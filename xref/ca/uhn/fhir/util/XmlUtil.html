198 	private static XMLInputFactory getOrCreateInputFactory() throws FactoryConfigurationError {
199 		if (ourInputFactory == null) {
200 			XMLInputFactory inputFactory = XMLInputFactory.newInstance();
201 
202 			if (!ourHaveLoggedStaxImplementation) {
203 				logStaxImplementation(inputFactory.getClass());
204 			}
205 
206 			/*
207 			 * In the following few lines, you can uncomment the first and comment the second to disable automatic
208 			 * parsing of extended entities, e.g. &sect;
209 			 * 
210 			 * Note that these properties are Woodstox specific and they cause a crash in environments where SJSXP is
211 			 * being used (e.g. glassfish) so we don't set them there.
212 			 */
213 
214 			if (inputFactory instanceof com.ctc.wstx.stax.WstxInputFactory) {
215 				// inputFactory.setProperty(WstxInputFactory.IS_REPLACING_ENTITY_REFERENCES, false);
216 				inputFactory.setProperty(WstxInputProperties.P_UNDECLARED_ENTITY_RESOLVER, XML_RESOLVER);
217 			}
218 			ourInputFactory = inputFactory;
219 		}
220 		return ourInputFactory;
221 	}

223 	private static void logStaxImplementation(Class<?> theClass) {
224 		try {
225 			URL rootUrl = getRootUrlForClass(theClass);
226 			if (rootUrl == null) {
227 				ourLog.info("Unable to determine location of StAX implementation containing class");
228 			} else {
229 				Manifest manifest;
230 				URL metaInfUrl = new URL(rootUrl, "META-INF/MANIFEST.MF");
231 				InputStream is = metaInfUrl.openStream();
232 				try {
233 					manifest = new Manifest(is);
234 				} finally {
235 					is.close();
236 				}
237 				Attributes attrs = manifest.getMainAttributes();
238 				String title = attrs.getValue(IMPLEMENTATION_TITLE);
239 				String symbolicName = attrs.getValue(BUNDLE_SYMBOLIC_NAME);
240 				if (symbolicName != null) {
241 					int i = symbolicName.indexOf(';');
242 					if (i != -1) {
243 						symbolicName = symbolicName.substring(0, i);
244 					}
245 				}
246 				String vendor = attrs.getValue(IMPLEMENTATION_VENDOR);
247 				if (vendor == null) {
248 					vendor = attrs.getValue(BUNDLE_VENDOR);
249 				}
250 				String version = attrs.getValue(IMPLEMENTATION_VERSION);
251 				if (version == null) {
252 					version = attrs.getValue(BUNDLE_VERSION);
253 				}
254 				if (ourLog.isDebugEnabled()) {
255 					ourLog.debug("FHIR XML procesing will use StAX implementation at {}\n  Title:         {}\n  Symbolic name: {}\n  Vendor:        {}\n  Version:       {}", rootUrl, title, symbolicName, vendor, version);
256 				} else {
257 					ourLog.info("FHIR XML procesing will use StAX implementation '{}' version '{}'", title, version);
258 				}
259 			}
260 		} catch (Throwable e) {
261 			ourLog.info("Unable to determine StAX implementation: " + e.getMessage());
262 		} finally {
263 			ourHaveLoggedStaxImplementation = true;
264 		}
265 	}

267 	private static URL getRootUrlForClass(Class<?> cls) {
268 		ClassLoader classLoader = cls.getClassLoader();
269 		String resource = cls.getName().replace('.', '/') + ".class";
270 		if (classLoader == null) {
271 			// A null class loader means the bootstrap class loader. In this case we use the
272 			// system class loader. This is safe since we can assume that the system class
273 			// loader uses parent first as delegation policy.
274 			classLoader = ClassLoader.getSystemClassLoader();
275 		}
276 		URL url = classLoader.getResource(resource);
277 		if (url == null) {
278 			return null;
279 		}
280 		String file = url.getFile();
281 		if (file.endsWith(resource)) {
282 			try {
283 				return new URL(url.getProtocol(), url.getHost(), url.getPort(), file.substring(0, file.length() - resource.length()));
284 			} catch (MalformedURLException ex) {
285 				return null;
286 			}
287 		} else {
288 			return null;
289 		}
290 	}

308 	private static XMLOutputFactory getOrCreateOutputFactory() throws FactoryConfigurationError {
309 		if (ourOutputFactory == null) {
310 			XMLOutputFactory outputFactory = XMLOutputFactory.newInstance();
311 
312 			if (!ourHaveLoggedStaxImplementation) {
313 				logStaxImplementation(outputFactory.getClass());
314 			}
315 
316 			/*
317 			 * Note that these properties are Woodstox specific and they cause a crash in environments where SJSXP is
318 			 * being used (e.g. glassfish) so we don't set them there.
319 			 */
320 			if (outputFactory instanceof WstxOutputFactory) {
321 				outputFactory.setProperty(XMLOutputFactory2.P_TEXT_ESCAPER, new MyEscaper());
322 			}
323 			ourOutputFactory = outputFactory;
324 		}
325 		return ourOutputFactory;
326 	}

328 	public static class MyEscaper implements EscapingWriterFactory {
329 
330 		@Override
331 		public Writer createEscapingWriterFor(OutputStream theOut, String theEnc) throws UnsupportedEncodingException {
332 			return createEscapingWriterFor(new OutputStreamWriter(theOut), theEnc);
333 		}
334 
335 		@Override
336 		public Writer createEscapingWriterFor(final Writer theW, String theEnc) throws UnsupportedEncodingException {
337 			return new Writer() {
338 
339 				@Override
340 				public void close() throws IOException {
341 					theW.close();
342 				}
343 
344 				@Override
345 				public void flush() throws IOException {
346 					theW.flush();
347 				}
348 
349 				@Override
350 				public void write(char[] theCbuf, int theOff, int theLen) throws IOException {
351 					boolean hasEscapable = false;
352 					for (int i = 0; i < theLen && !hasEscapable; i++) {
353 						char nextChar = theCbuf[i + theOff];
354 						switch (nextChar) {
355 						case '<':
356 						case '>':
357 						case '"':
358 						case '&':
359 							hasEscapable = true;
360 						}
361 					}
362 
363 					if (!hasEscapable) {
364 						theW.write(theCbuf, theOff, theLen);
365 						return;
366 					}
367 
368 					String escaped = StringEscapeUtils.escapeXml10(new String(theCbuf, theOff, theLen));
369 					theW.write(escaped.toCharArray());
370 				}
371 			};
372 		}

376 	private static final class ExtendedEntityReplacingXmlResolver implements XMLResolver {
377 		@Override
378 		public Object resolveEntity(String thePublicID, String theSystemID, String theBaseURI, String theNamespace) throws XMLStreamException {
379 			if (thePublicID == null && theSystemID == null) {
380 				if (theNamespace != null && VALID_ENTITY_NAMES.containsKey(theNamespace)) {
381 					return new String(Character.toChars(VALID_ENTITY_NAMES.get(theNamespace)));
382 				}
383 			}
384 
385 			return null;
386 		}
